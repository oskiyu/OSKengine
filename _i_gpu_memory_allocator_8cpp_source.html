170
171 if (info.sharedType == GpuSharedMemoryType::GPU_AND_CPU
172 && EFTraits::HasFlag(info.usage, GpuBufferUsage::UPLOAD_ONLY)) {
173
174 DynamicArray<UIndex64> freeableIndices{};
175
176 for (UIndex64 b = 0; b < bufferMemoryBlocks[i].GetSize(); b++) {

178
179 if (block->IsUnused())

182

194void IGpuMemoryAllocator::LoadDefaultNormalTexture() {
195 IRenderer* renderer = Engine::GetRenderer();
196 OwnedPtr<ICommandList> uploadCmdList = renderer->CreateSingleUseCommandList(GpuQueueType::ASYNC_TRANSFER);
197 uploadCmdList->Start();
198
199 TByte data[4] = {
200 127, // R: x
201 127, // G: y
202 255, // B: z
203 255 // A
204 };
205
206 OwnedPtr<GpuBuffer> stagingBuffer = CreateStagingBuffer(sizeof(data), uploadCmdList->GetCompatibleQueueType());
207 stagingBuffer->MapMemory();
208 stagingBuffer->Write(data, sizeof(data));
209 stagingBuffer->Unmap();
210
211 GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(
212 Vector2ui(1, 1),
213 Format::RGBA8_UNORM,
214 GpuImageUsage::SAMPLED | GpuImageUsage::TRANSFER_DESTINATION);
215 imageInfo.samplerDesc.mipMapMode = GpuImageMipmapMode::NONE;
216 imageInfo.samplerDesc.addressMode = GpuImageAddressMode::REPEAT;
217 m_defaultNormalTexture = CreateImage(imageInfo).GetPointer();
218
219 uploadCmdList->SetGpuImageBarrier(
220 m_defaultNormalTexture.GetPointer(),
221 GpuImageLayout::TRANSFER_DESTINATION,
222 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
223
224 uploadCmdList->CopyBufferToImage(stagingBuffer.GetValue(), m_defaultNormalTexture.GetPointer());
225
226 uploadCmdList->SetGpuImageBarrier(
227 m_defaultNormalTexture.GetPointer(),
228 GpuImageLayout::SAMPLED,
229 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE));
230
231 uploadCmdList->Close();
232
233 renderer->SubmitSingleUseCommandList(uploadCmdList);
234}
