54void CameraComponent3D::Rotate(float angleX, float angleY) {
55 const Vector2f prevAccumAngles = accumulatedAngles;
56 angles = Vector2f(angleX, angleY) * 0.25f;
57 accumulatedAngles += angles;
58
59
60 const static float middleAngle = 0.0f;
61 const static float diffAngle = 180.0f / 2;
62
63 if (accumulatedAngles.y > middleAngle + diffAngle)
64 angles.y = middleAngle + diffAngle - prevAccumAngles.y;
65 if (accumulatedAngles.y < middleAngle - diffAngle)
66 angles.y = middleAngle - diffAngle - prevAccumAngles.y;
67
68 accumulatedAngles = prevAccumAngles + Vector2f(angles.x, angles.y);
69}

78AnyFrustum CameraComponent3D::GetFrustum(const Transform3D& transform) const {
79 AnyFrustum output{};
80
81 Vector2f nearSize_half = Vector2f::Zero;
82
83 nearSize_half.y = nearPlane * glm::tan(glm::radians(fov) * 0.5f);
84 nearSize_half.x = nearSize_half.y * Engine::GetDisplay()->GetScreenRatio();
85
86 const Vector3f up = transform.GetTopVector().GetNormalized();
87 const Vector3f right = transform.GetRightVector().GetNormalized();
88 const Vector3f forward = transform.GetForwardVector().GetNormalized();
89
90 const Vector3f nearLeftPosition = forward * nearPlane - right * nearSize_half.x;
91 const Vector3f nearRightPosition = forward * nearPlane + right * nearSize_half.x;
92
93 const Vector3f nearTopPosition = forward * nearPlane + up * nearSize_half.y;
94 const Vector3f nearBottomPosition = forward * nearPlane - up * nearSize_half.y;
95
96 // Near
97 output.Insert({
98 .point = transform.GetPosition() + forward * nearPlane,
99 .normal = forward
100 });
101
102 // Left
103 output.Insert({
104 .point = transform.GetPosition(),
105 .normal = -up.Cross(nearLeftPosition).GetNormalized()
106 });
107
108 // Right
109 output.Insert({
110 .point = transform.GetPosition(),
111 .normal = up.Cross(nearRightPosition).GetNormalized()
112 });
113
114 // Top
115 output.Insert({
116 .point = transform.GetPosition(),
117 .normal = -right.Cross(nearTopPosition).GetNormalized()
118 });
119
120 // Bottom
121 output.Insert({
122 .point = transform.GetPosition(),
123 .normal = right.Cross(nearBottomPosition).GetNormalized()
124 });
125
126 return output;
127}
