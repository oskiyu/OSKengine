73void BloomPass::SetupMaterialInstances() {
74 GpuImageViewConfig sourceViewConfig = GpuImageViewConfig::CreateSampled_MipLevelRanged(0, 0);
75 GpuImageViewConfig destViewConfig = GpuImageViewConfig::CreateStorage_Default();
76
77 const UIndex32 lowestLevel = GetNumPasses();
78
79 for (UIndex32 pass = 0; pass < lowestLevel - 1; pass++) {
80
81 sourceViewConfig.baseMipLevel = pass;
82 sourceViewConfig.topMipLevel = pass;
83
84 destViewConfig.baseMipLevel = pass + 1;
85 destViewConfig.topMipLevel = pass + 1;
86
87 const auto& sourceView = *GetOutput().GetTargetImage()->GetView(sourceViewConfig);
88 const auto& targetView = *GetOutput().GetTargetImage()->GetView(destViewConfig);
89
90 m_downscalingMaterialInstances[pass]->GetSlot("texture")->SetGpuImage("inputImg", sourceView);
91 m_downscalingMaterialInstances[pass]->GetSlot("texture")->SetStorageImage("outputImg", targetView);
92 m_downscalingMaterialInstances[pass]->GetSlot("texture")->FlushUpdate();
93 }
94
95 for (UIndex32 pass = lowestLevel - 2; pass > 0; pass--) {
96
97 sourceViewConfig.baseMipLevel = pass + 1;
98 sourceViewConfig.topMipLevel = pass + 1;
99
100 destViewConfig.baseMipLevel = pass;
101 destViewConfig.topMipLevel = pass;
102
103 const auto& sourceView = GetOutput().GetTargetImage()->GetView(sourceViewConfig);
104 const auto& targetView = GetOutput().GetTargetImage()->GetView(destViewConfig);
105
106 m_upscalingMaterialInstances[pass]->GetSlot("texture")->SetGpuImage("inputImg", *sourceView);
107 m_upscalingMaterialInstances[pass]->GetSlot("texture")->SetStorageImage("outputImg", *targetView);
108 m_upscalingMaterialInstances[pass]->GetSlot("texture")->FlushUpdate();
109 }
110
111 // Resolve
112 sourceViewConfig.baseMipLevel = 1;
113 sourceViewConfig.topMipLevel = 1;
114
115 destViewConfig.baseMipLevel = 0;
116 destViewConfig.topMipLevel = 0;
117
118 const auto& sourceView = *GetOutput().GetTargetImage()->GetView(sourceViewConfig);
119 const auto& sceneView = *m_inputImage->GetView(GpuImageViewConfig::CreateSampled_SingleMipLevel(0));
120 const auto& targetView = *GetOutput().GetTargetImage()->GetView(destViewConfig);
121
122 m_resolveInstance->GetSlot("texture")->SetGpuImage("bloomImg", sourceView);
123 m_resolveInstance->GetSlot("texture")->SetGpuImage("sceneImg", sceneView);
124 m_resolveInstance->GetSlot("texture")->SetStorageImage("outputImg", targetView);
125 m_resolveInstance->GetSlot("texture")->FlushUpdate();
126}

145void BloomPass::DownscaleBloom(GRAPHICS::ICommandList* computeCmdList) {
146 computeCmdList->StartDebugSection("Downscale", Color::Purple);
147
148 computeCmdList->BindMaterial(*m_downscaleMaterial);
149 for (UIndex32 i = 0; i < GetNumPasses() - 1; i++) {
150 computeCmdList->BindMaterialSlot(*m_downscalingMaterialInstances[i]->GetSlot("texture"));
151
152 if (i == 0)
153 computeCmdList->PushMaterialConstants("info", 1);
154 else
155 computeCmdList->PushMaterialConstants("info", 0);
156
157 const UIndex32 sourceMipLevel = i;
158 const UIndex32 destinationMipLevel = sourceMipLevel + 1;
159
160 // Nivel anterior debe ser SAMPLED.
161 computeCmdList->SetGpuImageBarrier(
162 GetOutput().GetTargetImage(),
163 GpuImageLayout::SAMPLED,
164 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ),
165 { .baseLayer = 0, .numLayers = 1, .baseMipLevel = sourceMipLevel, .numMipLevels = 1 });
166
167 // Nivel siguiente ser√° escrito.
168 // Al no haber sido usado previamente en este frame,
169 // layout inicial es UNDEFINED y barrier es NONE.
170 computeCmdList->SetGpuImageBarrier(
171 GetOutput().GetTargetImage(),
172 GpuImageLayout::UNDEFINED,
173 GpuImageLayout::GENERAL,
174 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
175 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE),
176 { .baseLayer = 0, .numLayers = 1, .baseMipLevel = destinationMipLevel, .numMipLevels = 1 });
177
178 ExecuteSinglePass(computeCmdList, sourceMipLevel, destinationMipLevel);
179 }
180
181 computeCmdList->EndDebugSection();
182}

215void BloomPass::InitialCopy(ICommandList* computeCmdList) {
216 computeCmdList->AddDebugMarker("Source Copy", Color::Yellow);
217
218 // La imagen de la escena se copia a la cadena de mipmaps:
219 // debe estar en transfer source.
220 computeCmdList->SetGpuImageBarrier(
221 m_inputImage,
222 GpuImageLayout::TRANSFER_SOURCE,
223 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
224
225 // La imagen de destino (mip 0) debe ser transfer dest. para efectuar la copia.
226 // Al no haber sido usada previamente, el layout original es UNDEFINED y
227 // barrier info es NONE.
228 computeCmdList->SetGpuImageBarrier(
229 GetOutput().GetTargetImage(),
230 GpuImageLayout::UNDEFINED,
231 GpuImageLayout::TRANSFER_DESTINATION,
232 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
233 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
234 { .baseLayer = 0, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = 1 });
235
236 const Vector2ui imgSize = m_inputImage->GetSize2D();
237 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(imgSize);
238 copyInfo.destinationMipLevel = 0;

240 computeCmdList->RawCopyImageToImage(

242 GetOutput().GetTargetImage(),

271 static_cast<UIndex32>(glm::ceil(GetOutput().GetSize().y / BLOCK_SIZE))
272 };

276 computeCmdList->DispatchCompute({ dispatchRes.x, dispatchRes.y, 1 });
