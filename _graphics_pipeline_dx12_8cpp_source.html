39 layout = new PipelineLayoutDx12(materialLayout);

41 D3D12_GRAPHICS_PIPELINE_STATE_DESC createInfo{};
42
43 auto layoutDesc = GetInputLayoutDescDx12(vertexInfo);

65 createInfo.HS = tesselationControlShader.bytecode;
66
67 topologyType = D3D_PRIMITIVE_TOPOLOGY_3_CONTROL_POINT_PATCHLIST;
68 }
69 if (info.tesselationEvaluationPath != "") {

71 createInfo.DS = tesselationEvaluationShader.bytecode;
72 }
73
74 createInfo.RasterizerState = GetRasterizerDesc(info);
75 createInfo.BlendState = GetBlendDesc(info);
76 createInfo.DepthStencilState = GetDepthStencilDesc(info);
77 createInfo.SampleMask = UINT_MAX;

79 createInfo.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_PATCH;

81 createInfo.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
82
83 createInfo.NumRenderTargets = static_cast<UINT>(info.formats.GetSize());
84 for (UIndex32 i = 0; i < info.formats.GetSize(); i++)
85 createInfo.RTVFormats[i] = GetFormatDx12(info.formats[i]);

88 createInfo.SampleDesc.Count = 1;

169DXGI_FORMAT GraphicsPipelineDx12::GetVertexAttribFormatDx12(const VertexInfo::Entry& entry) const {
170 switch (entry.type) {
171
172 case VertexInfo::Entry::Type::INT:
173 if (entry.size == 2 * sizeof(int)) return DXGI_FORMAT_R32G32_SINT;
174 if (entry.size == 3 * sizeof(int)) return DXGI_FORMAT_R32G32B32_SINT;
175 if (entry.size == 4 * sizeof(int)) return DXGI_FORMAT_R32G32B32A32_SINT;
176 break;
177
178 case VertexInfo::Entry::Type::FLOAT:
179 if (entry.size == 2 * sizeof(float)) return DXGI_FORMAT_R32G32_FLOAT;
180 if (entry.size == 3 * sizeof(float)) return DXGI_FORMAT_R32G32B32_FLOAT;
181 if (entry.size == 4 * sizeof(float)) return DXGI_FORMAT_R32G32B32A32_FLOAT;
182 break;
183 }
184
185 OSK_ASSERT(false, NotImplementedException());
186 return DXGI_FORMAT_UNKNOWN;
187}

189DynamicArray<D3D12_INPUT_ELEMENT_DESC> GraphicsPipelineDx12::GetInputLayoutDescDx12(const VertexInfo& info) const {
190 DynamicArray<D3D12_INPUT_ELEMENT_DESC> output{};
191
192 USize64 offset = 0;
193 for (UIndex32 i = 0; i < info.entries.GetSize(); i++) {
194 const VertexInfo::Entry& entry = info.entries.At(i);
195
196 D3D12_INPUT_ELEMENT_DESC desc = {
197 NULL,
198 0,
199 GetVertexAttribFormatDx12(entry),
200 0,
201 offset,
202 D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA,
203 0
204 };
205
206 output.Insert(desc);
207
208 offset += entry.size;
209 }
210
211 return output;
