167DXGI_FORMAT GraphicsPipelineDx12::GetVertexAttribFormatDx12(const VertexInfo::Entry& entry) const {
168 switch (entry.type) {
169
170 case VertexInfo::Entry::Type::INT:
171 if (entry.size == 2 * sizeof(int)) return DXGI_FORMAT_R32G32_SINT;
172 if (entry.size == 3 * sizeof(int)) return DXGI_FORMAT_R32G32B32_SINT;
173 if (entry.size == 4 * sizeof(int)) return DXGI_FORMAT_R32G32B32A32_SINT;
174 break;
175
176 case VertexInfo::Entry::Type::FLOAT:
177 if (entry.size == 2 * sizeof(float)) return DXGI_FORMAT_R32G32_FLOAT;
178 if (entry.size == 3 * sizeof(float)) return DXGI_FORMAT_R32G32B32_FLOAT;
179 if (entry.size == 4 * sizeof(float)) return DXGI_FORMAT_R32G32B32A32_FLOAT;
180 break;
181 }
182
183 OSK_ASSERT(false, NotImplementedException());
184 return DXGI_FORMAT_UNKNOWN;
185}
