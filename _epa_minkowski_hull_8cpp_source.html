17OSK::Vector3f GetBarycentricCoordinates(const OSK::Vector3f& p, const OSK::Vector3f& a, const OSK::Vector3f& b, const OSK::Vector3f& c) {
18 auto output = OSK::Vector3f::Zero;
19
20 auto v0 = b - a, v1 = c - a, v2 = p - a;
21 float d00 = v0.Dot(v0);
22 float d01 = v0.Dot(v1);
23 float d11 = v1.Dot(v1);
24 float d20 = v2.Dot(v0);
25 float d21 = v2.Dot(v1);
26 float denom = d00 * d11 - d01 * d01;
27
28 output.y = (d11 * d20 - d01 * d21) / denom;
29 output.z = (d00 * d21 - d01 * d20) / denom;
30 output.x = 1.0f - output.y - output.z;
31
32 return output;
33}
