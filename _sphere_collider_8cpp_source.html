63RayCastResult SphereCollider::CastRay(const Ray& ray) const {
64 // https://www.lighthouse3d.com/tutorials/maths/ray-sphere-intersection/
65
66 // ray.origin -> position
67 Vector3f originsVec = Vector3f::Zero;
68 if (GetBroadOwner().has_value()) {
69 originsVec = GetBroadOwner().value()->GetPosition();
70 }
71 else {
72 originsVec = GetNarrowOwner().value()->GetTransform().GetPosition();
73 }
74
75 // La esfera está detras del rayo
76 if (ray.direction.Dot(originsVec) < 0.0f) {
77 return RayCastResult::False();
78 }
79
80 const float centerProjectionDistance = ray.direction.Dot(originsVec);
81 const Vector3f centerProjection = ray.origin + ray.direction * centerProjectionDistance;
82
83 const float distance = centerProjection.GetDistanceTo(originsVec);
84 if (distance < m_radius)
85 return RayCastResult::False();
86
87 const float distanceToIntersection = centerProjection.GetDistanceTo(ray.origin)
88 - glm::sqrt(m_radius * m_radius - distance * distance);
89
90 return RayCastResult::True(ray.origin + ray.direction * distanceToIntersection, ECS::EMPTY_GAME_OBJECT);
91}
