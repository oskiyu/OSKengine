28SpecularMapLoader::SpecularMapLoader() {
29 generationMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_gen.json");
30 prefilterMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_prefilter.json");
31 lutGenerationMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_lut_gen.json");
32
33 generationMaterialInstance = generationMaterial->CreateInstance().GetPointer();
34 prefilterMaterialInstance = prefilterMaterial->CreateInstance().GetPointer();
35 lutGenerationMaterialInstance = lutGenerationMaterial->CreateInstance().GetPointer();
36
37 RenderTargetAttachmentInfo colorInfo{ .format = Format::RGBA16_SFLOAT, .usage = GpuImageUsage::TRANSFER_SOURCE, .name = "Cubemap Render Color" };
38 RenderTargetAttachmentInfo depthInfo{ .format = Format::D16_UNORM, .name = "Cubemap Depth Color" };
39 cubemapRenderTarget.Create(maxResolution, { colorInfo }, depthInfo);
40
41 RenderTargetAttachmentInfo lutInfo{ .format = Format::RG16_SFLOAT, .usage = GpuImageUsage::TRANSFER_SOURCE, .name = "Specular Look-Up Table Generation" };
42 lookUpTable.Create(maxResolution, lutInfo);
43
44 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateStorage_Default();
45 const IGpuImageView* lutGenImage = lookUpTable.GetTargetImage()->GetView(viewConfig);
46
47 lutGenerationMaterialInstance->GetSlot("global")->SetStorageImage("finalImage", lutGenImage);
48 lutGenerationMaterialInstance->GetSlot("global")->FlushUpdate();
49

53void SpecularMapLoader::Load(const std::string& assetFilePath, SpecularMap* asset) {
54 // Asset file.
55 nlohmann::json assetInfo;
56 try {
57 assetInfo = ValidateDescriptionFile(assetFilePath);
58 }
59 catch (const EngineException& e) { throw e; }
60
61
62 std::string texturePath = assetInfo["raw_asset_path"];
63 asset->SetName(assetInfo["name"]);
64
65 // Original texture loading
66 int width = 0;
67 int height = 0;
68 int numChannels = 0;
69
70 const float* pixels = stbi_loadf(texturePath.c_str(), &width, &height, &numChannels, 4);
71
72 const Vector2ui originalImageSize = {
73 (uint32_t)width,
74 (uint32_t)height
75 };
76
77
78 // Carga de la imagen HDR original
79 const GpuImageUsage originalImageUsage = GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED;
80 const GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(originalImageSize, Format::RGBA32_SFLOAT, originalImageUsage);
81 UniquePtr<GpuImage> originalImage = Engine::GetRenderer()->GetAllocator()->CreateImage(imageInfo).GetPointer();
82 originalImage->SetDebugName("Original Specular 2D");
83
84 UploadImage(originalImage.GetPointer(), pixels, { originalImageSize.x, originalImageSize.y, 1 });
85
86 // Creación del cubemap.
87 // Cada mip level representa el reflejo para una rugosidad específica.
88 const UIndex32 maxMipLevel = 5;
89
90 GpuImageSamplerDesc origianlSampler{};
91 origianlSampler.mipMapMode = GpuImageMipmapMode::NONE;
92
93 GpuImageSamplerDesc prefilterSampler{};
94 prefilterSampler.mipMapMode = GpuImageMipmapMode::CUSTOM;
95 prefilterSampler.minMipLevel = 0;
96 prefilterSampler.maxMipLevel = maxMipLevel;
97 prefilterSampler.filteringType = GpuImageFilteringType::LIENAR;
98
99 UniquePtr<GpuImage> originalCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
100 maxResolution,
101 Format::RGBA16_SFLOAT,
102 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
103 GpuSharedMemoryType::GPU_ONLY,
104 origianlSampler).GetPointer();
105
106 OwnedPtr<GpuImage> targetCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
107 maxResolution,
108 Format::RGBA16_SFLOAT,
109 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
110 GpuSharedMemoryType::GPU_ONLY,
111 prefilterSampler);
112
113 originalCubemap->SetDebugName("Original Specular Cubemap");
114 targetCubemap->SetDebugName("Prefiltered Specular Cubemap");
115
116 generationMaterialInstance->GetSlot("global")->SetGpuImage("image", originalImage->GetView(GpuImageViewConfig::CreateSampled_SingleMipLevel(0)));
117 prefilterMaterialInstance->GetSlot("global")->SetGpuImage("image", originalCubemap->GetView(GpuImageViewConfig::CreateSampled_Cubemap()));
118
119 generationMaterialInstance->GetSlot("global")->FlushUpdate();
120 prefilterMaterialInstance->GetSlot("global")->FlushUpdate();
121
122 // Renderizar a cubemap
123 OwnedPtr<ICommandList> originalDrawCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
124 originalDrawCmdList->SetDebugName("Specular Cubemap Cmd List");
125
126 originalDrawCmdList->Reset();
127 originalDrawCmdList->Start();
128
129 DrawOriginal(originalCubemap.GetPointer(), originalDrawCmdList.GetPointer());
130
131 originalDrawCmdList->Close();
132
133 Engine::GetRenderer()->SubmitSingleUseCommandList(originalDrawCmdList);
134
135 // Preprocesado
136
137 OwnedPtr<ICommandList> prefilterCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
138 prefilterCmdList->SetDebugName("Prefilter and LUT generation");
139
140 prefilterCmdList->Reset();
141 prefilterCmdList->Start();
142
143 for (UIndex32 mipLevel = 0; mipLevel < maxMipLevel; mipLevel++) {
144 const float roughness = (float)mipLevel / (maxMipLevel - 1.0f);
145 DrawPreFilter(targetCubemap.GetPointer(), prefilterCmdList.GetPointer(), mipLevel, roughness);
146 }
147
148 // LUT
149 GenerateLut(prefilterCmdList.GetPointer());
150 GpuImageCreateInfo lutCreateInfo = GpuImageCreateInfo::CreateDefault2D(
151 lookUpTable.GetSize(),
152 lookUpTable.GetTargetImage()->GetFormat(),
153 GpuImageUsage::SAMPLED | GpuImageUsage::TRANSFER_DESTINATION);
154 OwnedPtr<GpuImage> lut = Engine::GetRenderer()->GetAllocator()->CreateImage(lutCreateInfo);
155 lut->SetDebugName("Specular Look-Up Table");
156
157 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::TRANSFER_DESTINATION,
158 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
159
160 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(lookUpTable.GetSize());
161 prefilterCmdList->RawCopyImageToImage(*lookUpTable.GetTargetImage(), lut.GetPointer(), copyInfo);
162
163 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::SAMPLED,
164 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SAMPLED_READ));
165
166 prefilterCmdList->Close();
167
168 Engine::GetRenderer()->SubmitSingleUseCommandList(prefilterCmdList);
169
170 asset->_SetCubemapImage(targetCubemap);
171 asset->_SetLookUpTable(lut);
172}

174void SpecularMapLoader::DrawOriginal(GRAPHICS::GpuImage* cubemap, GRAPHICS::ICommandList* cmdList) {
175 struct {
176 glm::mat4 cameraProjection = SpecularMapLoader::cameraProjection;
177 glm::mat4 cameraView = glm::mat4(1.0f);
178 } renderInfo;
179
180 for (UIndex32 faceId = 0; faceId < 6; faceId++) {
181 cmdList->BeginGraphicsRenderpass(&cubemapRenderTarget);
182
183 cmdList->BindMaterial(*generationMaterial);
184 cmdList->BindMaterialSlot(*generationMaterialInstance->GetSlot("global"));
185
186 const Viewport viewport{
187 .rectangle = {
188 0, 0,
189 (uint32_t)(cubemapRenderTarget.GetSize().x),
190 (uint32_t)(cubemapRenderTarget.GetSize().y)
191 }
192 };
193
194 cmdList->SetViewport(viewport);
195 cmdList->SetScissor(viewport.rectangle);
196
197 renderInfo.cameraView = cameraViews[faceId];
198 cmdList->PushMaterialConstants("info", renderInfo);
199
200 cmdList->BindVertexBuffer(*cubemapModel->GetVertexBuffer());
201 cmdList->BindIndexBuffer(*cubemapModel->GetIndexBuffer());
202 cmdList->DrawSingleInstance(cubemapModel->GetIndexCount());
203
204 cmdList->EndGraphicsRenderpass();
205
206 cmdList->SetGpuImageBarrier(cubemapRenderTarget.GetMainColorImage(),
207 GpuImageLayout::TRANSFER_SOURCE,
208 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
209 GpuImageRange{ .baseLayer = 0, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
210
211 cmdList->SetGpuImageBarrier(cubemap,
212 GpuImageLayout::TRANSFER_DESTINATION,
213 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
214 GpuImageRange{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
215
216 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(viewport.rectangle.GetRectangleSize());
217 copyInfo.destinationArrayLevel = faceId;
218 cmdList->RawCopyImageToImage(*cubemapRenderTarget.GetMainColorImage(), cubemap, copyInfo);
219
220 cmdList->SetGpuImageBarrier(cubemap,
221 GpuImageLayout::SAMPLED,
222 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ),
223 GpuImageRange{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
224 }

291void SpecularMapLoader::GenerateLut(ICommandList* cmdList) {
292 const auto resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
293 GpuImage* img = lookUpTable.GetTargetImage();
294
295 cmdList->SetGpuImageBarrier(img, GpuImageLayout::GENERAL,
296 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
297
298 cmdList->BindMaterial(*lutGenerationMaterial);
299 cmdList->BindMaterialSlot(*lutGenerationMaterialInstance->GetSlot("global"));
300
301 const Vector2ui imageResolution = lookUpTable.GetSize();
302 const Vector2ui threadBlockSize = { 8u, 8u };
303 const Vector2ui dispatchResolution = imageResolution / threadBlockSize;
304
305 cmdList->DispatchCompute({ dispatchResolution.x, dispatchResolution.y, 1 });
306
307 cmdList->SetGpuImageBarrier(img, GpuImageLayout::TRANSFER_SOURCE,
308 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
309}
