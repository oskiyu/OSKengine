29SpecularMapLoader::SpecularMapLoader() {
30 generationMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_gen.json");
31 prefilterMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_prefilter.json");
32 lutGenerationMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/Materials/AssetsGen/material_specular_lut_gen.json");
33
34 generationMaterialInstance = generationMaterial->CreateInstance();
35 prefilterMaterialInstance = prefilterMaterial->CreateInstance();
36 lutGenerationMaterialInstance = lutGenerationMaterial->CreateInstance();
37
38 RenderTargetAttachmentInfo colorInfo{ .format = Format::RGBA16_SFLOAT, .usage = GpuImageUsage::TRANSFER_SOURCE, .name = "Cubemap Render Color" };
39 RenderTargetAttachmentInfo depthInfo{ .format = Format::D16_UNORM, .name = "Cubemap Depth Color" };
40 cubemapRenderTarget.Create(maxResolution, { colorInfo }, depthInfo);
41
42 RenderTargetAttachmentInfo lutInfo{ .format = Format::RG16_SFLOAT, .usage = GpuImageUsage::TRANSFER_SOURCE, .name = "Specular Look-Up Table Generation" };
43 lookUpTable.Create(maxResolution, lutInfo);
44
45 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateStorage_Default();
46 const IGpuImageView* lutGenImage = lookUpTable.GetTargetImage()->GetView(viewConfig);
47
48 lutGenerationMaterialInstance->GetSlot("global")->SetStorageImage("finalImage", *lutGenImage);
49 lutGenerationMaterialInstance->GetSlot("global")->FlushUpdate();
50
51 cubemapModel = Engine::GetAssetManager()->Load<ASSETS::Model3D>("Resources/Assets/Models/cube.json");
52}

54void SpecularMapLoader::Load(const std::string& assetFilePath, SpecularMap* asset) {
55 // Asset file.
56 nlohmann::json assetInfo;
57 try {
58 assetInfo = ValidateDescriptionFile(assetFilePath);
59 }
60 catch (const EngineException& e) { throw e; }
61
62
63 std::string texturePath = assetInfo["raw_asset_path"];
64 asset->SetName(assetInfo["name"]);
65
66 // Original texture loading
67 int width = 0;
68 int height = 0;
69 int numChannels = 0;
70
71 const float* pixels = stbi_loadf(texturePath.c_str(), &width, &height, &numChannels, 4);
72
73 const Vector2ui originalImageSize = {
74 (uint32_t)width,
75 (uint32_t)height
76 };
77
78
79 // Carga de la imagen HDR original
80 const GpuImageUsage originalImageUsage = GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED;
81 const GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(originalImageSize, Format::RGBA32_SFLOAT, originalImageUsage);
82 UniquePtr<GpuImage> originalImage = Engine::GetRenderer()->GetAllocator()->CreateImage(imageInfo);
83 originalImage->SetDebugName("Original Specular 2D");
84
85 UploadImage(originalImage.GetPointer(), pixels, { originalImageSize.x, originalImageSize.y, 1 });
86
87 // Creación del cubemap.
88 // Cada mip level representa el reflejo para una rugosidad específica.
89 const UIndex32 maxMipLevel = 5;
90
91 GpuImageSamplerDesc origianlSampler{};
92 origianlSampler.mipMapMode = GpuImageMipmapMode::NONE;
93
94 GpuImageSamplerDesc prefilterSampler{};
95 prefilterSampler.mipMapMode = GpuImageMipmapMode::CUSTOM;
96 prefilterSampler.minMipLevel = 0;
97 prefilterSampler.maxMipLevel = maxMipLevel;
98 prefilterSampler.filteringType = GpuImageFilteringType::LINEAR;
99
100 UniquePtr<GpuImage> originalCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
101 maxResolution,
102 Format::RGBA16_SFLOAT,
103 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
104 GpuSharedMemoryType::GPU_ONLY,
105 GpuQueueType::MAIN,
106 origianlSampler);
107
108 UniquePtr<GpuImage> targetCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
109 maxResolution,
110 Format::RGBA16_SFLOAT,
111 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
112 GpuSharedMemoryType::GPU_ONLY,
113 GpuQueueType::MAIN,
114 prefilterSampler);
115
116 originalCubemap->SetDebugName("Original Specular Cubemap");
117 targetCubemap->SetDebugName("Prefiltered Specular Cubemap");
118
119 generationMaterialInstance->GetSlot("global")->SetGpuImage("image", *originalImage->GetView(GpuImageViewConfig::CreateSampled_SingleMipLevel(0)), GpuImageSamplerDesc::CreateDefault_NoMipMap());
120 prefilterMaterialInstance->GetSlot("global")->SetGpuImage("image", *originalCubemap->GetView(GpuImageViewConfig::CreateSampled_Cubemap()), GpuImageSamplerDesc::CreateDefault_NoMipMap());
121
122 generationMaterialInstance->GetSlot("global")->FlushUpdate();
123 prefilterMaterialInstance->GetSlot("global")->FlushUpdate();
124
125 // Renderizar a cubemap
126 UniquePtr<ICommandList> originalDrawCmdList = Engine::GetRenderer()->CreateSingleUseCommandList(GpuQueueType::MAIN);
127
128 originalDrawCmdList->SetDebugName("Specular Cubemap Cmd List");
129
130 originalDrawCmdList->Reset();
131 originalDrawCmdList->Start();
132
133 DrawOriginal(originalCubemap.GetPointer(), originalDrawCmdList.GetPointer());
134
135 originalDrawCmdList->Close();
136
137 Engine::GetRenderer()->SubmitSingleUseCommandList(std::move(originalDrawCmdList));
138
139 // Preprocesado
140
141 UniquePtr<ICommandList> prefilterCmdList = Engine::GetRenderer()->CreateSingleUseCommandList(GpuQueueType::MAIN);
142
143 prefilterCmdList->SetDebugName("Prefilter and LUT generation");
144
145 prefilterCmdList->Reset();
146 prefilterCmdList->Start();
147
148 for (UIndex32 mipLevel = 0; mipLevel < maxMipLevel; mipLevel++) {
149 const float roughness = (float)mipLevel / (maxMipLevel - 1.0f);
150 DrawPreFilter(targetCubemap.GetPointer(), prefilterCmdList.GetPointer(), mipLevel, roughness);
151 }
152
153 // LUT
154 GenerateLut(prefilterCmdList.GetPointer());
155 GpuImageCreateInfo lutCreateInfo = GpuImageCreateInfo::CreateDefault2D(
156 lookUpTable.GetSize(),
157 lookUpTable.GetTargetImage()->GetFormat(),
158 GpuImageUsage::SAMPLED | GpuImageUsage::TRANSFER_DESTINATION);
159 UniquePtr<GpuImage> lut = Engine::GetRenderer()->GetAllocator()->CreateImage(lutCreateInfo);
160 lut->SetDebugName("Specular Look-Up Table");
161
162 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::TRANSFER_DESTINATION,
163 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
164
165 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(lookUpTable.GetSize());
166 prefilterCmdList->RawCopyImageToImage(*lookUpTable.GetTargetImage(), lut.GetPointer(), copyInfo);
167
168 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::SAMPLED,
169 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SAMPLED_READ));
170
171 prefilterCmdList->Close();
172
173 Engine::GetRenderer()->SubmitSingleUseCommandList(std::move(prefilterCmdList));
174
175 asset->_SetCubemapImage(std::move(targetCubemap));
176 asset->_SetLookUpTable(std::move(lut));
177}

179void SpecularMapLoader::DrawOriginal(GRAPHICS::GpuImage* cubemap, GRAPHICS::ICommandList* cmdList) {
180 struct {
181 glm::mat4 cameraProjection = SpecularMapLoader::cameraProjection;
182 glm::mat4 cameraView = glm::mat4(1.0f);
183 } renderInfo;
184
185 for (UIndex32 faceId = 0; faceId < 6; faceId++) {
186 cmdList->BeginGraphicsRenderpass(&cubemapRenderTarget);
187
188 cmdList->BindMaterial(*generationMaterial);
189 cmdList->BindMaterialSlot(*generationMaterialInstance->GetSlot("global"));
190
191 const Viewport viewport{
192 .rectangle = {
193 0, 0,
194 (uint32_t)(cubemapRenderTarget.GetSize().x),
195 (uint32_t)(cubemapRenderTarget.GetSize().y)
196 }
197 };
198
199 cmdList->SetViewport(viewport);
200 cmdList->SetScissor(viewport.rectangle);
201
202 renderInfo.cameraView = cameraViews[faceId];
203 cmdList->PushMaterialConstants("info", renderInfo);
204
205 cmdList->BindVertexBuffer(cubemapModel->GetModel().GetVertexBuffer());
206 cmdList->BindIndexBuffer(cubemapModel->GetModel().GetIndexBuffer());
207 cmdList->DrawSingleInstance(cubemapModel->GetModel().GetTotalIndexCount());
208
209 cmdList->EndGraphicsRenderpass();
210
211 cmdList->SetGpuImageBarrier(cubemapRenderTarget.GetMainColorImage(),
212 GpuImageLayout::TRANSFER_SOURCE,
213 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
214 GpuImageRange{ .baseLayer = 0, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
215
216 cmdList->SetGpuImageBarrier(cubemap,
217 GpuImageLayout::TRANSFER_DESTINATION,
218 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
219 GpuImageRange{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
220
221 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(viewport.rectangle.GetRectangleSize());
222 copyInfo.destinationArrayLevel = faceId;
223 cmdList->RawCopyImageToImage(*cubemapRenderTarget.GetMainColorImage(), cubemap, copyInfo);
224
225 cmdList->SetGpuImageBarrier(cubemap,
226 GpuImageLayout::SAMPLED,
227 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ),
228 GpuImageRange{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
229 }

296void SpecularMapLoader::GenerateLut(ICommandList* cmdList) {
297 const auto resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
298 GpuImage* img = lookUpTable.GetTargetImage();
299
300 cmdList->SetGpuImageBarrier(img, GpuImageLayout::GENERAL,
301 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
302
303 cmdList->BindMaterial(*lutGenerationMaterial);
304 cmdList->BindMaterialSlot(*lutGenerationMaterialInstance->GetSlot("global"));
305
306 const Vector2ui imageResolution = lookUpTable.GetSize();
307 const Vector2ui threadBlockSize = { 8u, 8u };
308 const Vector2ui dispatchResolution = imageResolution / threadBlockSize;
309
310 cmdList->DispatchCompute({ dispatchResolution.x, dispatchResolution.y, 1 });
311
312 cmdList->SetGpuImageBarrier(img, GpuImageLayout::TRANSFER_SOURCE,
313 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
314}
