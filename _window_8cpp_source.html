38 // Creación de la ventana.

95void Window::SetFullscreen(bool fullscreen) {
96 if (isFullscreen == fullscreen)
97 return;
98
99 if (fullscreen) {
100 // Obtenemos la posición de la ventana,
101 // para poder reestablecerla después de salir de
102 // la ventana completa.
103 glfwGetWindowPos(window.GetPointer(), &previous.position.x, &previous.position.y);
104 glfwGetWindowSize(window.GetPointer(), &previous.size.x, &previous.size.y);
105
106 glfwSetWindowMonitor(window.GetPointer(), monitor.GetPointer(), 0, 0, monitorInfo->width, monitorInfo->height, monitorInfo->refreshRate);
107
108 resolution.x = monitorInfo->width;
109 resolution.y = monitorInfo->height;
110 }
111 else {
112 glfwSetWindowMonitor(window.GetPointer(), nullptr, previous.position.x, previous.position.y, previous.size.x, previous.size.y, monitorInfo->refreshRate);
113 }
114
115 isFullscreen = fullscreen;
116}
