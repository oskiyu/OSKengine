65void SkyboxRenderSystem::Render(ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
66 if (cameraObject == EMPTY_GAME_OBJECT)
67 return;
68
69 const auto resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
70
71 const auto& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
72 const auto& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
73
74 m_cameraUbos[resourceIndex]->MapMemory();
75 m_cameraUbos[resourceIndex]->Write(camera.GetProjectionMatrix());
76 m_cameraUbos[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
77 m_cameraUbos[resourceIndex]->Unmap();
78
79 commandList->StartDebugSection("Skybox Rendering", Color::Red);
80
81 commandList->BeginGraphicsRenderpass(&m_renderTarget, Color::Black * 0.0f);
82 this->SetupViewport(commandList);
83 commandList->BindMaterial(*m_skyboxMaterial);
84 commandList->BindMaterialInstance(*m_skyboxMaterialInstances[Engine::GetRenderer()->GetCurrentResourceIndex()]);
85 commandList->BindVertexBuffer(m_cubemapModel->GetModel().GetVertexBuffer());
86 commandList->BindIndexBuffer(m_cubemapModel->GetModel().GetIndexBuffer());
87 commandList->PushMaterialConstants("brightness", 1.0f);
88 commandList->DrawSingleInstance(m_cubemapModel->GetModel().GetTotalIndexCount());
89 commandList->EndGraphicsRenderpass();
90
91 commandList->EndDebugSection();
92}
