70 VkImageMemoryBarrier2 barrier{};

95 barrier.dstAccessMask = GetPipelineAccess(next.accessStage);
96
97 barrier.srcStageMask = GetPipelineStage(previous.stage);
98 barrier.dstStageMask = GetPipelineStage(next.stage);
99
100 VkDependencyInfo info{};
101 info.sType = VK_STRUCTURE_TYPE_DEPENDENCY_INFO;
102
103 info.imageMemoryBarrierCount = 1;
104 info.pImageMemoryBarriers = &barrier;
105 info.dependencyFlags = 0;
106
107 vkCmdPipelineBarrier2(commandBuffers[_GetCommandListIndex()], &info);
108
109 image->SetCurrentBarrier(next);

113 imageInfo.baseMipLevel,
114 imageInfo.numMipLevels,

118void CommandListVk::CopyBufferToImage(const GpuBuffer& source, GpuImage* dest, UIndex32 layer, USize64 offset) {
119 VkBufferImageCopy region{};
120
121 region.bufferOffset = source.GetMemorySubblock()->GetOffsetFromBlock() + offset;
122 region.bufferRowLength = 0;
123 region.bufferImageHeight = 0;
124
125 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
126 region.imageSubresource.mipLevel = 0;
127 region.imageSubresource.baseArrayLayer = layer;
128 region.imageSubresource.layerCount = 1;
129
130 region.imageOffset = { 0, 0, 0 };
131 region.imageExtent = {
132 dest->GetSize2D().x,
133 dest->GetSize2D().y,
134 1
135 };
136
137 // Layout[0] = TRANSFER_DEST
138 vkCmdCopyBufferToImage(
139 commandBuffers[_GetCommandListIndex()],
140 source.GetMemoryBlock()->As<GpuMemoryBlockVk>()->GetVulkanBuffer(),
141 dest->As<GpuImageVk>()->GetVkImage(),
142 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
143
144 if (dest->GetMipLevels() > 1) {
145 // Mip levels
146 Vector2i mipSize = dest->GetSize2D().ToVector2i();
147 // El nivel 0 ya está lleno.
148 for (UIndex32 mipLevel = 1; mipLevel < dest->GetMipLevels(); mipLevel++) {
149
150 // Mip anterior: layout DESTINATION -> SOURCE
151 SetGpuImageBarrier(
152 dest,
153 GpuImageLayout::TRANSFER_DESTINATION,
154 GpuImageLayout::TRANSFER_SOURCE,
155 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
156 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
157 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = mipLevel - 1, .numMipLevels = 1 });
158
159 VkImageBlit blit{};
160
161 blit.srcOffsets[0] = { 0, 0, 0 };
162 blit.srcOffsets[1] = { mipSize.x, mipSize.y, 1 };
163
164 // Origen: mip level anterior.
165 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
166 blit.srcSubresource.mipLevel = mipLevel - 1;
167 blit.srcSubresource.baseArrayLayer = layer;
168 blit.srcSubresource.layerCount = 1;
169
170 blit.dstOffsets[0] = { 0, 0, 0 };
171 blit.dstOffsets[1] = { mipSize.x > 1 ? mipSize.x / 2 : 1, mipSize.y > 1 ? mipSize.y / 2 : 1, 1 };
172
173 // Destino.
174 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
175 blit.dstSubresource.mipLevel = mipLevel;
176 blit.dstSubresource.baseArrayLayer = layer;
177 blit.dstSubresource.layerCount = 1;
178
179 // Mip actual: layout DESTINATION
180 vkCmdBlitImage(commandBuffers[_GetCommandListIndex()],
181 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
182 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
183 1, &blit, VK_FILTER_LINEAR);
184
185 if (mipSize.x > 1)
186 mipSize.x /= 2;
187 if (mipSize.y > 1)
188 mipSize.y /= 2;
189 }
190
191 // Establecer layout del último nivel = TRANSFER_SOURCE
192 SetGpuImageBarrier(
193 dest,
194 GpuImageLayout::TRANSFER_DESTINATION,
195 GpuImageLayout::TRANSFER_SOURCE,
196 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
197 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
198 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = dest->GetMipLevels() - 1, .numMipLevels = 1 });
199
200 // Establecer todo TRANSFER_SOURCE -> TRANSFER_DESTINATION
201 SetGpuImageBarrier(
202 dest,
203 GpuImageLayout::TRANSFER_SOURCE,
204 GpuImageLayout::TRANSFER_DESTINATION,
205 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
206 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
207 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
208 }

339void CommandListVk::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color, bool autoSync) {
340 const Vector2ui targetSize = colorImages[0].targetImage->GetSize2D();
341
342 if (autoSync) {
343 for (const auto& img : colorImages) {
344 if (img.clear) {
345 SetGpuImageBarrier(
346 img.targetImage,
347 GpuImageLayout::UNDEFINED,
348 GpuImageLayout::COLOR_ATTACHMENT,
349 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
350 GpuBarrierInfo(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE),
351 { .baseLayer = img.arrayLevel,
352 .numLayers = 1,
353 .baseMipLevel = 0,
354 .numMipLevels = ALL_MIP_LEVELS, });
355 }
356 else {
357 const GpuImageLayout previousLayout = img.targetImage->_GetLayout(img.arrayLevel, 0);
358 SetGpuImageBarrier(
359 img.targetImage,
360 previousLayout,
361 GpuImageLayout::COLOR_ATTACHMENT,
362 img.targetImage->GetCurrentBarrier(),
363 GpuBarrierInfo(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE),
364 GpuImageRange{ .baseLayer = img.arrayLevel,
365 .numLayers = 1,
366 .baseMipLevel = 0,
367 .numMipLevels = ALL_MIP_LEVELS, });
368 }
369 }
370
371 SampledChannel depthChannel = SampledChannel::DEPTH;
372 if (FormatSupportsStencil(depthImage.targetImage->GetFormat()))
373 depthChannel |= SampledChannel::STENCIL;
374
375 if (depthImage.clear) {
376 SetGpuImageBarrier(
377 depthImage.targetImage,
378 GpuImageLayout::UNDEFINED,
379 GpuImageLayout::DEPTH_STENCIL_TARGET,
380 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
381 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
382 { .baseLayer = depthImage.arrayLevel,
383 .numLayers = 1,
384 .baseMipLevel = 0,
385 .numMipLevels = ALL_MIP_LEVELS,
386 .channel = depthChannel });
387 }
388 else {
389 const GpuImageLayout previousLayout = depthImage.targetImage->_GetLayout(depthImage.arrayLevel, 0);
390 SetGpuImageBarrier(
391 depthImage.targetImage,
392 previousLayout,
393 GpuImageLayout::DEPTH_STENCIL_TARGET,
394 depthImage.targetImage->GetCurrentBarrier(),
395 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
396 { .baseLayer = depthImage.arrayLevel,
397 .numLayers = 1,
398 .baseMipLevel = 0,
399 .numMipLevels = ALL_MIP_LEVELS,
400 .channel = depthChannel });
401 }
402 }
403
404 GpuImageViewConfig colorAttachmentConfig = GpuImageViewConfig::CreateTarget_Color();
405 DynamicArray<VkRenderingAttachmentInfo> colorAttachments = DynamicArray<VkRenderingAttachmentInfo>::CreateResizedArray(colorImages.GetSize());
406 for (UIndex64 i = 0; i < colorImages.GetSize(); i++) {
407 colorAttachmentConfig.baseArrayLevel = colorImages[i].arrayLevel;
408
409 colorAttachments[i] = {};
410 colorAttachments[i].sType = VK_STRUCTURE_TYPE_RENDERING_ATTACHMENT_INFO_KHR;
411 colorAttachments[i].loadOp = colorImages[i].clear ? VK_ATTACHMENT_LOAD_OP_CLEAR : VK_ATTACHMENT_LOAD_OP_LOAD;
412 colorAttachments[i].storeOp = VK_ATTACHMENT_STORE_OP_STORE;
413 colorAttachments[i].imageLayout = VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL;
414 colorAttachments[i].imageView = colorImages[i].targetImage->GetView(colorAttachmentConfig)->As<GpuImageViewVk>()->GetVkView();
415 colorAttachments[i].resolveMode = VK_RESOLVE_MODE_NONE;
416 colorAttachments[i].resolveImageLayout = VK_IMAGE_LAYOUT_UNDEFINED;
417 colorAttachments[i].resolveImageView = VK_NULL_HANDLE;
418 colorAttachments[i].clearValue.color = { color.red, color.green, color.blue, color.alpha };
419 }
420
421 GpuImageViewConfig dephtAttachmentConfig = FormatSupportsStencil(depthImage.targetImage->GetFormat())
422 ? GpuImageViewConfig::CreateTarget_DepthStencil()
423 : GpuImageViewConfig::CreateTarget_Depth();
424 dephtAttachmentConfig.baseArrayLevel = depthImage.arrayLevel;
425
426 VkRenderingAttachmentInfo depthAttachment{};
427 depthAttachment.sType = VK_STRUCTURE_TYPE_RENDERING_ATTACHMENT_INFO_KHR;
428 depthAttachment.loadOp = depthImage.clear ? VK_ATTACHMENT_LOAD_OP_CLEAR : VK_ATTACHMENT_LOAD_OP_LOAD;
429 depthAttachment.storeOp = VK_ATTACHMENT_STORE_OP_STORE;
430 depthAttachment.imageLayout = FormatSupportsStencil(depthImage.targetImage->GetFormat())
431 ? VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL
432 : VK_IMAGE_LAYOUT_DEPTH_ATTACHMENT_OPTIMAL;
433 depthAttachment.imageView = depthImage.targetImage->GetView(dephtAttachmentConfig)->As<GpuImageViewVk>()->GetVkView();
434 depthAttachment.resolveMode = VK_RESOLVE_MODE_NONE;
435 depthAttachment.resolveImageLayout = VK_IMAGE_LAYOUT_UNDEFINED;
436 depthAttachment.resolveImageView = VK_NULL_HANDLE;
437 depthAttachment.clearValue.depthStencil = { 0.0f, 0 };
438
439 VkRenderingInfo renderpassInfo{};
440 renderpassInfo.sType = VK_STRUCTURE_TYPE_RENDERING_INFO;
441 renderpassInfo.renderArea = { 0, 0, targetSize.x, targetSize.y };
442 renderpassInfo.layerCount = 1;
443 renderpassInfo.colorAttachmentCount = static_cast<uint32_t>(colorAttachments.GetSize());
444 renderpassInfo.pColorAttachments = colorAttachments.GetData();
445 renderpassInfo.pDepthAttachment = &depthAttachment;
446 renderpassInfo.pStencilAttachment = FormatSupportsStencil(depthImage.targetImage->GetFormat())
447 ? &depthAttachment
448 : VK_NULL_HANDLE;
449
450 RendererVk::pvkCmdBeginRendering(commandBuffers.At(_GetCommandListIndex()), &renderpassInfo);
451
452 currentColorImages = colorImages;
453 currentDepthImage = depthImage;
454}
