61static VkPipelineStageFlagBits2 GetPipelineStageVk(GpuCommandStage stage) {
62 switch (stage) {
63 case GpuCommandStage::VERTEX_SHADER:
64 return VK_PIPELINE_STAGE_2_VERTEX_SHADER_BIT;
65 case GpuCommandStage::FRAGMENT_SHADER:
66 return VK_PIPELINE_STAGE_2_FRAGMENT_SHADER_BIT;
67
68 case GpuCommandStage::TESSELATION_CONTROL:
69 return VK_PIPELINE_STAGE_2_TESSELLATION_CONTROL_SHADER_BIT;
70 case GpuCommandStage::TESSELATION_EVALUATION:
71 return VK_PIPELINE_STAGE_2_TESSELLATION_EVALUATION_SHADER_BIT;
72
73 case GpuCommandStage::RAYTRACING_SHADER:
74 return VK_PIPELINE_STAGE_2_RAY_TRACING_SHADER_BIT_KHR;
75 case GpuCommandStage::RT_AS_BUILD:
76 return VK_PIPELINE_STAGE_2_ACCELERATION_STRUCTURE_BUILD_BIT_KHR;
77
78 case GpuCommandStage::COLOR_ATTACHMENT_OUTPUT:
79 return VK_PIPELINE_STAGE_2_COLOR_ATTACHMENT_OUTPUT_BIT;
80 case GpuCommandStage::DEPTH_STENCIL_START:
81 return VK_PIPELINE_STAGE_2_EARLY_FRAGMENT_TESTS_BIT;
82 case GpuCommandStage::DEPTH_STENCIL_END:
83 return VK_PIPELINE_STAGE_2_LATE_FRAGMENT_TESTS_BIT;
84
85 case GpuCommandStage::TRANSFER:
86 return VK_PIPELINE_STAGE_2_ALL_TRANSFER_BIT;
87
88 case GpuCommandStage::COMPUTE_SHADER:
89 return VK_PIPELINE_STAGE_2_COMPUTE_SHADER_BIT;
90
91 case GpuCommandStage::NONE:
92 return VK_PIPELINE_STAGE_2_NONE;
93 case GpuCommandStage::ALL:
94 return VK_PIPELINE_STAGE_2_ALL_COMMANDS_BIT;

97 OSK_ASSERT(false, NotImplementedException());
98 return VK_PIPELINE_STAGE_2_NONE;

305void CommandListVk::CopyBufferToImage(const GpuBuffer& source, GpuImage* dest, UIndex32 layer, USize64 offset) {
306 VkBufferImageCopy region{};
307
308 region.bufferOffset = source.GetMemorySubblock()->GetOffsetFromBlock() + offset;
309 region.bufferRowLength = 0;
310 region.bufferImageHeight = 0;
311
312 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
313 region.imageSubresource.mipLevel = 0;
314 region.imageSubresource.baseArrayLayer = layer;
315 region.imageSubresource.layerCount = 1;
316
317 region.imageOffset = { 0, 0, 0 };
318 region.imageExtent = {
319 dest->GetSize2D().x,
320 dest->GetSize2D().y,
321 1
322 };
323
324 // `Layout[0] = TRANSFER_DEST`
325 vkCmdCopyBufferToImage(
326 m_commandBuffers[_GetCommandListIndex()],
327 source.GetMemoryBlock()->As<GpuMemoryBlockVk>()->GetVulkanBuffer(),
328 dest->As<GpuImageVk>()->GetVkImage(),
329 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
330
331 if (dest->GetMipLevels() > 1) {
332 // Mip levels
333 Vector2i mipSize = dest->GetSize2D().ToVector2i();
334 // El nivel 0 ya está lleno.
335 for (UIndex32 mipLevel = 1; mipLevel < dest->GetMipLevels(); mipLevel++) {
336
337 // Mip anterior: layout DESTINATION -> SOURCE
338 SetGpuImageBarrier(
339 dest,
340 GpuImageLayout::TRANSFER_DESTINATION,
341 GpuImageLayout::TRANSFER_SOURCE,
342 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
343 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
344 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = mipLevel - 1, .numMipLevels = 1 },
345 ResourceQueueTransferInfo::Empty());
346
347 VkImageBlit blit{};
348
349 blit.srcOffsets[0] = { 0, 0, 0 };
350 blit.srcOffsets[1] = { mipSize.x, mipSize.y, 1 };
351
352 // Origen: mip level anterior.
353 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
354 blit.srcSubresource.mipLevel = mipLevel - 1;
355 blit.srcSubresource.baseArrayLayer = layer;
356 blit.srcSubresource.layerCount = 1;
357
358 blit.dstOffsets[0] = { 0, 0, 0 };
359 blit.dstOffsets[1] = { mipSize.x > 1 ? mipSize.x / 2 : 1, mipSize.y > 1 ? mipSize.y / 2 : 1, 1 };
360
361 // Destino.
362 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
363 blit.dstSubresource.mipLevel = mipLevel;
364 blit.dstSubresource.baseArrayLayer = layer;
365 blit.dstSubresource.layerCount = 1;
366
367 // Mip actual: layout DESTINATION
368 vkCmdBlitImage(m_commandBuffers[_GetCommandListIndex()],
369 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
370 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
371 1, &blit, VK_FILTER_LINEAR);
372
373 if (mipSize.x > 1)
374 mipSize.x /= 2;
375 if (mipSize.y > 1)
376 mipSize.y /= 2;
377 }
378
379 // Establecer layout del último nivel = TRANSFER_SOURCE
380 SetGpuImageBarrier(
381 dest,
382 GpuImageLayout::TRANSFER_DESTINATION,
383 GpuImageLayout::TRANSFER_SOURCE,
384 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
385 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
386 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = dest->GetMipLevels() - 1, .numMipLevels = 1 },
387 ResourceQueueTransferInfo::Empty());
388
389 // Establecer todo TRANSFER_SOURCE -> TRANSFER_DESTINATION
390 SetGpuImageBarrier(
391 dest,
392 GpuImageLayout::TRANSFER_SOURCE,
393 GpuImageLayout::TRANSFER_DESTINATION,
394 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
395 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
396 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS },
397 ResourceQueueTransferInfo::Empty());
398 }
