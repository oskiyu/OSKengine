71static VkPipelineStageFlagBits2 GetPipelineStageVk(GpuCommandStage stage) {
72 switch (stage) {
73 case GpuCommandStage::VERTEX_SHADER:
74 return VK_PIPELINE_STAGE_2_VERTEX_SHADER_BIT;
75 case GpuCommandStage::FRAGMENT_SHADER:
76 return VK_PIPELINE_STAGE_2_FRAGMENT_SHADER_BIT;
77
78 case GpuCommandStage::TESSELATION_CONTROL:
79 return VK_PIPELINE_STAGE_2_TESSELLATION_CONTROL_SHADER_BIT;
80 case GpuCommandStage::TESSELATION_EVALUATION:
81 return VK_PIPELINE_STAGE_2_TESSELLATION_EVALUATION_SHADER_BIT;
82
83 case GpuCommandStage::RAYTRACING_SHADER:
84 return VK_PIPELINE_STAGE_2_RAY_TRACING_SHADER_BIT_KHR;
85 case GpuCommandStage::RT_AS_BUILD:
86 return VK_PIPELINE_STAGE_2_ACCELERATION_STRUCTURE_BUILD_BIT_KHR;
87
88 case GpuCommandStage::COLOR_ATTACHMENT_OUTPUT:
89 return VK_PIPELINE_STAGE_2_COLOR_ATTACHMENT_OUTPUT_BIT;
90 case GpuCommandStage::DEPTH_STENCIL_START:
91 return VK_PIPELINE_STAGE_2_EARLY_FRAGMENT_TESTS_BIT;
92 case GpuCommandStage::DEPTH_STENCIL_END:
93 return VK_PIPELINE_STAGE_2_LATE_FRAGMENT_TESTS_BIT;
94
95 case GpuCommandStage::TRANSFER:
96 return VK_PIPELINE_STAGE_2_ALL_TRANSFER_BIT;
97
98 case GpuCommandStage::COMPUTE_SHADER:
99 return VK_PIPELINE_STAGE_2_COMPUTE_SHADER_BIT;
100
101 case GpuCommandStage::NONE:
102 return VK_PIPELINE_STAGE_2_NONE;
103 case GpuCommandStage::ALL:
104 return VK_PIPELINE_STAGE_2_ALL_COMMANDS_BIT;

107 OSK_ASSERT(false, NotImplementedException());
108 return VK_PIPELINE_STAGE_2_NONE;

318void CommandListVk::CopyBufferToImage(const GpuBuffer& source, GpuImage* dest, UIndex32 layer, USize64 offset) {
319 VkBufferImageCopy region{};
320
321 region.bufferOffset = source.GetMemorySubblock()->GetOffsetFromBlock() + offset;
322 region.bufferRowLength = 0;
323 region.bufferImageHeight = 0;
324
325 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
326 region.imageSubresource.mipLevel = 0;
327 region.imageSubresource.baseArrayLayer = layer;
328 region.imageSubresource.layerCount = 1;
329
330 region.imageOffset = { 0, 0, 0 };
331 region.imageExtent = {
332 dest->GetSize2D().x,
333 dest->GetSize2D().y,
334 1
335 };
336
337 // `Layout[0] = TRANSFER_DEST`
338 vkCmdCopyBufferToImage(
339 m_commandBuffers[_GetCommandListIndex()],
340 source.GetMemoryBlock()->As<GpuMemoryBlockVk>()->GetVulkanBuffer(),
341 dest->As<GpuImageVk>()->GetVkImage(),
342 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
343
344 if (dest->GetMipLevels() > 1) {
345 // Mip levels
346 Vector2i mipSize = dest->GetSize2D().ToVector2i();
347 // El nivel 0 ya está lleno.
348 for (UIndex32 mipLevel = 1; mipLevel < dest->GetMipLevels(); mipLevel++) {
349
350 // Mip anterior: layout DESTINATION -> SOURCE
351 SetGpuImageBarrier(
352 dest,
353 GpuImageLayout::TRANSFER_DESTINATION,
354 GpuImageLayout::TRANSFER_SOURCE,
355 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
356 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
357 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = mipLevel - 1, .numMipLevels = 1 },
358 ResourceQueueTransferInfo::Empty());
359
360 VkImageBlit blit{};
361
362 blit.srcOffsets[0] = { 0, 0, 0 };
363 blit.srcOffsets[1] = { mipSize.x, mipSize.y, 1 };
364
365 // Origen: mip level anterior.
366 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
367 blit.srcSubresource.mipLevel = mipLevel - 1;
368 blit.srcSubresource.baseArrayLayer = layer;
369 blit.srcSubresource.layerCount = 1;
370
371 blit.dstOffsets[0] = { 0, 0, 0 };
372 blit.dstOffsets[1] = { mipSize.x > 1 ? mipSize.x / 2 : 1, mipSize.y > 1 ? mipSize.y / 2 : 1, 1 };
373
374 // Destino.
375 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
376 blit.dstSubresource.mipLevel = mipLevel;
377 blit.dstSubresource.baseArrayLayer = layer;
378 blit.dstSubresource.layerCount = 1;
379
380 // Mip actual: layout DESTINATION
381 vkCmdBlitImage(m_commandBuffers[_GetCommandListIndex()],
382 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
383 dest->As<GpuImageVk>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
384 1, &blit, VK_FILTER_LINEAR);
385
386 if (mipSize.x > 1)
387 mipSize.x /= 2;
388 if (mipSize.y > 1)
389 mipSize.y /= 2;
390 }
391
392 // Establecer layout del último nivel = TRANSFER_SOURCE
393 SetGpuImageBarrier(
394 dest,
395 GpuImageLayout::TRANSFER_DESTINATION,
396 GpuImageLayout::TRANSFER_SOURCE,
397 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
398 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
399 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = dest->GetMipLevels() - 1, .numMipLevels = 1 },
400 ResourceQueueTransferInfo::Empty());
401
402 // Establecer todo TRANSFER_SOURCE -> TRANSFER_DESTINATION
403 SetGpuImageBarrier(
404 dest,
405 GpuImageLayout::TRANSFER_SOURCE,
406 GpuImageLayout::TRANSFER_DESTINATION,
407 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ),
408 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE),
409 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS },
410 ResourceQueueTransferInfo::Empty());
411 }
