912 hbaoPass = new HbaoPass();
913 hbaoPass->Create(Engine::GetDisplay()->GetResolution());
914 }
915
916 if (!bloomPass.HasValue()) {
917 bloomPass = new BloomPass();

919 }
920
921 if (!toneMappingPass.HasValue()) {
922 toneMappingPass = new ToneMappingPass();
923 toneMappingPass->Create(Engine::GetDisplay()->GetResolution());
924
925 const GpuBuffer* epxBuffers[NUM_RESOURCES_IN_FLIGHT]{};
926 for (UIndex32 i = 0; i < exposureBuffers.size(); i++) {
927 exposureBuffers[i] = Engine::GetRenderer()->GetAllocator()->CreateStorageBuffer(sizeof(float)).GetPointer();
928 epxBuffers[i] = exposureBuffers[i].GetPointer();
929
930 exposureBuffers[i]->MapMemory();
931 exposureBuffers[i]->Write(toneMappingPass->GetExposure());
932 exposureBuffers[i]->Unmap();
933 }
934
935 toneMappingPass->SetExposureBuffers(epxBuffers);
936 }
937
938 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateSampled_SingleMipLevel(0);
939
940 auto& preEffectsSource = preEffectsFrameCombiner->GetRenderTarget();
941
942 auto* renderSystem = Engine::GetEcs()->GetSystem<OSK_CURRENT_RSYSTEM>();

945 hbaoPass->SetInputTarget(preEffectsSource, viewConfig);

947 hbaoPass->SetDepthInput(renderSystem->GetGbuffer().GetImage(GBuffer::Target::DEPTH));
948 }

951 bloomPass->SetInputTarget(hbaoPass->GetOutput(), viewConfig);
952 toneMappingPass->SetInputTarget(bloomPass->GetOutput(), viewConfig);
953 } else
954 if (config.useBloom) {
955 bloomPass->SetInputTarget(preEffectsSource, viewConfig);
956 toneMappingPass->SetInputTarget(bloomPass->GetOutput(), viewConfig);
957 }
958 else
959 if (config.useHbao) {
960 toneMappingPass->SetInputTarget(hbaoPass->GetOutput(), viewConfig);
961 }
962 else {
963 toneMappingPass->SetInputTarget(preEffectsSource, viewConfig);
964 }
965
966 Engine::GetRenderer()->WaitForCompletion();
967
968 if (config.useHbao)
969 hbaoPass->UpdateMaterialInstance();
970 if (config.useBloom)
971 bloomPass->UpdateMaterialInstance();
972 toneMappingPass->UpdateMaterialInstance();
973 }
