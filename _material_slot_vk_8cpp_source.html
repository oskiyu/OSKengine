26MaterialSlotVk::MaterialSlotVk(const std::string& name, const MaterialLayout* layout)
27 : IMaterialSlot(layout, name),
28 m_descLayout(new DescriptorLayoutVk(&layout->GetSlot(name), 0))
29{
30 m_pool = new DescriptorPoolVk(m_descLayout.GetValue(), 1);
31
32 std::array<USize32, NUM_RESOURCES_IN_FLIGHT> maxCount{};
33 maxCount.fill(1);
34
35 std::array<VkDescriptorSetLayout, NUM_RESOURCES_IN_FLIGHT> descriptorLayout{};
36 descriptorLayout.fill(m_descLayout->GetLayout());
37
38 VkDescriptorSetVariableDescriptorCountAllocateInfo variableSizeDescriptorInfo{};
39 variableSizeDescriptorInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_VARIABLE_DESCRIPTOR_COUNT_ALLOCATE_INFO;
40 variableSizeDescriptorInfo.descriptorSetCount = static_cast<uint32_t>(m_descriptorSets.size());
41 variableSizeDescriptorInfo.pDescriptorCounts = maxCount.data();
42 variableSizeDescriptorInfo.pNext = VK_NULL_HANDLE;
43
44 VkDescriptorSetAllocateInfo allocInfo{};
45 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
46 allocInfo.descriptorPool = m_pool->GetPool();
47 allocInfo.descriptorSetCount = m_descriptorSets.size();
48 allocInfo.pSetLayouts = descriptorLayout.data();
49 allocInfo.pNext = &variableSizeDescriptorInfo;
50
51 std::array<VkDescriptorSetLayout, NUM_RESOURCES_IN_FLIGHT> layouts{};
52 layouts.fill(m_descLayout->GetLayout());
53
54 allocInfo.pSetLayouts = layouts.data();
55
56 VkResult result = vkAllocateDescriptorSets(GetDevice(), &allocInfo, m_descriptorSets.data());
57 OSK_ASSERT(result == VK_SUCCESS, MaterialSlotCreationException(result));
58
59 SetDebugName(name);
60}

209void MaterialSlotVk::FlushUpdate() {
210 VkDevice device = GetDevice();
211
212 m_descriptorWrites.Empty();
213
214 for (const auto& bindingsPerFrame : m_bindings) {
215 for (const auto& [bindingIndex, bindingData] : bindingsPerFrame) {
216 for (const auto& [arrayIndex, descriptorWrite] : bindingData.descriptorWrites) {
217 m_descriptorWrites.Insert(descriptorWrite);
218 }
219 }
220 }
221
222 vkUpdateDescriptorSets(device, static_cast<uint32_t>(m_descriptorWrites.GetSize()), m_descriptorWrites.GetData(), 0, nullptr);
