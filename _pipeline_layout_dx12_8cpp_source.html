18D3D12_SHADER_VISIBILITY GetShaderStageDx12(ShaderStage stage) {
19 if (OSK::EFTraits::HasFlag(stage, ShaderStage::VERTEX)
20 && OSK::EFTraits::HasFlag(stage, ShaderStage::FRAGMENT))
21 return D3D12_SHADER_VISIBILITY_ALL;
22
23 switch (stage) {
24 case ShaderStage::VERTEX:
25 return D3D12_SHADER_VISIBILITY_VERTEX;

27 return D3D12_SHADER_VISIBILITY_PIXEL;
28 case ShaderStage::TESSELATION_CONTROL:
29 return D3D12_SHADER_VISIBILITY_ALL;
30 case ShaderStage::TESSELATION_EVALUATION:
