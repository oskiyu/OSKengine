15D3D12_SHADER_VISIBILITY GetShaderStageDx12(ShaderStage stage) {
16 if (OSK::EFTraits::HasFlag(stage, ShaderStage::VERTEX)
17 && OSK::EFTraits::HasFlag(stage, ShaderStage::FRAGMENT))
18 return D3D12_SHADER_VISIBILITY_ALL;

23 case ShaderStage::FRAGMENT:
24 return D3D12_SHADER_VISIBILITY_PIXEL;

26 return D3D12_SHADER_VISIBILITY_ALL;
27 case ShaderStage::TESSELATION_EVALUATION:
