91 template <typename Op>
92 void walk_from(std::unordered_set<uint32_t> &seen_blocks, uint32_t block, const Op &op) const
93 {
94 if (seen_blocks.count(block))

98 if (op(block))
99 {
100 for (auto b : get_succeeding_edges(block))
