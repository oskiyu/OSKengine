38void ISwapchain::TakeScreenshot(std::string_view path) {
39 // Obtenemos la imagen a guardar.
40 GpuImage* swapchainImage = m_images[Engine::GetRenderer()->GetCurrentFrameIndex()].GetPointer();
41
42 // Creamos una imagen intermedia con tiling lineal,
43 // necesario para poder pasar los datos a la CPU.
44 GpuImageCreateInfo intermediateImageInfo = GpuImageCreateInfo::CreateDefault2D(
45 swapchainImage->GetSize2D(),
46 swapchainImage->GetFormat(),
47 GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION
48 );
49 intermediateImageInfo.memoryType = GpuSharedMemoryType::GPU_AND_CPU;
50 intermediateImageInfo.tilingType = GpuImageTiling::LINEAL;
51 intermediateImageInfo.msaaSamples = 1;
52 intermediateImageInfo.samplerDesc = GpuImageSamplerDesc::CreateDefault();
53
54 UniquePtr<GpuImage> intermediateImage = Engine::GetRenderer()->GetAllocator()->CreateImage(intermediateImageInfo).GetPointer();
55
56
57 // Comenzamos el proceso de copia.
58
59 // Preferir NO usar cola de transferencia, para no tener que transferir la propiedad de la imagen.
60 OwnedPtr<ICommandList> commandList = Engine::GetRenderer()->CreateSingleUseCommandList(GpuQueueType::MAIN);
61 commandList->Reset();
62 commandList->Start();
63
64 commandList->SetGpuImageBarrier(
65 swapchainImage,
66 GpuImageLayout::TRANSFER_SOURCE,
67 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ)
68 );
69
70 commandList->SetGpuImageBarrier(
71 intermediateImage.GetPointer(),
72 GpuImageLayout::TRANSFER_DESTINATION,
73 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE)
74 );
75
76 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(swapchainImage->GetSize2D());
77 commandList->RawCopyImageToImage(*swapchainImage, intermediateImage.GetPointer(), copyInfo);
78
79 commandList->SetGpuImageBarrier(
80 intermediateImage.GetPointer(),
81 GpuImageLayout::GENERAL,
82 GpuBarrierInfo(GpuCommandStage::ALL, GpuAccessStage::MEMORY_WRITE));
83
84 commandList->Close();
85
86 Engine::GetRenderer()->SubmitSingleUseCommandList(commandList);
87
88
89 // Obtenemos los píxeles de la imagen lineal, leyendo la memoria directamente.
90
91 DynamicArray<TByte> cpuImage = DynamicArray<TByte>::CreateReserved(
92 intermediateImage->GetSize2D().x * intermediateImage->GetSize2D().y * 4
93 );
94
95 intermediateImage->GetBuffer()->MapMemory();
96 const TByte* mappedImage = intermediateImage->GetBuffer()->GetMappedData();
97
98 const TByte* pointer = mappedImage;
99 for (UIndex32 y = 0; y < intermediateImage->GetSize2D().y; y++) {
100 for (UIndex32 x = 0; x < intermediateImage->GetSize2D().x; x++) {
101 // Debemos cambiar el formato:
102 // BGRA -> RGBA
103 cpuImage.Insert(pointer[2]);
104 cpuImage.Insert(pointer[1]);
105 cpuImage.Insert(pointer[0]);
106 cpuImage.Insert(pointer[3]);
107 pointer += 4;
108 }
109 }
110
111 intermediateImage->GetBuffer()->Unmap();
112
113
114 // Obtenemos la fecha de la captura.
115
116 const auto now = std::chrono::system_clock::now();
117 const auto time = std::chrono::system_clock::to_time_t(now);
118 std::tm now2{};
119 localtime_s(&now2, &time);
120
121 const std::string nowText = std::format(
122 "{}_{}_{}_{}{}{}",
123 now2.tm_year + 1900,
124 now2.tm_mon + 1,
125 now2.tm_mday,
126 now2.tm_hour,
127 now2.tm_min,
128 now2.tm_sec);
129
130
131 // Guardamos la imagen.
132
133 const auto name = std::format("{}_{}.png", path, nowText);
134 stbi_write_png(name.c_str(), intermediateImage->GetSize2D().x, intermediateImage->GetSize2D().y, 4, cpuImage.GetData(), intermediateImage->GetSize2D().x * 4);
135
136}
