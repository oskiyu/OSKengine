31void ISwapchain::TakeScreenshot(std::string_view path) {
32 // Obtenemos la imagen a guardar.
33 GpuImage* swapchainImage = m_images[Engine::GetRenderer()->GetCurrentFrameIndex()].GetPointer();
34
35 // Creamos una imagen intermedia con tiling lineal,
36 // necesario para poder pasar los datos a la CPU.
37 GpuImageCreateInfo intermediateImageInfo = GpuImageCreateInfo::CreateDefault2D(
38 swapchainImage->GetSize2D(),
39 swapchainImage->GetFormat(),
40 GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION
41 );
42 intermediateImageInfo.memoryType = GpuSharedMemoryType::GPU_AND_CPU;
43 intermediateImageInfo.tilingType = GpuImageTiling::LINEAL;
44 intermediateImageInfo.msaaSamples = 1;
45 intermediateImageInfo.samplerDesc = GpuImageSamplerDesc::CreateDefault_NoMipMap();
46
47 UniquePtr<GpuImage> intermediateImage = Engine::GetRenderer()->GetAllocator()->CreateImage(intermediateImageInfo);
48
49
50 // Comenzamos el proceso de copia.
51
52 // Preferir NO usar cola de transferencia, para no tener que transferir la propiedad de la imagen.
53 UniquePtr<ICommandList> commandList = Engine::GetRenderer()->CreateSingleUseCommandList(GpuQueueType::MAIN);
54 commandList->Reset();
55 commandList->Start();
56
57 commandList->SetGpuImageBarrier(
58 swapchainImage,
59 GpuImageLayout::TRANSFER_SOURCE,
60 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ)
61 );
62
63 commandList->SetGpuImageBarrier(
64 intermediateImage.GetPointer(),
65 GpuImageLayout::TRANSFER_DESTINATION,
66 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE)
67 );
68
69 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(swapchainImage->GetSize2D());
70 commandList->RawCopyImageToImage(*swapchainImage, intermediateImage.GetPointer(), copyInfo);
71
72 commandList->SetGpuImageBarrier(
73 intermediateImage.GetPointer(),
74 GpuImageLayout::GENERAL,
75 GpuBarrierInfo(GpuCommandStage::ALL, GpuAccessStage::MEMORY_WRITE));
76
77 commandList->Close();
78
79 Engine::GetRenderer()->SubmitSingleUseCommandList(std::move(commandList));
80
81
82 // Obtenemos los píxeles de la imagen lineal, leyendo la memoria directamente.
83
84 DynamicArray<TByte> cpuImage = DynamicArray<TByte>::CreateReserved(
85 intermediateImage->GetSize2D().x * intermediateImage->GetSize2D().y * 4
86 );
87
88 intermediateImage->GetBuffer()->MapMemory();
89 const TByte* mappedImage = intermediateImage->GetBuffer()->GetMappedData();
90
91 const TByte* pointer = mappedImage;
92 for (UIndex32 y = 0; y < intermediateImage->GetSize2D().y; y++) {
93 for (UIndex32 x = 0; x < intermediateImage->GetSize2D().x; x++) {
94 // Debemos cambiar el formato:
95 // BGRA -> RGBA
96 cpuImage.Insert(pointer[2]);
97 cpuImage.Insert(pointer[1]);
98 cpuImage.Insert(pointer[0]);
99 cpuImage.Insert(pointer[3]);
100 pointer += 4;
101 }
102 }
103
104 intermediateImage->GetBuffer()->Unmap();
105
106
107 // Obtenemos la fecha de la captura.
108
109 const auto now = std::chrono::system_clock::now();
110 const auto time = std::chrono::system_clock::to_time_t(now);
111 std::tm now2{};
112 localtime_s(&now2, &time);
113
114 const std::string nowText = std::format(
115 "{}_{}_{}_{}{}{}",
116 now2.tm_year + 1900,
117 now2.tm_mon + 1,
118 now2.tm_mday,
119 now2.tm_hour,
120 now2.tm_min,
121 now2.tm_sec);
122
123
124 // Guardamos la imagen.
125
126 const auto name = std::format("{}_{}.png", path, nowText);
127 stbi_write_png(name.c_str(), intermediateImage->GetSize2D().x, intermediateImage->GetSize2D().y, 4, cpuImage.GetData(), intermediateImage->GetSize2D().x * 4);
128
129}
