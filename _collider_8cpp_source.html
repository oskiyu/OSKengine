37CollisionInfo Collider::GetCollisionInfo(const Collider& other, const Transform3D& thisTransform, const Transform3D& otherTransform) const {
38 const auto otherTopLevel = other.GetTopLevelCollider();
39 if (!otherTopLevel)
40 return CollisionInfo::False();
41
42 const Vector3f thisPosition = thisTransform.GetPosition();
43 const Vector3f otherPosition = otherTransform.GetPosition();
44
45 if (!m_topLevelCollider->IsColliding(*otherTopLevel))
46 return CollisionInfo::False();
47
48 DynamicArray<DetailedCollisionInfo> bottomLevelCollisions{};
49
50 for (const auto& bottomLevelA : m_bottomLevelColliders) {
51 for (const auto& bottomLevelB : other.m_bottomLevelColliders) {
52
53 const auto collisionInfo = bottomLevelA->GetCollisionInfo(*bottomLevelB.GetPointer(), thisTransform, otherTransform);
54
55 if (collisionInfo.IsColliding()) {
56 bottomLevelCollisions.Insert(collisionInfo);
57 }
58 }
59 }
60
61 return bottomLevelCollisions.IsEmpty()
62 ? CollisionInfo::TopLevelOnly()
63 : CollisionInfo::True(bottomLevelCollisions);
64}
