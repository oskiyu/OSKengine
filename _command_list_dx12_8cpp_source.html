178void CommandListDx12::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color, bool autoSync) {
179 DynamicArray<D3D12_CPU_DESCRIPTOR_HANDLE> colorAttachments{};
180 D3D12_CPU_DESCRIPTOR_HANDLE depthStencilDesc{};
181
182 if (m_currentlyBoundRenderTargetType == RenderpassType::FINAL) {
183 const SwapchainDx12* swapchain = Engine::GetRenderer()->_GetSwapchain()->As<SwapchainDx12>();
184
185 colorAttachments.Insert(swapchain->GetRenderTargetMemory()->GetCPUDescriptorHandleForHeapStart());
186 colorAttachments[0].ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
187 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
188
189 depthStencilDesc = swapchain->GetDepthStencilMemory()->GetCPUDescriptorHandleForHeapStart();
190 depthStencilDesc.ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
191 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
192 }
193 else {
194 GpuImageViewConfig colorView = GpuImageViewConfig::CreateTarget_Color();
195 GpuImageViewConfig depthView = GpuImageViewConfig::CreateTarget_DepthStencil();
196 depthView.baseArrayLevel = depthImage.arrayLevel;
197
198 for (const auto& [image, layer, clear] : colorImages) {
199 colorView.baseArrayLevel = layer;
200 colorAttachments.Insert(image->GetView(colorView)
201 ->As<GpuImageViewDx12>()->GetCpuAddress());
202 }
203
204 depthStencilDesc = depthImage.targetImage->GetView(depthView)
205 ->As<GpuImageViewDx12>()->GetCpuAddress();
206 }
207
208 for (const auto& img : colorImages) {
209 const GpuBarrierInfo prev(GpuCommandStage::NONE, GpuAccessStage::NONE);
210 const GpuBarrierInfo next(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE);
211 const GpuImageRange imgInfo = {
212 .baseLayer = img.arrayLevel,
213 .numLayers = 1,
214 .baseMipLevel = 0,
215 .numMipLevels = 1,
216 };
217
218 if (autoSync)
219 SetGpuImageBarrier(img.targetImage, GpuImageLayout::UNDEFINED, GpuImageLayout::COLOR_ATTACHMENT, prev, next, imgInfo, ResourceQueueTransferInfo::Empty());
220 }
221
222 const FLOAT clearValue[] = { color.red, color.green, color.blue, color.alpha };
223
224 for (auto colorAttachment : colorAttachments)
225 commandList->ClearRenderTargetView(colorAttachment, clearValue, 0, nullptr);
226 commandList->ClearDepthStencilView(depthStencilDesc, D3D12_CLEAR_FLAG_DEPTH, 1.0f, 0, 0, nullptr);
227
228 commandList->OMSetRenderTargets(static_cast<UINT>(colorAttachments.GetSize()), colorAttachments.GetData(), FALSE, &depthStencilDesc);
229
230 m_currentlyBoundColorImages = std::move(colorImages);
231 m_currentlyBoundDepthImage = depthImage;
232}

234void CommandListDx12::EndGraphicsRenderpass(bool autoSync) {
235 GpuImageLayout finalLayout = GpuImageLayout::SAMPLED;
236 if (m_currentlyBoundRenderTargetType == RenderpassType::FINAL)
237 finalLayout = GpuImageLayout::PRESENT;
238
239 for (const auto& img : m_currentlyBoundColorImages) {
240 GpuBarrierInfo prevBarrier{};
241 prevBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
242 prevBarrier.accessStage = GpuAccessStage::COLOR_ATTACHMENT_WRITE;
243
244 GpuBarrierInfo nextBarrier{};
245 nextBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
246 nextBarrier.accessStage = GpuAccessStage::SHADER_READ;
247
248 GpuImageRange imgBarrier{};
249 imgBarrier.baseLayer = img.arrayLevel;
250 imgBarrier.numLayers = 1;
251 imgBarrier.baseMipLevel = 0;
252 imgBarrier.numMipLevels = ALL_MIP_LEVELS;
253
254 if (autoSync)
255 SetGpuImageBarrier(img.targetImage, GpuImageLayout::COLOR_ATTACHMENT, finalLayout, prevBarrier, nextBarrier, imgBarrier, ResourceQueueTransferInfo::Empty());
256 }
257
258 if (m_currentlyBoundRenderTargetType != RenderpassType::INTERMEDIATE) {
259 m_currentlyBoundRenderTargetType = RenderpassType::INTERMEDIATE;
260 }
