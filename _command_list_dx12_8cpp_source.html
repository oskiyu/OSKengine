173void CommandListDx12::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color, bool autoSync) {
174 DynamicArray<D3D12_CPU_DESCRIPTOR_HANDLE> colorAttachments{};
175 D3D12_CPU_DESCRIPTOR_HANDLE depthStencilDesc{};
176
177 if (m_currentlyBoundRenderTargetType == RenderpassType::FINAL) {
178 const SwapchainDx12* swapchain = Engine::GetRenderer()->_GetSwapchain()->As<SwapchainDx12>();
179
180 colorAttachments.Insert(swapchain->GetRenderTargetMemory()->GetCPUDescriptorHandleForHeapStart());
181 colorAttachments[0].ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
182 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
183
184 depthStencilDesc = swapchain->GetDepthStencilMemory()->GetCPUDescriptorHandleForHeapStart();
185 depthStencilDesc.ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
186 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
187 }
188 else {
189 GpuImageViewConfig colorView = GpuImageViewConfig::CreateTarget_Color();
190 GpuImageViewConfig depthView = GpuImageViewConfig::CreateTarget_DepthStencil();
191 depthView.baseArrayLevel = depthImage.arrayLevel;
192
193 for (const auto& [image, layer, clear] : colorImages) {
194 colorView.baseArrayLevel = layer;
195 colorAttachments.Insert(image->GetView(colorView)
196 ->As<GpuImageViewDx12>()->GetCpuAddress());
197 }
198
199 depthStencilDesc = depthImage.targetImage->GetView(depthView)
200 ->As<GpuImageViewDx12>()->GetCpuAddress();
201 }
202
203 for (const auto& img : colorImages) {
204 const GpuBarrierInfo prev(GpuCommandStage::NONE, GpuAccessStage::NONE);
205 const GpuBarrierInfo next(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE);
206 const GpuImageRange imgInfo = {
207 .baseLayer = img.arrayLevel,
208 .numLayers = 1,
209 .baseMipLevel = 0,
210 .numMipLevels = 1,
211 };
212
213 if (autoSync)
214 SetGpuImageBarrier(img.targetImage, GpuImageLayout::UNDEFINED, GpuImageLayout::COLOR_ATTACHMENT, prev, next, imgInfo, ResourceQueueTransferInfo::Empty());
215 }
216
217 const FLOAT clearValue[] = { color.red, color.green, color.blue, color.alpha };
218
219 for (auto colorAttachment : colorAttachments)
220 commandList->ClearRenderTargetView(colorAttachment, clearValue, 0, nullptr);
221 commandList->ClearDepthStencilView(depthStencilDesc, D3D12_CLEAR_FLAG_DEPTH, 1.0f, 0, 0, nullptr);
222
223 commandList->OMSetRenderTargets(static_cast<UINT>(colorAttachments.GetSize()), colorAttachments.GetData(), FALSE, &depthStencilDesc);
224
225 m_currentlyBoundColorImages = std::move(colorImages);
226 m_currentlyBoundDepthImage = depthImage;
227}

229void CommandListDx12::EndGraphicsRenderpass(bool autoSync) {
230 GpuImageLayout finalLayout = GpuImageLayout::SAMPLED;
231 if (m_currentlyBoundRenderTargetType == RenderpassType::FINAL)
232 finalLayout = GpuImageLayout::PRESENT;
233
234 for (const auto& img : m_currentlyBoundColorImages) {
235 GpuBarrierInfo prevBarrier{};
236 prevBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
237 prevBarrier.accessStage = GpuAccessStage::COLOR_ATTACHMENT_WRITE;
238
239 GpuBarrierInfo nextBarrier{};
240 nextBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
241 nextBarrier.accessStage = GpuAccessStage::SHADER_READ;
242
243 GpuImageRange imgBarrier{};
244 imgBarrier.baseLayer = img.arrayLevel;
245 imgBarrier.numLayers = 1;
246 imgBarrier.baseMipLevel = 0;
247 imgBarrier.numMipLevels = ALL_MIP_LEVELS;
248
249 if (autoSync)
250 SetGpuImageBarrier(img.targetImage, GpuImageLayout::COLOR_ATTACHMENT, finalLayout, prevBarrier, nextBarrier, imgBarrier, ResourceQueueTransferInfo::Empty());
251 }
252
253 if (m_currentlyBoundRenderTargetType != RenderpassType::INTERMEDIATE) {
254 m_currentlyBoundRenderTargetType = RenderpassType::INTERMEDIATE;
255 }
