142void CommandListDx12::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color, bool autoSync) {
143 DynamicArray<D3D12_CPU_DESCRIPTOR_HANDLE> colorAttachments{};
144 D3D12_CPU_DESCRIPTOR_HANDLE depthStencilDesc{};
145
146 if (currentRenderpassType== RenderpassType::FINAL) {
147 const SwapchainDx12* swapchain = Engine::GetRenderer()->_GetSwapchain()->As<SwapchainDx12>();
148
149 colorAttachments.Insert(swapchain->GetRenderTargetMemory()->GetCPUDescriptorHandleForHeapStart());
150 colorAttachments[0].ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
151 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
152
153 depthStencilDesc = swapchain->GetDepthStencilMemory()->GetCPUDescriptorHandleForHeapStart();
154 depthStencilDesc.ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
155 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
156 }
157 else {
158 GpuImageViewConfig colorView = GpuImageViewConfig::CreateTarget_Color();
159 GpuImageViewConfig depthView = GpuImageViewConfig::CreateTarget_DepthStencil();
160 depthView.baseArrayLevel = depthImage.arrayLevel;
161
162 for (const auto& [image, layer, clear] : colorImages) {
163 colorView.baseArrayLevel = layer;
164 colorAttachments.Insert(image->GetView(colorView)
165 ->As<GpuImageViewDx12>()->GetCpuAddress());
166 }
167
168 depthStencilDesc = depthImage.targetImage->GetView(depthView)
169 ->As<GpuImageViewDx12>()->GetCpuAddress();
170 }
171
172 for (const auto& img : colorImages) {
173 const GpuBarrierInfo prev(GpuCommandStage::NONE, GpuAccessStage::NONE);
174 const GpuBarrierInfo next(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE);
175 const GpuImageRange imgInfo = {
176 .baseLayer = img.arrayLevel,
177 .numLayers = 1,
178 .baseMipLevel = 0,
179 .numMipLevels = 1,
180 };
181
182 if (autoSync)
183 SetGpuImageBarrier(img.targetImage, GpuImageLayout::UNDEFINED, GpuImageLayout::COLOR_ATTACHMENT, prev, next, imgInfo);
184 }
185
186 const FLOAT clearValue[] = { color.red, color.green, color.blue, color.alpha };
187
188 for (auto colorAttachment : colorAttachments)
189 commandList->ClearRenderTargetView(colorAttachment, clearValue, 0, nullptr);
190 commandList->ClearDepthStencilView(depthStencilDesc, D3D12_CLEAR_FLAG_DEPTH, 1.0f, 0, 0, nullptr);
191
192 commandList->OMSetRenderTargets(static_cast<UINT>(colorAttachments.GetSize()), colorAttachments.GetData(), FALSE, &depthStencilDesc);
193
194 currentColorImages = std::move(colorImages);
195 currentDepthImage = depthImage;
196}

198void CommandListDx12::EndGraphicsRenderpass(bool autoSync) {
199 GpuImageLayout finalLayout = GpuImageLayout::SAMPLED;
200 if (currentRenderpassType == RenderpassType::FINAL)
201 finalLayout = GpuImageLayout::PRESENT;
202
203 for (const auto& img : currentColorImages) {
204 GpuBarrierInfo prevBarrier{};
205 prevBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
206 prevBarrier.accessStage = GpuAccessStage::COLOR_ATTACHMENT_WRITE;
207
208 GpuBarrierInfo nextBarrier{};
209 nextBarrier.stage = GpuCommandStage::COLOR_ATTACHMENT_OUTPUT;
210 nextBarrier.accessStage = GpuAccessStage::SHADER_READ;
211
212 GpuImageRange imgBarrier{};
213 imgBarrier.baseLayer = img.arrayLevel;
214 imgBarrier.numLayers = 1;
215 imgBarrier.baseMipLevel = 0;
216 imgBarrier.numMipLevels = ALL_MIP_LEVELS;
217
218 if (autoSync)
219 SetGpuImageBarrier(img.targetImage, GpuImageLayout::COLOR_ATTACHMENT, finalLayout, prevBarrier, nextBarrier, imgBarrier);
220 }
221
222 if (currentRenderpassType != RenderpassType::INTERMEDIATE)
223 currentRenderpassType = RenderpassType::INTERMEDIATE;
224}
