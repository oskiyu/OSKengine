140UniquePtr<IGpuImageView> GpuImageVk::CreateView(const GpuImageViewConfig& viewConfig) const {

142
143 VkImageViewCreateInfo viewInfo{};
144
145 viewInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
146 viewInfo.image = m_image;

148 viewInfo.format = GetFormatVk(GetFormat());

154 aspectMask |= VK_IMAGE_ASPECT_DEPTH_BIT;
155 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::STENCIL))
156 aspectMask |= VK_IMAGE_ASPECT_STENCIL_BIT;
157
158 viewInfo.subresourceRange.aspectMask = aspectMask;
159
160 viewInfo.subresourceRange.baseMipLevel =
161 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
162 ? 0
163 : viewConfig.baseMipLevel;
164 viewInfo.subresourceRange.levelCount =
165 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
166 ? GetMipLevels()
167 : viewConfig.topMipLevel - viewConfig.baseMipLevel + 1;
168
169 viewInfo.subresourceRange.baseArrayLayer = viewConfig.baseArrayLevel;
170 viewInfo.subresourceRange.layerCount = viewConfig.arrayLevelCount;
171
172 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
173 const VkResult result = vkCreateImageView(device, &viewInfo, nullptr, &view);
174
175 OSK_ASSERT(result == VK_SUCCESS, ImageViewCreationException(result));
176
177 return MakeUnique<GpuImageViewVk>(view, *this, viewConfig);
