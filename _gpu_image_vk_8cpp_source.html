40void GpuImageVk::CreateVkImage() {
41 bool usesMultipleQueues = EFTraits::HasFlag(GetUsage(), GpuImageUsage::COMPUTE) && GetUsage() != GpuImageUsage::COMPUTE;
42
43 VkImageCreateInfo imageInfo{};
44
45 imageInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
46 imageInfo.imageType = GetVkImageType();
47
48 imageInfo.extent.width = GetSize3D().x;
49 imageInfo.extent.height = GetSize3D().y;
50 imageInfo.extent.depth = GetSize3D().z;
51
52 imageInfo.mipLevels = GetMipLevels();
53 imageInfo.arrayLayers = GetNumLayers();
54
55 imageInfo.format = GetFormatVk(GetFormat());
56 imageInfo.tiling = VK_IMAGE_TILING_OPTIMAL;
57 imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
58 imageInfo.usage = GetGpuImageUsageVk(GetUsage());
59 imageInfo.samples = (VkSampleCountFlagBits)GetNumSamples();
60
61 DynamicArray<uint32_t> queueIndices;
62 if (usesMultipleQueues) {
63 queueIndices.Insert(Engine::GetRenderer()->GetGraphicsCommandQueue()->As<CommandQueueVk>()->GetQueueIndex());
64 queueIndices.Insert(Engine::GetRenderer()->GetComputeCommandQueue()->As<CommandQueueVk>()->GetQueueIndex());
65
66 if (queueIndices[0] == queueIndices[1])
67 usesMultipleQueues = false;
68 }
69 else if (EFTraits::HasFlag(GetUsage(), GpuImageUsage::COMPUTE)) {
70 queueIndices.Insert(Engine::GetRenderer()->GetComputeCommandQueue()->As<CommandQueueVk>()->GetQueueIndex());
71 }
72 else {
73 queueIndices.Insert(Engine::GetRenderer()->GetGraphicsCommandQueue()->As<CommandQueueVk>()->GetQueueIndex());
74 }
75
76 imageInfo.queueFamilyIndexCount = usesMultipleQueues ? static_cast<uint32_t>(queueIndices.GetSize()) : 1;
77 imageInfo.pQueueFamilyIndices = queueIndices.GetData();
78 imageInfo.sharingMode = usesMultipleQueues
79 ? VK_SHARING_MODE_CONCURRENT
80 : VK_SHARING_MODE_EXCLUSIVE;
81
82 // Si la imagen se usará como cubemap, debemos especificarlo.
83 imageInfo.flags = EFTraits::HasFlag(GetUsage(), GpuImageUsage::CUBEMAP) ? VK_IMAGE_CREATE_CUBE_COMPATIBLE_BIT : 0;
84
85 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
86 const VkResult result = vkCreateImage(device, &imageInfo, nullptr, &image);
87
88 OSK_ASSERT(result == VK_SUCCESS, ImageCreationException(result));
89}

95void GpuImageVk::CreateVkSampler(const GpuImageSamplerDesc& samplerDesc) {
96 //Info del sampler.
97 VkSamplerCreateInfo samplerInfo{};
98 samplerInfo.sType = VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO;
99 //Filtro:
100 // VK_FILTER_LINEAR: suavizado.
101 // VK_FILTER_NEAREST: pixelado.
102 samplerInfo.minFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
103 samplerInfo.magFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
104 //AddressMode: como se accede a la imagen con TexCoords fuera de los límites.
105 samplerInfo.addressModeU = GetAddressModeVulkan(samplerDesc.addressMode);
106 samplerInfo.addressModeV = GetAddressModeVulkan(samplerDesc.addressMode);
107 samplerInfo.addressModeW = GetAddressModeVulkan(samplerDesc.addressMode);
108
109 samplerInfo.anisotropyEnable = VK_TRUE;
110 samplerInfo.maxAnisotropy = 16.0f;
111
112 samplerInfo.borderColor = VK_BORDER_COLOR_FLOAT_OPAQUE_BLACK;
113 if (samplerDesc.addressMode == GpuImageAddressMode::BACKGROUND_WHITE)
114 samplerInfo.borderColor = VK_BORDER_COLOR_FLOAT_OPAQUE_WHITE;
115
116 samplerInfo.unnormalizedCoordinates = VK_FALSE;
117
118 samplerInfo.compareEnable = VK_FALSE;
119 samplerInfo.compareOp = VK_COMPARE_OP_ALWAYS;
120
121 switch (samplerDesc.mipMapMode) {
122 case GpuImageMipmapMode::AUTO:
123 samplerInfo.minLod = 0.0f;
124 samplerInfo.maxLod = static_cast<float>(GetMipLevels());
125 break;
126
127 case GpuImageMipmapMode::CUSTOM:
128 samplerInfo.minLod = static_cast<float>(samplerDesc.minMipLevel);
129 samplerInfo.maxLod = static_cast<float>(samplerDesc.maxMipLevel);
130 break;
131
132 case GpuImageMipmapMode::NONE:
133 samplerInfo.minLod = 0.0f;
134 samplerInfo.maxLod = 0.0f;
135 break;
136 }
137
138 samplerInfo.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
139 samplerInfo.mipLodBias = -0.5f;
140
141 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
142 vkCreateSampler(device, &samplerInfo, nullptr, &sampler);
143}

161OwnedPtr<IGpuImageView> GpuImageVk::CreateView(const GpuImageViewConfig& viewConfig) const {
162 VkImageView view = VK_NULL_HANDLE;
163
164 VkImageViewCreateInfo viewInfo{};
165
166 viewInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
167 viewInfo.image = image;
168 viewInfo.viewType = viewConfig.arrayType == SampledArrayType::SINGLE_LAYER ? GetVkImageViewType() : GetVkImageArrayViewType();
169 viewInfo.format = GetFormatVk(GetFormat());
170
171 VkImageAspectFlags aspectMask = 0;
172 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::COLOR))
173 aspectMask |= VK_IMAGE_ASPECT_COLOR_BIT;
174 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::DEPTH))
175 aspectMask |= VK_IMAGE_ASPECT_DEPTH_BIT;
176 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::STENCIL))
177 aspectMask |= VK_IMAGE_ASPECT_STENCIL_BIT;
178
179 viewInfo.subresourceRange.aspectMask = aspectMask;
180
181 viewInfo.subresourceRange.baseMipLevel =
182 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
183 ? 0
184 : viewConfig.baseMipLevel;
185 viewInfo.subresourceRange.levelCount =
186 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
187 ? GetMipLevels()
188 : viewConfig.topMipLevel - viewConfig.baseMipLevel + 1;
189
190 viewInfo.subresourceRange.baseArrayLayer = viewConfig.baseArrayLevel;
191 viewInfo.subresourceRange.layerCount = viewConfig.arrayLevelCount;
192
193 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
194 const VkResult result = vkCreateImageView(device, &viewInfo, nullptr, &view);
195
196 OSK_ASSERT(result == VK_SUCCESS, ImageViewCreationException(result));
197
198 return new GpuImageViewVk(view, *this, viewConfig);
199}
