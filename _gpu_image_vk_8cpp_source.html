84void GpuImageVk::CreateVkImage() {
85 VkImageCreateInfo imageInfo{};
86
87 imageInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
88 imageInfo.imageType = GetVkImageType();
89
90 imageInfo.extent.width = GetSize3D().x;
91 imageInfo.extent.height = GetSize3D().y;
92 imageInfo.extent.depth = GetSize3D().z;
93
94 imageInfo.mipLevels = GetMipLevels();
95 imageInfo.arrayLayers = GetNumLayers();
96
97 imageInfo.format = GetFormatVk(GetFormat());
98 imageInfo.tiling = GetImageTiling() == GpuImageTiling::OPTIMAL ? VK_IMAGE_TILING_OPTIMAL : VK_IMAGE_TILING_LINEAR;
99 imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
100 imageInfo.usage = GetGpuImageUsageVk(GetUsage());
101 imageInfo.samples = (VkSampleCountFlagBits)GetMsaaSampleCount();
102
103 std::set<UIndex32> queueIndices;
104
105 const auto* renderer = Engine::GetRenderer();
106
107 if (GetOwnerQueue()->GetQueueType() == GpuQueueType::ASYNC_TRANSFER && renderer->HasTransferOnlyQueue()) {
108 queueIndices.insert(renderer->GetTransferOnlyQueue()->GetFamily().familyIndex);
109 }
110 else {
111 if (GetOwnerQueue()->GetQueueType() == GpuQueueType::PRESENTATION) {
112 if (renderer->UseUnifiedCommandQueue()) {
113 queueIndices.insert(renderer->GetUnifiedQueue()->GetFamily().familyIndex);
114 }
115 else {
116 queueIndices.insert(renderer->GetGraphicsComputeQueue()->GetFamily().familyIndex);
117 queueIndices.insert(renderer->GetPresentationQueue()->GetFamily().familyIndex);
118 }
119 }
120
121 if (GetOwnerQueue()->GetQueueType() == GpuQueueType::MAIN) {
122 if (renderer->UseUnifiedCommandQueue()) {
123 queueIndices.insert(renderer->GetUnifiedQueue()->GetFamily().familyIndex);
124 }
125 else {
126 queueIndices.insert(renderer->GetGraphicsComputeQueue()->GetFamily().familyIndex);
127 }
128 }
129 }
130
131 DynamicArray<USize32> indices = DynamicArray<USize32>::CreateReservedArray(static_cast<USize32>(queueIndices.size()));
132 for (const auto& index : queueIndices) {
133 indices.Insert(index);
134 }
135
136 imageInfo.queueFamilyIndexCount = static_cast<USize32>(queueIndices.size());
137 imageInfo.pQueueFamilyIndices = indices.GetData();
138 imageInfo.sharingMode = queueIndices.size() == 1
139 ? VK_SHARING_MODE_EXCLUSIVE
140 : VK_SHARING_MODE_CONCURRENT;
141
142 // Si la imagen se usará como cubemap, debemos especificarlo.
143 imageInfo.flags = EFTraits::HasFlag(GetUsage(), GpuImageUsage::CUBEMAP) ? VK_IMAGE_CREATE_CUBE_COMPATIBLE_BIT : 0;
144
145 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
146 const VkResult result = vkCreateImage(device, &imageInfo, nullptr, &m_image);
147
148 OSK_ASSERT(result == VK_SUCCESS, ImageCreationException(result));
149}

155void GpuImageVk::CreateVkSampler(const GpuImageSamplerDesc& samplerDesc) {
156 //Info del sampler.
157 VkSamplerCreateInfo samplerInfo{};
158 samplerInfo.sType = VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO;
159 //Filtro:
160 // VK_FILTER_LINEAR: suavizado.
161 // VK_FILTER_NEAREST: pixelado.
162 samplerInfo.minFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
163 samplerInfo.magFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
164 //AddressMode: como se accede a la imagen con TexCoords fuera de los límites.
165 samplerInfo.addressModeU = GetAddressModeVulkan(samplerDesc.addressMode);
166 samplerInfo.addressModeV = GetAddressModeVulkan(samplerDesc.addressMode);
167 samplerInfo.addressModeW = GetAddressModeVulkan(samplerDesc.addressMode);
168
169 samplerInfo.anisotropyEnable = VK_TRUE;
170 samplerInfo.maxAnisotropy = 16.0f;
171
172 samplerInfo.borderColor = VK_BORDER_COLOR_FLOAT_OPAQUE_BLACK;
173 if (samplerDesc.addressMode == GpuImageAddressMode::BACKGROUND_WHITE)
174 samplerInfo.borderColor = VK_BORDER_COLOR_FLOAT_OPAQUE_WHITE;
175
176 samplerInfo.unnormalizedCoordinates = VK_FALSE;
177
178 samplerInfo.compareEnable = VK_FALSE;
179 samplerInfo.compareOp = VK_COMPARE_OP_ALWAYS;
180
181 switch (samplerDesc.mipMapMode) {
182 case GpuImageMipmapMode::AUTO:
183 samplerInfo.minLod = 0.0f;
184 samplerInfo.maxLod = static_cast<float>(GetMipLevels());
185 break;
186
187 case GpuImageMipmapMode::CUSTOM:
188 samplerInfo.minLod = static_cast<float>(samplerDesc.minMipLevel);
189 samplerInfo.maxLod = static_cast<float>(samplerDesc.maxMipLevel);
190 break;
191
192 case GpuImageMipmapMode::NONE:
193 samplerInfo.minLod = 0.0f;
194 samplerInfo.maxLod = 0.0f;
195 break;
196 }
197
198 samplerInfo.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
199 samplerInfo.mipLodBias = -0.5f;
200
201 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
202 vkCreateSampler(device, &samplerInfo, nullptr, &m_sampler);
203}

221OwnedPtr<IGpuImageView> GpuImageVk::CreateView(const GpuImageViewConfig& viewConfig) const {
222 VkImageView view = VK_NULL_HANDLE;
223
224 VkImageViewCreateInfo viewInfo{};
225
226 viewInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
227 viewInfo.image = m_image;
228 viewInfo.viewType = viewConfig.arrayType == SampledArrayType::SINGLE_LAYER ? GetVkImageViewType() : GetVkImageArrayViewType();
229 viewInfo.format = GetFormatVk(GetFormat());
230
231 VkImageAspectFlags aspectMask = 0;
232 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::COLOR))
233 aspectMask |= VK_IMAGE_ASPECT_COLOR_BIT;
234 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::DEPTH))
235 aspectMask |= VK_IMAGE_ASPECT_DEPTH_BIT;
236 if (EFTraits::HasFlag(viewConfig.channel, SampledChannel::STENCIL))
237 aspectMask |= VK_IMAGE_ASPECT_STENCIL_BIT;
238
239 viewInfo.subresourceRange.aspectMask = aspectMask;
240
241 viewInfo.subresourceRange.baseMipLevel =
242 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
243 ? 0
244 : viewConfig.baseMipLevel;
245 viewInfo.subresourceRange.levelCount =
246 viewConfig.mipMapMode == GpuImageViewConfig::MipMapMode::ALL_LEVELS
247 ? GetMipLevels()
248 : viewConfig.topMipLevel - viewConfig.baseMipLevel + 1;
249
250 viewInfo.subresourceRange.baseArrayLayer = viewConfig.baseArrayLevel;
251 viewInfo.subresourceRange.layerCount = viewConfig.arrayLevelCount;
252
253 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice();
254 const VkResult result = vkCreateImageView(device, &viewInfo, nullptr, &view);
255
256 OSK_ASSERT(result == VK_SUCCESS, ImageViewCreationException(result));
257
258 return new GpuImageViewVk(view, *this, viewConfig);
259}
