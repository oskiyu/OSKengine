480Material* MaterialSystem::LoadMaterial(const std::string& path) {
481 if (materialsPathTable.contains(path))
482 return materialsPathTable.at(path);
483
484 MaterialLayout* layout = new MaterialLayout;
485
486 // Material file.
487 nlohmann::json materialInfo = nlohmann::json::parse(IO::FileIO::ReadFromFile(path));
488
489 OSK_ASSERT(materialInfo.contains("file_type"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'file_type'.", ""));
490 OSK_ASSERT(materialInfo.contains("name"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'name'.", ""));
491
492 MaterialType type = MaterialType::GRAPHICS;
493 if (materialInfo["file_type"] == "MATERIAL_RT") {
494 OSK_ASSERT(Engine::GetRenderer()->SupportsRaytracing(), RayTracingNotSupportedException());
495 type = MaterialType::RAYTRACING;
496 }
497 else if (materialInfo["file_type"] == "MATERIAL_COMPUTE") {
498 type = MaterialType::COMPUTE;
499 }
500
501 VertexInfo vertexType = {};
502
503 if (type == MaterialType::GRAPHICS) {
504 OSK_ASSERT(materialInfo.contains("vertex_type"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'vertex_type'.", path));
505 vertexType = vertexTypesTable.at(materialInfo["vertex_type"]);
506 }
507
508 int fileVersion = materialInfo["spec_ver"];
509
510 PipelineCreateInfo info{};
511 info.cullMode = PolygonCullMode::BACK;
512 info.frontFaceType = PolygonFrontFaceType::COUNTERCLOCKWISE;
513 info.isRaytracing = type == MaterialType::RAYTRACING;
514
515 if (fileVersion == 0)
516 LoadMaterialV0(layout, materialInfo, &info);
517 else if (fileVersion == 1)
518 LoadMaterialV1(layout, materialInfo, &info, type);
519 else
520 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("La versión del archivo de material no está soportada.", path))
521
522 if (materialInfo.find("config") != materialInfo.end()) {
523 if (materialInfo["config"].contains("depth_testing")) {
524 if (materialInfo["config"]["depth_testing"] == "none")
525 info.depthTestingType = DepthTestingType::NONE;
526 else if (materialInfo["config"]["depth_testing"] == "read")
527 info.depthTestingType = DepthTestingType::READ;
528 else if (materialInfo["config"]["depth_testing"] == "read/write")
529 info.depthTestingType = DepthTestingType::READ_WRITE;
530 else
531 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("Error en el archivo de material: config depth_testing inválido.", path));
532 }
533
534 if (materialInfo["config"].contains("cull_mode")) {
535 if (materialInfo["config"]["cull_mode"] == "none")
536 info.cullMode = PolygonCullMode::NONE;
537 else if (materialInfo["config"]["cull_mode"] == "front")
538 info.cullMode = PolygonCullMode::FRONT;
539 else if (materialInfo["config"]["cull_mode"] == "back")
540 info.cullMode = PolygonCullMode::BACK;
541 else
542 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("Error en el archivo de material: config cull_mode inválido.", path));
543 }
544
545 if (materialInfo["config"].contains("disable_alpha_blending"))
546 info.useAlphaBlending = false;
547 }
548
549 if (materialInfo.contains("polygon_mode"))
550 info.polygonMode = GetPolygonMode(materialInfo["polygon_mode"]);
551
552 auto output = new Material(info, layout, vertexType, type);
553 output->SetName(materialInfo["name"]);
554
555 materials.Insert(output);
556
557 materialsPathTable[path] = output;
558 materialsNameTable[materialInfo["name"]] = output;
559
560 return output;
561}
