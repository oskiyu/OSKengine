65void MaterialSystem::LoadMaterialV0(MaterialLayout* layout, const nlohmann::json& materialInfo, PipelineCreateInfo* info) {
66 OSK_ASSERT(materialInfo.contains("shader_file"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'shader_file'.", ""));
67 OSK_ASSERT(materialInfo.contains("layout"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'layout'.", ""));
68
69 info->precompiledHlslShaders = true;
70
71 // Material file
72 std::string materialName = materialInfo["name"];
73 std::string shaderFilePath = materialInfo["shader_file"];
74
75 for (auto& slotInfo : materialInfo["layout"]["slots"]) {
76 MaterialLayoutSlot slot{};
77 slot.name = slotInfo["name"];
78
79 for (auto& bindingInfo : slotInfo["bindings"]) {
80 MaterialLayoutBinding binding{};
81
82 binding.name = bindingInfo["name"];
83 binding.type = GetShaderBindingType(bindingInfo["type"]);
84
85 slot.bindings[binding.name] = binding;
86 }
87
88 layout->AddSlot(slot);
89 }
90
91 for (auto& pushConstantInfo : materialInfo["layout"]["push_constants"]) {
92 MaterialLayoutPushConstant pushConst{};
93
94 pushConst.name = pushConstantInfo["name"];
95 pushConst.size = pushConstantInfo["size"];
96
97 layout->AddPushConstant(pushConst);
98 }
99
100 // Shader file.
101 {
102 nlohmann::json shaderInfo = nlohmann::json::parse(IO::FileIO::ReadFromFile(shaderFilePath));
103
104 int fileVersion = shaderInfo["spec_ver"];
105
106 if (Engine::GetRenderer()->GetRenderApi() == RenderApiType::VULKAN) {
107 info->vertexPath = shaderInfo["glsl_shaders"]["vertex"];
108 info->fragmentPath = shaderInfo["glsl_shaders"]["fragment"];
109
110 if (shaderInfo["glsl_shaders"].contains("tesselation_control"))
111 info->tesselationControlPath = shaderInfo["glsl_shaders"]["tesselation_control"];
112
113 if (shaderInfo["glsl_shaders"].contains("tesselation_evaluation"))
114 info->tesselationEvaluationPath = shaderInfo["glsl_shaders"]["tesselation_evaluation"];
115 }
116 else if (Engine::GetRenderer()->GetRenderApi() == RenderApiType::DX12) {
117 info->vertexPath = shaderInfo["hlsl_shaders"]["vertex"];
118 info->fragmentPath = shaderInfo["hlsl_shaders"]["fragment"];
119
120 if (shaderInfo["hlsl_shaders"].contains("tesselation_control"))
121 info->tesselationControlPath = shaderInfo["hlsl_shaders"]["tesselation_control"];
122
123 if (shaderInfo["hlsl_shaders"].contains("tesselation_evaluation"))
124 info->tesselationEvaluationPath = shaderInfo["hlsl_shaders"]["tesselation_evaluation"];
125 }
126
127 for (auto& slotInfo : shaderInfo["slots"]) {
128 std::string slotName = slotInfo["name"];
129 auto& slot = layout->GetSlot(slotName);
130 slot.glslSetIndex = slotInfo["glsl_set"];
131
132 for (auto& stage : slotInfo["shader_stages"])
133 EFTraits::AddFlag(&slot.stage, GetShaderStage(stage));
134
135 for (auto& bindingInfo : slotInfo["bindings"]) {
136 std::string bindingName = bindingInfo["name"];
137
138 slot.bindings[bindingName].glslIndex = bindingInfo["glsl_index"];
139 slot.bindings[bindingName].hlslIndex = bindingInfo["hlsl_index"];
140 }
141 }
142
143 for (auto& pushConstantInfo : shaderInfo["push_constants"]) {
144 auto& pushConst = layout->GetPushConstant(pushConstantInfo["name"]);
145
146 for (auto& stage : pushConstantInfo["shader_stages"])
147 EFTraits::AddFlag(&pushConst.stage, GetShaderStage(stage));
148
149 pushConst.hlslIndex = pushConstantInfo["hlsl_index"];
150 }
151 }
152
153 UIndex32 hlslDescIndex = 0;
154 for (auto& [name, slot] : layout->GetAllSlots()) {
155 for (auto& binding : slot.bindings) {
156 binding.second.hlslDescriptorIndex = hlslDescIndex;
157
158 hlslDescIndex++;
159 }
160 }
161
162 for (auto& pushConstInfo : layout->GetAllPushConstants()) {
163 pushConstInfo.second.hlslBindingIndex = hlslDescIndex;
164 hlslDescIndex++;
165 }

169Material* MaterialSystem::LoadMaterial(const std::string& path) {
170 if (materialsPathTable.contains(path))
171 return materialsPathTable.at(path);
172
173 UniquePtr<MaterialLayout> layout = MakeUnique<MaterialLayout>(path);
174
175 // Material file.
176 nlohmann::json materialInfo = nlohmann::json::parse(IO::FileIO::ReadFromFile(path));
177
178 OSK_ASSERT(materialInfo.contains("file_type"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'file_type'.", ""));
179 OSK_ASSERT(materialInfo.contains("name"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'name'.", ""));
180
181 MaterialType type = MaterialType::GRAPHICS;
182 if (materialInfo["file_type"] == "MATERIAL_RT") {
183 OSK_ASSERT(Engine::GetRenderer()->SupportsRaytracing(), RayTracingNotSupportedException());
184 type = MaterialType::RAYTRACING;
185 }
186 else if (materialInfo["file_type"] == "MATERIAL_COMPUTE") {
187 type = MaterialType::COMPUTE;
188 }
189 else if (materialInfo["file_type"] == "MATERIAL_MESH") {
190 type = MaterialType::MESH;
191 }
192
193 VertexInfo vertexType = {};
194
195 if (type == MaterialType::GRAPHICS) {
196 OSK_ASSERT(materialInfo.contains("vertex_type"), ASSETS::InvalidDescriptionFileException("Archivo de material incorrecto: no se encuentra 'vertex_type'.", path));
197 vertexType = vertexTypesTable.at(materialInfo["vertex_type"]);
198 }
199
200 int fileVersion = materialInfo["spec_ver"];
201
202 PipelineCreateInfo info{};
203 info.cullMode = PolygonCullMode::BACK;
204 info.frontFaceType = PolygonFrontFaceType::COUNTERCLOCKWISE;
205
206 if (fileVersion == 0)
207 LoadMaterialV0(layout.GetPointer(), materialInfo, &info);
208 else if (fileVersion == 1)
209 layout = LoadMaterialLayoutV1(materialInfo, &info, type);
210 else
211 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("La versión del archivo de material no está soportada.", path))
212
213 if (materialInfo.find("config") != materialInfo.end()) {
214 if (materialInfo["config"].contains("depth_testing")) {
215 if (materialInfo["config"]["depth_testing"] == "none")
216 info.depthTestingType = DepthTestingType::NONE;
217 else if (materialInfo["config"]["depth_testing"] == "read")
218 info.depthTestingType = DepthTestingType::READ;
219 else if (materialInfo["config"]["depth_testing"] == "read/write")
220 info.depthTestingType = DepthTestingType::READ_WRITE;
221 else
222 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("Error en el archivo de material: config depth_testing inválido.", path));
223 }
224
225 if (materialInfo["config"].contains("cull_mode")) {
226 if (materialInfo["config"]["cull_mode"] == "none")
227 info.cullMode = PolygonCullMode::NONE;
228 else if (materialInfo["config"]["cull_mode"] == "front")
229 info.cullMode = PolygonCullMode::FRONT;
230 else if (materialInfo["config"]["cull_mode"] == "back")
231 info.cullMode = PolygonCullMode::BACK;
232 else
233 OSK_ASSERT(false, ASSETS::InvalidDescriptionFileException("Error en el archivo de material: config cull_mode inválido.", path));
234 }
235
236 if (materialInfo["config"].contains("disable_alpha_blending"))
237 info.useAlphaBlending = false;
238 }
239
240 if (materialInfo.contains("polygon_mode"))
241 info.polygonMode = GetPolygonMode(materialInfo["polygon_mode"]);
242
243 // Engine::GetLogger()->DebugLog(ToString(*layout));
244
245 auto material = MakeUnique<Material>(info, std::move(layout), vertexType, type);
246 auto* output = material.GetPointer();
247
248 material->SetName(materialInfo["name"]);
249
250 materials.Insert(std::move(material));
251
252 materialsPathTable[path] = output;
253 materialsNameTable[materialInfo["name"]] = output;
254
255 return output;
256}
