25GpuMemoryUsageInfo GpuVk::GetMemoryUsageInfo() const {
26 VkPhysicalDeviceProperties properties{};
27 vkGetPhysicalDeviceProperties(physicalDevice, &properties);
28
29 VkPhysicalDeviceMemoryBudgetPropertiesEXT memoryBudget{};
30 memoryBudget.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_MEMORY_BUDGET_PROPERTIES_EXT;
31 memoryBudget.pNext = NULL;
32
33 VkPhysicalDeviceMemoryProperties2 memoryProperties{};
34 memoryProperties.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_MEMORY_PROPERTIES_2;
35 memoryProperties.pNext = &memoryBudget;
36
37 vkGetPhysicalDeviceMemoryProperties2(physicalDevice, &memoryProperties);
38
39 GpuMemoryUsageInfo output{};
40
41 for (UIndex32 i = 0; i < memoryProperties.memoryProperties.memoryHeapCount; i++) {
42 const GpuSharedMemoryType memoryType = memoryProperties.memoryProperties.memoryHeaps[i].flags & VK_MEMORY_HEAP_DEVICE_LOCAL_BIT
43 ? GpuSharedMemoryType::GPU_ONLY
44 : GpuSharedMemoryType::GPU_AND_CPU;
45
46 GpuHeapMemoryUsageInfo heapInfo{};
47 heapInfo.maxCapacity = memoryBudget.heapBudget[i];
48 heapInfo.usedSpace = memoryBudget.heapUsage[i];
49 heapInfo.memoryType = memoryType;
50
51 output.heapsInfo.Insert(heapInfo);
52
53 if (memoryType == GpuSharedMemoryType::GPU_ONLY) {
54 output.gpuOnlyMemoryInfo.maxCapacity += heapInfo.maxCapacity;
55 output.gpuOnlyMemoryInfo.usedSpace += heapInfo.usedSpace;
56 }
57 else {
58 output.gpuAndCpuMemoryInfo.maxCapacity += heapInfo.maxCapacity;
59 output.gpuAndCpuMemoryInfo.usedSpace += heapInfo.usedSpace;
60 }
61 }
62
63 return output;

117void GpuVk::CreateLogicalDevice() {
118 const QueueFamiles families = GetQueueFamilyIndices();
119
120 const auto graphicsAndComputeFamiliesList = families.GetFamilies(
121 CommandsSupport::GRAPHICS | CommandsSupport::COMPUTE);
122 const auto transferOnlyQueues = families.GetFamilies(CommandsSupport::TRANSFER);
123
124 std::set<uint32_t> uniqueQueueFamilies;
125
126 const QueueFamily graphicsAndComputeFamily = graphicsAndComputeFamiliesList.At(0);
127 uniqueQueueFamilies.insert(graphicsAndComputeFamily.familyIndex);
128
129 if (!EFTraits::HasFlag(graphicsAndComputeFamily.support, CommandsSupport::PRESENTATION)) {
130 uniqueQueueFamilies.insert(families.GetFamilies(CommandsSupport::PRESENTATION).AtCpy(0).familyIndex);
131 }
132
133 for (const auto& family : transferOnlyQueues) {
134 // Sólamente transfer.
135 if (family.support == CommandsSupport::TRANSFER) {
136 uniqueQueueFamilies.insert(family.familyIndex);
137 break;
138 }
139 }
140
141 // Creación de las colas.
142 DynamicArray<VkDeviceQueueCreateInfo> createInfos;
143 float_t qPriority = 1.0f;
144 for (auto& qFamily : uniqueQueueFamilies) {
145 VkDeviceQueueCreateInfo qCreateInfo{};
146 qCreateInfo.sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
147 qCreateInfo.queueFamilyIndex = qFamily;
148 qCreateInfo.queueCount = 1;
149
150 // Prioridades.
151 qCreateInfo.pQueuePriorities = &qPriority;
152
153 createInfos.Insert(std::move(qCreateInfo));
154 }
155
156 // Características que vamos a usar.
157 VkPhysicalDeviceFeatures features{};
158 features.samplerAnisotropy = VK_TRUE;
159 features.tessellationShader = VK_TRUE;
160 features.fillModeNonSolid = VK_TRUE;
161
162 DynamicArray<VkExtensionProperties> availableExtensions = GetAvailableExtensions(physicalDevice);
163 DynamicArray<const char*> gpuExtensions = { VK_KHR_SWAPCHAIN_EXTENSION_NAME };
164
165 gpuExtensions.Insert(VK_EXT_MEMORY_BUDGET_EXTENSION_NAME);
166
167 // RT
168 if (info.IsRtCompatible()) {
169 gpuExtensions.Insert(VK_KHR_ACCELERATION_STRUCTURE_EXTENSION_NAME);
170 gpuExtensions.Insert(VK_KHR_RAY_TRACING_PIPELINE_EXTENSION_NAME);
171
172 gpuExtensions.Insert(VK_KHR_BUFFER_DEVICE_ADDRESS_EXTENSION_NAME);
173 gpuExtensions.Insert(VK_KHR_DEFERRED_HOST_OPERATIONS_EXTENSION_NAME);
174 gpuExtensions.Insert(VK_EXT_DESCRIPTOR_INDEXING_EXTENSION_NAME);
175
176 gpuExtensions.Insert(VK_KHR_SPIRV_1_4_EXTENSION_NAME);
177 gpuExtensions.Insert(VK_KHR_SHADER_FLOAT_CONTROLS_EXTENSION_NAME);
178 }
179
180 // Crear el logical device.
181 VkDeviceCreateInfo createInfo{};
182 createInfo.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO;
183 createInfo.queueCreateInfoCount = static_cast<uint32_t>(createInfos.GetSize());
184 createInfo.pQueueCreateInfos = createInfos.GetData();
185 createInfo.pEnabledFeatures = &features;
186 createInfo.enabledExtensionCount = static_cast<uint32_t>(gpuExtensions.GetSize());
187 createInfo.ppEnabledExtensionNames = gpuExtensions.GetData();
188
189 createInfo.pEnabledFeatures = nullptr;
190 createInfo.pNext = &info.extendedFeatures;
191 info.extendedFeatures.features = features;
192 info.extendedFeatures.pNext = &info.dynamicRenderingFeatures;
193
194 info.dynamicRenderingFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_DYNAMIC_RENDERING_FEATURES;
195 info.dynamicRenderingFeatures.dynamicRendering = VK_TRUE;
196 info.dynamicRenderingFeatures.pNext = &info.syncFeatures;
197
198 info.syncFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_SYNCHRONIZATION_2_FEATURES;
199 info.syncFeatures.synchronization2 = VK_TRUE;
200 info.syncFeatures.pNext = nullptr;
201
202 if (info.IsRtCompatible()) {
203 info.syncFeatures.pNext = &info.rtAccelerationStructuresFeatures;
204
205 info.rtAccelerationStructuresFeatures = {};
206 info.rtAccelerationStructuresFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_ACCELERATION_STRUCTURE_FEATURES_KHR,
207 info.rtAccelerationStructuresFeatures.accelerationStructure = VK_TRUE;
208 info.rtAccelerationStructuresFeatures.pNext = &info.rtDeviceAddressFeatures;
209
210 info.rtDeviceAddressFeatures = {};
211 info.rtDeviceAddressFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_BUFFER_DEVICE_ADDRESS_FEATURES,
212 info.rtDeviceAddressFeatures.bufferDeviceAddress = VK_TRUE;
213 info.rtDeviceAddressFeatures.pNext = &info.rtPipelineFeatures;
214
215 info.rtPipelineFeatures = {};
216 info.rtPipelineFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_RAY_TRACING_PIPELINE_FEATURES_KHR,
217 info.rtPipelineFeatures.rayTracingPipeline = VK_TRUE;
218 info.rtPipelineFeatures.pNext = &info.bindlessTexturesSets;
219
220 info.bindlessTexturesSets.shaderSampledImageArrayNonUniformIndexing = VK_TRUE;
221 info.bindlessTexturesSets.runtimeDescriptorArray = VK_TRUE;
222 info.bindlessTexturesSets.descriptorBindingVariableDescriptorCount = VK_TRUE;
223 info.bindlessTexturesSets.pNext = nullptr;
224 }
225
226 // Crear el logical device.
227 VkResult result = vkCreateDevice(physicalDevice, &createInfo, nullptr, &logicalDevice);
228
229 // Error-handling.
230 OSK_ASSERT(result == VK_SUCCESS, LogicalDeviceCreationException(result));
231}

241QueueFamiles GpuVk::GetQueueFamilyIndices() const {
242 QueueFamiles output{};
243
244 // Número de familias.
245 uint32_t queueFamilyCount = 0;
246 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, nullptr);
247
248 // Obtener las familias.
249 auto queueFamilies = DynamicArray<VkQueueFamilyProperties>::CreateResizedArray(queueFamilyCount);
250
251 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, queueFamilies.GetData());
252
253 // Comprobar el soporte de distintos tipos de cola.
254 int familyIndex = 0;
255 for (const auto& q : queueFamilies) {
256 QueueFamily family{};
257 family.familyIndex = familyIndex;
258 family.numQueues = q.queueCount;
259
260 if (q.queueFlags & VK_QUEUE_GRAPHICS_BIT)
261 family.support |= CommandsSupport::GRAPHICS;
262
263 if (q.queueFlags & VK_QUEUE_COMPUTE_BIT)
264 family.support |= CommandsSupport::COMPUTE;
265
266 if (q.queueFlags & VK_QUEUE_TRANSFER_BIT)
267 family.support |= CommandsSupport::TRANSFER;
268
269 // Según el spec, si una familia soporta tanto GRAPHICS como COMPUTE, también debe soportar transfer, aunque no esté especificado explícitamente en el struct.
270 if (EFTraits::HasFlag(family.support, CommandsSupport::COMPUTE) && EFTraits::HasFlag(family.support, CommandsSupport::GRAPHICS))
271 family.support |= CommandsSupport::TRANSFER;
272
273 // Soporte para presentación.
274 VkBool32 presentSupport = false;
275 vkGetPhysicalDeviceSurfaceSupportKHR(physicalDevice, familyIndex, surface, &presentSupport);
276
277 if (presentSupport)
278 family.support |= CommandsSupport::PRESENTATION;
279
280 output.families.Insert(family);
281
282 familyIndex++;
283 }
284
285 return output;
286}
