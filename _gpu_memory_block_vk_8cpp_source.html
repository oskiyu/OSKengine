107VkMemoryAllocateFlags GpuMemoryBlockVk::GetMemoryAllocateFlags(GpuBufferUsage usage) {
108 VkMemoryAllocateFlags output = 0;
109
110 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE))
111 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
112
113 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE_BUILDING))
114 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
115
116 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_SHADER_BINDING_TABLE))
117 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
118
119 if (Engine::GetRenderer()->SupportsRaytracing() &&
120 (EFTraits::HasFlag(usage, GpuBufferUsage::VERTEX_BUFFER) || EFTraits::HasFlag(usage, GpuBufferUsage::INDEX_BUFFER)))
121 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
122
123 return output;
124}

126uint32_t GpuMemoryBlockVk::GetMemoryType(uint32_t memoryTypeFilter, GpuVk* device, GpuSharedMemoryType type) {
127 auto nativeSharedMode = GetGpuSharedMemoryTypeVk(type);
128
129 for (uint32_t i = 0; i < device->GetInfo().memoryProperties.memoryTypeCount; i++)
130 if ((memoryTypeFilter & (1 << i)) && ((device->GetInfo().memoryProperties.memoryTypes[i].propertyFlags & nativeSharedMode) == nativeSharedMode))
131 return i;
132
133 OSK_ASSERT(false, GpuMemoryAllocException(0));
134
135 return -1;
