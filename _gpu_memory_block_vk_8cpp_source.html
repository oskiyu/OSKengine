112VkMemoryAllocateFlags GpuMemoryBlockVk::GetMemoryAllocateFlags(GpuBufferUsage usage) {
113 VkMemoryAllocateFlags output = 0;
114
115 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE))
116 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
117
118 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE_BUILDING))
119 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
120
121 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_SHADER_BINDING_TABLE))
122 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
123
124 if (Engine::GetRenderer()->SupportsRaytracing() &&
125 (EFTraits::HasFlag(usage, GpuBufferUsage::VERTEX_BUFFER) || EFTraits::HasFlag(usage, GpuBufferUsage::INDEX_BUFFER)))
126 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
127
128 return output;
129}

131uint32_t GpuMemoryBlockVk::GetMemoryType(uint32_t memoryTypeFilter, GpuVk* device, GpuSharedMemoryType type) {
132 auto nativeSharedMode = GetGpuSharedMemoryTypeVk(type);
133
134 for (uint32_t i = 0; i < device->GetInfo().memoryProperties.memoryTypeCount; i++)

136 return i;

138 OSK_ASSERT(false, GpuMemoryAllocException(0));
139
140 return -1;
