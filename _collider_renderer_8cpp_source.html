34ColliderRenderSystem::ColliderRenderSystem() {
35 Signature signature{};
36 signature.SetTrue(Engine::GetEcs()->GetComponentType<Transform3D>());
37 signature.SetTrue(Engine::GetEcs()->GetComponentType<CollisionComponent>());
38 _SetSignature(signature);
39
40 // Material load
41 material = Engine::GetRenderer()->GetMaterialSystem()
42 ->LoadMaterial("Resources/Materials/Collision/collision_material.json");
43 lowLevelMaterial = Engine::GetRenderer()->GetMaterialSystem()
44 ->LoadMaterial("Resources/Materials/Collision/lowlevel_collision_material.json");
45 pointMaterial = Engine::GetRenderer()->GetMaterialSystem()
46 ->LoadMaterial("Resources/Materials/Collision/collision_point_material.json");
47
48 materialInstance = material->CreateInstance().GetPointer();
49 pointsMaterialInstance = pointMaterial->CreateInstance().GetPointer();
50
51 // Asset load
52 cubeModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/cube.json");
53 sphereModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/sphere.json");
54
55 // Material setup
56 const GpuBuffer* _cameraUbos[NUM_RESOURCES_IN_FLIGHT]{};
57 for (UIndex32 i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
58 cameraUbos[i] = Engine::GetRenderer()->GetAllocator()
59 ->CreateUniformBuffer(sizeof(glm::mat4) * 2 + sizeof(glm::vec4))
60 .GetPointer();
61 _cameraUbos[i] = cameraUbos[i].GetPointer();
62 }
63
64 materialInstance->GetSlot("global")->SetUniformBuffers("camera", _cameraUbos);
65 materialInstance->GetSlot("global")->FlushUpdate();
66
67 pointsMaterialInstance->GetSlot("global")->SetUniformBuffers("camera", _cameraUbos);
68 pointsMaterialInstance->GetSlot("global")->FlushUpdate();
69}

81void ColliderRenderSystem::Render(GRAPHICS::ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
82 if (cameraObject == EMPTY_GAME_OBJECT)
83 return;
84
85 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
86
87 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
88 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
89
90 cameraUbos[resourceIndex]->MapMemory();
91 cameraUbos[resourceIndex]->Write(camera.GetProjectionMatrix());
92 cameraUbos[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
93 cameraUbos[resourceIndex]->Write(cameraTransform.GetPosition());
94 cameraUbos[resourceIndex]->Unmap();
95
96 // Información por cada entidad.
97 struct RenderInfo {
98 glm::mat4 modelMatrix;
99 Color color;
100 } renderInfo {};
101
102 commandList->StartDebugSection("Collision Render", Color::Red);
103
104 commandList->BeginGraphicsRenderpass(&m_renderTarget, Color::Black * 0.0f);
105 SetupViewport(commandList);
106
107
108 // Obtenemos todas las entidades que están colisionando,
109 // para poder cambiarles de color.
110 const auto& eventQueue = Engine::GetEcs()->GetEventQueue<CollisionEvent>();
111 DynamicArray<Vector3f> contactPoints;
112 DynamicArray<Vector3f> singleContactPoints;
113 DynamicArray<Vector3f> rayCastContactPoints;
114 std::set<GameObjectIndex> collisionObjects;
115 std::set<GameObjectIndex> rayCastedObjects;
116
117 for (const CollisionEvent& ev : eventQueue) {
118 collisionObjects.insert(ev.firstEntity);
119 collisionObjects.insert(ev.secondEntity);
120 contactPoints.InsertAll(ev.collisionInfo.GetContactPoints());
121 singleContactPoints.Insert(ev.collisionInfo.GetSingleContactPoint());
122 }
123
124 for (GameObjectIndex gameObject : objects) {
125 commandList->BindMaterial(*material);
126 commandList->BindMaterialSlot(*materialInstance->GetSlot("global"));
127
128 const Transform3D& originalTransform = Engine::GetEcs()->GetComponent<Transform3D>(gameObject);
129 Transform3D topLevelTransform = originalTransform;
130 topLevelTransform.SetRotation({});
131
132 if (gameObject == 1) {
133 COLLISION::Ray ray{};
134 ray.origin = originalTransform.GetPosition() + Vector3f(0.0f, 0.11f, 0.0f);
135 ray.direction = originalTransform.GetForwardVector();
136
137 const auto rayResult = Engine::GetEcs()->GetSystem<CollisionSystem>()->CastRay(ray, gameObject);
138 if (rayResult.Result()) {
139 rayCastContactPoints.Insert(rayResult.GetIntersectionPoint());
140 rayCastedObjects.insert(rayResult.GetObject());
141 }
142 }
143
144 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(gameObject).GetCollider();
145 const ITopLevelCollider* topLevelCollider = collider.GetTopLevelCollider();
146
147 if (rayCastedObjects.contains(gameObject))
148 renderInfo.color = Color::Green * 0.75f;
149 // else if (collisionObjects.contains(gameObject))
150 // renderInfo.color = Color::Red * 0.75f;
151 else
152 renderInfo.color = Color::Yellow * 0.75f;
153
154 if (auto* box = dynamic_cast<const AxisAlignedBoundingBox*>(topLevelCollider)) {
155 topLevelTransform.SetScale(box->GetSize());
156 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
157 commandList->PushMaterialConstants("pushConstants", renderInfo);
158
159 commandList->BindVertexBuffer(cubeModel.GetAsset()->GetModel().GetVertexBuffer());
160 commandList->BindIndexBuffer(cubeModel.GetAsset()->GetModel().GetIndexBuffer());
161 commandList->DrawSingleInstance(cubeModel.GetAsset()->GetModel().GetTotalIndexCount());
162 } else
163 if (auto* sphere = dynamic_cast<const SphereCollider*>(topLevelCollider)) {
164 topLevelTransform.SetScale(Vector3f(sphere->GetRadius()));
165 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
166 commandList->PushMaterialConstants("pushConstants", renderInfo);
167
168 commandList->BindVertexBuffer(sphereModel.GetAsset()->GetModel().GetVertexBuffer());
169 commandList->BindIndexBuffer(sphereModel.GetAsset()->GetModel().GetIndexBuffer());
170 commandList->DrawSingleInstance(sphereModel.GetAsset()->GetModel().GetTotalIndexCount());
171 }
172
173 if (bottomLevelVertexBuffers.contains(gameObject)) {
174 const auto& vertexBuffers = bottomLevelVertexBuffers.at(gameObject);
175 const auto& indexBuffers = bottomLevelIndexBuffers.at(gameObject);
176
177 if (rayCastedObjects.contains(gameObject))
178 renderInfo.color = Color::Green * 0.75f;
179 // else if (collisionObjects.contains(gameObject))
180 // renderInfo.color = Color::Red * 0.75f;
181 else
182 renderInfo.color = Color::Yellow * 0.75f;
183
184 renderInfo.modelMatrix = originalTransform.GetAsMatrix();
185 commandList->PushMaterialConstants("pushConstants", renderInfo);
186
187 commandList->BindMaterial(*lowLevelMaterial);
188 for (UIndex64 i = 0; i < vertexBuffers.GetSize(); i++) {
189 commandList->BindVertexBuffer(*vertexBuffers[i].GetPointer());
190 commandList->BindIndexBuffer(*indexBuffers[i].GetPointer());
191 commandList->DrawSingleInstance(indexBuffers[i]->GetIndexView().numIndices);
192 }
193 }
194 }
195
196 commandList->BindVertexBuffer(cubeModel.GetAsset()->GetModel().GetVertexBuffer());
197 commandList->BindIndexBuffer(cubeModel.GetAsset()->GetModel().GetIndexBuffer());
198
199 commandList->BindMaterial(*pointMaterial);
200 commandList->BindMaterialSlot(*pointsMaterialInstance->GetSlot("global"));
201
202 struct {
203 Vector4f point;
204 Color color = Color::Red;
205 } pushConstant;
206
207 for (const Vector3f point : contactPoints) {
208 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
209 commandList->PushMaterialConstants("pushConstants", pushConstant);
210 commandList->DrawSingleInstance(1);
211 }
212
213 pushConstant.color = Color::Purple;
214 for (const Vector3f point : singleContactPoints) {
215 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
216 commandList->PushMaterialConstants("pushConstants", pushConstant);
217 commandList->DrawSingleInstance(1);
218 }
219
220 pushConstant.color = Color::Green;
221 for (const Vector3f point : rayCastContactPoints) {
222 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
223 commandList->PushMaterialConstants("pushConstants", pushConstant);
224 commandList->DrawSingleInstance(1);
225 }
226
227 commandList->EndGraphicsRenderpass();
228
229 commandList->EndDebugSection();
230}

243void ColliderRenderSystem::SetupBottomLevelModel(GameObjectIndex obj) {
244 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(obj).GetCollider();
245
246 if (bottomLevelVertexBuffers.contains(obj)) {
247 bottomLevelVertexBuffers.at(obj).Empty();
248 bottomLevelIndexBuffers.at(obj).Empty();
249 }
250
251 // Listas con los vertex e index buffers de la entidad.
252 // Cada ConvexVolume tendrá uno.
253 DynamicArray<OwnedPtr<GpuBuffer>> vertexBuffers;
254 DynamicArray<OwnedPtr<GpuBuffer>> indexBuffers;
255
256
257 for (UIndex32 c = 0; c < collider.GetBottomLevelCollidersCount(); c++) {
258 const auto& blc = *collider.GetBottomLevelCollider(c)->As<ConvexVolume>();
259
260 // "Convertimos" los vértices del collider
261 // en vértices de renderizado.
262 DynamicArray<Vertex3D> vertices;
263 for (const auto& cVertex : blc.GetLocalSpaceVertices()){
264 Vertex3D vertex{};
265
266 vertex.position = cVertex;
267 vertex.normal = Vector3f::Zero;
268 vertex.color = Color::White;
269 vertex.texCoords = Vector2f::Zero;
270
271 vertices.Insert(vertex);
272 }
273
274 // Índices de vértices.
275 DynamicArray<TIndexSize> indices;
276 for (const auto& face : blc.GetFaceIndices()) {
277 for (const auto index : face)
278 indices.Insert(index);
279
280 // Re-añadimos el último vértice de la geometría,
281 // para "cerrar" el polígono, creando una
282 // línea entre el primer y el último vértice.
283 indices.Insert(face[0]);
284 }
285
286 // Creamos los buffers.
287 vertexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateVertexBuffer(vertices, Vertex3D::GetVertexInfo()).GetPointer());
288 indexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateIndexBuffer(indices).GetPointer());
289 }
290
291 // Si no existían las listas, las introducimos primero.
292 if (!bottomLevelVertexBuffers.contains(obj)) {
293 bottomLevelVertexBuffers[obj] = {};
294 bottomLevelIndexBuffers[obj] = {};
295 }
296
297 for (const auto& v : vertexBuffers)
298 bottomLevelVertexBuffers.at(obj).Insert(v.GetPointer());
299
300 for (const auto& i : indexBuffers)
301 bottomLevelIndexBuffers.at(obj).Insert(i.GetPointer());
302}
