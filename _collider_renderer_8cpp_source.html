47ColliderRenderSystem::ColliderRenderSystem() {
48 Signature signature{};
49 signature.SetTrue(Engine::GetEcs()->GetComponentType<TransformComponent3D>());
50 signature.SetTrue(Engine::GetEcs()->GetComponentType<CollisionComponent>());
51 _SetSignature(signature);
52
53 // Material load
54 {
55 material = Engine::GetRenderer()->GetMaterialSystem()
56 ->LoadMaterial("Resources/Materials/Collision/collision_material.json");
57 lowLevelMaterial = Engine::GetRenderer()->GetMaterialSystem()
58 ->LoadMaterial("Resources/Materials/Collision/lowlevel_collision_material.json");
59 pointMaterial = Engine::GetRenderer()->GetMaterialSystem()
60 ->LoadMaterial("Resources/Materials/Collision/collision_point_material.json");
61
62 for (UIndex32 i = 0; i < MAX_RESOURCES_IN_FLIGHT; i++) {
63 m_materialInstances[i] = material->CreateInstance();
64 m_pointsMaterialInstances[i] = pointMaterial->CreateInstance();
65 }
66 }
67
68 // Asset load
69 {
70 m_cubeModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/cube.json");
71 m_sphereModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/sphere.json");
72 }
73
74 // Material setup
75 {
76 constexpr USize64 cameraBufferSize = sizeof(glm::mat4) * 2 + sizeof(glm::vec4);
77
78 for (UIndex32 i = 0; i < MAX_RESOURCES_IN_FLIGHT; i++) {
79 IGpuMemoryAllocator* allocator = Engine::GetRenderer()->GetAllocator();
80 UniquePtr<GpuBuffer> buffer = allocator->CreateUniformBuffer(cameraBufferSize, GpuQueueType::MAIN);
81
82 m_materialInstances[i]->GetSlot("global")->SetUniformBuffer("camera", buffer.GetValue());
83 m_materialInstances[i]->GetSlot("global")->FlushUpdate();
84
85 m_pointsMaterialInstances[i]->GetSlot("global")->SetUniformBuffer("camera", buffer.GetValue());
86 m_pointsMaterialInstances[i]->GetSlot("global")->FlushUpdate();

88 m_cameraGpuBuffers[i] = std::move(buffer);
89 }
90 }

103void ColliderRenderSystem::Render(GRAPHICS::ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
104 if (m_cameraObject == EMPTY_GAME_OBJECT) {
105 return;
106 }
107
108 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
109
110 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
111 const auto& cameraTransform = Engine::GetEcs()->GetComponent<TransformComponent3D>(m_cameraObject).GetTransform();
112
113 m_cameraGpuBuffers[resourceIndex]->MapMemory();
114 m_cameraGpuBuffers[resourceIndex]->Write(camera.GetProjectionMatrix());
115 m_cameraGpuBuffers[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
116 m_cameraGpuBuffers[resourceIndex]->Write(cameraTransform.GetPosition());
117 m_cameraGpuBuffers[resourceIndex]->Unmap();
118
119 // Informaci√≥n por cada entidad.
120 struct RenderInfo {
121 glm::mat4 modelMatrix;
122 Color color;
123 } renderInfo {};
124
125 commandList->StartDebugSection("Collision Render", Color::Red);
126
127 commandList->BeginGraphicsRenderpass(&m_renderTarget, Color::Black * 0.0f);
128 SetupViewport(commandList);
129
130
131 // Obtenemos todas las entidades que est√°n colisionando,
132 // para poder cambiarles de color.
133 const auto& eventQueue = Engine::GetEcs()->GetEventQueue<CollisionEvent>();
134 DynamicArray<Vector3f> contactPoints;
135 DynamicArray<Vector3f> singleContactPoints;
136 DynamicArray<Vector3f> rayCastContactPoints;
137 std::set<GameObjectIndex> collisionObjects;
138 std::set<GameObjectIndex> rayCastedObjects;
139
140 for (const CollisionEvent& ev : eventQueue) {
141 collisionObjects.insert(ev.firstEntity);
142 collisionObjects.insert(ev.secondEntity);
143 contactPoints.InsertAll(ev.collisionInfo.GetContactPoints());
144 singleContactPoints.Insert(ev.collisionInfo.GetSingleContactPoint());
145 }
146
147 for (GameObjectIndex gameObject : objects) {
148 commandList->BindMaterial(*material);
149 commandList->BindMaterialSlot(*m_materialInstances[resourceIndex]->GetSlot("global"));
150
151 const TransformComponent3D& originalTransform = Engine::GetEcs()->GetComponent<TransformComponent3D>(gameObject);
152 Transform3D topLevelTransform = originalTransform.GetTransform();
153 topLevelTransform.SetRotation({});
154 topLevelTransform._ApplyChanges();
155
156 singleContactPoints.Insert(originalTransform.GetTransform().GetPosition());
157
158 if (gameObject.Get() == 1) {
159 COLLISION::Ray ray{};
160 ray.origin = originalTransform.GetTransform().GetPosition() + Vector3f(0.0f, 0.11f, 0.0f);
161 ray.direction = originalTransform.GetTransform().GetForwardVector();
162
163 const auto rayResult = Engine::GetEcs()->GetSystem<CollisionSystem>()->CastRay(ray, gameObject);
164 if (rayResult.Result()) {
165 rayCastContactPoints.Insert(rayResult.GetIntersectionPoint());
166 rayCastedObjects.insert(rayResult.GetObject());
167 }
168 }
169
170 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(gameObject).GetCollider();
171 const auto* topLevelCollider = collider.GetTopLevelCollider()->GetCollider();
172
173 if (rayCastedObjects.contains(gameObject))
174 renderInfo.color = Color::Green * 0.35f;
175 // else if (collisionObjects.contains(gameObject))
176 // renderInfo.color = Color::Red * 0.75f;
177 else
178 renderInfo.color = Color::Yellow * 0.35f;
179
180 if (auto* box = dynamic_cast<const AxisAlignedBoundingBox*>(topLevelCollider)) {
181 topLevelTransform.SetScale(box->GetSize());
182 topLevelTransform._ApplyChanges();
183 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
184 commandList->PushMaterialConstants("pushConstants", renderInfo);
185
186 commandList->BindVertexBuffer(m_cubeModel.GetAsset()->GetModel().GetVertexBuffer());
187 commandList->BindIndexBuffer(m_cubeModel.GetAsset()->GetModel().GetIndexBuffer());
188 commandList->DrawSingleInstance(m_cubeModel.GetAsset()->GetModel().GetTotalIndexCount());
189 } else
190 if (auto* sphere = dynamic_cast<const SphereCollider*>(topLevelCollider)) {
191 topLevelTransform.SetScale(Vector3f(sphere->GetRadius()));
192 topLevelTransform._ApplyChanges();
193 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
194 commandList->PushMaterialConstants("pushConstants", renderInfo);
195
196 commandList->BindVertexBuffer(m_sphereModel.GetAsset()->GetModel().GetVertexBuffer());
197 commandList->BindIndexBuffer(m_sphereModel.GetAsset()->GetModel().GetIndexBuffer());
198 commandList->DrawSingleInstance(m_sphereModel.GetAsset()->GetModel().GetTotalIndexCount());
199 }
200
201 // TODO: ¬øvarios blc?
202 if (m_bottomLevelVertexBuffers.contains(gameObject)) {
203 const auto& vertexBuffers = m_bottomLevelVertexBuffers.at(gameObject);
204 const auto& indexBuffers = m_bottomLevelIndexBuffers.at(gameObject);
205
206 if (rayCastedObjects.contains(gameObject))
207 renderInfo.color = Color::Green * 0.75f;
208 // else if (collisionObjects.contains(gameObject))
209 // renderInfo.color = Color::Red * 0.75f;
210 else
211 renderInfo.color = Color::Yellow * 0.75f;
212
213 renderInfo.modelMatrix = originalTransform.GetTransform().GetAsMatrix();
214 commandList->PushMaterialConstants("pushConstants", renderInfo);
215
216 commandList->BindMaterial(*lowLevelMaterial);
217 for (UIndex64 i = 0; i < vertexBuffers.GetSize(); i++) {
218 commandList->BindVertexBuffer(*vertexBuffers[i].GetPointer());
219 commandList->BindIndexBuffer(*indexBuffers[i].GetPointer());
220 commandList->DrawSingleInstance(indexBuffers[i]->GetIndexView().numIndices);
221 }
222 }
223
224 for (UIndex64 i = 0; i < collider.GetBottomLevelCollidersCount(); i++) {
225 const auto* blc = collider.GetBottomLevelCollider(i);
226
227 if (!blc->GetCollider()->Is<SphereCollider>()) {
228 continue;
229 }
230
231 topLevelTransform.SetPosition(blc->GetTransform().GetPosition());
232 topLevelTransform.SetScale(Vector3f(blc->GetCollider()->As<SphereCollider>()->GetRadius()));
233 topLevelTransform._ApplyChanges();
234 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
235 commandList->PushMaterialConstants("pushConstants", renderInfo);
236
237 commandList->BindVertexBuffer(m_sphereModel.GetAsset()->GetModel().GetVertexBuffer());
238 commandList->BindIndexBuffer(m_sphereModel.GetAsset()->GetModel().GetIndexBuffer());
239 commandList->DrawSingleInstance(m_sphereModel.GetAsset()->GetModel().GetTotalIndexCount());
240 }
241 }
242
243 commandList->BindVertexBuffer(m_cubeModel.GetAsset()->GetModel().GetVertexBuffer());
244 commandList->BindIndexBuffer(m_cubeModel.GetAsset()->GetModel().GetIndexBuffer());
245
246 commandList->BindMaterial(*pointMaterial);
247 commandList->BindMaterialSlot(*m_pointsMaterialInstances[resourceIndex]->GetSlot("global"));
248
249 struct {
250 Vector4f point;
251 Color color = Color::Red;
252 } pushConstant;
253
254 for (const Vector3f point : contactPoints) {
255 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
256 commandList->PushMaterialConstants("pushConstants", pushConstant);
257 commandList->DrawSingleInstance(1);
258 }
259
260 pushConstant.color = Color::Purple;
261 for (const Vector3f point : singleContactPoints) {
262 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
263 commandList->PushMaterialConstants("pushConstants", pushConstant);
264 commandList->DrawSingleInstance(1);
265 }
266
267 pushConstant.color = Color::Green;
268 for (const Vector3f point : rayCastContactPoints) {
269 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
270 commandList->PushMaterialConstants("pushConstants", pushConstant);
271 commandList->DrawSingleInstance(1);
272 }
273
274 commandList->EndGraphicsRenderpass();
275
276 commandList->EndDebugSection();
277}

290void ColliderRenderSystem::SetupBottomLevelModel(GameObjectIndex obj) {
291 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(obj).GetCollider();
292
293 if (m_bottomLevelVertexBuffers.contains(obj)) {
294 m_bottomLevelVertexBuffers.at(obj).Empty();
295 m_bottomLevelIndexBuffers.at(obj).Empty();
296 }
297
298 // Listas con los vertex e index buffers de la entidad.
299 // Cada ConvexVolume tendr√° uno.
300 DynamicArray<UniquePtr<GpuBuffer>> vertexBuffers;
301 DynamicArray<UniquePtr<GpuBuffer>> indexBuffers;
302
303
304 for (UIndex32 c = 0; c < collider.GetBottomLevelCollidersCount(); c++) {
305 if (!collider.GetBottomLevelCollider(c)->GetCollider()->Is<ConvexVolume>()) {
306 continue;
307 }
308
309 const auto& blc = *collider.GetBottomLevelCollider(c)->GetCollider()->As<ConvexVolume>();
310
311 // "Convertimos" los v√©rtices del collider
312 // en v√©rtices de renderizado.
313 DynamicArray<Vertex3D> vertices;
314 for (const auto& cVertex : blc.GetLocalSpaceVertices()){
315 Vertex3D vertex{};
316
317 vertex.position = cVertex;
318 vertex.normal = Vector3f::Zero;
319 vertex.color = Color::White;
320 vertex.texCoords = Vector2f::Zero;
321
322 vertices.Insert(vertex);
323 }
324
325 // √çndices de v√©rtices.
326 DynamicArray<TIndexSize> indices;
327 for (const auto& face : blc.GetFaceIndices()) {
328 for (const auto index : face)
329 indices.Insert(index);
330
331 // Re-a√±adimos el √∫ltimo v√©rtice de la geometr√≠a,
332 // para "cerrar" el pol√≠gono, creando una
333 // l√≠nea entre el primer y el √∫ltimo v√©rtice.
334 indices.Insert(face[0]);
335 }
336
337 // Creamos los buffers.
338 vertexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateVertexBuffer(
339 vertices,
340 Vertex3D::GetVertexInfo(),
341 GpuQueueType::MAIN));
342
343 indexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateIndexBuffer(indices, GpuQueueType::MAIN));
344 }
345
346 // Si no exist√≠an las listas, las introducimos primero.
347 if (!m_bottomLevelVertexBuffers.contains(obj)) {
348 m_bottomLevelVertexBuffers[obj] = {};
349 m_bottomLevelIndexBuffers[obj] = {};
350 }
351
352 for (auto& v : vertexBuffers) {
353 m_bottomLevelVertexBuffers.at(obj).Insert(std::move(v));
354 }
355
356 for (auto& i : indexBuffers) {
357 m_bottomLevelIndexBuffers.at(obj).Insert(std::move(i));
358 }
359}
