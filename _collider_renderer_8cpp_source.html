47ColliderRenderSystem::ColliderRenderSystem() {
48 Signature signature{};
49 signature.SetTrue(Engine::GetEcs()->GetComponentType<Transform3D>());
50 signature.SetTrue(Engine::GetEcs()->GetComponentType<CollisionComponent>());
51 _SetSignature(signature);
52
53 // Material load
54 {
55 material = Engine::GetRenderer()->GetMaterialSystem()
56 ->LoadMaterial("Resources/Materials/Collision/collision_material.json");
57 lowLevelMaterial = Engine::GetRenderer()->GetMaterialSystem()
58 ->LoadMaterial("Resources/Materials/Collision/lowlevel_collision_material.json");
59 pointMaterial = Engine::GetRenderer()->GetMaterialSystem()
60 ->LoadMaterial("Resources/Materials/Collision/collision_point_material.json");
61
62 for (UIndex32 i = 0; i < MAX_RESOURCES_IN_FLIGHT; i++) {
63 m_materialInstances[i] = material->CreateInstance().GetPointer();
64 m_pointsMaterialInstances[i] = pointMaterial->CreateInstance().GetPointer();
65 }
66 }
67
68 // Asset load
69 {
70 m_cubeModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/cube.json");
71 m_sphereModel = Engine::GetAssetManager()->Load<Model3D>("Resources/Assets/Models/Colliders/sphere.json");
72 }
73
74 // Material setup
75 {
76 constexpr USize64 cameraBufferSize = sizeof(glm::mat4) * 2 + sizeof(glm::vec4);
77
78 for (UIndex32 i = 0; i < MAX_RESOURCES_IN_FLIGHT; i++) {
79 IGpuMemoryAllocator* allocator = Engine::GetRenderer()->GetAllocator();
80 OwnedPtr<GpuBuffer> buffer = allocator->CreateUniformBuffer(cameraBufferSize, GpuQueueType::MAIN);
81
82 m_materialInstances[i]->GetSlot("global")->SetUniformBuffer("camera", buffer.GetValue());
83 m_materialInstances[i]->GetSlot("global")->FlushUpdate();
84
85 m_pointsMaterialInstances[i]->GetSlot("global")->SetUniformBuffer("camera", buffer.GetValue());
86 m_pointsMaterialInstances[i]->GetSlot("global")->FlushUpdate();
87
88 m_cameraGpuBuffers[i] = buffer.GetPointer();
89 }
90 }
91}

103void ColliderRenderSystem::Render(GRAPHICS::ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
104 if (m_cameraObject == EMPTY_GAME_OBJECT) {
105 return;
106 }
107
108 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
109
110 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
111 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
112
113 m_cameraGpuBuffers[resourceIndex]->MapMemory();
114 m_cameraGpuBuffers[resourceIndex]->Write(camera.GetProjectionMatrix());
115 m_cameraGpuBuffers[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
116 m_cameraGpuBuffers[resourceIndex]->Write(cameraTransform.GetPosition());
117 m_cameraGpuBuffers[resourceIndex]->Unmap();
118
119 // Información por cada entidad.
120 struct RenderInfo {
121 glm::mat4 modelMatrix;
122 Color color;
123 } renderInfo {};
124
125 commandList->StartDebugSection("Collision Render", Color::Red);
126
127 commandList->BeginGraphicsRenderpass(&m_renderTarget, Color::Black * 0.0f);
128 SetupViewport(commandList);
129
130
131 // Obtenemos todas las entidades que están colisionando,
132 // para poder cambiarles de color.
133 const auto& eventQueue = Engine::GetEcs()->GetEventQueue<CollisionEvent>();
134 DynamicArray<Vector3f> contactPoints;
135 DynamicArray<Vector3f> singleContactPoints;
136 DynamicArray<Vector3f> rayCastContactPoints;
137 std::set<GameObjectIndex> collisionObjects;
138 std::set<GameObjectIndex> rayCastedObjects;
139
140 for (const CollisionEvent& ev : eventQueue) {
141 collisionObjects.insert(ev.firstEntity);
142 collisionObjects.insert(ev.secondEntity);
143 contactPoints.InsertAll(ev.collisionInfo.GetContactPoints());
144 singleContactPoints.Insert(ev.collisionInfo.GetSingleContactPoint());
145 }
146
147 for (GameObjectIndex gameObject : objects) {
148 commandList->BindMaterial(*material);
149 commandList->BindMaterialSlot(*m_materialInstances[resourceIndex]->GetSlot("global"));
150
151 const Transform3D& originalTransform = Engine::GetEcs()->GetComponent<Transform3D>(gameObject);
152 Transform3D topLevelTransform = originalTransform;
153 topLevelTransform.SetRotation({});
154
155 singleContactPoints.Insert(originalTransform.GetPosition());
156
157 if (gameObject.Get() == 1) {
158 COLLISION::Ray ray{};
159 ray.origin = originalTransform.GetPosition() + Vector3f(0.0f, 0.11f, 0.0f);
160 ray.direction = originalTransform.GetForwardVector();
161
162 const auto rayResult = Engine::GetEcs()->GetSystem<CollisionSystem>()->CastRay(ray, gameObject);
163 if (rayResult.Result()) {
164 rayCastContactPoints.Insert(rayResult.GetIntersectionPoint());
165 rayCastedObjects.insert(rayResult.GetObject());
166 }
167 }
168
169 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(gameObject).GetCollider();
170 const ITopLevelCollider* topLevelCollider = collider.GetTopLevelCollider();
171
172 if (rayCastedObjects.contains(gameObject))
173 renderInfo.color = Color::Green * 0.75f;
174 // else if (collisionObjects.contains(gameObject))
175 // renderInfo.color = Color::Red * 0.75f;
176 else
177 renderInfo.color = Color::Yellow * 0.75f;
178
179 if (auto* box = dynamic_cast<const AxisAlignedBoundingBox*>(topLevelCollider)) {
180 topLevelTransform.SetScale(box->GetSize());
181 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
182 commandList->PushMaterialConstants("pushConstants", renderInfo);
183
184 commandList->BindVertexBuffer(m_cubeModel.GetAsset()->GetModel().GetVertexBuffer());
185 commandList->BindIndexBuffer(m_cubeModel.GetAsset()->GetModel().GetIndexBuffer());
186 commandList->DrawSingleInstance(m_cubeModel.GetAsset()->GetModel().GetTotalIndexCount());
187 } else
188 if (auto* sphere = dynamic_cast<const SphereCollider*>(topLevelCollider)) {
189 topLevelTransform.SetScale(Vector3f(sphere->GetRadius()));
190 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
191 commandList->PushMaterialConstants("pushConstants", renderInfo);
192
193 commandList->BindVertexBuffer(m_sphereModel.GetAsset()->GetModel().GetVertexBuffer());
194 commandList->BindIndexBuffer(m_sphereModel.GetAsset()->GetModel().GetIndexBuffer());
195 commandList->DrawSingleInstance(m_sphereModel.GetAsset()->GetModel().GetTotalIndexCount());
196 }
197
198 if (m_bottomLevelVertexBuffers.contains(gameObject)) {
199 const auto& vertexBuffers = m_bottomLevelVertexBuffers.at(gameObject);
200 const auto& indexBuffers = m_bottomLevelIndexBuffers.at(gameObject);
201
202 if (rayCastedObjects.contains(gameObject))
203 renderInfo.color = Color::Green * 0.75f;
204 // else if (collisionObjects.contains(gameObject))
205 // renderInfo.color = Color::Red * 0.75f;
206 else
207 renderInfo.color = Color::Yellow * 0.75f;
208
209 renderInfo.modelMatrix = originalTransform.GetAsMatrix();
210 commandList->PushMaterialConstants("pushConstants", renderInfo);
211
212 commandList->BindMaterial(*lowLevelMaterial);
213 for (UIndex64 i = 0; i < vertexBuffers.GetSize(); i++) {
214 commandList->BindVertexBuffer(*vertexBuffers[i].GetPointer());
215 commandList->BindIndexBuffer(*indexBuffers[i].GetPointer());
216 commandList->DrawSingleInstance(indexBuffers[i]->GetIndexView().numIndices);
217 }
218 }
219 }
220
221 commandList->BindVertexBuffer(m_cubeModel.GetAsset()->GetModel().GetVertexBuffer());
222 commandList->BindIndexBuffer(m_cubeModel.GetAsset()->GetModel().GetIndexBuffer());
223
224 commandList->BindMaterial(*pointMaterial);
225 commandList->BindMaterialSlot(*m_pointsMaterialInstances[resourceIndex]->GetSlot("global"));
226
227 struct {
228 Vector4f point;
229 Color color = Color::Red;
230 } pushConstant;
231
232 for (const Vector3f point : contactPoints) {
233 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
234 commandList->PushMaterialConstants("pushConstants", pushConstant);
235 commandList->DrawSingleInstance(1);
236 }
237
238 pushConstant.color = Color::Purple;
239 for (const Vector3f point : singleContactPoints) {
240 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
241 commandList->PushMaterialConstants("pushConstants", pushConstant);
242 commandList->DrawSingleInstance(1);
243 }
244
245 pushConstant.color = Color::Green;
246 for (const Vector3f point : rayCastContactPoints) {
247 pushConstant.point = Vector4f(point.x, point.y, point.z, 1.0f);
248 commandList->PushMaterialConstants("pushConstants", pushConstant);
249 commandList->DrawSingleInstance(1);
250 }
251
252 commandList->EndGraphicsRenderpass();
253
254 commandList->EndDebugSection();
255}

268void ColliderRenderSystem::SetupBottomLevelModel(GameObjectIndex obj) {
269 const Collider& collider = *Engine::GetEcs()->GetComponent<CollisionComponent>(obj).GetCollider();
270
271 if (m_bottomLevelVertexBuffers.contains(obj)) {
272 m_bottomLevelVertexBuffers.at(obj).Empty();
273 m_bottomLevelIndexBuffers.at(obj).Empty();
274 }
275
276 // Listas con los vertex e index buffers de la entidad.
277 // Cada ConvexVolume tendrá uno.
278 DynamicArray<OwnedPtr<GpuBuffer>> vertexBuffers;
279 DynamicArray<OwnedPtr<GpuBuffer>> indexBuffers;
280
281
282 for (UIndex32 c = 0; c < collider.GetBottomLevelCollidersCount(); c++) {
283 const auto& blc = *collider.GetBottomLevelCollider(c)->As<ConvexVolume>();
284
285 // "Convertimos" los vértices del collider
286 // en vértices de renderizado.
287 DynamicArray<Vertex3D> vertices;
288 for (const auto& cVertex : blc.GetLocalSpaceVertices()){
289 Vertex3D vertex{};
290
291 vertex.position = cVertex;
292 vertex.normal = Vector3f::Zero;
293 vertex.color = Color::White;
294 vertex.texCoords = Vector2f::Zero;
295
296 vertices.Insert(vertex);
297 }
298
299 // Índices de vértices.
300 DynamicArray<TIndexSize> indices;
301 for (const auto& face : blc.GetFaceIndices()) {
302 for (const auto index : face)
303 indices.Insert(index);
304
305 // Re-añadimos el último vértice de la geometría,
306 // para "cerrar" el polígono, creando una
307 // línea entre el primer y el último vértice.
308 indices.Insert(face[0]);
309 }
310
311 // Creamos los buffers.
312 vertexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateVertexBuffer(
313 vertices,
314 Vertex3D::GetVertexInfo(),
315 GpuQueueType::MAIN).GetPointer());
316
317 indexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateIndexBuffer(indices, GpuQueueType::MAIN).GetPointer());
318 }
319
320 // Si no existían las listas, las introducimos primero.
321 if (!m_bottomLevelVertexBuffers.contains(obj)) {
322 m_bottomLevelVertexBuffers[obj] = {};
323 m_bottomLevelIndexBuffers[obj] = {};
324 }
325
326 for (const auto& v : vertexBuffers) {
327 m_bottomLevelVertexBuffers.at(obj).Insert(v.GetPointer());
328 }
329
330 for (const auto& i : indexBuffers) {
331 m_bottomLevelIndexBuffers.at(obj).Insert(i.GetPointer());
332 }
333}
