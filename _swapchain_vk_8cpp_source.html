63
64 // Formato del swapchain.

67 surfaceFormat.format = GetFormatVk(GetColorFormat());

78 if (
79 info.swapchainSupportDetails.surfaceCapabilities.minImageCount <= MAX_RESOURCES_IN_FLIGHT &&
80 info.swapchainSupportDetails.surfaceCapabilities.maxImageCount >= MAX_RESOURCES_IN_FLIGHT) {
81 SetNumImagesInFlight(MAX_RESOURCES_IN_FLIGHT);

83 else {
84 SetNumImagesInFlight(info.swapchainSupportDetails.surfaceCapabilities.maxImageCount);
85 }

87 // Create info.

93 createInfo.imageColorSpace = surfaceFormat.colorSpace;
94 createInfo.imageExtent = extent;

98 // Colas.
99 // C√≥mo se maneja el swapchain.
100 if (queueIndices.size() == 1) {
101 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
102 createInfo.queueFamilyIndexCount = 0;
103 createInfo.pQueueFamilyIndices = nullptr;
104 }
105 else {
106 bool areAllEqual = true;
107 const auto firstIndex = queueIndices[0];
108
109 for (const auto& index : queueIndices) {
110 if (firstIndex != index) {
111 areAllEqual = false;
112 break;
113 }
114 }
115
116 if (areAllEqual) {
117 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
118 createInfo.queueFamilyIndexCount = 0;
119 createInfo.pQueueFamilyIndices = nullptr;
120 }
121 else {
122 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
123 createInfo.queueFamilyIndexCount = static_cast<uint32_t>(queueIndices.size());
124 createInfo.pQueueFamilyIndices = queueIndices.data();
125 }
126 }

129 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR; // ¬øDeber√≠a mostrarse lo que hay detr√°s de la ventana?
130 createInfo.presentMode = presentMode;

132 createInfo.oldSwapchain = nullptr;

134 // Crearlo y error-handling.

136 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("No se ha podido crear el swapchain", result));
137

152void SwapchainVk::AcquireImages(const GpuVk& device) {
153 auto imageCount = GetImageCount();

155 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("Error al adquirir imagenes del swapchain", result));

159 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("Error al adquirir imagenes del swapchain", result));

179 createInfo.subresourceRange.baseMipLevel = 0;

182 createInfo.subresourceRange.layerCount = 1;
