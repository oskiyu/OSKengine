54void SwapchainVk::CreationLogic(const GpuVk& device, const Vector2ui& resolution) {
55 auto& info = device.GetInfo();
56
57 // Formato del swapchain.
58 VkSurfaceFormatKHR surfaceFormat{};
59 surfaceFormat.colorSpace = GetSupportedColorSpace(device);
60 surfaceFormat.format = GetFormatVk(GetColorFormat());
61
62 // Modo de pressentaci√≥n.
63 VkPresentModeKHR presentMode = GetPresentModeVk(GetCurrentPresentMode());
64
65 // Tama√±o.
66 VkExtent2D extent{};
67 extent.width = resolution.x;
68 extent.height = resolution.y;
69
70 // N√∫mero de im√°genes en el swapchain.
71 if (
72 info.swapchainSupportDetails.surfaceCapabilities.minImageCount <= MAX_RESOURCES_IN_FLIGHT &&
73 info.swapchainSupportDetails.surfaceCapabilities.maxImageCount >= MAX_RESOURCES_IN_FLIGHT) {
74 SetNumImagesInFlight(MAX_RESOURCES_IN_FLIGHT);
75 }
76 else {
77 SetNumImagesInFlight(info.swapchainSupportDetails.surfaceCapabilities.maxImageCount);
78 }
79
80 // Create info.
81 VkSwapchainCreateInfoKHR createInfo{};
82 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
83 createInfo.surface = device.GetSurface();
84 createInfo.minImageCount = GetImageCount();
85 createInfo.imageFormat = surfaceFormat.format;
86 createInfo.imageColorSpace = surfaceFormat.colorSpace;
87 createInfo.imageExtent = extent;
88 createInfo.imageArrayLayers = 1;
89 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
90
91 // Colas.
92 // C√≥mo se maneja el swapchain.
93 if (GetQueueIndices().size() == 1) {
94 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
95 createInfo.queueFamilyIndexCount = 0;
96 createInfo.pQueueFamilyIndices = nullptr;
97 }
98 else {
99 bool areAllEqual = true;
100 const auto firstIndex = GetQueueIndices()[0];
101
102 for (const auto& index : GetQueueIndices()) {
103 if (firstIndex != index) {
104 areAllEqual = false;
105 break;
106 }
107 }
108
109 if (areAllEqual) {
110 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
111 createInfo.queueFamilyIndexCount = 0;
112 createInfo.pQueueFamilyIndices = nullptr;
113 }
114 else {
115 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
116 createInfo.queueFamilyIndexCount = GetQueueIndices().size();
117 createInfo.pQueueFamilyIndices = GetQueueIndices().data();
118 }
119 }
120
121 createInfo.preTransform = info.swapchainSupportDetails.surfaceCapabilities.currentTransform;
122 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR; // ¬øDeber√≠a mostrarse lo que hay detr√°s de la ventana?
123 createInfo.presentMode = presentMode;
124 createInfo.clipped = VK_TRUE; // Si hay algo tapando la ventana, no se renderiza.
125 createInfo.oldSwapchain = nullptr;
126
127 // Crearlo y error-handling.
128 VkResult result = vkCreateSwapchainKHR(device.GetLogicalDevice(), &createInfo, nullptr, &m_swapchain);
129 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("No se ha podido crear el swapchain", result));
130
131 for (UIndex32 i = 0; i < GetImageCount(); i++) {
132 GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(resolution, GetColorFormat(), GpuImageUsage::COLOR);
133 const auto* queue = Engine::GetRenderer()->UseUnifiedCommandQueue()
134 ? Engine::GetRenderer()->GetUnifiedQueue()
135 : Engine::GetRenderer()->GetPresentationQueue();
136
137 SetImage(new GpuImageVk(imageInfo, queue), i);
138 }
139
140 AcquireImages(device, { extent.width, extent.height });
141 AcquireViews(device);
142}

158void SwapchainVk::AcquireViews(const GpuVk& device) {

160
161 for (UIndex32 i = 0; i < GetImageCount(); i++) {

164 createInfo.image = GetImage(i)->As<GpuImageVk>()->GetVkImage();

167 createInfo.components.r = VK_COMPONENT_SWIZZLE_IDENTITY;
168 createInfo.components.g = VK_COMPONENT_SWIZZLE_IDENTITY;
169 createInfo.components.b = VK_COMPONENT_SWIZZLE_IDENTITY;
170 createInfo.components.a = VK_COMPONENT_SWIZZLE_IDENTITY;
171 createInfo.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
172 createInfo.subresourceRange.baseMipLevel = 0;
173 createInfo.subresourceRange.levelCount = 1;
174 createInfo.subresourceRange.baseArrayLayer = 0;
175 createInfo.subresourceRange.layerCount = 1;
176
177 VkResult result = vkCreateImageView(device.GetLogicalDevice(), &createInfo, nullptr, &tempViews[i]);
178 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("Error al crear view de imagen del swapchain", result));
179
180 if (GetImage(i)->As<GpuImageVk>()->GetSwapchainView() != VK_NULL_HANDLE) {
181 vkDestroyImageView(

183 GetImage(i)->As<GpuImageVk>()->GetSwapchainView(), 0);

186 GetImage(i)->As<GpuImageVk>()->SetSwapchainView(tempViews[i]);
