133void SwapchainVk::AcquireViews() {
134 auto tempViews = new VkImageView[m_imageCount];
135
136 for (UIndex32 i = 0; i < m_imageCount; i++) {
137 VkImageViewCreateInfo createInfo{};
138 createInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
139 createInfo.image = m_images[i]->As<GpuImageVk>()->GetVkImage();
140 createInfo.viewType = VK_IMAGE_VIEW_TYPE_2D;
141 createInfo.format = GetFormatVk(m_colorFormat);
142 createInfo.components.r = VK_COMPONENT_SWIZZLE_IDENTITY;
143 createInfo.components.g = VK_COMPONENT_SWIZZLE_IDENTITY;
144 createInfo.components.b = VK_COMPONENT_SWIZZLE_IDENTITY;
145 createInfo.components.a = VK_COMPONENT_SWIZZLE_IDENTITY;
146 createInfo.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
147 createInfo.subresourceRange.baseMipLevel = 0;
148 createInfo.subresourceRange.levelCount = 1;
149 createInfo.subresourceRange.baseArrayLayer = 0;
150 createInfo.subresourceRange.layerCount = 1;
151
152 VkResult result = vkCreateImageView(m_device->GetLogicalDevice(), &createInfo, nullptr, &tempViews[i]);
153 OSK_ASSERT(result == VK_SUCCESS, SwapchainCreationException("Error al crear view de imagen del swapchain", result));
154
155 if (m_images[i]->As<GpuImageVk>()->GetSwapchainView() != VK_NULL_HANDLE)
156 vkDestroyImageView(Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice(), m_images[i]->As<GpuImageVk>()->GetSwapchainView(), 0);
157 m_images[i]->As<GpuImageVk>()->SetSwapchainView(tempViews[i]);
158 }
159
160 delete[] tempViews;
