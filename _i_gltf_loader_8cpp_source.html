59void IGltfLoader::SetupModel(Model3D* model) {
60 model->_SetIndexBuffer(Engine::GetRenderer()->GetAllocator()->CreateIndexBuffer(m_indices));
61 model->_SetIndexCount(static_cast<USize32>(m_indices.GetSize()));
62
63 for (UIndex32 i = 0; i < m_meshes.GetSize(); i++) {
64 MeshMetadata meshMetadata{};
65
66 if (m_meshIdToMaterialId.contains(i)) {
67 const auto& materialInfo = m_modelInfo.materialInfos.At(m_meshIdToMaterialId.at(i));
68
69 if (materialInfo.hasColorTexture)
70 meshMetadata.textureTable[(std::string)ModelMetadata::BaseColorTextureName] = materialInfo.colorTextureIndex;
71
72 if (materialInfo.hasNormalTexture)

74

103glm::mat4 IGltfLoader::GetNodeMatrix(const tinygltf::Node& node) {
104 // Obtenemos el transform del nodo.
105 // Puede estar definido de varias maneras:
106
107 glm::mat4 nodeMatrix = glm::mat4(1.0f);
108
109 // Definido por una posición.
110 if (node.translation.size() == 3) {
111 nodeMatrix = glm::translate(nodeMatrix, glm::vec3(node.translation[0], node.translation[1], node.translation[2]));
112 }
113 // Definido por una rotación.
114 if (node.rotation.size() == 4) {
115 glm::quat quaternion = glm::make_quat(node.rotation.data());
116 nodeMatrix *= glm::mat4(quaternion);
117 }
118 // Definido por una escala.
119 if (node.scale.size() == 3) {
120 nodeMatrix = glm::scale(nodeMatrix, glm::vec3(glm::make_vec3(node.scale.data())));
121 }
122 // Definido por una matriz.
123 if (node.matrix.size() == 16) {
124 nodeMatrix = glm::make_mat4x4(node.matrix.data());
125 }
126
127 return nodeMatrix;
128}

130std::unordered_map<std::string, std::string, StringHasher, std::equal_to<>> IGltfLoader::GetCustomProperties(const tinygltf::Mesh& mesh) const {
131 std::unordered_map<std::string, std::string, StringHasher, std::equal_to<>> output{};
132 const auto& extras = mesh.extras;
133
134 if (extras.IsObject()) {
135 for (const auto& key : extras.Keys()) {
136 const auto& value = extras.Get(key);
137
138 switch (value.Type()) {
139 case tinygltf::Type::NULL_TYPE:
140 output[key] = "NULL";
141 break;
142
143 case tinygltf::Type::REAL_TYPE:
144 output[key] = std::format("{}", value.GetNumberAsDouble());
145 break;
146
147 case tinygltf::Type::INT_TYPE:
148 output[key] = std::format("{}", value.GetNumberAsInt());
149 break;
150
151 case tinygltf::Type::BOOL_TYPE:
152 output[key] = value.Get<bool>();
153 break;
154
155 case tinygltf::Type::STRING_TYPE:
156 output[key] = value.Get<std::string>();
157 break;
158
159 case tinygltf::Type::ARRAY_TYPE:
160 throw UnsupportedMeshProperties("ARRAY_TYPE");
161 case tinygltf::Type::BINARY_TYPE:
162 throw UnsupportedMeshProperties("BINARY_TYPE");
163 case tinygltf::Type::OBJECT_TYPE:
164 throw UnsupportedMeshProperties("OBJECT_TYPE");

172DynamicArray<OwnedPtr<GpuImage>> IGltfLoader::LoadImages(const tinygltf::Model& model) {
173 DynamicArray<OwnedPtr<GpuImage>> output;
174
176
177 for (UIndex32 i = 0; i < model.images.size(); i++) {
178 const tinygltf::Image& originalImage = model.images[i];
179
180 OwnedPtr<GpuImage> image = Engine::GetRenderer()->GetAllocator()->CreateImage(
181 GpuImageCreateInfo::CreateDefault2D({
182 (unsigned int)originalImage.width,
183 (unsigned int)originalImage.height },
184 Format::RGBA8_UNORM,
185 GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED)
186 );
187 image->SetDebugName(std::format("Mesh texture {}", i));
188
189 OwnedPtr<ICommandList> uploadCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
190 uploadCmdList->Reset();
191 uploadCmdList->Start();
192
193 const USize64 numBytes = originalImage.width * originalImage.height * 4;
194 if (originalImage.component == 3) {
195 auto dataContainer = DynamicArray<TByte>::CreateReservedArray(numBytes);
196 auto* data = dataContainer.GetData();
197
198 memset(data, 255, numBytes);
199
200 UIndex64 rgbPos = 0;
201 for (UIndex64 i = 0; i < numBytes; i += 4) {
202 memcpy(&data[i], originalImage.image.data() + rgbPos, 3);
203 rgbPos += 3;
204 }
205
206 uploadCmdList->SetGpuImageBarrier(
207 image.GetPointer(),
208 GpuImageLayout::TRANSFER_DESTINATION,
209 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
210
211 Engine::GetRenderer()->UploadImageToGpu(
212 image.GetPointer(),
213 originalImage.image.data(),
214 numBytes,
215 uploadCmdList.GetPointer());
216
217 uploadCmdList->SetGpuImageBarrier(
218 image.GetPointer(),
219 GpuImageLayout::SAMPLED,
220 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ));
221 }
222 else if (originalImage.component == 4) {
223 uploadCmdList->SetGpuImageBarrier(
224 image.GetPointer(),
225 GpuImageLayout::TRANSFER_DESTINATION,
226 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
227
228 Engine::GetRenderer()->UploadImageToGpu(
229 image.GetPointer(),
230 originalImage.image.data(),
231 numBytes,
232 uploadCmdList.GetPointer());
233
234 uploadCmdList->SetGpuImageBarrier(
235 image.GetPointer(),
236 GpuImageLayout::SAMPLED,
237 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ));
238 }
239 else {
240 OSK_ASSERT(false, UnsupportedModelImageFormatException());
241 }
242
243 uploadCmdList->Close();
244 Engine::GetRenderer()->SubmitSingleUseCommandList(uploadCmdList.GetPointer());
245
246 output.Insert(image);
247 }
248
249 return output;
250}

253 auto output = DynamicArray<GltfMaterialInfo>::CreateResizedArray(model.materials.size());

255 for (UIndex64 i = 0; i < model.materials.size(); i++) {

259 static_cast<float>(model.materials[i].pbrMetallicRoughness.baseColorFactor[1]),

266 static_cast<float>(model.materials[i].pbrMetallicRoughness.metallicFactor)

268

272 static_cast<float>(model.materials[i].emissiveFactor[2])
273 };
274
275 if (model.materials[i].pbrMetallicRoughness.baseColorTexture.index != -1) {
276 info.hasColorTexture = true;
277 info.colorTextureIndex = model.textures[model.materials[i].pbrMetallicRoughness.baseColorTexture.index].source;
278 }
279
280 if (model.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index != -1) {
281 info.hasMetallicTexture = true;
282 info.metallicTextureIndex = model.textures[model.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
283 }

286 info.hasRoughnessTexture = true;
287 info.metallicTextureIndex = model.textures[model.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
288 }
289
290 if (model.materials[i].normalTexture.index != -1) {
291 info.hasNormalTexture = true;
292 info.normalTextureIndex = model.textures[model.materials[i].normalTexture.index].source;

333DynamicArray<Vector3f> IGltfLoader::GetVertexPositions(const tinygltf::Primitive& primitive, const glm::mat4& nodeMatrix, const tinygltf::Model& model) const {
334 // Comprobamos que tiene almacenado info de posición.
335 OSK_ASSERT(HasPositions(primitive), NoVertexPositionsFoundException());
336
337 // Para poder acceder a la información en forma de buffer.
338 const tinygltf::Accessor& accessor = model.accessors[primitive.attributes.find("POSITION")->second];
339 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
340
341 // Leemos el buffer.
342 const float* positionsBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
343 const USize32 numVertices = static_cast<USize32>(accessor.count);
344
345
346 // Transformamos a vértices
347 DynamicArray<Vector3f> output = DynamicArray<Vector3f>::CreateResizedArray(numVertices);
348
349 for (UIndex32 v = 0; v < numVertices; v++) {
350 const glm::vec4 vertexPosition = glm::vec4(
351 positionsBuffer[v * 3 + 0],
352 positionsBuffer[v * 3 + 1],
353 positionsBuffer[v * 3 + 2],
354 m_globalScale
355 );
356
357 output[v] = Vector3f(glm::vec3(nodeMatrix * vertexPosition));
358 }
359
360 return output;
361}

418DynamicArray<Vector3f> IGltfLoader::GenerateTangetVectors(const DynamicArray<Vector2f>& texCoords, const DynamicArray<Vector3f>& _positions, const DynamicArray<TIndexSize>& indices, UIndex32 indicesStartOffset) const {
419 DynamicArray<Vector3f> output = DynamicArray<Vector3f>::CreateResizedArray(texCoords.GetSize());
420
421 for (UIndex32 i = 0; i < indices.GetSize(); i += 3) {
422 const GRAPHICS::TIndexSize indexA = indices[i + 0] - indicesStartOffset;
423 const GRAPHICS::TIndexSize indexB = indices[i + 1] - indicesStartOffset;
424 const GRAPHICS::TIndexSize indexC = indices[i + 2] - indicesStartOffset;
425
426 const std::array<const Vector2f, 3> tCoords = {
427 texCoords[indexA],
428 texCoords[indexB],
429 texCoords[indexC]
430 };
431 const std::array<const Vector3f, 3> positions = {
432 _positions[indexA],
433 _positions[indexB],
434 _positions[indexC]
435 };
436
437 const Vector3f edge1 = positions[1] - positions[0];
438 const Vector3f edge2 = positions[2] - positions[0];
439
440 const Vector2f uvDelta1 = tCoords[1] - tCoords[0];
441 const Vector2f uvDelta2 = tCoords[2] - tCoords[0];
442
443 const float f = 1.0f / (uvDelta1.x * uvDelta2.y - uvDelta2.x * uvDelta1.y);
444
445 const Vector3f tangent = Vector3f(
446 f * (uvDelta2.y * edge1.x - uvDelta1.y * edge2.x),
447 f * (uvDelta2.y * edge1.y - uvDelta1.y * edge2.y),
448 f * (uvDelta2.y * edge1.z - uvDelta1.y * edge2.z)
449 ).GetNormalized();
450
451 output[indexA] = tangent;
452 output[indexB] = tangent;
453 output[indexC] = tangent;
454 }
455
456 return output;
457}

483DynamicArray<Color> IGltfLoader::GetVertexColors(const tinygltf::Primitive& primitive, const tinygltf::Model& model) const {
484 if (!HasColors(primitive))
485 return DynamicArray<Color>{};
486
487 // Para poder acceder a la información en forma de buffer.
488 const tinygltf::Accessor& accessor = model.accessors[primitive.attributes.find("COLOR_0")->second];
489 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
490
491 // Leemos el buffer.
492 const void* colorBuffer = &(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]);
493 const auto numVertices = static_cast<USize32>(accessor.count);
494
495 const auto colorVectorType = accessor.type;
496 const auto colorDataType = accessor.componentType;
497
498 const int stride = colorVectorType;
499
500 DynamicArray<Color> output = DynamicArray<Color>::CreateResizedArray(numVertices);
501
502 for (UIndex32 v = 0; v < numVertices; v++) {
503
504 switch (colorDataType) {
505
506 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
507 constexpr float denom = std::numeric_limits<uint8_t>::max();
508 output[v] = {
509 static_cast<const uint8_t*>(colorBuffer)[v * stride + 0] / denom,
510 static_cast<const uint8_t*>(colorBuffer)[v * stride + 1] / denom,
511 static_cast<const uint8_t*>(colorBuffer)[v * stride + 2] / denom,
512 1.0f
513 };
514
515 break;
516 }
517
518 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
519 constexpr float denom = std::numeric_limits<uint16_t>::max();
520 output[v] = {
521 static_cast<const uint16_t*>(colorBuffer)[v * stride + 0] / denom,
522 static_cast<const uint16_t*>(colorBuffer)[v * stride + 1] / denom,
523 static_cast<const uint16_t*>(colorBuffer)[v * stride + 2] / denom,
524 1.0f
525 };
526
527 break;
528 }
529
530 case TINYGLTF_PARAMETER_TYPE_FLOAT: {
531 output[v] = Color{
532 static_cast<const float*>(colorBuffer)[v * stride + 0],
533 static_cast<const float*>(colorBuffer)[v * stride + 1],
534 static_cast<const float*>(colorBuffer)[v * stride + 2],
535 1.0f
536 };
537
538 break;
539 }
540 }

544DynamicArray<TIndexSize> IGltfLoader::GetIndices(const tinygltf::Primitive& primitive, TIndexSize startOffset, const tinygltf::Model& model) const {
545 // Índices
546 const tinygltf::Accessor& indicesAccesor = model.accessors[primitive.indices];
547 const tinygltf::BufferView& indicesView = model.bufferViews[indicesAccesor.bufferView];
548 const tinygltf::Buffer& indicesBuffer = model.buffers[indicesView.buffer];
549
550 const auto numIndices = static_cast<USize32>(indicesAccesor.count);
551
552 DynamicArray<TIndexSize> output = DynamicArray<TIndexSize>::CreateResizedArray(numIndices);
553
554 // Los índices también se guardan en buffers.
555 // Necesitamos saber su tipo para procesar el buffer.
556 // El índice es el ID del vértice dentro de la primitiva.
557 // Para obtener el ID real, debemos tener en cuenta todos los vértices anteriormente procesados.
558 switch (indicesAccesor.componentType) {
559 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_INT: {
560 const uint32_t* indBuffer = (const uint32_t*)(&indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset]);
561 for (UIndex32 index = 0; index < indicesAccesor.count; index++)
562 output[index] = (indBuffer[index] + startOffset);
563
564 break;
565 }
566
567 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
568 const uint16_t* indBuffer = (const uint16_t*)(&indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset]);
569 for (UIndex32 index = 0; index < indicesAccesor.count; index++)
570 output[index] = ((TIndexSize)indBuffer[index] + startOffset);
571
572 break;
573 }
574
575 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
576 const uint8_t* indBuffer = &indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset];
577 for (UIndex32 index = 0; index < indicesAccesor.count; index++)
578 output[index] = ((TIndexSize)indBuffer[index] + startOffset);
579
580 break;
581 }
582
583 default:
584 OSK_ASSERT(false, UnsupportedIndexTypeException(std::to_string(indicesAccesor.componentType)));
585 break;
586
587 }
588
589 return output;
590}

592DynamicArray<Vector4f> IGltfLoader::GetJoints(const tinygltf::Primitive& primitive, const tinygltf::Model& model) const {
593 if (!HasJoints(primitive))
594 return DynamicArray<Vector4f>();
595
596 // Para poder acceder a la información en forma de buffer.
597 const tinygltf::Accessor& accessor = model.accessors[primitive.attributes.find("JOINTS_0")->second];
598 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
599
600 // Leemos el buffer.
601 const void* boneIds = &(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]);
602 const int jointsDataType = accessor.componentType;
603
604 const auto numVertices = static_cast<USize32>(accessor.count);
605
606 DynamicArray<Vector4f> output = DynamicArray<Vector4f>::CreateResizedArray(numVertices);
607
608 for (UIndex32 v = 0; v < numVertices; v++) {
609 switch (jointsDataType) {
610
611 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_INT: {
612 output[v] = {
613 static_cast<float>(static_cast<const uint32_t*>(boneIds)[v * 4 + 0]),

654DynamicArray<Vector4f> IGltfLoader::GetBoneWeights(const tinygltf::Primitive& primitive, const tinygltf::Model& model) const {

656 return DynamicArray<Vector4f>();
657
658 // Para poder acceder a la información en forma de buffer.
659 const tinygltf::Accessor& accessor = model.accessors[primitive.attributes.find("WEIGHTS_0")->second];
660 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
661
662 // Leemos el buffer.
663 const float* boneWeights = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
664
665 const USize32 numVertices = static_cast<USize32>(accessor.count);
666
667 DynamicArray<Vector4f> output = DynamicArray<Vector4f>::CreateResizedArray(numVertices);
668
669 for (UIndex32 v = 0; v < numVertices; v++) {
670 output[v] = Vector4f(
671 boneWeights[v * 4 + 0],
672 boneWeights[v * 4 + 1],
673 boneWeights[v * 4 + 2],
674 boneWeights[v * 4 + 3]
675 );
