78void ShadowMap::UpdateLightMatrixBuffer() {
79 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
80
81 ShadowsBufferContent bufferContent{};
82
83 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
84 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
85
86 bufferContent.cascadeSplits = m_splitsVec;
87
88 float lastClip = camera.GetNearPlane();
89
90 // Matriz vista de la cámara del jugador.
91 const glm::mat4 viewMatrix = camera.GetViewMatrix(cameraTransform);
92
93 for (UIndex32 i = 0; i < numMaps; i++) {
94
95 // Creamos un frustum con NEAR = CascadeSplits[i - 1] y FAR = CascadeSplits[i],
96 // de tal manera que quede dentro toda la zona renderizada que está dentro del split.
97 CameraComponent3D clipCamera = camera;
98 clipCamera.SetPlanes(lastClip, m_splits[i]);
99
100 // Matriz proyección de la cámara del jugadro, en la que los planos near y far
101 // han sido recortados para representar el nivel de shadowmap.
102 const glm::mat4 croppedProjection = clipCamera.GetProjectionMatrix_UnreversedZ();
103
104 // Obtiene las esquinas en el espacio del mundo.
105 DynamicArray<Vector3f> worldSpaceFrustumCorners =
106 GetFrustumCorners(croppedProjection * viewMatrix);
107
108 // Calculamos el centro del frustum.
109 Vector3f frustumCenter = Vector3f::Zero;
110 for (const auto& corner : worldSpaceFrustumCorners)
111 frustumCenter += corner;
112 frustumCenter /= static_cast<float>(worldSpaceFrustumCorners.GetSize());
113
114 // Movemos para que el centro este en Y = 0.
115 for (auto& corner : worldSpaceFrustumCorners)
116 corner += m_lightDirection * frustumCenter.y;
117 frustumCenter += m_lightDirection * frustumCenter.y;
118
119 // Calculamos el radio, que será el radio del frustum que renderizará el mapa de sombras.
120 float radius = 0.0f;
121 for (const auto& corner : worldSpaceFrustumCorners)
122 radius = glm::max(radius, frustumCenter.GetDistanceTo(corner));
123 radius = glm::ceil(radius * 16.0f) / 16.0f;
124
125 // Extents del frustum
126 const Vector2f maxExtent = Vector2f(radius * 1.2f);
127 const Vector2f minExtent = -maxExtent;
128
129 // 'Cámara' virtual para renderizar el mapa de sombras.
130 const glm::mat4 lightProjection = glm::orthoRH_ZO(minExtent.x, maxExtent.x, maxExtent.y, minExtent.y, m_farPlane * 6, m_nearPlane * 12);
131 const glm::mat4 lightView = glm::lookAt((frustumCenter - m_lightDirection).ToGlm(), frustumCenter.ToGlm(), glm::vec3(0.0f, 1.0f, 0.0f));
132
133 bufferContent.matrices[i] = lightProjection * lightView;
134
135 lastClip = m_splits[i];
136 }
137
138 lightUniformBuffer[resourceIndex]->ResetCursor();
139 lightUniformBuffer[resourceIndex]->MapMemory();
140 lightUniformBuffer[resourceIndex]->Write(bufferContent);
141 lightUniformBuffer[resourceIndex]->Unmap();
142}

192DynamicArray<Vector3f> ShadowMap::GetFrustumCorners(const glm::mat4& cameraMatrix) {
193 const glm::mat4 inverseMatrix = glm::inverse(cameraMatrix);
194
195 // Normalized Device Coordinates
196 DynamicArray<Vector3f> corners = {
197 Vector3f( 1, 1, 1),
198 Vector3f( 1, 1, -1),
199 Vector3f(-1, 1, 1),
200 Vector3f(-1, 1, -1),
201 Vector3f( 1, -1, 1),
202 Vector3f( 1, -1, -1),
203 Vector3f(-1, -1, 1),
204 Vector3f(-1, -1, -1),
205 };
206
207 for (auto& corner : corners) {
208 const glm::vec4 inverseResult = inverseMatrix * glm::vec4(corner.ToGlm(), 1.0f);
209 corner = Vector3f(glm::vec3(inverseResult / inverseResult.w));
210 }
211
212 return corners;
213}
