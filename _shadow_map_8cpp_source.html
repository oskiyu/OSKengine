69void ShadowMap::UpdateLightMatrixBuffer() {
70 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
71
72 ShadowsBufferContent bufferContent{};
73
74 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
75 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
76
77 bufferContent.cascadeSplits = m_splitsVec;
78
79 float lastClip = camera.GetNearPlane();
80
81 // Matriz vista de la c√°mara del jugador.
82 const glm::mat4 viewMatrix = camera.GetViewMatrix(cameraTransform);
83
84 for (UIndex32 i = 0; i < numMaps; i++) {
85
86 // Creamos un frustum con NEAR = CascadeSplits[i - 1] y FAR = CascadeSplits[i],
87 // de tal manera que quede dentro toda la zona renderizada que est√° dentro del split.
88 CameraComponent3D clipCamera = camera;
89 clipCamera.SetPlanes(lastClip, m_splits[i]);
90
91 // Matriz proyecci√≥n de la c√°mara del jugadro, en la que los planos near y far
92 // han sido recortados para representar el nivel de shadowmap.
93 const glm::mat4 croppedProjection = clipCamera.GetProjectionMatrix_UnreversedZ();
94
95 // Obtiene las esquinas en el espacio del mundo.
96 DynamicArray<Vector3f> worldSpaceFrustumCorners =
97 GetFrustumCorners(croppedProjection * viewMatrix);
98
99 // Calculamos el centro del frustum.
100 Vector3f frustumCenter = Vector3f::Zero;
101 for (const auto& corner : worldSpaceFrustumCorners)
102 frustumCenter += corner;
103 frustumCenter /= static_cast<float>(worldSpaceFrustumCorners.GetSize());
104
105 // Movemos para que el centro este en Y = 0.
106 for (auto& corner : worldSpaceFrustumCorners)
107 corner += m_lightDirection * frustumCenter.y;
108 frustumCenter += m_lightDirection * frustumCenter.y;
109
110 // Calculamos el radio, que ser√° el radio del frustum que renderizar√° el mapa de sombras.
111 float radius = 0.0f;
112 for (const auto& corner : worldSpaceFrustumCorners)
113 radius = glm::max(radius, frustumCenter.GetDistanceTo(corner));
114 radius = glm::ceil(radius * 16.0f) / 16.0f;
115
116 // Extents del frustum
117 const Vector2f maxExtent = Vector2f(radius * 1.2f);
118 const Vector2f minExtent = -maxExtent;
119
120 // 'C√°mara' virtual para renderizar el mapa de sombras.
121 const glm::mat4 lightProjection = glm::orthoRH_ZO(minExtent.x, maxExtent.x, maxExtent.y, minExtent.y, m_farPlane * 6, m_nearPlane * 12);
122 const glm::mat4 lightView = glm::lookAt((frustumCenter - m_lightDirection).ToGlm(), frustumCenter.ToGlm(), glm::vec3(0.0f, 1.0f, 0.0f));
123
124 bufferContent.matrices[i] = lightProjection * lightView;
125
126 lastClip = m_splits[i];
127 }
128
129 lightUniformBuffer[resourceIndex]->ResetCursor();
130 lightUniformBuffer[resourceIndex]->MapMemory();
131 lightUniformBuffer[resourceIndex]->Write(bufferContent);
132 lightUniformBuffer[resourceIndex]->Unmap();
133}

182DynamicArray<Vector3f> ShadowMap::GetFrustumCorners(const glm::mat4& cameraMatrix) {
183 const glm::mat4 inverseMatrix = glm::inverse(cameraMatrix);
184
185 // Normalized Device Coordinates
186 DynamicArray<Vector3f> corners = {
187 Vector3f( 1, 1, 1),
188 Vector3f( 1, 1, -1),
189 Vector3f(-1, 1, 1),
190 Vector3f(-1, 1, -1),
191 Vector3f( 1, -1, 1),
192 Vector3f( 1, -1, -1),
193 Vector3f(-1, -1, 1),
194 Vector3f(-1, -1, -1),
195 };
196
197 for (auto& corner : corners) {
198 const glm::vec4 inverseResult = inverseMatrix * glm::vec4(corner.ToGlm(), 1.0f);
199 corner = Vector3f(glm::vec3(inverseResult / inverseResult.w));
200 }
201
202 return corners;
203}
