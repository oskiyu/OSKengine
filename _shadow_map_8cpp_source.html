75void ShadowMap::UpdateLightMatrixBuffer() {
76 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
77
78 ShadowsBufferContent bufferContent{};
79
80 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
81 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
82
83 bufferContent.cascadeSplits = m_splitsVec;
84
85 float lastClip = camera.GetNearPlane();
86
87 // Matriz vista de la cámara del jugador.
88 const glm::mat4 viewMatrix = camera.GetViewMatrix(cameraTransform);
89
90 for (UIndex32 i = 0; i < numMaps; i++) {
91
92 // Creamos un frustum con NEAR = CascadeSplits[i - 1] y FAR = CascadeSplits[i],
93 // de tal manera que quede dentro toda la zona renderizada que está dentro del split.
94 CameraComponent3D clipCamera = camera;
95 clipCamera.SetPlanes(lastClip, m_splits[i]);
96
97 // Matriz proyección de la cámara del jugadro, en la que los planos near y far
98 // han sido recortados para representar el nivel de shadowmap.
99 const glm::mat4 croppedProjection = clipCamera.GetProjectionMatrix_UnreversedZ();
100
101 // Obtiene las esquinas en el espacio del mundo.
102 DynamicArray<Vector3f> worldSpaceFrustumCorners =
103 GetFrustumCorners(croppedProjection * viewMatrix);
104
105 // Calculamos el centro del frustum.
106 Vector3f frustumCenter = Vector3f::Zero;
107 for (const auto& corner : worldSpaceFrustumCorners)
108 frustumCenter += corner;
109 frustumCenter /= static_cast<float>(worldSpaceFrustumCorners.GetSize());
110
111 // Movemos para que el centro este en Y = 0.
112 for (auto& corner : worldSpaceFrustumCorners)
113 corner += m_lightDirection * frustumCenter.y;
114 frustumCenter += m_lightDirection * frustumCenter.y;
115
116 // Calculamos el radio, que será el radio del frustum que renderizará el mapa de sombras.
117 float radius = 0.0f;
118 for (const auto& corner : worldSpaceFrustumCorners)
119 radius = glm::max(radius, frustumCenter.GetDistanceTo(corner));
120 radius = glm::ceil(radius * 16.0f) / 16.0f;
121
122 // Extents del frustum
123 const Vector2f maxExtent = Vector2f(radius * 1.2f);
124 const Vector2f minExtent = -maxExtent;
125
126 // 'Cámara' virtual para renderizar el mapa de sombras.
127 const glm::mat4 lightProjection = glm::orthoRH_ZO(minExtent.x, maxExtent.x, maxExtent.y, minExtent.y, m_farPlane * 6, m_nearPlane * 12);
128 const glm::mat4 lightView = glm::lookAt((frustumCenter - m_lightDirection).ToGlm(), frustumCenter.ToGlm(), glm::vec3(0.0f, 1.0f, 0.0f));
129
130 bufferContent.matrices[i] = lightProjection * lightView;
131
132 lastClip = m_splits[i];
133 }
134
135 lightUniformBuffer[resourceIndex]->ResetCursor();
136 lightUniformBuffer[resourceIndex]->MapMemory();
137 lightUniformBuffer[resourceIndex]->Write(bufferContent);
138 lightUniformBuffer[resourceIndex]->Unmap();
139}

188DynamicArray<Vector3f> ShadowMap::GetFrustumCorners(const glm::mat4& cameraMatrix) {
189 const glm::mat4 inverseMatrix = glm::inverse(cameraMatrix);
190
191 // Normalized Device Coordinates
192 DynamicArray<Vector3f> corners = {
193 Vector3f( 1, 1, 1),
194 Vector3f( 1, 1, -1),
195 Vector3f(-1, 1, 1),
196 Vector3f(-1, 1, -1),
197 Vector3f( 1, -1, 1),
198 Vector3f( 1, -1, -1),
199 Vector3f(-1, -1, 1),
200 Vector3f(-1, -1, -1),
201 };
202
203 for (auto& corner : corners) {
204 const glm::vec4 inverseResult = inverseMatrix * glm::vec4(corner.ToGlm(), 1.0f);
205 corner = Vector3f(glm::vec3(inverseResult / inverseResult.w));
206 }
207
208 return corners;
209}
