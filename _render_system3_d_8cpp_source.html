212void RenderSystem3D::GenerateShadows(ICommandList* commandList) {
213 const Viewport viewport {
214 .rectangle = { 0u, 0u,
215 m_shadowMap.GetColorImage()->GetSize2D().x,
216 m_shadowMap.GetColorImage()->GetSize2D().y }
217 };
218
219 commandList->StartDebugSection("Shadows", Color::Black);
220
221 commandList->SetGpuImageBarrier(
222 m_shadowMap.GetShadowImage(),
223 GpuImageLayout::UNDEFINED,
224 GpuImageLayout::DEPTH_STENCIL_TARGET,
225 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
226 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
227 GpuImageRange {
228 .baseLayer = 0,
229 .numLayers = m_shadowMap.GetNumCascades(),
230 .baseMipLevel = 0,
231 .numMipLevels = ALL_MIP_LEVELS,
232 .channel = SampledChannel::DEPTH
233 });
234
235 commandList->SetViewport(viewport);
236 commandList->SetScissor(viewport.rectangle);
237
238 for (UIndex32 i = 0; i < m_shadowMap.GetNumCascades(); i++) {
239
240 commandList->StartDebugSection(std::format("Cascade {}", i), Color::Black);
241
242 RenderPassImageInfo colorInfo{};
243 colorInfo.arrayLevel = i;
244 colorInfo.targetImage = m_shadowMap.GetColorImage();
245
246 RenderPassImageInfo depthInfo{};
247 depthInfo.arrayLevel = i;
248 depthInfo.targetImage = m_shadowMap.GetShadowImage();
249
250 commandList->BeginGraphicsRenderpass({ colorInfo }, depthInfo, { 1.0f, 1.0f, 1.0f, 1.0f }, false);
251
252 commandList->StartDebugSection("Static Meshes", Color::Black);
253
254 for (auto& pass : m_shadowsPasses.GetAllPasses()) {
255 pass->RenderLoop(
256 commandList,
257 m_shadowsPasses.GetCompatibleObjects(pass->GetTypeName()),
258 &m_meshMapping,
259 0,
260 m_shadowMap.GetShadowImage()->GetSize2D());
261 }
262
263 commandList->EndDebugSection();
264
265 commandList->EndGraphicsRenderpass(false);
266
267 commandList->EndDebugSection();
268 }
269
270 commandList->EndDebugSection();
271}

348void RenderSystem3D::CopyTaaResult(GRAPHICS::ICommandList* commandList) {

350 GpuImage* destinationImage = m_renderTarget.GetColorImage(COLOR_IMAGE_INDEX);
351

363
364 commandList->RawCopyImageToImage(
365 *sourceImage,
366 destinationImage,
367 CopyImageInfo::CreateDefault2D(m_renderTarget.GetSize()));

370void RenderSystem3D::Render(GRAPHICS::ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
371 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
372
373 const auto& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
374 const auto& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
375
376 const glm::mat4 currentProjection = camera.GetProjectionMatrix();
377 const glm::mat4 currentView = camera.GetViewMatrix(cameraTransform);
378
379 m_cameraBuffers[resourceIndex]->MapMemory();
380 m_cameraBuffers[resourceIndex]->Write(currentProjection);
381 m_cameraBuffers[resourceIndex]->Write(currentView);
382 m_cameraBuffers[resourceIndex]->Write(cameraTransform.GetPosition());
383 m_cameraBuffers[resourceIndex]->Unmap();
384
385 m_previousCameraBuffers[resourceIndex]->MapMemory();
386 m_previousCameraBuffers[resourceIndex]->Write(m_previousCameraProjection);
387 m_previousCameraBuffers[resourceIndex]->Write(m_previousCameraView);
388 m_previousCameraBuffers[resourceIndex]->Unmap();
389
390 m_dirLightUbos[resourceIndex]->MapMemory();
391 m_dirLightUbos[resourceIndex]->Write(m_dirLight);
392 m_dirLightUbos[resourceIndex]->Unmap();
393
394 m_previousCameraProjection = currentProjection;
395 m_previousCameraView = currentView;
396
397 m_shadowMap.SetDirectionalLight(m_dirLight);
398
399 GenerateShadows(commandList);
400 RenderScene(commandList);
401 ExecuteTaa(commandList);
402
403 for (const GameObjectIndex obj : objects)
404 m_previousModelMatrices[obj] = Engine::GetEcs()->GetComponent<Transform3D>(obj).GetAsMatrix();
405}
