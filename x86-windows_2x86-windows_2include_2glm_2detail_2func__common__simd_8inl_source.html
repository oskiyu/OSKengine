190 struct compute_mix_vector<4, float, bool, Q, true>
191 {
192 GLM_FUNC_QUALIFIER static vec<4, float, Q> call(vec<4, float, Q> const& x, vec<4, float, Q> const& y, vec<4, bool, Q> const& a)
193 {
194 __m128i const Load = _mm_set_epi32(-static_cast<int>(a.w), -static_cast<int>(a.z), -static_cast<int>(a.y), -static_cast<int>(a.x));
195 __m128 const Mask = _mm_castsi128_ps(Load);
196
197 vec<4, float, Q> Result;
198# if 0 && GLM_ARCH & GLM_ARCH_AVX
199 Result.data = _mm_blendv_ps(x.data, y.data, Mask);
200# else
201 Result.data = _mm_or_ps(_mm_and_ps(Mask, y.data), _mm_andnot_ps(Mask, x.data));
202# endif
203 return Result;
204 }
