39 DynamicArray<Vertex3D> vertices{};

43 TIndexSize meshesOffset = 0;
44 for (const auto& mesh : model.GetMeshes()) {
45 for (const auto& vertex : mesh.GetVertices()) {
46 OSK_ASSERT(vertex.position.has_value(), NoVertexPositionsFoundException())
47 OSK_ASSERT(vertex.normal.has_value(), NoVertexNormalsFoundException())
48 OSK_ASSERT(vertex.tangent.has_value(), NoVertexTangentsFoundException())

54 gpuVertex.normal = vertex.normal.value();

57 gpuVertex.color = vertex.color.value();

59 vertices.Insert(gpuVertex);

66 gpuVertex.tangent

73 indices.Insert(static_cast<TIndexSize>(triangles[2] + meshesOffset));

103 output->AddMesh(gpuMesh);

114 GpuQueueType::ASYNC_TRANSFER,
115 GpuQueueType::MAIN));

118 indices,

120 GpuQueueType::MAIN));
121
122 output->SetIndexCount(static_cast<USize32>(indices.GetSize()));
