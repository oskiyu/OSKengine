92 float closestDistance = std::numeric_limits<float>::max();
93 RayCastResult closest = RayCastResult::False();

95 for (const GameObjectIndex i : GetAllCompatibleObjects()) {
96 if (i == sendingObject) {

101
102 const auto topLevelResult = collider.GetTopLevelCollider()->CastRay(ray);
103
104 if (!topLevelResult.Result()) {
105 // continue;
106 }
107
108 for (UIndex32 blc_i = 0; blc_i < collider.GetBottomLevelCollidersCount(); blc_i++) {
109 const auto& blc = collider.GetBottomLevelCollider(blc_i);
110 const auto result = blc->CastRay(ray);
111
112 if (!result.Result()) {
113 continue;
114 }
115
116 const float nDistance = result.GetIntersectionPoint().GetDistanceTo2(ray.origin);
117 if (nDistance < closestDistance) {

119 closest = RayCastResult::True(result.GetIntersectionPoint(), i);
120 }
121 }
122 }
123
