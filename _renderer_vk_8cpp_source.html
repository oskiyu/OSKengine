68VKAPI_ATTR VkBool32 VKAPI_CALL DebugCallback(VkDebugUtilsMessageSeverityFlagBitsEXT, VkDebugUtilsMessageTypeFlagsEXT messageType, const VkDebugUtilsMessengerCallbackDataEXT* pCallbackData, void*) {
69 // Message severity:
70 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT: "diagnostic" message.
71 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT: informaci√≥n.
72 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT: warning.
73 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT: error.
74
75 // Tipos de mensaje:
76 // VK_DEBUG_UTILS_MESSAGE_TYPE_GENERAL_BIT_EXT: algo ha ocurrido, no tiene que ver con la especificaci√≥n o el rendimiento.
77 // VK_DEBUG_UTILS_MESSAGE_TYPE_VALIDATION_BIT_EXT: algo ha ocurrido, incumple la especificaci√≥n.
78 // VK_DEBUG_UTILS_MESSAGE_TYPE_PERFORMANCE_BIT_EXT: algo ha ocurrido, uso no √≥ptimo de vulkan.
79
80 if (ignoredValidationLayersMessages.ContainsElement(pCallbackData->messageIdNumber))
81 return 0;
82
83 IO::LogLevel level = IO::LogLevel::WARNING;
84
85 switch (messageType) {
86 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
87 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
88 return 0;
89
90 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
91 level = IO::LogLevel::WARNING;
92 break;
93
94 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
95 level = IO::LogLevel::L_ERROR;
96 break;
97
98 default:
99 level = IO::LogLevel::WARNING;
100 break;
101 }
102
103 Engine::GetLogger()->Log(level, std::string(pCallbackData->pMessage) + "\n");
104
105 return VK_FALSE;
106}

211void RendererVk::SubmitSingleUseCommandList(OwnedPtr<ICommandList> commandList) {
212 std::lock_guard lock(m_queueSubmitMutex.mutex);
213
214 const auto cmdIndex = commandList->_GetCommandListIndex();
215
216 // Debemos saber cual es la cola compatible con el pool usado.
217 VkQueue queue = VK_NULL_HANDLE;
218
219 if (UseUnifiedCommandQueue() && commandList->GetOwnerPool()->GetLinkedQueueType() == GetUnifiedQueue()->GetQueueType()) {
220 // Cola unificada.
221 queue = GetUnifiedQueue()->As<CommandQueueVk>()->GetQueue();
222 // Engine::GetLogger()->InfoLog(std::format("\tInsertada lista en cola unificada."));
223 }
224 else if (!UseUnifiedCommandQueue() && commandList->GetOwnerPool()->GetLinkedQueueType() == GetGraphicsComputeQueue()->GetQueueType()) {
225 // Cola principal.
226 queue = GetGraphicsComputeQueue()->As<CommandQueueVk>()->GetQueue();
227 // Engine::GetLogger()->InfoLog(std::format("\tInsertada lista en cola principal."));
228 }
229 else if (HasTransferOnlyCommandPool() && commandList->GetOwnerPool()->GetLinkedQueueType() == GetTransferOnlyQueue()->GetQueueType()) {
230 // Cola de transferencia.
231 queue = GetTransferOnlyQueue()->As<CommandQueueVk>()->GetQueue();
232 // Engine::GetLogger()->InfoLog(std::format("\tInsertada lista en cola exclusiva de transferencia."));
233 }
234
235 const VkCommandBuffer cmdBuffer = commandList->As<CommandListVk>()->GetCommandBuffers()[cmdIndex];
236
237 VkSubmitInfo submitInfo{};
238 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
239 submitInfo.commandBufferCount = 1;
240 submitInfo.pCommandBuffers = &cmdBuffer;
241
242 VkResult result = vkQueueSubmit(queue, 1, &submitInfo, VK_NULL_HANDLE);
243 OSK_ASSERT(result == VK_SUCCESS, CommandQueueSubmitException(result));
244 result = vkQueueWaitIdle(queue);
245 OSK_ASSERT(result == VK_SUCCESS, CommandQueueSubmitException(result));
246
247 vkFreeCommandBuffers(
248 GetGpu()->As<GpuVk>()->GetLogicalDevice(),
249 commandList->GetOwnerPool()->As<CommandPoolVk>()->GetCommandPool(),
250 1, &cmdBuffer);
251
252 m_singleTimeCommandLists.Insert(commandList.GetPointer());
253}

368void RendererVk::ChooseGpu() {
369 // --------------- Physical ----------------- //
370
371 // Obtiene el n√∫mero de GPUs disponibles.
372 uint32_t count = 0;
373 vkEnumeratePhysicalDevices(m_instance, &count, nullptr);
374
375 OSK_ASSERT(count != 0, GpuNotFoundException());
376
377 // Obtiene los handlers de las GPUs.
378 auto devices = DynamicArray<VkPhysicalDevice>::CreateResized(count);
379 vkEnumeratePhysicalDevices(m_instance, &count, devices.GetData());
380
381 // Comprobar la compatibilidad de las GPUs.
382 // Obtener una GPU compatible.
383 DynamicArray<GpuVk::Info> gpus;
384 for (const VkPhysicalDevice gpu : devices) {
385 const GpuVk::Info info = GpuVk::Info::Get(gpu, m_surface);
386
387 if (info.isSuitable)
388 gpus.Insert(info);
389 }
390
391 OSK_ASSERT(!gpus.IsEmpty(), GpuNotFoundException());
392
393 VkPhysicalDevice gpu = devices[0];
394
395 GpuVk::Info info = gpus[0];
396
397 bool hasDiscrete = false;
398 for (UIndex64 i = 0; i < devices.GetSize(); i++) {
399 if (gpus[i].properties.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) {
400 gpu = devices[i];
401 info = gpus[i];
402
403 hasDiscrete = true;
404 }
405 }
406
407 Engine::GetLogger()->InfoLog("GPU elegida: " + std::string(info.properties.deviceName));
408
409 auto* gpuVk = new GpuVk(gpu, m_surface);
410 gpuVk->CreateLogicalDevice();
411
412 _SetGpu(gpuVk);
413}

415void RendererVk::CreateCommandQueues() {
416 GpuVk& gpu = *GetGpu()->As<GpuVk>();
417
418 const QueueFamiles queueFamilies = gpu.GetQueueFamilyIndices();
419
420 // Obtener las colas.
421
422 // Preferir cola √∫nica para comandos y gr√°ficos.
423 // Seg√∫n el spec, si hay al menos una familia de comandos gr√°ficos, tambi√©n debe soportar computaci√≥n y transfer.
424
425 const DynamicArray<QueueFamily> unifiedFamiliesList = queueFamilies.GetFamilies(
426 CommandsSupport::GRAPHICS |
427 CommandsSupport::COMPUTE |
428 CommandsSupport::TRANSFER |
429 CommandsSupport::PRESENTATION);
430
431 if (!unifiedFamiliesList.IsEmpty()) {
432 // Existe una familia con soporte para todos los comandos:
433 // usar cola unificada.
434
435 _SetUnifiedCommandQueue(new CommandQueueVk(unifiedFamiliesList[0], 0, GpuQueueType::MAIN, gpu));
436
437 RegisterUnifiedCommandPool(GetUnifiedQueue());
438
439 _SetMainCommandList(GetUnifiedCommandPool(std::this_thread::get_id())->CreateCommandList(gpu));
440
441 OSK::Engine::GetLogger()->InfoLog("Uso de cola GPU unificada.");
442 }
443 else {
444 // Colas potencialmente exclusivas para cada tarea.
445 //
446 // Una para graphics + compute + transfer (debe existir seg√∫n spec.).
447 const QueueFamily graphicsAndComputeFamily = queueFamilies.GetFamilies(
448 CommandsSupport::GRAPHICS |
449 CommandsSupport::COMPUTE |
450 CommandsSupport::TRANSFER)[0];
451
452 _SetGraphicsCommputeCommandQueue(new CommandQueueVk(graphicsAndComputeFamily, 0, GpuQueueType::MAIN, gpu));
453
454 RegisterGraphicsCommputeCommandPool(GetGraphicsComputeQueue());
455
456 // Una para presentaci√≥n.
457 const QueueFamily presentationFamily = queueFamilies.GetFamilies(
458 CommandsSupport::PRESENTATION)[0];
459
460 _SetPresentationCommandQueue(new CommandQueueVk(presentationFamily, 0, GpuQueueType::PRESENTATION, gpu));
461
462 _SetMainCommandList(GetGraphicsComputeCommandPool(std::this_thread::get_id())->CreateCommandList(gpu));
463
464 OSK::Engine::GetLogger()->InfoLog("Uso de colas GPU de renderizado y presentaci√≥n.");
465 }
466
467 // Buscamos una cola exclusivamente de transferencia.
468 const DynamicArray<QueueFamily> transferFamilies = queueFamilies.GetFamilies(
469 CommandsSupport::TRANSFER);
470
471 for (const auto& family : transferFamilies) {
472 if (family.support == CommandsSupport::TRANSFER) {
473 _SetTransferOnlyCommandQueue(new CommandQueueVk(family, 0, GpuQueueType::ASYNC_TRANSFER, gpu));
474 RegisterTransferOnlyCommandPool(GetTransferOnlyQueue());
475 OSK::Engine::GetLogger()->InfoLog("Uso de cola GPU de transferencia.");
476
477 break;
478 }
479 }

614void RendererVk::SubmitMainCommandList() {
615 // Esperar a que se completen todos los comandos.
616 const VkPipelineStageFlags waitStage = VK_PIPELINE_STAGE_ALL_COMMANDS_BIT;
617

637void RendererVk::SubmitFrame() {
638 const VkDevice logicalDevice = GetGpu()->As<GpuVk>()->GetLogicalDevice();
639
640 const VkSwapchainKHR swapChains = _GetSwapchain()->As<SwapchainVk>()->GetSwapchain();
641 // Esperamos a que la imagen haya terminado de rendereizarse.
642 const VkSemaphore waitSemaphores = m_imageFinishedSemaphores[currentCommandBufferIndex];
643
644 VkPresentInfoKHR presentInfo{};
645 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
646 presentInfo.waitSemaphoreCount = 1;
647 presentInfo.pWaitSemaphores = &waitSemaphores; // Debemos esperar a que la imagen actual termine de renderizarse.
648 presentInfo.swapchainCount = 1;
649 presentInfo.pSwapchains = &swapChains;
650 presentInfo.pImageIndices = &currentFrameIndex; // Lo presentamos en esta imagen.
651 presentInfo.pResults = nullptr;
652
653 const VkQueue queue = UseUnifiedCommandQueue()
654 ? GetUnifiedQueue()->As<CommandQueueVk>()->GetQueue()
655 : GetPresentationQueue()->As<CommandQueueVk>()->GetQueue();
656
657 VkResult result = vkQueuePresentKHR(queue, &presentInfo);
658 const bool resized = result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR;
659 if (resized) {
660 // Esperamos a que se terminen todas las listas de comandos para
661 // poder cambiar de tama√±o los render targets.
662 vkDeviceWaitIdle(logicalDevice);
664
665 const auto& resolution = Engine::GetDisplay()->GetResolution();
666
667 if (resolution.x > 0 && resolution.y > 0) {
668 _GetSwapchain()->As<SwapchainVk>()->Resize(*GetGpu(), resolution);
669 Engine::GetRenderer()->As<RendererVk>()->HandleResize(resolution);
670 }
671 }
672
673 currentCommandBufferIndex = (currentCommandBufferIndex + 1) % MAX_RESOURCES_IN_FLIGHT;
674
675 // Si la siguiente imagen est√° siendo procesada, esperar a que termine.
676 vkWaitForFences(logicalDevice, 1, &m_fullyRenderedFences[currentCommandBufferIndex], VK_TRUE, UINT64_MAX);
677 vkResetFences(logicalDevice, 1, &m_fullyRenderedFences[currentCommandBufferIndex]);
678}
