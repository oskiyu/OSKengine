209void DeferredRenderSystem::UpdateResolveMaterial() {
210 if (!m_resolverPass.GetPointer()) {
211 return;
212 }
213
214 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateSampled_MipLevelRanged(0, 0);
215
216 GpuImageViewConfig depthViewConfig = viewConfig;
217 depthViewConfig.channel = SampledChannel::DEPTH;
218
219 for (UIndex32 i = 0; i < MAX_RESOURCES_IN_FLIGHT; i++) {
220 const auto* depthView = m_gBuffer.GetImage(GBuffer::Target::DEPTH)->GetView(depthViewConfig);
221 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("depthTexture", *depthView);
222
223 const auto* colorView = m_gBuffer.GetImage(GBuffer::Target::COLOR)->GetView(viewConfig);
224 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("colorTexture", *colorView);
225
226 const auto* normalView = m_gBuffer.GetImage(GBuffer::Target::NORMAL)->GetView(viewConfig);
227 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("normalTexture", *normalView);
228
229 const auto* metallicView = m_gBuffer.GetImage(GBuffer::Target::METALLIC_ROUGHNESS)->GetView(viewConfig);
230 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("metallicRoughnessTexture", *metallicView);
231
232 const auto* emissiveView = m_gBuffer.GetImage(GBuffer::Target::EMISSIVE)->GetView(viewConfig);
233 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("emissiveTexture", *emissiveView);
234
235 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->FlushUpdate();
236
237 GpuImageViewConfig computeOutputConfig = GpuImageViewConfig::CreateStorage_Default();
238
239 m_resolverPass->GetMaterialInstance(i)->GetSlot("output")->SetStorageImage("finalImage",
240 *m_resolveRenderTarget.GetTargetImage()->GetView(computeOutputConfig));
241 m_resolverPass->GetMaterialInstance(i)->GetSlot("output")->FlushUpdate();
242 }

304void DeferredRenderSystem::Render(ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
305 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
306
307 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
308 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
309
310 CameraInfo currentCameraInfo{};
311 currentCameraInfo.projectionMatrix = camera.GetProjectionMatrix();
312 currentCameraInfo.viewMatrix = camera.GetViewMatrix(cameraTransform);
313 currentCameraInfo.projectionViewMatrix = currentCameraInfo.projectionMatrix * currentCameraInfo.viewMatrix;
314 currentCameraInfo.position = glm::vec4(cameraTransform.GetPosition().ToGlm(), 1.0f);
315 currentCameraInfo.nearFarPlanes = { camera.GetNearPlane(), camera.GetFarPlane() };
316
317 PreviousCameraInfo previousCameraInfo{};
318 previousCameraInfo.projectionMatrix = m_previousCameraProjection;
319 previousCameraInfo.viewMatrix = m_previousCameraView;
320 previousCameraInfo.projectionViewMatrix = m_previousCameraProjection * m_previousCameraView;
321
322 m_cameraBuffers[resourceIndex]->MapMemory();
323 m_cameraBuffers[resourceIndex]->Write(currentCameraInfo);
324 m_cameraBuffers[resourceIndex]->Unmap();
325
326 m_previousFrameCameraBuffers[resourceIndex]->MapMemory();
327 m_previousFrameCameraBuffers[resourceIndex]->Write(previousCameraInfo);
328 m_previousFrameCameraBuffers[resourceIndex]->Unmap();
329
330 m_previousCameraProjection = camera.GetProjectionMatrix();
331 m_previousCameraView = camera.GetViewMatrix(cameraTransform);
332
333 m_directionalLightBuffers[resourceIndex]->MapMemory();
334 m_directionalLightBuffers[resourceIndex]->Write(m_directionalLight);
335 m_directionalLightBuffers[resourceIndex]->Unmap();
336
337 m_iblConfigBuffers[resourceIndex]->MapMemory();
338 m_iblConfigBuffers[resourceIndex]->Write(m_iblConfig);
339 m_iblConfigBuffers[resourceIndex]->Unmap();
340
341 m_shadowMap.SetDirectionalLight(m_directionalLight);
342
343 m_gBuffer.BindPipelineBarriers(commandList);
344
345 UpdatePerPassObjectLists(objects);
346
347 // MeshMapping
348 for (auto& pass : m_shaderPasses.GetAllPasses()) {
349 pass->UpdateLocalMeshMapping(m_shaderPasses.GetCompatibleObjects(pass->GetTypeName()));
350 }
351
352 for (auto& pass : m_shadowsPasses.GetAllPasses()) {
353 pass->UpdateLocalMeshMapping(m_shadowsPasses.GetCompatibleObjects(pass->GetTypeName()));
354 }
355
356 for (const auto& obj : objects) {
357 if (!m_meshMapping.HasPreviousModelMatrix(obj)) {
358 const auto& transform = Engine::GetEcs()->GetComponent<Transform3D>(obj);
359 m_meshMapping.SetPreviousModelMatrix(obj, transform.GetAsMatrix());
360 }
361 }
362
363 GenerateShadows(commandList);
364
365 commandList->StartDebugSection("PBR Deferred", Color::Red);
366
367 RenderGBuffer(commandList);
368 ResolveGBuffer(commandList);
369
370 commandList->EndDebugSection();
371
372 ExecuteTaa(commandList);
373 CopyFinalImages(commandList);
374
375 for (const auto& obj : objects) {
376 const auto& transform = Engine::GetEcs()->GetComponent<Transform3D>(obj);
377 m_meshMapping.SetPreviousModelMatrix(obj, transform.GetAsMatrix());
378 }
379}

381void DeferredRenderSystem::GenerateShadows(ICommandList* commandList) {
382 const Viewport viewport{
383 .rectangle = { 0u, 0u,
384 m_shadowMap.GetColorImage()->GetSize2D().x,
385 m_shadowMap.GetColorImage()->GetSize2D().y }
386 };
387
388 commandList->StartDebugSection("Shadows", Color::Black);
389
390 commandList->SetGpuImageBarrier(
391 m_shadowMap.GetShadowImage(),
392 GpuImageLayout::UNDEFINED,
393 GpuImageLayout::DEPTH_STENCIL_TARGET,
394 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
395 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
396 GpuImageRange{
397 .baseLayer = 0,
398 .numLayers = m_shadowMap.GetNumCascades(),
399 .baseMipLevel = 0,
400 .numMipLevels = ALL_MIP_LEVELS,
401 .channel = SampledChannel::DEPTH });
402
403 commandList->SetViewport(viewport);
404 commandList->SetScissor(viewport.rectangle);
405
406 for (UIndex32 i = 0; i < m_shadowMap.GetNumCascades(); i++) {
407 commandList->StartDebugSection(std::format("Cascade {}", i), Color::Black);
408
409 RenderPassImageInfo colorInfo{};
410 colorInfo.arrayLevel = i;
411 colorInfo.targetImage = m_shadowMap.GetColorImage();
412
413 RenderPassImageInfo depthInfo{};
414 depthInfo.arrayLevel = i;
415 depthInfo.targetImage = m_shadowMap.GetShadowImage();
416
417 commandList->BeginGraphicsRenderpass({ colorInfo }, depthInfo, { 1.0f, 1.0f, 1.0f, 1.0f }, false);
418
419 for (auto& pass : m_shadowsPasses.GetAllPasses()) {
420 pass->As<IShadowsPass>()->ShadowsRenderLoop(
421 commandList,
422 m_shadowsPasses.GetCompatibleObjects(pass->GetTypeName()),
423 i,
424 m_shadowMap);
425 }
426
427 commandList->EndGraphicsRenderpass();
428
429 commandList->EndDebugSection();
430 }
431
432 commandList->EndDebugSection();
433}

435void DeferredRenderSystem::RenderGBuffer(ICommandList* commandList) {
436 commandList->StartDebugSection("PBR GBuffer", Color::Red);
437
438 m_gBuffer.BeginRenderpass(commandList, Color::Black * 0.0f, false);
439
440 SetupViewport(commandList);
441
442 // Bindeamos el slot de la cámara.
443 commandList->BindMaterial(*m_gBufferCameraInstances[Engine::GetRenderer()->GetCurrentResourceIndex()]->GetMaterial());
444
445 static UIndex32 jitterIndex = 0;
446 constexpr UIndex32 maxJitter = 4;
447
448 jitterIndex = m_taaProvider.GetCurrentFrameJitterIndex();
449
450 const Vector2ui resolution = m_gBuffer.GetImage(GBuffer::Target::COLOR)->GetSize2D();
451
452 for (auto& pass : m_shaderPasses.GetAllPasses()) {
453 // OSK_ASSERT(
454 // m_objectsPerPass.contains(pass->GetTypeName()),
455 // InvalidObjectStateException(std::format("No se encuentra el pase {}", pass->GetTypeName())));
456
457 commandList->BindMaterialSlot(*m_gBufferCameraInstances[Engine::GetRenderer()->GetCurrentResourceIndex()]->GetSlot("global"));
458 const auto& objectList = m_shaderPasses.GetCompatibleObjects(pass->GetTypeName());
459 pass->RenderLoop(commandList, objectList, &m_meshMapping, jitterIndex, resolution);
460 }
461
462 commandList->EndGraphicsRenderpass();
463
464 commandList->EndDebugSection();

467void DeferredRenderSystem::ResolveGBuffer(ICommandList* commandList) {
468 commandList->StartDebugSection("PBR Resolve", Color::Purple);
469
470 // Sincronización de mapa de sombras.
471 commandList->SetGpuImageBarrier(
472 m_shadowMap.GetShadowImage(),
473 GpuImageLayout::SAMPLED,
474 GpuBarrierInfo(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE),
475 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SAMPLED_READ),
476 GpuImageRange{
477 .baseLayer = 0,
478 .numLayers = m_shadowMap.GetNumCascades(),
479 .baseMipLevel = 0,
480 .numMipLevels = ALL_MIP_LEVELS,
481 .channel = SampledChannel::DEPTH });
482
483 // Sincronización de render target.
484 commandList->SetGpuImageBarrier(
485 m_resolveRenderTarget.GetTargetImage(),
486 GpuImageLayout::UNDEFINED,
487 GpuImageLayout::GENERAL,
488 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
489 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
490
491
492 // Sincronización con todos los targets de color.
493 for (const auto type : GBuffer::ColorTargetTypes)
494 commandList->SetGpuImageBarrier(
495 m_gBuffer.GetImage(type),
496 GpuImageLayout::SAMPLED,
497 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SAMPLED_READ));
498
499 // Sincronización con depth.
500 commandList->SetGpuImageBarrier(
501 m_gBuffer.GetImage(GBuffer::Target::DEPTH),
502 GpuImageLayout::SAMPLED,
503 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ),
504 { .channel = SampledChannel::DEPTH });
505
506 m_resolverPass->RenderLoop(
507 commandList, {}, &m_meshMapping,
508 m_taaProvider.GetCurrentFrameJitterIndex(),
509 m_resolveRenderTarget.GetSize());
510
511 commandList->EndDebugSection();

534void DeferredRenderSystem::CopyFinalImages(ICommandList* cmdList) {
535 cmdList->StartDebugSection("Final Image Copy", Color::Blue);
536
537 GpuImage* sourceImage = m_taaProvider.IsActive()
538 ? m_taaProvider.GetTaaOutput().GetTargetImage()
539 : m_resolveRenderTarget.GetTargetImage();
540 GpuImage* targetImage = m_renderTarget.GetMainColorImage();
541
542
543 // Imagen original: transfer source.
544 cmdList->SetGpuImageBarrier(
545 sourceImage,
546 GpuImageLayout::TRANSFER_SOURCE,
547 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
548
549 // Imagen del render target final: transfer destination.
550 cmdList->SetGpuImageBarrier(
551 targetImage,
552 GpuImageLayout::TRANSFER_DESTINATION,
553 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
554
555 cmdList->RawCopyImageToImage(
556 *sourceImage,
557 targetImage,
558 CopyImageInfo::CreateDefault2D(m_resolveRenderTarget.GetSize()));
559
560 cmdList->EndDebugSection();

563void DeferredRenderSystem::Execute(TDeltaTime deltaTime, std::span<const ECS::GameObjectIndex> objects) {
564 for (const GameObjectIndex obj : objects) {
565 auto& model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj);
566 Transform3D transformCopy = Engine::GetEcs()->GetComponent<Transform3D>(obj);
567 transformCopy.SetScale(Vector3f::One);
568
569 for (UIndex32 i = 0; i < model.GetModel()->GetMeshes().GetSize(); i++) {
570 auto& mesh = model.GetModel()->GetMeshes()[i];
571 const Vector3f spherePosition = Math::TransformPoint(mesh.GetSphereCenter(), transformCopy.GetAsMatrix());
572 mesh.GetBounds().SetPosition(spherePosition);
573 }
574
575 model.GetAnimator().Update(deltaTime);
576 }
