268void DeferredRenderSystem::Render(ICommandList* commandList) {
269 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
270
271 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
272 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(m_cameraObject);
273
274 CameraInfo currentCameraInfo{};
275 currentCameraInfo.projectionMatrix = camera.GetProjectionMatrix();
276 currentCameraInfo.viewMatrix = camera.GetViewMatrix(cameraTransform);
277 currentCameraInfo.projectionViewMatrix = currentCameraInfo.projectionMatrix * currentCameraInfo.viewMatrix;
278 currentCameraInfo.position = glm::vec4(cameraTransform.GetPosition().ToGlm(), 1.0f);
279 currentCameraInfo.nearFarPlanes = { camera.GetNearPlane(), camera.GetFarPlane() };
280
281 PreviousCameraInfo previousCameraInfo{};
282 previousCameraInfo.projectionMatrix = m_previousCameraProjection;
283 previousCameraInfo.viewMatrix = m_previousCameraView;
284 previousCameraInfo.projectionViewMatrix = m_previousCameraProjection * m_previousCameraView;
285
286 m_cameraBuffers[resourceIndex]->MapMemory();
287 m_cameraBuffers[resourceIndex]->Write(currentCameraInfo);
288 m_cameraBuffers[resourceIndex]->Unmap();
289
290 m_previousFrameCameraBuffers[resourceIndex]->MapMemory();
291 m_previousFrameCameraBuffers[resourceIndex]->Write(previousCameraInfo);
292 m_previousFrameCameraBuffers[resourceIndex]->Unmap();
293
294 m_previousCameraProjection = camera.GetProjectionMatrix();
295 m_previousCameraView = camera.GetViewMatrix(cameraTransform);
296
297 m_directionalLightBuffers[resourceIndex]->MapMemory();
298 m_directionalLightBuffers[resourceIndex]->Write(m_directionalLight);
299 m_directionalLightBuffers[resourceIndex]->Unmap();
300
301 m_iblConfigBuffers[resourceIndex]->MapMemory();
302 m_iblConfigBuffers[resourceIndex]->Write(m_iblConfig);
303 m_iblConfigBuffers[resourceIndex]->Unmap();
304
305 m_shadowMap.SetDirectionalLight(m_directionalLight);
306
307 m_gBuffer.BindPipelineBarriers(commandList);
308
309 GenerateShadows(commandList);
310
311 commandList->StartDebugSection("PBR Deferred", Color::Red);
312
313 UpdatePerPassObjectLists();
314
315 RenderGBuffer(commandList);
316 ResolveGBuffer(commandList);
317
318 commandList->EndDebugSection();
319
320 ExecuteTaa(commandList);
321 CopyFinalImages(commandList);
322}

324void DeferredRenderSystem::GenerateShadows(ICommandList* commandList) {
325 const Viewport viewport{
326 .rectangle = { 0u, 0u,
327 m_shadowMap.GetColorImage()->GetSize2D().x,
328 m_shadowMap.GetColorImage()->GetSize2D().y }
329 };
330
331 commandList->StartDebugSection("Shadows", Color::Black);
332
333 commandList->SetGpuImageBarrier(
334 m_shadowMap.GetShadowImage(),
335 GpuImageLayout::UNDEFINED,
336 GpuImageLayout::DEPTH_STENCIL_TARGET,
337 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
338 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
339 GpuImageRange{
340 .baseLayer = 0,
341 .numLayers = m_shadowMap.GetNumCascades(),
342 .baseMipLevel = 0,
343 .numMipLevels = ALL_MIP_LEVELS,
344 .channel = SampledChannel::DEPTH });
345
346 commandList->SetViewport(viewport);
347 commandList->SetScissor(viewport.rectangle);
348
349 for (UIndex32 i = 0; i < m_shadowMap.GetNumCascades(); i++) {
350 commandList->StartDebugSection(std::format("Cascade {}", i), Color::Black);
351
352 RenderPassImageInfo colorInfo{};
353 colorInfo.arrayLevel = i;
354 colorInfo.targetImage = m_shadowMap.GetColorImage();
355
356 RenderPassImageInfo depthInfo{};
357 depthInfo.arrayLevel = i;
358 depthInfo.targetImage = m_shadowMap.GetShadowImage();
359
360 commandList->BeginGraphicsRenderpass({ colorInfo }, depthInfo, { 1.0f, 1.0f, 1.0f, 1.0f }, false);
361
362 commandList->BindMaterial(*m_shadowMap.GetShadowsMaterial(ASSETS::ModelType::STATIC_MESH));
363 commandList->BindMaterialSlot(*m_shadowMap.GetShadowsMaterialInstance()->GetSlot("global"));
364
365 ShadowsRenderLoop(ModelType::STATIC_MESH, commandList, i);
366
367 commandList->BindMaterial(*m_shadowMap.GetShadowsMaterial(ModelType::ANIMATED_MODEL));
368 commandList->BindMaterialSlot(*m_shadowMap.GetShadowsMaterialInstance()->GetSlot("global"));
369
370 ShadowsRenderLoop(ModelType::ANIMATED_MODEL, commandList, i);
371
372 commandList->EndGraphicsRenderpass();
373
374 commandList->EndDebugSection();
375 }

377 commandList->EndDebugSection();

380void DeferredRenderSystem::ShadowsRenderLoop(ModelType modelType, ICommandList* commandList, UIndex32 cascadeIndex) {
381 for (const GameObjectIndex obj : GetObjects()) {
382 const ModelComponent3D& model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj);
383 const Transform3D& transform = Engine::GetEcs()->GetComponent<Transform3D>(obj);
384
385 if (model.GetModel()->GetType() != modelType)
386 continue;
387
388 if (!model.CastsShadows())
389 continue;
390
391 commandList->BindVertexBuffer(*model.GetModel()->GetVertexBuffer());
392 commandList->BindIndexBuffer(*model.GetModel()->GetIndexBuffer());
393
394 if (modelType == ModelType::ANIMATED_MODEL)
395 commandList->BindMaterialSlot(*model.GetModel()->GetAnimator()->GetMaterialInstance()->GetSlot("animation"));
396
397 struct {
398 glm::mat4 model;
399 int cascadeIndex;
400 } const pushConstant{
401 .model = transform.GetAsMatrix(),
402 .cascadeIndex = static_cast<int>(cascadeIndex)
403 };
404 commandList->PushMaterialConstants("model", pushConstant);
405
406 for (UIndex32 i = 0; i < model.GetModel()->GetMeshes().GetSize(); i++)
407 commandList->DrawSingleMesh(
408 model.GetModel()->GetMeshes()[i].GetFirstIndexId(),
409 model.GetModel()->GetMeshes()[i].GetNumberOfIndices());
410 }

416 m_gBuffer.BeginRenderpass(commandList, Color::Black * 0.0f, false);

512void DeferredRenderSystem::CopyFinalImages(ICommandList* cmdList) {
513 cmdList->StartDebugSection("Final Image Copy", Color::Blue);
514

541void DeferredRenderSystem::OnTick(TDeltaTime deltaTime) {
542 for (const GameObjectIndex obj : GetObjects()) {
543 Model3D* model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj).GetModel();
544 Transform3D transformCopy = Engine::GetEcs()->GetComponent<Transform3D>(obj);

546
547 for (UIndex32 i = 0; i < model->GetMeshes().GetSize(); i++) {

551 }
552
553 if (model->GetType() == ModelType::STATIC_MESH)
554 continue;
