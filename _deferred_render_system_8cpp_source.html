165void DeferredRenderSystem::SetupGlobalMeshMapping() {
166 for (const auto& obj : GetAllCompatibleObjects()) {
167 const auto* model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj).GetModel();
168
169 if (!m_meshMapping.HasModel(model->GetUuid())) {
170 m_meshMapping.RegisterModel(model->GetUuid());
171
172 if (model->HasAnimator()) {
173 auto& modelData = m_meshMapping.GetModelData(model->GetUuid());
174
175 // Buffer de huesos.
176 {
177 const auto boneCount = model->GetAnimator().GetFinalSkeletonMatrices().GetSize();
178 const auto bonesSize = sizeof(glm::mat4) * boneCount;
179 auto buffer = Engine::GetRenderer()->GetAllocator()->CreateStorageBuffer(bonesSize, GpuQueueType::MAIN);
180
181 modelData.SetBonesBuffer(std::move(buffer));
182 }
183
184 // Instancia de material.
185 {
186 auto* material = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial(AnimatedGBufferPass::MaterialPath);
187 auto instance = material->CreateInstance();
188 instance->GetSlot("animation")->SetStorageBuffer("animation", *modelData.GetBonesBuffer());
189 instance->GetSlot("animation")->FlushUpdate();
190
191 modelData.SetAnimationMaterialInstance(std::move(instance));
192 }
193 }
194 }
195 }
196
197 // Update bones:
198 for (const auto& obj : GetAllCompatibleObjects()) {
199 auto* model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj).GetModel();
200
201 if (model->HasAnimator()) {
202 const auto& animator = model->GetAnimator();
203
204 auto& modelData = m_meshMapping.GetModelData(model->GetUuid());
205 auto* boneBuffer = modelData.GetBonesBuffer();
206
207 boneBuffer->ResetCursor();
208 boneBuffer->MapMemory();
209 for (const auto& boneMatrix : animator.GetFinalSkeletonMatrices()) {
210 boneBuffer->Write(boneMatrix);

212 boneBuffer->Unmap();
213 }

270void DeferredRenderSystem::UpdateResolveMaterial() {
271 if (!m_resolverPass.GetPointer()) {
272 return;
273 }
274
275 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateSampled_MipLevelRanged(0, 0);
276
277 GpuImageViewConfig depthViewConfig = viewConfig;
278 depthViewConfig.channel = SampledChannel::DEPTH;
279

283
284 const auto* colorView = m_gBuffer.GetImage(GBuffer::Target::COLOR)->GetView(viewConfig);
285 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("colorTexture", *colorView, GpuImageSamplerDesc::CreateDefault_NoMipMap());
286
287 const auto* normalView = m_gBuffer.GetImage(GBuffer::Target::NORMAL)->GetView(viewConfig);
288 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("normalTexture", *normalView, GpuImageSamplerDesc::CreateDefault_NoMipMap());
289
290 const auto* metallicView = m_gBuffer.GetImage(GBuffer::Target::METALLIC_ROUGHNESS)->GetView(viewConfig);
291 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->SetGpuImage("metallicRoughnessTexture", *metallicView, GpuImageSamplerDesc::CreateDefault_NoMipMap());
292
293 const auto* emissiveView = m_gBuffer.GetImage(GBuffer::Target::EMISSIVE)->GetView(viewConfig);

295
296 m_resolverPass->GetMaterialInstance(i)->GetSlot("gbuffer")->FlushUpdate();
297
298 GpuImageViewConfig computeOutputConfig = GpuImageViewConfig::CreateStorage_Default();

300 m_resolverPass->GetMaterialInstance(i)->GetSlot("output")->SetStorageImage("finalImage",
301 *m_resolveRenderTarget.GetTargetImage()->GetView(computeOutputConfig));

365void DeferredRenderSystem::Render(ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
366 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
367
368 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(m_cameraObject);
369 const auto& cameraTransform = Engine::GetEcs()->GetComponent<TransformComponent3D>(m_cameraObject).GetTransform();
370
371 CameraInfo currentCameraInfo{};
372 currentCameraInfo.projectionMatrix = camera.GetProjectionMatrix();
373 currentCameraInfo.viewMatrix = camera.GetViewMatrix(cameraTransform);
374 currentCameraInfo.projectionViewMatrix = currentCameraInfo.projectionMatrix * currentCameraInfo.viewMatrix;
375 currentCameraInfo.position = glm::vec4(cameraTransform.GetPosition().ToGlm(), 1.0f);
376 currentCameraInfo.nearFarPlanes = { camera.GetNearPlane(), camera.GetFarPlane() };
377
378 PreviousCameraInfo previousCameraInfo{};
379 previousCameraInfo.projectionMatrix = m_previousCameraProjection;
380 previousCameraInfo.viewMatrix = m_previousCameraView;
381 previousCameraInfo.projectionViewMatrix = m_previousCameraProjection * m_previousCameraView;
382
383 m_cameraBuffers[resourceIndex]->MapMemory();
384 m_cameraBuffers[resourceIndex]->Write(currentCameraInfo);
385 m_cameraBuffers[resourceIndex]->Unmap();
386
387 m_previousFrameCameraBuffers[resourceIndex]->MapMemory();
388 m_previousFrameCameraBuffers[resourceIndex]->Write(previousCameraInfo);
389 m_previousFrameCameraBuffers[resourceIndex]->Unmap();
390
391 m_previousCameraProjection = camera.GetProjectionMatrix();
392 m_previousCameraView = camera.GetViewMatrix(cameraTransform);
393
394 m_directionalLightBuffers[resourceIndex]->MapMemory();
395 m_directionalLightBuffers[resourceIndex]->Write(m_directionalLight);
396 m_directionalLightBuffers[resourceIndex]->Unmap();
397
398 m_iblConfigBuffers[resourceIndex]->MapMemory();
399 m_iblConfigBuffers[resourceIndex]->Write(m_iblConfig);
400 m_iblConfigBuffers[resourceIndex]->Unmap();
401
402 m_shadowMap.SetDirectionalLight(m_directionalLight);
403
404 m_gBuffer.BindPipelineBarriers(commandList);
405
406 UpdatePerPassObjectLists(objects);
407
408 // MeshMapping
409 for (auto& pass : m_shaderPasses.GetAllPasses()) {
410 pass->UpdateLocalMeshMapping(m_shaderPasses.GetCompatibleObjects(pass->GetTypeName()));
411 }
412
413 for (auto& pass : m_shadowsPasses.GetAllPasses()) {
414 pass->UpdateLocalMeshMapping(m_shadowsPasses.GetCompatibleObjects(pass->GetTypeName()));
415 }
416
417 SetupGlobalMeshMapping();
418
419 for (const auto& obj : objects) {
420 if (!m_meshMapping.HasPreviousModelMatrix(obj)) {
421 const auto& transform = Engine::GetEcs()->GetComponent<TransformComponent3D>(obj).GetTransform();
422 m_meshMapping.SetPreviousModelMatrix(obj, transform.GetAsMatrix());
423 }
424 }
425
426 GenerateShadows(commandList);
427
428 commandList->StartDebugSection("PBR Deferred", Color::Red);
429
430 RenderGBuffer(commandList);
431 ResolveGBuffer(commandList);
432
433 commandList->EndDebugSection();
434
435 ExecuteTaa(commandList);
436 CopyFinalImages(commandList);
437
438 for (const auto& obj : objects) {
439 const auto& transform = Engine::GetEcs()->GetComponent<TransformComponent3D>(obj).GetTransform();
440 m_meshMapping.SetPreviousModelMatrix(obj, transform.GetAsMatrix());
441 }
442}

444void DeferredRenderSystem::GenerateShadows(ICommandList* commandList) {
445 const Viewport viewport{
446 .rectangle = { 0u, 0u,
447 m_shadowMap.GetColorImage()->GetSize2D().x,
448 m_shadowMap.GetColorImage()->GetSize2D().y }
449 };
450
451 commandList->StartDebugSection("Shadows", Color::Black);
452
453 commandList->SetGpuImageBarrier(
454 m_shadowMap.GetShadowImage(),
455 GpuImageLayout::UNDEFINED,
456 GpuImageLayout::DEPTH_STENCIL_TARGET,
457 GpuBarrierInfo(GpuCommandStage::NONE, GpuAccessStage::NONE),
458 GpuBarrierInfo(GpuCommandStage::DEPTH_STENCIL_START, GpuAccessStage::DEPTH_STENCIL_READ | GpuAccessStage::DEPTH_STENCIL_WRITE),
459 GpuImageRange{
460 .baseLayer = 0,
461 .numLayers = m_shadowMap.GetNumCascades(),
462 .baseMipLevel = 0,
463 .numMipLevels = ALL_MIP_LEVELS,
464 .channel = SampledChannel::DEPTH });
465
466 commandList->SetViewport(viewport);
467 commandList->SetScissor(viewport.rectangle);
468
469 for (UIndex32 i = 0; i < m_shadowMap.GetNumCascades(); i++) {
470 commandList->StartDebugSection(std::format("Cascade {}", i), Color::Black);
471
472 RenderPassImageInfo colorInfo{};
473 colorInfo.arrayLevel = i;
474 colorInfo.targetImage = m_shadowMap.GetColorImage();
475
476 RenderPassImageInfo depthInfo{};
477 depthInfo.arrayLevel = i;
478 depthInfo.targetImage = m_shadowMap.GetShadowImage();
479
480 commandList->BeginGraphicsRenderpass({ colorInfo }, depthInfo, { 1.0f, 1.0f, 1.0f, 1.0f }, false);
481
482 for (auto& pass : m_shadowsPasses.GetAllPasses()) {
483 pass->As<IShadowsPass>()->ShadowsRenderLoop(
484 commandList,
485 m_shadowsPasses.GetCompatibleObjects(pass->GetTypeName()),
486 i,
487 m_shadowMap);
488 }
489
490 commandList->EndGraphicsRenderpass();
491
492 commandList->EndDebugSection();
493 }
494
495 commandList->EndDebugSection();
496}

597void DeferredRenderSystem::CopyFinalImages(ICommandList* cmdList) {
598 cmdList->StartDebugSection("Final Image Copy", Color::Blue);
599
600 GpuImage* sourceImage = m_taaProvider.IsActive()
601 ? m_taaProvider.GetTaaOutput().GetTargetImage()
602 : m_resolveRenderTarget.GetTargetImage();
603 GpuImage* targetImage = m_renderTarget.GetMainColorImage();
604
605
606 // Imagen original: transfer source.
607 cmdList->SetGpuImageBarrier(
608 sourceImage,
609 GpuImageLayout::TRANSFER_SOURCE,
610 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
611
612 // Imagen del render target final: transfer destination.
