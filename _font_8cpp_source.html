33void Font::LoadSizedFont(USize32 fontSize) {
34 if (ContainsInstance(fontSize))
35 return;

37 // Carga de la librería.
38 static FT_Library freeType = nullptr;
39 if (freeType == nullptr) {
40 const auto result = FT_Init_FreeType(&freeType);
41 OSK_ASSERT(result == 0, FontLibraryInitializationException(result));
42 }
43
44 // Un FT_Face describe la tipografía de la fuente.
45 FT_Face face = nullptr;
46
47 // Carga de la fuente.
48 FT_Error result = FT_New_Face(freeType, m_fontFile.c_str(), 0, &face);
49 OSK_ASSERT(result == 0, FontLodaingException(result));
50
51 // Establece el tamaño de esta instancia en concreto.
52 FT_Set_Pixel_Sizes(face, 0, fontSize);
53
54
55 // Definirá cada uno de los caracteres.
56 struct FtChar {
57 UniquePtr<TByte> data;
58 USize32 sizeX = 0;
59 USize32 sizeY = 0;
60 USize32 left = 0;
61 USize32 top = 0;
62 USize32 advanceX = 0;
63 };
64

66
67 Vector2ui gpuImageSize = { 0, 0 };
68
69 // Cargamos todos los caracteres.
70 // Actualizamos el tamaño de la imagen.
71 for (USize32 i = 0; i < 255; i++) {
72 // Carga / renderizado del char.
73 result = FT_Load_Char(face, i, FT_LOAD_RENDER);
74 OSK_ASSERT(!result, FontCharacterLodaingException(i, result));
75
76 ftCharacters[i].sizeX = face->glyph->bitmap.width;
77 ftCharacters[i].sizeY = face->glyph->bitmap.rows;
78
79 ftCharacters[i].left = face->glyph->bitmap_left;
80 ftCharacters[i].top = face->glyph->bitmap_top;
81
82 ftCharacters[i].advanceX = face->glyph->advance.x;
83
84 ftCharacters[i].data = new TByte[ftCharacters[i].sizeX * ftCharacters[i].sizeY];
85
86 // Copiamos los datos al buffer de la imagen de la GPU.
87 const size_t bitmapSize = ftCharacters[i].sizeX * ftCharacters[i].sizeY * sizeof(TByte);
88 memcpy(
89 ftCharacters[i].data.GetPointer(),
90 face->glyph->bitmap.buffer,
91 bitmapSize);
92
93 // Tamaño de la imagen.
94 gpuImageSize.x += ftCharacters[i].sizeX;
95 if (ftCharacters[i].sizeY > gpuImageSize.y)
96 gpuImageSize.y = ftCharacters[i].sizeY;
97 }
98
99 // Creación de la imagen.
100 GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(
101 gpuImageSize, Format::RGBA8_SRGB,
102 GpuImageUsage::SAMPLED | GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION);
103
104 OwnedPtr<GpuImage> gpuImage = Engine::GetRenderer()->GetAllocator()->CreateImage(imageInfo);
105 gpuImage->SetDebugName(std::format("Font {} size {}", GetName(), fontSize));
106
107 FontInstance instance = {};
108
109 instance.image = gpuImage.GetPointer();
110
111 const auto numBytes = gpuImage->GetNumberOfBytes();
112 TByte* data = new TByte[numBytes];
113
114 USize32 currentX = 0;
115 for (USize32 c = 0; c < 255; c++) {
116 for (USize32 i = 0; i < ftCharacters[c].sizeY; i++) {
117 memcpy(&data[currentX + gpuImageSize.x * i],
118 &ftCharacters[c].data.GetPointer()[ftCharacters[c].sizeX * i],
119 ftCharacters[c].sizeX);
120 }
121
122 FontCharacter fontChar{};
123
124 fontChar.size = {
125 static_cast<float>(ftCharacters[c].sizeX),
126 static_cast<float>(ftCharacters[c].sizeY) };
127
128 fontChar.bearing = {
129 static_cast<float>(ftCharacters[c].left),
130 static_cast<float>(ftCharacters[c].top) };
131
132 fontChar.advance = ftCharacters[c].advanceX;
133
134 fontChar.texCoords = {
135 static_cast<int>(currentX),
136 0,
137 static_cast<int>(fontChar.size.x),
138 static_cast<int>(fontChar.size.y) };
139
140 instance.characters[static_cast<char>(c)] = fontChar;
141
142 currentX += ftCharacters[c].sizeX;
143 }
144
145 auto finalPixels = DynamicArray<TByte>::CreateReservedArray(numBytes);
146 for (USize32 i = 0; i < numBytes / 4; i++) {
147 finalPixels.Insert(255);
148 finalPixels.Insert(255);
149 finalPixels.Insert(255);
150 finalPixels.Insert(data[i]);
151 }
152
153 OwnedPtr<ICommandList> copyCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
154 copyCmdList->Reset();
155 copyCmdList->Start();
156
157 copyCmdList->SetGpuImageBarrier(
158 gpuImage.GetPointer(),
159 GpuImageLayout::TRANSFER_DESTINATION,
160 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
161
162 Engine::GetRenderer()->UploadImageToGpu(gpuImage.GetPointer(), finalPixels.GetData(), numBytes, copyCmdList.GetPointer());
163
164 copyCmdList->SetGpuImageBarrier(
165 gpuImage.GetPointer(),
166 GpuImageLayout::SAMPLED,
167 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ));
168 copyCmdList->Close();
169 Engine::GetRenderer()->SubmitSingleUseCommandList(copyCmdList.GetPointer());
170
171 const GpuImageViewConfig viewConfig = GpuImageViewConfig::CreateSampled_SingleMipLevel(0);
172
173 instance.sprite = new Sprite;
174 instance.sprite->SetImageView(gpuImage->GetView(viewConfig));
175
176 m_instances[fontSize] = std::move(instance);
177
178 delete[] data;
179 // Memory free
180 FT_Done_Face(face);
181}
