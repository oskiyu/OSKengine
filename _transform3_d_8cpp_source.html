221void Transform3D::UpdateModel(std::optional<const Transform3D*> parent) {
222 m_matrix = glm::mat4(1.0f);
223
224 if (parent.has_value()) {
225 const Transform3D& parentTransform = *parent.value();
226
227 m_globalRotation = m_inheritRotation
228 ? parentTransform.m_globalRotation * m_localRotation.ToMat4()
229 : m_localRotation.ToMat4();
230
231 if (m_inheritPosition) {
232 m_matrix = glm::translate(m_matrix, parentTransform.GetPosition().ToGlm());
233 }
234
235 if (m_inheritRotation) {
236 m_matrix = m_matrix * parentTransform.m_globalRotation;
237 }
238 }
239 else {
240 m_globalRotation = m_localRotation.ToMat4();
241 }
242
243 // Posici√≥n local.
244 m_matrix = glm::translate(m_matrix, m_localPosition.ToGlm());
245
246 // Rotaci√≥n local.
247 m_matrix = m_matrix * m_localRotation.ToMat4();
248
249 // Escala local.
250 m_matrix = glm::scale(m_matrix, m_localScale.ToGlm());
251
252
253 // Obtener posici√≥n final.
254 m_globalPosition = Vector3f(m_matrix * glm::vec4(0, 0, 0, 1));
255}
