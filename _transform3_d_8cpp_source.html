193void Transform3D::UpdateModel(std::optional<const Transform3D*> parent) {
194 m_matrix = glm::mat4(1.0f);
195
196 if (parent.has_value()) {
197 const Transform3D& parentTransform = *parent.value();
198
199 m_globalRotation = m_inheritRotation
200 ? parentTransform.m_globalRotation * m_localRotation.ToMat4()
201 : m_localRotation.ToMat4();
202
203 if (m_inheritPosition) {
204 m_matrix = glm::translate(m_matrix, parentTransform.GetPosition().ToGlm());
205 }
206
207 if (m_inheritRotation) {
208 m_matrix = m_matrix * parentTransform.m_globalRotation;
209 }
210 }
211 else {
212 m_globalRotation = m_localRotation.ToMat4();
213 }
214
215 // Posición local.
216 m_matrix = glm::translate(m_matrix, m_localPosition.ToGlm());
217
218 // Rotación local.
219 m_matrix = m_matrix * m_localRotation.ToMat4();
220
221 // Escala local.
222 m_matrix = glm::scale(m_matrix, m_localScale.ToGlm());
223
224
225 // Obtener posición final.
226 m_globalPosition = Vector3f(m_matrix * glm::vec4(0, 0, 0, 1));
227}
