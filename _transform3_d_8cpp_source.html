223void Transform3D::UpdateModel(std::optional<const Transform3D*> parent) {
224 m_matrix = glm::mat4(1.0f);
225
226 if (parent.has_value()) {
227 const Transform3D& parentTransform = *parent.value();
228
229 m_globalRotation = m_inheritRotation
230 ? parentTransform.m_globalRotation * m_localRotation.ToMat4()
231 : m_localRotation.ToMat4();
232
233 if (m_inheritPosition) {
234 m_matrix = glm::translate(m_matrix, parentTransform.GetPosition().ToGlm());
235 }
236
237 if (m_inheritRotation) {
238 m_matrix = m_matrix * parentTransform.m_globalRotation;
239 }
240 }
241 else {
242 m_globalRotation = m_localRotation.ToMat4();
243 }
244
245 // Posici√≥n local.
246 m_matrix = glm::translate(m_matrix, m_localPosition.ToGlm());
247
248 // Rotaci√≥n local.
249 m_matrix = m_matrix * m_localRotation.ToMat4();
250
251 // Escala local.
252 m_matrix = glm::scale(m_matrix, m_localScale.ToGlm());
253
254
255 // Obtener posici√≥n final.
256 m_globalPosition = Vector3f(m_matrix * glm::vec4(0, 0, 0, 1));
257}
