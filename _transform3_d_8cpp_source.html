95void Transform3D::UpdateModel() {
96 m_matrix = glm::mat4(1.0f);
97
98 if (parent) {
99 const Transform3D& parentTransform = GetEcs()->GetComponent<Transform3D>(parent);
100
101 m_globalRotation = m_inheritRotation
102 ? parentTransform.m_globalRotation * m_localRotation.ToMat4()
103 : m_localRotation.ToMat4();
104
105 if (m_inheritPosition) {
106 m_matrix = glm::translate(m_matrix, parentTransform.GetPosition().ToGlm());
107 }
108
109 if (m_inheritRotation) {
110 m_matrix = m_matrix * parentTransform.m_globalRotation;
111 }
112 }
113 else {
114 m_globalRotation = m_localRotation.ToMat4();
115 }
116
117 // Posición local.
118 m_matrix = glm::translate(m_matrix, m_localPosition.ToGlm());
119
120 // Rotación local.
121 m_matrix = m_matrix * m_localRotation.ToMat4();
122
123 // Escala local.
124 m_matrix = glm::scale(m_matrix, m_localScale.ToGlm());
125
126
127 // Obtener posición final.
128 m_globalPosition = Vector3f(m_matrix * glm::vec4(0, 0, 0, 1));
129
130 for (UIndex64 i = 0; i < childTransforms.GetSize(); i++) {
131 Transform3D& child = GetEcs()->GetComponent<Transform3D>(childTransforms[i]);
132
133 if (child.parent == owner) {
134 child.UpdateModel();
135 }
136 else {
137 childTransforms[i] = childTransforms.Pop();
138 }
139 }
140}
