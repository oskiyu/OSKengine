46VkShaderModule IPipelineVk::CreateShaderModule(std::span<const char> code, std::string_view name, VkDevice logicalDevice) {

48
49 VkShaderModuleCreateInfo createInfo{};
50 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;

60 nameInfo.objectType = VK_OBJECT_TYPE_SHADER_MODULE;

62 nameInfo.pObjectName = name.data();
63
64 RendererVk::pvkSetDebugUtilsObjectNameEXT(logicalDevice, &nameInfo);
65 }
66
67 return output;

70VkFormat IPipelineVk::GetVertexAttribFormat(const VertexInfo::Entry& entry) const {
71 switch (entry.type) {
72
73 case VertexInfo::Entry::Type::INT:
74 if (entry.size == 2 * sizeof(int)) return VK_FORMAT_R32G32_SINT;
75 if (entry.size == 3 * sizeof(int)) return VK_FORMAT_R32G32B32_SINT;
76 if (entry.size == 4 * sizeof(int)) return VK_FORMAT_R32G32B32A32_SINT;
77 break;
78
79 case VertexInfo::Entry::Type::FLOAT:
80 if (entry.size == 2 * sizeof(float)) return VK_FORMAT_R32G32_SFLOAT;
81 if (entry.size == 3 * sizeof(float)) return VK_FORMAT_R32G32B32_SFLOAT;
82 if (entry.size == 4 * sizeof(float)) return VK_FORMAT_R32G32B32A32_SFLOAT;
83 break;
84
85 case VertexInfo::Entry::Type::UNSIGNED_INT:
86 if (entry.size == 1 * sizeof(unsigned int)) return VK_FORMAT_R32_UINT;
87 if (entry.size == 2 * sizeof(unsigned int)) return VK_FORMAT_R32G32_UINT;
88 if (entry.size == 3 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32_UINT;
89 if (entry.size == 4 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32A32_UINT;
90 break;
91 }
92
93 OSK_ASSERT(false, NotImplementedException());
94 return VK_FORMAT_MAX_ENUM;
95}

97DynamicArray<VkVertexInputAttributeDescription> IPipelineVk::GetAttributeDescription(const VertexInfo& info) const {
98 DynamicArray<VkVertexInputAttributeDescription> output;

100 USize32 offset = 0;

102 VkVertexInputAttributeDescription desc{};

106 desc.offset = offset;

110

112 }

195ShaderStageVk IPipelineVk::LoadShader(const std::string& path, ShaderStage stage) {
196 ShaderStageVk output{};
197
198 // Lee el código SPIR-V.
199 const DynamicArray<char> code = IO::FileIO::ReadBinaryFromFile(path);
200
201 VkShaderModuleCreateInfo createInfo{};
202 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
203 createInfo.codeSize = code.GetSize();
204 createInfo.pCode = reinterpret_cast<const uint32_t*>(code.GetData());
205
206 VkResult result = vkCreateShaderModule(Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice(),
207 &createInfo, nullptr, &output.shaderModule);
208 OSK_ASSERT(result == VK_SUCCESS, ShaderLoadingException(result));
209
210 // Insertar el stage en el array, para poder insertarlo al crear el pipeline.
211 output.shaderCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
212 output.shaderCreateInfo.stage = GetShaderStageVk(stage);
213 output.shaderCreateInfo.module = output.shaderModule;
214 output.shaderCreateInfo.pName = "main";
215
216 shaderStagesInfo.Insert(output.shaderCreateInfo);
217 shaderModulesToDelete.Insert(output.shaderModule);
218
219 return output;
220}
