67VkFormat IPipelineVk::GetVertexAttribFormat(const VertexInfo::Entry& entry) const {
68 switch (entry.type) {
69
70 case VertexInfo::Entry::Type::INT:
71 if (entry.size == 2 * sizeof(int)) return VK_FORMAT_R32G32_SINT;
72 if (entry.size == 3 * sizeof(int)) return VK_FORMAT_R32G32B32_SINT;
73 if (entry.size == 4 * sizeof(int)) return VK_FORMAT_R32G32B32A32_SINT;
74 break;
75
76 case VertexInfo::Entry::Type::FLOAT:
77 if (entry.size == 2 * sizeof(float)) return VK_FORMAT_R32G32_SFLOAT;
78 if (entry.size == 3 * sizeof(float)) return VK_FORMAT_R32G32B32_SFLOAT;
79 if (entry.size == 4 * sizeof(float)) return VK_FORMAT_R32G32B32A32_SFLOAT;
80 break;
81
82 case VertexInfo::Entry::Type::UNSIGNED_INT:
83 if (entry.size == 1 * sizeof(unsigned int)) return VK_FORMAT_R32_UINT;
84 if (entry.size == 2 * sizeof(unsigned int)) return VK_FORMAT_R32G32_UINT;
85 if (entry.size == 3 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32_UINT;
86 if (entry.size == 4 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32A32_UINT;
87 break;
88 }
89
90 OSK_ASSERT(false, NotImplementedException());
91 return VK_FORMAT_MAX_ENUM;
92}
