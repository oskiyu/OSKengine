41VkFormat IPipelineVk::GetVertexAttribFormat(const VertexInfo::Entry& entry) const {
42 switch (entry.type) {
43
44 case VertexInfo::Entry::Type::INT:
45 if (entry.size == 2 * sizeof(int)) return VK_FORMAT_R32G32_SINT;
46 if (entry.size == 3 * sizeof(int)) return VK_FORMAT_R32G32B32_SINT;
47 if (entry.size == 4 * sizeof(int)) return VK_FORMAT_R32G32B32A32_SINT;
48 break;
49
50 case VertexInfo::Entry::Type::FLOAT:
51 if (entry.size == 2 * sizeof(float)) return VK_FORMAT_R32G32_SFLOAT;
52 if (entry.size == 3 * sizeof(float)) return VK_FORMAT_R32G32B32_SFLOAT;
53 if (entry.size == 4 * sizeof(float)) return VK_FORMAT_R32G32B32A32_SFLOAT;
54 break;
55
56 case VertexInfo::Entry::Type::UNSIGNED_INT:
57 if (entry.size == 1 * sizeof(unsigned int)) return VK_FORMAT_R32_UINT;
58 if (entry.size == 2 * sizeof(unsigned int)) return VK_FORMAT_R32G32_UINT;
59 if (entry.size == 3 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32_UINT;
60 if (entry.size == 4 * sizeof(unsigned int)) return VK_FORMAT_R32G32B32A32_UINT;
61 break;
62 }
63
64 OSK_ASSERT(false, NotImplementedException());
65 return VK_FORMAT_MAX_ENUM;
66}

166ShaderStageVk IPipelineVk::LoadShader(const std::string& path, ShaderStage stage) {
167 ShaderStageVk output{};
168
169 // Lee el código SPIR-V.
170 const DynamicArray<char> code = IO::FileIO::ReadBinaryFromFile(path);
171
172 VkShaderModuleCreateInfo createInfo{};
173 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
174 createInfo.codeSize = code.GetSize();
175 createInfo.pCode = reinterpret_cast<const uint32_t*>(code.GetData());
176
177 VkResult result = vkCreateShaderModule(Engine::GetRenderer()->GetGpu()->As<GpuVk>()->GetLogicalDevice(),
178 &createInfo, nullptr, &output.shaderModule);
179 OSK_ASSERT(result == VK_SUCCESS, ShaderLoadingException(result));
180
181 // Insertar el stage en el array, para poder insertarlo al crear el pipeline.
182 output.shaderCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
183 output.shaderCreateInfo.stage = GetShaderStageVk(stage);
184 output.shaderCreateInfo.module = output.shaderModule;
185 output.shaderCreateInfo.pName = "main";
186
187 shaderStagesInfo.Insert(output.shaderCreateInfo);
188 shaderModulesToDelete.Insert(output.shaderModule);
189
190 return output;
191}
