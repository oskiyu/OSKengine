105void TaaProvider::ExecuteTaaFirstPass(ICommandList* commandList) {
106 commandList->StartDebugSection(TaaAccumulationDebugName, Color::Purple);
107
108 GpuImage* historicalImage = m_taaHistoricalImage.GetTargetImage();
109 GpuImage* finalImage = m_taaRenderTarget.GetTargetImage();
110
111 const GpuImageLayout previousLayout = historicalImage->_GetLayout(0, 0);
112 commandList->SetGpuImageBarrier(

115 GpuImageLayout::SAMPLED,

124 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));

129
130 commandList->BindMaterial(*m_taaMaterialInstance->GetMaterial());
