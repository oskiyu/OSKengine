107bool UI::Dropdown::UpdateByCursor(Vector2f cursor, bool isPressed) {
108 if (!IsVisible() || IsLocked() || m_elements.IsEmpty())
109 return false;
110
111 if (m_state == State::DEFAULT) {
112 const bool isMouseOver = GetRectangle().ContainsPoint(cursor);
113
114 m_elements[m_selectedElement].SetState(isMouseOver ? Element::State::HOVERED : Element::State::DEFAULT);
115
116 if (isMouseOver && isPressed && !m_wasPreviousFramePressed) {
117 m_state = State::EXPANDED;
118 m_wasPreviousFramePressed = true;
119 } else
120 if (!isMouseOver || !isPressed) {
121 m_wasPreviousFramePressed = false;
122 }
123
124 return isMouseOver;
125 }
126
127 else if (m_state == State::EXPANDED) {
128 Vector2f offset = GetContentTopLeftPosition();
129
130 Element& firstElement = m_elements[m_selectedElement];
131 bool isMouseOver = firstElement.GetRectangle().ContainsPoint(cursor);
132
133 firstElement.SetState(isMouseOver ? Element::State::HOVERED : Element::State::DEFAULT);
134
135 if (isMouseOver && isPressed && !m_wasPreviousFramePressed) {
136 m_state = State::DEFAULT;
137 m_wasPreviousFramePressed = true;
138
139 return true;
140 }
141
142 m_wasPreviousFramePressed = isMouseOver && isPressed;
143
144 offset.y += firstElement.GetSize().y;
145
146 for (UIndex64 i = 0; i < m_elements.GetSize(); i++) {
147 if (i == m_selectedElement)
148 continue;
149
150 Element& e = m_elements[i];
151
152 isMouseOver = e.GetRectangle().ContainsPoint(cursor);
153
154 e.SetState(isMouseOver ? Element::State::HOVERED : Element::State::DEFAULT);
155
156 if (isMouseOver && isPressed) {
157 SelectOption(e.GetText());
158 m_wasPreviousFramePressed = true;
159
160 return true;
161 }
162
163 offset.y += e.GetSize().y;
164
165 if (isMouseOver)
166 return true;
167 }

169 return false;
