82 void RemoveComponent(GameObjectIndex obj) {
83 OSK_ASSERT(
84 m_objectToComponent.contains(obj),
85 InvalidArgumentException(std::format("El objeto {} no contiene el componente.", obj.Get())))
86
87 ComponentIndex compIndex = m_objectToComponent.at(obj);
88 const UIndex64 indexOfLast = m_components.GetSize() - 1;
89
90 // Hacemos que el componente a eliminar esté en la última posición.
91 // El último componente se coloca en el hueco del componente eliminado,
92 // para evitar dejar huecos en el array.
93 std::swap(m_components[compIndex], m_components[indexOfLast]);
94
95 // Obtenemos el objeto dueño del componente que acabamos de colocar
96 // donde el componente eliminado.
97 // Actualizamos las tablas para que apunten al nuevo lugar que ocupa.
98 GameObjectIndex objectOfLast = m_componentToObject.at(indexOfLast);
99 m_objectToComponent.at(objectOfLast) = compIndex;
100 m_componentToObject.at(compIndex) = objectOfLast;
101
102 // Eliminación final.
103 m_components.RemoveLast();
104
105 m_objectToComponent.erase(obj);
106 m_componentToObject.erase(indexOfLast);
107 }

212 PERSISTENCE::BinaryBlock BinarySerializeAll() const override {
213 PERSISTENCE::BinaryBlock output{};
214
215 if constexpr (IsSerializableComponent<TComponent>) {
216
217 output.Write<USize64>(m_components.GetSize());
218
219 // Datos.
220 auto datasBlock = PERSISTENCE::BinaryBlock::Empty();
221 for (const auto& component : m_components) {
222 datasBlock.AppendBlock(PERSISTENCE::BinarySerializeComponent<TComponent>(component));
223 }
224
225 // Offset del bloque de objetos enlazados.
226 // = tamaño de datos + lo escrito hasta ahora.
227 output.Write<USize64>(datasBlock.GetCurrentSize() + output.GetCurrentSize() + sizeof(USize64));
228
229 output.AppendBlock(datasBlock);
230
231 // Mapa ID componente -> ID objeto.
232 for (UIndex64 compIdx = 0; compIdx < m_components.GetSize(); compIdx++) {
233 output.Write<GameObjectIndex::TUnderlyingType>(m_componentToObject.at(compIdx).Get());
234 }
235
236 }
