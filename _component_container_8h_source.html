81 void RemoveComponent(GameObjectIndex obj) {
82 OSK_ASSERT(
83 m_objectToComponent.contains(obj),
84 InvalidArgumentException(std::format("El objeto {} no contiene el componente.", obj.Get())))
85
86 ComponentIndex compIndex = m_objectToComponent.at(obj);
87 const UIndex64 indexOfLast = m_components.GetSize() - 1;
88
89 // Hacemos que el componente a eliminar esté en la última posición.
90 // El último componente se coloca en el hueco del componente eliminado,
91 // para evitar dejar huecos en el array.
92 MEMORY::MemorySwap(&m_components[compIndex], &m_components[indexOfLast], sizeof(TComponent));
93
94 // Obtenemos el objeto dueño del componente que acabamos de colocar
95 // donde el componente eliminado.
96 // Actualizamos las tablas para que apunten al nuevo lugar que ocupa.
97 GameObjectIndex objectOfLast = m_componentToObject.at(indexOfLast);
98 m_objectToComponent.at(objectOfLast) = compIndex;
99 m_componentToObject.at(compIndex) = objectOfLast;
100
101 // Eliminación final.
102 m_components.RemoveLast();
103
104 m_objectToComponent.erase(obj);
105 m_componentToObject.erase(indexOfLast);
106 }

187 PERSISTENCE::BinaryBlock BinarySerializeAll() const override {
188 PERSISTENCE::BinaryBlock output{};
189
190 output.Write<USize64>(m_components.GetSize());
191
192 // Datos.
193 auto datasBlock = PERSISTENCE::BinaryBlock::Empty();
194 for (const auto& component : m_components) {
195 datasBlock.AppendBlock(PERSISTENCE::BinarySerializeComponent<TComponent>(component));
196 }
197
198 // Offset del bloque de objetos enlazados.
199 // = tamaño de datos + lo escrito hasta ahora.
200 output.Write<USize64>(datasBlock.GetCurrentSize() + output.GetCurrentSize() + sizeof(USize64));
201
202 output.AppendBlock(datasBlock);
203
204 // Mapa ID componente -> ID objeto.
205 for (UIndex64 compIdx = 0; compIdx < m_components.GetSize(); compIdx++) {
206 output.Write<GameObjectIndex::TUnderlyingType>(m_componentToObject.at(compIdx).Get());
207 }
208
