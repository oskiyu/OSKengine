166bool ConvexVolume::AreFacesEquivalent(UIndex32 firstFace, UIndex32 secondFace) const {
167 const Axis axis1 = GetLocalSpaceAxis(firstFace);
168 const Axis axis2 = GetLocalSpaceAxis(secondFace);
169
170 const bool isSameAxis = axis1 == axis2 || axis1 == -axis2;
171
172 bool hasCommonPoints = false;
173
174 for (const auto& indexA : m_faces[firstFace]) {
175 for (const auto& indexB : m_faces[secondFace]) {
176 if (indexA == indexB) {
177 hasCommonPoints = true;
178 break;
179 }
180 }
181
182 if (hasCommonPoints) {
183 break;
184 }
185 }
186
187 return isSameAxis && hasCommonPoints;

190void ConvexVolume::MergeVertices() {
191 if (m_vertices.IsEmpty())
192 return;
193
194 // Array con los vértices finales (sin duplicados).
195 DynamicArray<Vector3f> nVertices = DynamicArray<Vector3f>::CreateReserved(m_vertices.GetSize());
196
197 // Old vertex ID -> new vertex ID.
198 // Para poder actualizar los índices de
199 // las caras.
200 std::unordered_map<UIndex32, UIndex32> oldToNewId;
201
202 for (UIndex32 originalId = 0; originalId < m_vertices.GetSize(); originalId++) {
203 const Vector3f originalVertex = m_vertices[originalId];
204
205 // Sólamente añadimos el vértice al array si no estaba previamente.
206 // Ningún vértice se añade más de una vez.
207 bool containsVertex = false;
208 for (const auto& v : nVertices) {
209 if (v.Equals(originalVertex, 0.00001f)) {
210 containsVertex = true;
211 break;
212 }
213 }
214
215 if (!containsVertex) {
216 nVertices.Insert(originalVertex);
217 }
218
219 // Índice del vértice en el nuevo array.
220 UIndex32 newId = 0;
221 for (; newId < nVertices.GetSize(); newId++) {
222 if (nVertices[newId].Equals(originalVertex, 0.00001f)) {
223 break;
224 }
225 }
226
227 oldToNewId[originalId] = newId;
228 }
229
230 // Actualización de las caras.
231 for (auto& face : m_faces) {
232 for (auto& index : face) {
233 index = oldToNewId[index];
234 }
235 }
236
237 m_vertices = nVertices;
238 m_transformedVertices = nVertices;
239
240 RecalculateCenter();
241}

249void ConvexVolume::NormalizeFace(UIndex32 faceIndex) {
250
251 // Cara procesada
252 FaceIndices& face = m_faces[faceIndex];
253
254 // Vector normal de la cara.
255 const Vector3f faceNormal = GetLocalSpaceAxis(faceIndex);
256
257
258 // Nueva lista con los índices de los vértices.
259 DynamicArray<UIndex32> newIndices = {};
260
261
262 // Centro de la cara.
263 Vector3f center = Vector3f::Zero;
264
265 // Vértices que faltan por procesar.
266 std::set<UIndex32> remainingVertices = {};
267
268
269 // Calculamos centro y vértices por procesar.
270 for (UIndex32 i : face) {
271 center += m_vertices[i];
272 remainingVertices.insert(i);
273 }
274
275 center /= static_cast<float>(m_vertices.GetSize());
276
277
278 // ID del vértice que está siendo procesado.
279 UIndex32 currentVertexId = *remainingVertices.begin();
280
281 remainingVertices.erase(currentVertexId);
282
283 newIndices.Insert(currentVertexId);
284
285 Vector3f currentVertex = m_vertices[currentVertexId];
286
287 while (!remainingVertices.empty()) {
288 float minAngle = std::numeric_limits<float>::max();
289 UIndex32 nextVertexId = 0;
290
291 for (const UIndex32 i : remainingVertices) {
292 const Vector3f newVertex = m_vertices[i];
293
294 const Vector3f refToCenter = center - currentVertex;
295 const Vector3f newToCenter = center - newVertex;
296
297 // Get direction
298 const Vector3f direction = refToCenter.Cross(faceNormal).GetNormalized();
299 const float newDot = newToCenter.Dot(direction);
300
301 float angleDeg = refToCenter.GetAngle(newToCenter);
302
303 if (newDot < 0.0f)
304 angleDeg = (180.0f - angleDeg) + 180.0f;
305
306 if (angleDeg < minAngle) {
307 minAngle = angleDeg;
308 nextVertexId = i;
309 }
310 }
311
312 remainingVertices.erase(nextVertexId);
313 newIndices.Insert(nextVertexId);
314
315 currentVertex = m_vertices[nextVertexId];
316 }
317
318 face = newIndices;
319}

321void ConvexVolume::MergeFaces() {
322 if (m_faces.IsEmpty())
323 return;
324
325 MergeVertices();
326
327 // ID de las caras que deben eliminarse y que serán
328 // reemplazadas.
329 std::unordered_set<UIndex32> facesToDelete = {};
330
331 // Contiene los índices de cada nueva cara.
332 // Índices de caras.
333 DynamicArray<std::unordered_set<UIndex32>> newFaces = {};
334
335
336 for (UIndex32 faceA = 0; faceA < m_faces.GetSize() - 1; faceA++) {
337 if (facesToDelete.contains(faceA))
338 continue;
339
340 // IDs de las caras equivalentes a faceA.
341 std::unordered_set<UIndex32> equivalentFaces{};
342
343 // Por cada eje original, comprobamos si hay otros
344 // ejes equivalentes.
345 for (UIndex32 faceB = faceA + 1; faceB < m_faces.GetSize(); faceB++) {
346 if (facesToDelete.contains(faceB))
347 continue;
348
349 // Si el otro eje es igual, lo marcamos para no añadirlo
350 // a la lista optimizada.
351 if (AreFacesEquivalent(faceA, faceB)) {
352 equivalentFaces.insert(faceA);
353 equivalentFaces.insert(faceB);
354
355 facesToDelete.insert(faceA);
356 facesToDelete.insert(faceB);
357 }
358 }
359
360 if (!equivalentFaces.empty()) {
361 newFaces.Insert(equivalentFaces);
362 }
363 }
364
365 DynamicArray<FaceIndices> optimizedFaces{};
366
367 // Añadimos las caras que no han sido eliminadas.
368 for (UIndex32 i = 0; i < m_faces.GetSize(); i++) {
369 if (!facesToDelete.contains(i)) {
370 optimizedFaces.Insert(m_faces[i]);
371 }
372 }
373
374 // Por cada nueva cara...
375 for (const auto& nFace : newFaces) {
376 FaceIndices newFaceIndices = {};
377
378 // ...obtenemos todos sus índices.
379 for (const auto& faceIndex : nFace) {
380 newFaceIndices.InsertAll(m_faces[faceIndex]);
381 }
382
383 optimizedFaces.Insert(newFaceIndices);
384 }
385
386 m_faces = optimizedFaces;
387
388 m_axes.Empty();
389
390 for (UIndex64 i = 0; i < m_faces.GetSize(); i++) {
391 m_axes.Insert(GetWorldSpaceAxis(i));
392 }
393
394 NormalizeFaces();
395}

406ConvexVolume::Axis ConvexVolume::GetLocalSpaceAxis(UIndex32 faceId) const {
407 const FaceIndices& faceIndices = m_faces[faceId];
408
409 const Vector3f pointA = m_vertices.At(faceIndices.At(0));
410 const Vector3f pointB = m_vertices.At(faceIndices.At(1));
411 const Vector3f pointC = m_vertices.At(faceIndices.At(2));
412
413 const Vector3f vec1 = pointB - pointA;
414 const Vector3f vec2 = pointC - pointB;
415
416 const Vector3f cross = vec1.Cross(vec2).GetNormalized();
417
418 const float centerProjection = m_center.Dot(cross);
419 const float vecProjection = pointA.Dot(cross);
420
421 return cross * MATH::Sign(vecProjection - centerProjection);
422}

424ConvexVolume::Axis ConvexVolume::GetWorldSpaceAxis(UIndex32 faceId) const {
425 const FaceIndices& faceIndices = m_faces[faceId];
426
427 const Vector3f pointA = m_transformedVertices.At(faceIndices.At(0));
428 const Vector3f pointB = m_transformedVertices.At(faceIndices.At(1));
429 const Vector3f pointC = m_transformedVertices.At(faceIndices.At(2));
430
431 const Vector3f vec1 = pointB - pointA;
432 const Vector3f vec2 = pointC - pointB;
433
434 const Vector3f normal = vec1.Cross(vec2).GetNormalized();
435
436 const float centerProjection = m_transformedCenter.Dot(normal);
437 const float vecProjection = pointA.Dot(normal);
438
439 return normal * MATH::Sign(vecProjection - centerProjection);
440}

443UIndex32 ConvexVolume::GetMostParallelFaceIndex(const Vector3f& faceNormal, const DynamicArray<UIndex32>& candidates) const {
444 // Dot de dos vectores paralelos es 1.
445 float mostParallelDot = std::numeric_limits<float>::max();
446 UIndex32 output = 0;
447
448 for (UIndex32 faceId : candidates) {
449
450 const Vector3f nFaceNormal = GetWorldSpaceAxis(faceId);
451 const float dot = glm::abs(faceNormal.Dot(nFaceNormal));
452 const float diff = glm::abs(1.0f - dot);
453
454 if (diff < mostParallelDot) {
455 output = faceId;
456 mostParallelDot = dot;
457 }
458 }
459
460 return output;
461}

483DetailedCollisionInfo ConvexVolume::GetCollisionInfo(const IBottomLevelCollider& other,
484 const Transform3D& thisTransform, const Transform3D& otherTransform) const {
485
486 const auto& otherVolume = (const ConvexVolume&)other;
487
488#define GJK_POINTS
489#ifdef GJK_POINTS
490
491 const Simplex simplex = Simplex::GenerateFrom(*this, otherVolume);
492
493 if (!simplex.ContainsOrigin()) {
494 return DetailedCollisionInfo::False();
495 }
496
497 const auto minkowskiHull = MinkowskiHull::BuildFromVolumes(*this, otherVolume);
498
499 if (!minkowskiHull.SurpasesMinDepth()) {
500 return DetailedCollisionInfo::False();
501 }
502
503 Vector3f mtv = minkowskiHull.GetMtv();
504
505 const UIndex64 firstFaceId = GetFaceWithVertexIndices(minkowskiHull.GetFirstFaceVerticesIds(), minkowskiHull.GetMtv());
506 const UIndex64 secondFaceId = otherVolume.GetFaceWithVertexIndices(minkowskiHull.GetSecondFaceVerticesIds(), -minkowskiHull.GetMtv());
507
508#ifdef OSK_COLLISION_DEBUG
509 Engine::GetLogger()->DebugLog(std::format("First vertices count: {}", m_transformedVertices.GetSize()));
510 Engine::GetLogger()->DebugLog(std::format("Second vertices count: {}", otherVolume.m_transformedVertices.GetSize()));
511
512 Engine::GetLogger()->DebugLog(std::format("MTV: {:.3f} {:.3f} {:.3f}", mtv.x, mtv.y, mtv.z));
513 Engine::GetLogger()->DebugLog(std::format("\tOriginal mtv depth: {:.3f}", expandingPolytope.GetMtv().GetLenght()));
514#endif
515
516 const FaceIndices& faceA = m_faces[firstFaceId];
517 const FaceIndices& faceB = otherVolume.m_faces[secondFaceId];
518
519 DynamicArray<Vector3f> verticesA = DynamicArray<Vector3f>::CreateReserved(faceA.GetSize());
520 for (const auto index : faceA)
521 verticesA.Insert(m_transformedVertices[index]);
522
523 DynamicArray<Vector3f> verticesB = DynamicArray<Vector3f>::CreateReserved(faceB.GetSize());
524 for (const auto index : faceB)
525 verticesB.Insert(otherVolume.m_transformedVertices[index]);
526
527 // Get most incident face:
528 MinkowskiHull::Volume incident = MinkowskiHull::Volume::A;
529 {
530 const float firstDot = m_axes[firstFaceId].Dot(-mtv);
531 const float secondDot = otherVolume.m_axes[secondFaceId].Dot(mtv);
532
533 incident = firstDot < secondDot ? MinkowskiHull::Volume::A : MinkowskiHull::Volume::B;
534 }
535
536 const Vector3f normal = minkowskiHull.GetNormalizedMtv();
537 const Vector3f otherNormal = otherVolume.GetWorldSpaceAxis(secondFaceId);
538
539#ifdef OSK_COLLISION_DEBUG
540 Engine::GetLogger()->DebugLog(std::format("First normal: {:.3f} {:.3f} {:.3f}", normal.x, normal.y, normal.z));
541 Engine::GetLogger()->DebugLog(std::format("Second normal: {:.3f} {:.3f} {:.3f}", otherNormal.x, otherNormal.y, otherNormal.z));
542#endif
543
544 const DynamicArray<Vector3f> finalContactPoints = incident == MinkowskiHull::Volume::A
545 ? ClipFaces(verticesA, verticesB, normal, otherNormal)
546 : ClipFaces(verticesB, verticesA, otherNormal, normal);
547
548
549 return DetailedCollisionInfo::True(
550 minkowskiHull.GetMtv(),
551 finalContactPoints,
552 minkowskiHull.GetNormalizedMtv(),
553 otherNormal,
554 DetailedCollisionInfo::MtvDirection::A_TO_B
555 );
556
557#else
558 const auto mtvInfo = SatCollision(*this, other);
559
560 if (!mtvInfo.overlaps) {
561 return DetailedCollisionInfo::False();
562 }
563
564 const ConvexVolume& primaryCollider = mtvInfo.face.collider == SatInfo::Face::Collider::A
565 ? *this
566 : otherVolume;
567
568 const ConvexVolume& secondaryCollider = mtvInfo.face.collider == SatInfo::Face::Collider::A
569 ? otherVolume
570 : *this;
571
572
573 // MTV normalizado, expresado en la dirección Referencia -> Incidente.
574 const Vector3f normalizedDirectedMtv = mtvInfo.mtv.GetNormalized();
575
576
577 // Debemos obtener el punto del collider incidente que está
578 // más hacia el collider de referencia.
579 const UIndex32 furthestIncidentPointIndex = static_cast<UIndex32>(secondaryCollider.GetSupport(normalizedDirectedMtv).originalVertexId);
580
581
582 // Índices de las caras del segundo collider que contienen
583 // el vértice más hacia la referencia.
584 const DynamicArray<UIndex32> secondaryFacesIndices
585 = secondaryCollider.GetFaceIndicesWithVertex(furthestIncidentPointIndex);
586
587
588 // Obtenemos la cara del collider incidente que está más paralela a
589 // la cara del MTV.
590 const UIndex32 incidentFaceIndex = secondaryCollider.GetMostParallelFaceIndex(normalizedDirectedMtv, secondaryFacesIndices);
591 const Vector3f secondFaceNormal = secondaryCollider.GetWorldSpaceAxis(incidentFaceIndex);
592
593 DynamicArray<Vector3f> referencePoints = DynamicArray<Vector3f>::CreateReservedArray(primaryCollider.m_faces[mtvInfo.face.index].GetSize());
594 for (const UIndex32 index : primaryCollider.m_faces[mtvInfo.face.index])
595 referencePoints.Insert(primaryCollider.m_transformedVertices[index]);
596
597 DynamicArray<Vector3f> secondaryPoints = DynamicArray<Vector3f>::CreateReservedArray(secondaryCollider.m_faces[incidentFaceIndex].GetSize());
598 for (const UIndex32 index : secondaryCollider.m_faces[incidentFaceIndex])
599 referencePoints.Insert(secondaryCollider.m_transformedVertices[index]);
600
601 const DynamicArray<Vector3f> finalContactPoints = ClipFaces(referencePoints, secondaryPoints, normalizedDirectedMtv, secondFaceNormal);
602
603 const Vector3f finalMtv = mtvInfo.mtv * (mtvInfo.face.collider == SatInfo::Face::Collider::A ? 1.0f : -1.0f);
604 const Vector3f aToB = mtvInfo.face.collider == SatInfo::Face::Collider::A ? mtvInfo.mtv : secondFaceNormal;
605 const Vector3f bToA = mtvInfo.face.collider == SatInfo::Face::Collider::A ? secondFaceNormal : mtvInfo.mtv;
606
607 return DetailedCollisionInfo::True(
608 finalMtv,
609 finalContactPoints,
610 aToB,
611 bToA,
612 DetailedCollisionInfo::MtvDirection::A_TO_B);
613
614#endif
615}

637GjkSupport ConvexVolume::GetSupport(const Vector3f& direction) const {
638 GjkSupport output{};
639
640 double outputDistance = std::numeric_limits<double>::lowest();
641
642 for (UIndex64 v = 0; v < m_transformedVertices.GetSize(); v++) {
643 const Vector3f vertex = m_transformedVertices[v];
644 const float distance = vertex.Dot(direction);
645
646 if (distance > outputDistance) {
647 outputDistance = distance;
648
649 output.point = vertex;
650 output.originalVertexId = v;
651 }
652 }
653
654 return output;
655}

712UIndex64 ConvexVolume::GetFaceWithVertexIndices(std::span<const UIndex64, 3> indices, const Vector3f& refNormal) const {
713 USize64 bestMatchVertexCount = 0;
714
715 DynamicArray<UIndex64> bestMatches;
716
717 for (UIndex64 faceId = 0; faceId < m_faces.GetSize(); faceId++) {
718 const auto& face = m_faces[faceId];
719
720 const USize64 count =
721 static_cast<USize64>(face.ContainsElement(static_cast<UIndex32>(indices[0]))) +
722 static_cast<USize64>(face.ContainsElement(static_cast<UIndex32>(indices[1]))) +
723 static_cast<USize64>(face.ContainsElement(static_cast<UIndex32>(indices[2])));
724
725 if (count == 3) {
726 return faceId;
727 }
728 else if (count > bestMatchVertexCount) {
729 bestMatches = { faceId };
730 bestMatchVertexCount = count;
731 }
732 else if (count == bestMatchVertexCount) {
733 bestMatches.Insert(faceId);
734 }
735 }
736
737 UIndex64 bestMatchId = 0;
738 float bestDot = std::numeric_limits<float>::lowest();
739
740 for (UIndex64 i = 0; i < bestMatches.GetSize(); i++) {
741 const Vector3f normal = m_axes[bestMatches[i]];
742 const float dot = normal.Dot(refNormal);
743
744 if (dot > bestDot) {
745 bestDot = dot;
746 bestMatchId = i;
747 }
748 }
749
750 return bestMatches[bestMatchId];
751}

753RayCastResult ConvexVolume::CastRay(const Ray& ray) const {
754 float closestIntersection = std::numeric_limits<float>::max();
755 Vector3f intersectionPoint = Vector3f::Zero;
756 bool intersection = false;
757
758 for (UIndex64 i = 0; i < m_faces.GetSize(); i++) {
759 const auto& axis = m_axes[i];
760 const auto& face = m_faces[i];
761
762 const auto planeIntersectionResult = ray.IntersectionWithPlane(axis, m_transformedVertices[face[0]]);
763
764 if (!planeIntersectionResult.Result()) {
765 continue;
766 }
767
768
769 // Comprobamos si está dentro de la cara
770 bool insideFace = true;
771 for (UIndex64 v = 0; v < face.GetSize(); v++) {
772 const Vector3f pointA = m_transformedVertices[face[v]];
773 const Vector3f pointB = m_transformedVertices[face[(v + 1) % face.GetSize()]];
774
775 const Vector3f edge = pointB - pointA;
776
777 if (axis.Dot(edge.Cross(planeIntersectionResult.GetIntersectionPoint() - pointB)) > 0.0f) {
778 insideFace = false;
779 break;
780 }
781 }
782
783 if (insideFace) {
784 intersection = true;
785
786 const float nDistance = planeIntersectionResult.GetIntersectionPoint().GetDistanceTo2(ray.origin);
787 if (nDistance < closestIntersection) {
788 closestIntersection = nDistance;
789 intersectionPoint = planeIntersectionResult.GetIntersectionPoint();
790 }
791 }
792 }
793
794 return intersection
795 ? RayCastResult::True(intersectionPoint, ECS::EMPTY_GAME_OBJECT)
796 : RayCastResult::False();
797}
