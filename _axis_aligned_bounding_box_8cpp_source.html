60RayCastResult AxisAlignedBoundingBox::CastRay(const Ray& ray) const {
61 const std::array<Vector3f, 6> normals = {
62 Vector3f(1.0f, 0.0f, 0.0f),
63 Vector3f(-1.0f, 0.0f, 0.0f),
64 Vector3f(0.0f, 1.0f, 0.0f),
65 Vector3f(0.0f, -1.0f, 0.0f),
66 Vector3f(0.0f, 0.0f, 1.0f),
67 Vector3f(0.0f, 0.0f, -1.0f)
68 };
69
70 Vector3f position = Vector3f::Zero;
71 if (GetBroadOwner().has_value()) {
72 position = GetBroadOwner().value()->GetPosition();
73 }
74 else {
75 position = GetNarrowOwner().value()->GetTransform().GetPosition();
76 }
77
78 const std::array<Vector3f, 6> points = {
79 position + normals[0] * m_size * 0.5f,
80 position + normals[1] * m_size * 0.5f,
81 position + normals[2] * m_size * 0.5f,
82 position + normals[3] * m_size * 0.5f,
83 position + normals[4] * m_size * 0.5f,
84 position + normals[5] * m_size * 0.5f
85 };
86
87 float closestDistance = std::numeric_limits<float>::max();
88 Vector3f closestIntersection = Vector3f::Zero;
89 bool intersection = false;
90 for (UIndex64 i = 0; i < 6; i++) {
91 const auto result = ray.IntersectionWithPlane(normals[i], points[i]);
92
93 if (!result.Result())
94 continue;
95
96 intersection = true;
97
98 const float nDistance = result.GetIntersectionPoint().GetDistanceTo2(ray.origin);
99 if (nDistance < closestDistance) {
100 closestDistance = nDistance;
101 closestIntersection = result.GetIntersectionPoint();
102 }
103 }
104
105 return intersection
106 ? RayCastResult::True(closestIntersection, ECS::EMPTY_GAME_OBJECT)
107 : RayCastResult::False();
108}
