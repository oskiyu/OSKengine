39RayCastResult AxisAlignedBoundingBox::CastRay(const Ray& ray) const {
40 const std::array<Vector3f, 6> normals = {
41 Vector3f(1.0f, 0.0f, 0.0f),
42 Vector3f(-1.0f, 0.0f, 0.0f),
43 Vector3f(0.0f, 1.0f, 0.0f),
44 Vector3f(0.0f, -1.0f, 0.0f),
45 Vector3f(0.0f, 0.0f, 1.0f),
46 Vector3f(0.0f, 0.0f, -1.0f)
47 };
48
49 const std::array<Vector3f, 6> points = {
50 GetPosition() + normals[0] * m_size * 0.5f,
51 GetPosition() + normals[1] * m_size * 0.5f,
52 GetPosition() + normals[2] * m_size * 0.5f,
53 GetPosition() + normals[3] * m_size * 0.5f,
54 GetPosition() + normals[4] * m_size * 0.5f,
55 GetPosition() + normals[5] * m_size * 0.5f
56 };
57
58 float closestDistance = std::numeric_limits<float>::max();
59 Vector3f closestIntersection = Vector3f::Zero;
60 bool intersection = false;
61 for (UIndex64 i = 0; i < 6; i++) {
62 const auto result = ray.IntersectionWithPlane(normals[i], points[i]);
63
64 if (!result.Result())
65 continue;
66
67 intersection = true;
68
69 const float nDistance = result.GetIntersectionPoint().GetDistanceTo2(ray.origin);
70 if (nDistance < closestDistance) {
71 closestDistance = nDistance;
72 closestIntersection = result.GetIntersectionPoint();
73 }
74 }
75
76 return intersection
77 ? RayCastResult::True(closestIntersection, ECS::EMPTY_GAME_OBJECT)
78 : RayCastResult::False();
79}

81bool AxisAlignedBoundingBox::IsBehindPlane(Plane plane) const {
82 const Vector3f extent = m_size * 0.5f;
83
84 const float collapsedAabb =
85 extent.x * glm::abs(plane.normal.x) +
86 extent.y * glm::abs(plane.normal.y) +
87 extent.z * glm::abs(plane.normal.z);
88
89 const float signedDistanceToPlane = plane.normal.Dot(GetPosition() - plane.point);
90
91 return !(glm::abs(signedDistanceToPlane) <= collapsedAabb);
92}
