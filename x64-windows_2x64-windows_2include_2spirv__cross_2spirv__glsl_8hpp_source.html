443 template <typename... Ts>
444 inline void statement(Ts &&... ts)
445 {
446 if (is_forcing_recompilation())
447 {
448 // Do not bother emitting code while force_recompile is active.
449 // We will compile again.
450 statement_count++;
451 return;
452 }
453
454 if (redirect_statement)
455 {
456 redirect_statement->push_back(join(std::forward<Ts>(ts)...));
457 statement_count++;
458 }
459 else
460 {
461 for (uint32_t i = 0; i < indent; i++)
462 buffer << " ";
463 statement_inner(std::forward<Ts>(ts)...);
464 buffer << '\n';
465 }
466 }
467
468 template <typename... Ts>
469 inline void statement_no_indent(Ts &&... ts)
470 {
471 auto old_indent = indent;
472 indent = 0;
473 statement(std::forward<Ts>(ts)...);
474 indent = old_indent;
475 }
