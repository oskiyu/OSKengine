87void PcUserInput::UpdateMouseState(MouseState* state) {
88 *state = mouseCallbackState;

90 GLFWwindow* window = Engine::GetDisplay()->As<Window>()->_GetGlfw();

92 for (int i = 0; i < 8; i++)

96 state->_SetButtonState(MouseButton::BUTTON_RIGHT, state->GetButtonState(MouseButton::BUTTON_2));

100void PcUserInput::UpdateGamepadState(GamepadState* state) {
101 const auto id = GLFW_JOYSTICK_1 + state->GetIdentifier();
102
103 if (glfwJoystickPresent(id)) {
104 int count;
105 const float* axes = glfwGetJoystickAxes(id, &count);
106
107 state->_SetConnectionState(true);
108
109 for (int axis = 0; axis < 6; axis++)
110 state->_SetAxisState((GamepadAxis)axis, axes[axis]);
111
112 const unsigned char* buttons = glfwGetJoystickButtons(id, &count);
113 for (int button = 0; button < 4; button++)
114 state->_SetButtonState((GamepadButton)button, (bool)buttons[button] ? GamepadButtonState::PRESSED : GamepadButtonState::RELEASED);
115 }
116 else {
117 state->_SetConnectionState(false);
118 }
119}
