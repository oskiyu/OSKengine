166void AnimMeshLoader::LoadAnimations(const tinygltf::Model& model) {
167 for (UIndex32 animationId = 0; animationId < model.animations.size(); animationId++) {
168 tinygltf::Animation gltfAnimation = model.animations[animationId];
169 Animation animation{};
170
171 animation.name = gltfAnimation.name;
172 animation.samplers.Resize(gltfAnimation.samplers.size());
173 animation.channels.Resize(gltfAnimation.channels.size());
174
175 // Samplers
176 for (UIndex32 samplerId = 0; samplerId < gltfAnimation.samplers.size(); samplerId++) {
177 tinygltf::AnimationSampler gltfSampler = gltfAnimation.samplers[samplerId];
178
179 AnimationSampler& sampler = animation.samplers[samplerId];
180 // TODO: sampler interpolation type
181
182 // Establece los valores de entrada del sampler, es decir,
183 // los timestamps en el que se produce una transforamción de algún tipo.
184 {
185 const tinygltf::Accessor& accessor = model.accessors[gltfSampler.input];
186 const tinygltf::BufferView& bufferView = model.bufferViews[accessor.bufferView];
187 const tinygltf::Buffer& buffer = model.buffers[bufferView.buffer];
188
189 const void* data = &buffer.data[accessor.byteOffset + bufferView.byteOffset];
190 const float* timestampsBuffer = static_cast<const float*>(data);
191
192 for (UIndex32 i = 0; i < accessor.count; i++)
193 sampler.timestamps.Insert(timestampsBuffer[i]);
194
195 // Timestamps de inicio y fin de la animación.
196 for (const TDeltaTime timestamp : sampler.timestamps) {
197 animation.startTime = glm::min(animation.startTime, timestamp);
198 animation.endTime = glm::max(animation.endTime, timestamp);
199 }
200 }
201
202 // Establece los valores de salida del sampler, es decir,
203 // la transformación de cada timestamp.
204 {
205 const tinygltf::Accessor& accessor = model.accessors[gltfSampler.output];
206 const tinygltf::BufferView& bufferView = model.bufferViews[accessor.bufferView];
207 const tinygltf::Buffer& buffer = model.buffers[bufferView.buffer];
208 const void* data = &buffer.data[accessor.byteOffset + bufferView.byteOffset];
209
210 switch (accessor.type) {
211
212 case TINYGLTF_TYPE_VEC3: {
213 const glm::vec3* vecBuffer = static_cast<const glm::vec3*>(data);
214 for (UIndex32 i = 0; i < accessor.count; i++)
215 sampler.outputs.Insert(glm::vec4(vecBuffer[i], 0.0));
216 }
217 break;
218
219 case TINYGLTF_TYPE_VEC4: {
220 const glm::vec4* vecBuffer = static_cast<const glm::vec4*>(data);
221 for (UIndex32 i = 0; i < accessor.count; i++)
222 sampler.outputs.Insert(vecBuffer[i]);
223 }
224 break;
225
226 }
227 }
228 }
229
230 // Channels
231 for (UIndex32 channelId = 0; channelId < gltfAnimation.channels.size(); channelId++) {
232 tinygltf::AnimationChannel gltfChannel = gltfAnimation.channels[channelId];
233
234 AnimationChannel& channel = animation.channels[channelId];
235 channel.samplerIndex = channelId;
236 channel.nodeId = gltfChannel.target_node;
237
238 if (gltfChannel.target_path == "translation") {
239 channel.type = AnimationChannel::ChannelType::TRANSLATION;
240 }
241 else if (gltfChannel.target_path == "rotation") {
242 channel.type = AnimationChannel::ChannelType::ROTATION;
243 }
244 else if (gltfChannel.target_path == "scale") {
245 channel.type = AnimationChannel::ChannelType::SCALE;
246 }
247 else {
248 // OSK_CHECK(false, "El canal de animación " + gltfChannel.target_path + " no está soportado.");
249 }
250 }

252 // Engine::GetLogger()->InfoLog("Animación cargada: " + animation.name);

258void AnimMeshLoader::LoadSkins(const tinygltf::Model& model) {
259 for (UIndex32 skinId = 0; skinId < model.skins.size(); skinId++) {
260 const tinygltf::Skin& gltfSkin = model.skins[skinId];
261
262 AnimationSkin skin;
263 skin.name = gltfSkin.name;
264 skin.thisIndex = skinId;
265
266 if (gltfSkin.skeleton > -1)
267 skin.rootIndex = static_cast<UIndex32>(gltfSkin.skeleton);
268
269 for (const auto bone : gltfSkin.joints)
270 skin.bonesIds.Insert(bone);
271
272 if (gltfSkin.inverseBindMatrices > -1) {
273 const tinygltf::Accessor& accessor = model.accessors[gltfSkin.inverseBindMatrices];
274 const tinygltf::BufferView& bufferView = model.bufferViews[accessor.bufferView];
275 const tinygltf::Buffer& buffer = model.buffers[bufferView.buffer];
276
277 const UIndex32 matricesSize = static_cast<UIndex32>(accessor.count * sizeof(glm::mat4));
278
279 skin.inverseMatrices.Resize(accessor.count);
280 memcpy(skin.inverseMatrices.GetData(), &buffer.data[accessor.byteOffset + bufferView.byteOffset], matricesSize);
281 }
282
283 // Engine::GetLogger()->InfoLog("Skin cargada: " + skin.name + ", root: " + std::to_string(skin.rootIndex));
284 tempAnimator._AddSkin(std::move(skin));
285 if (skinId == 0)
286 tempAnimator.SetActiveSkin(gltfSkin.name);
287 }
