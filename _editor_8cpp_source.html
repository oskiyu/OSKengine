41void OSK::Editor::Editor::UpdateByObject(EntityComponentSystem* ecs) {
42 // Si el objeto seleccionado no está vivo,
43 // se vacía.
44 if (!ecs->IsGameObjectAlive(m_selectedObject)) {
45 m_selectedObject = ECS::GameObjectIndex::CreateEmpty();
46 m_isSelectedObjectDirty = true;
47 }
48
49 // Si el objeto seleccionado ha cambiado,
50 // se eliminan los controladores y las
51 // vistas relacionadas.
52 if (m_isSelectedObjectDirty) {
53 m_controllers.clear();
54 m_views.clear();
55
56 m_isSelectedObjectDirty = false;
57
58 m_editorUi->ClearAllComponentViews();
59 }
60
61 // Si no hay objeto, no se actualiza.
62 if (m_selectedObject.IsEmpty()) {
63 m_editorUi->ClearAllComponentViews();
64
65 return;
66 }
67
68 // Actualizar cada controlador de cada
69 // tipo de componente poseído.
70 for (const ComponentType type : ecs->GetObjectComponentsTypes(m_selectedObject)) {
71 auto controllerViewPairOpt = GetOrCreateControllerViewPair(ecs, type);
72
73 if (!controllerViewPairOpt.has_value()) {
74 continue;
75 }
76
77 auto& controllerViewPair = controllerViewPairOpt.value();
78
79 if (controllerViewPair.isNewInsertion) {
80 m_editorUi->AddComponentView(controllerViewPair.view);
81 }
82
83 // Actualizar la dirección de los
84 // datos del componente.
85 controllerViewPair.controller->Update(ecs->GetComponentAddress(m_selectedObject, type));
86
87 // Proceso de actualización de:
88 // - Componente (aplicando cambios en la interfaz).
89 // - Interfaz (aplicando cambios en el componente).
90 controllerViewPair.controller->Poll();

100std::optional<OSK::Editor::Editor::ControllerViewPair> OSK::Editor::Editor::GetOrCreateControllerViewPair(EntityComponentSystem* ecs, ECS::ComponentType type) {
101 auto iterator = m_controllers.find(type);
102
103 // Si no se encuentra...
104 if (iterator == m_controllers.end()) {
105 // ...crearlos.
106
107 const std::string componentName = ecs->GetComponentTypeName(type);
108
109 auto viewIterator = m_componentViewsFactoryMethods.find(componentName);
110 if (viewIterator == m_componentViewsFactoryMethods.end()) {
111 return std::optional<ControllerViewPair>{};
112 }
113
114 auto factoryIterator = m_componentControllersFactoryMethods.find(componentName);
115 if (factoryIterator == m_componentControllersFactoryMethods.end()) {
116 return std::optional<ControllerViewPair>{};
117 }
118
119 auto* view = viewIterator->second.operator()(Vector2f::One * 12.0f).GetPointer();
120 m_views.emplace(type, view);
121
122 const auto& factoryMethod = factoryIterator->second;
123
124 auto controller = factoryMethod(
125 m_selectedObject,
126 ecs->GetComponentAddress(m_selectedObject, type),
127 view).GetPointer();
128
129 m_controllers[type] = controller;
130
131 return ControllerViewPair{
132 .controller = controller,
133 .view = view,
134 .isNewInsertion = true
135 };
136 }
137
138 return ControllerViewPair{
139 .controller = iterator->second.GetPointer(),
140 .view = m_views.find(type)->second,
141 .isNewInsertion = false
142 };
143}

173 if (systemName.empty()) {

190 m_currentSystemController->_GetSystemName() != systemName;
191
192 if (!shouldCreateController) {
193 return;
194 }
195
196 if (!m_systemControllerFactoryMethods.contains(systemName)) {
197 return;
198 }
199

202 m_ecs->GetSystemByName(systemName),
