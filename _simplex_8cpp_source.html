76void Simplex::UpdateLine() {
77 m_containsOrigin = false;
78
79 // Sólamente tenemos una línea:
80 //
81 // - No puede estar en la dirección del vértice antiguo, ya que
82 // para encontrar el origen de coordenadas debimos buscar en la
83 // dirección contraria.
84 //
85 // - No puede estar más allá del nuevo vértice, ya que en cuyo
86 // caso no habría colisión.
87 //
88 // Por esto, simplemente buscamos un vector perpendicular a la
89 // línea, en dirección al origen de coordenadas.
