84void IRenderer::UploadLayeredImageToGpu(GpuImage* destination, const TByte* data, USize64 numBytes, USize32 numLayers, ICommandList* cmdList) {
85 OSK_ASSERT(numLayers > 0, InvalidArgumentException("El n√∫mero de capas debe ser > 0."));
86 OSK_ASSERT(numBytes > 0, InvalidArgumentException("El n√∫mero de bytes debe ser > 0."));
87
88 USize64 gpuNumBytes = destination->GetPhysicalNumberOfBytes() * numLayers;
89
90 const TByte* uploadableData = data;
91
92 auto stagingBuffer = GetAllocator()->CreateStagingBuffer(gpuNumBytes, cmdList->GetCompatibleQueueType());
93 stagingBuffer->MapMemory();
94 stagingBuffer->Write(uploadableData, gpuNumBytes);
95 stagingBuffer->Unmap();
96
97 USize64 offsetPerIteration = gpuNumBytes / numLayers;
98 for (UIndex32 i = 0; i < numLayers; i++)
99 cmdList->CopyBufferToImage(stagingBuffer.GetValue(), destination, i, offsetPerIteration * i);
100
101 cmdList->RegisterStagingBuffer(stagingBuffer);
102}

128OwnedPtr<ICommandList> IRenderer::CreateSingleUseCommandList(GpuQueueType queueType, std::thread::id threadId) {
129 ICommandPool* creator = nullptr;
130
131 switch (queueType) {
132 case OSK::GRAPHICS::GpuQueueType::MAIN:
133 creator = UseUnifiedCommandPool()
134 ? GetUnifiedCommandPool(threadId)
135 : GetGraphicsComputeCommandPool(threadId);
136 break;
137
138 case OSK::GRAPHICS::GpuQueueType::PRESENTATION:
139 // No hay lista de comandos de PRESENTATION.
140 OSK_ASSERT(false, InvalidArgumentException("No se puede crear una lista de comandos en la cola GpuQueueType::PRESENTATION."));
141 break;
142
143 case OSK::GRAPHICS::GpuQueueType::ASYNC_TRANSFER:
144 if (HasTransferOnlyCommandPool()) {
145 creator = GetTransferOnlyCommandPool(threadId);
146 }
147 else {
148 creator = UseUnifiedCommandPool()
149 ? GetUnifiedCommandPool(threadId)
150 : GetGraphicsComputeCommandPool(threadId);
151 }
152 break;
153
154 default:
155 break;
156 }
157
158 return creator->CreateSingleTimeCommandList(*GetGpu());
159}
