54void IRenderer::UploadLayeredImageToGpu(GpuImage* destination, const TByte* data, USize64 numBytes, USize32 numLayers, ICommandList* cmdList) {
55 OSK_ASSERT(numLayers > 0, InvalidArgumentException("El número de capas debe ser > 0."));
56 OSK_ASSERT(numBytes > 0, InvalidArgumentException("El número de bytes debe ser > 0."));
57
58 USize64 gpuNumBytes = destination->GetPhysicalNumberOfBytes() * numLayers;
59
60 const TByte* uploadableData = data;
61
62 auto stagingBuffer = GetAllocator()->CreateStagingBuffer(gpuNumBytes);
63 stagingBuffer->MapMemory();
64 stagingBuffer->Write(uploadableData, gpuNumBytes);
65 stagingBuffer->Unmap();
66
67 USize64 offsetPerIteration = gpuNumBytes / numLayers;
68 for (UIndex32 i = 0; i < numLayers; i++)
69 cmdList->CopyBufferToImage(stagingBuffer.GetValue(), destination, i, offsetPerIteration * i);
70
71 cmdList->RegisterStagingBuffer(stagingBuffer);
72}
