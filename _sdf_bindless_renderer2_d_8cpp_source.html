118void OSK::GRAPHICS::SdfBindlessRenderer2D::End() {
119 if (m_batches.IsEmpty()) {
120 return;
121 }
122
123 const auto currentResourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
124
125 // Flushes
126 for (auto& mInstance : m_drawCallsBuffersInstances[currentResourceIndex]) {
127 mInstance->GetSlot(DrawCallsSlotName)->FlushUpdate();
128 }
129
130 // Renderizado final
131 GetCurrentCommandList()->BindVertexBuffer(*Sprite::globalVertexBuffer);
132 GetCurrentCommandList()->BindIndexBuffer(*Sprite::globalIndexBuffer);
133
134 GetCurrentCommandList()->BindMaterial(*m_batches[0].material);
135 GetCurrentCommandList()->BindMaterialSlot(*GetGlobalInformationSlot(currentResourceIndex));
136
137 const Material* previousMaterial = m_batches[0].material;
138
139 for (UIndex64 i = 0; i < m_batches.GetSize(); i++) {
140 const SdfBindlessBatch2D& batch = m_batches[i];
141
142 if (batch.material != previousMaterial) {
143 GetCurrentCommandList()->BindMaterial(*batch.material);
144 previousMaterial = batch.material;
145 GetCurrentCommandList()->BindMaterialSlot(*GetGlobalInformationSlot(currentResourceIndex));
146 }
147
148 const IMaterialSlot* instanceSlot = m_drawCallsBuffersInstances[currentResourceIndex][i]->GetSlot(DrawCallsSlotName);
149
150 GetCurrentCommandList()->BindMaterialSlot(*instanceSlot);
151
152 GetCurrentCommandList()->DrawInstances(0, 6, 0, static_cast<USize32>(batch.drawCalls.GetSize()));
153 }
154
155 for (auto& buffer : m_drawCallsBuffers[currentResourceIndex]) {
156 buffer->ResetCursor();
157 }
