50void IShaderPass::SetupMaterialInstance(const GpuModel3D& model, const GpuMesh3D& mesh) {
51 m_localMeshMapping.RegisterModel(model.GetUuid());
52
53 auto& modelData = m_localMeshMapping.GetModelData(model.GetUuid());
54 modelData.RegisterMesh(mesh.GetUuid());
55
56 auto& meshData = modelData.GetMeshData(mesh.GetUuid());
57
58 meshData._SetMaterialInstance(m_passMaterial->CreateInstance());
59 meshData._SetMaterialBuffer(Engine::GetRenderer()->GetAllocator()->CreateBuffer(
60 sizeof(PbrMaterialInfo),
61 GPU_MEMORY_NO_ALIGNMENT,
62 GpuBufferUsage::UNIFORM_BUFFER,
63 GpuSharedMemoryType::GPU_AND_CPU,
64 GpuQueueType::MAIN));
65
66 // Material Instance
67 const GpuImageViewConfig view = GpuImageViewConfig::CreateSampled_Default();
68
69 const auto defaultNormalTextureView = Engine::GetRenderer()->GetAllocator()->GetDefaultNormalTexture()->GetView(view);
70 const auto defaultTexture = Engine::GetAssetManager()->Load<Texture>("Resources/Assets/Textures/texture0.json");
71 const auto defaultTextureView = defaultTexture->GetGpuImage()->GetView(view);
72
73 auto* mInstance = meshData.GetMaterialInstance();
74
75 const auto& material = model.GetMaterial(mesh.GetMaterialIndex());
76
77 auto* textureLoader = Engine::GetAssetManager()->GetLoader<TextureLoader>();
78 const auto& textureTable = model.GetTextureTable();
79
80 const auto* colorTexture = material.colorTextureIndex.has_value()
81 ? textureTable.GetImageView(material.colorTextureIndex.value())
82 : defaultTextureView;
83 const auto* normalTexture = material.normalTextureIndex.has_value()
84 ? textureTable.GetImageView(material.normalTextureIndex.value())
85 : defaultNormalTextureView;
86
87 auto* materialBuffer = meshData.GetMaterialBuffer();
88
89 mInstance->GetSlot("texture")->SetGpuImage("albedoTexture", *colorTexture, GpuImageSamplerDesc::CreateDefault_WithMipMap(colorTexture->GetImage()));
90 mInstance->GetSlot("texture")->SetGpuImage("normalTexture", *normalTexture, GpuImageSamplerDesc::CreateDefault_WithMipMap(normalTexture->GetImage()));
91 mInstance->GetSlot("texture")->SetUniformBuffer("materialInfo", *materialBuffer);
92 mInstance->GetSlot("texture")->FlushUpdate();
93
94 PbrMaterialInfo materialInfo{};
95 materialInfo.emissiveColor = material.emissiveColor;
96 materialInfo.roughnessMetallic = { material.roughnessFactor, material.metallicFactor };
97 materialInfo.hasNormalTexture = material.normalTextureIndex.has_value();
98
99 materialBuffer->GetMemorySubblock()->MapMemory();
100 materialBuffer->GetMemorySubblock()->Write(&materialInfo, sizeof(materialInfo));
101 materialBuffer->GetMemorySubblock()->Unmap();
102}
