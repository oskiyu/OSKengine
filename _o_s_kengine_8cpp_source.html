90void Engine::Create(GRAPHICS::RenderApiType type) {
91 logger = new IO::Logger;
92 logger->Start("LOG_LAST.txt");
93
94 display = new IO::Window;
95 entityComponentSystem = new ECS::EntityComponentSystem(logger.GetPointer());
96 input = new IO::PcUserInput;
97 inputManager = new IO::InputManager;
98 audioApi = new AUDIO::AudioApiAl;
99 audioApi->Initialize();
100 uuidProvider = new UuidProvider;
101 m_jobSystem = new JobSystem;
102 commandExecutor = new ConsoleCommandExecutor;
103
104 logger->InfoLog("Iniciando OSKengine.");
105 logger->InfoLog(std::format("\tVersion: {}.{}.{}",
106 GetVersion().mayor,
107 GetVersion().menor,
108 GetVersion().parche));
109 logger->InfoLog(std::format("\tBuild {}.",GetBuild()));
110
111 const nlohmann::json engineConfig = nlohmann::json::parse(OSK::IO::FileIO::ReadFromFile("engine_config.json"));
112 const bool requestRayTracing = engineConfig.contains("use_rt") && engineConfig["use_rt"] == 1;
113
114 switch (type) {
115
116 case OSK::GRAPHICS::RenderApiType::VULKAN:
117 renderer = new GRAPHICS::RendererVk(requestRayTracing);
118
119 break;
120
121 case OSK::GRAPHICS::RenderApiType::DX12:
122 renderer = new GRAPHICS::RendererDx12(requestRayTracing);
123
124 break;
125 }
126
127 assetManager = new ASSETS::AssetManager();
128}

171void Engine::RegisterBuiltinSystems() {
172#ifdef OSK_USE_FORWARD_RENDERER
173 entityComponentSystem->RegisterSystem<ECS::RenderSystem3D>(ECS::ISystem::DEFAULT_EXECUTION_ORDER);
174#elif defined(OSK_USE_DEFERRED_RENDERER)
175 entityComponentSystem->RegisterSystem<ECS::TreeNormalsRenderSystem>(ECS::SystemDependencies::Empty());
176
177 ECS::SystemDependencies deferredRenderDependencies{};
178 deferredRenderDependencies.executeAfterThese.insert(static_cast<std::string>(ECS::TreeNormalsRenderSystem::GetSystemName()));
179 deferredRenderDependencies.executeAfterThese.insert(static_cast<std::string>(TransformApplierSystem::GetSystemName()));
180 entityComponentSystem->RegisterSystem<ECS::DeferredRenderSystem>(deferredRenderDependencies);
181#elif defined(OSK_USE_HYBRID_RENDERER)
182 entityComponentSystem->RegisterSystem<ECS::HybridRenderSystem>();
183#elif defined(OSK_USE_GDR_RENDERER)
184 entityComponentSystem->RegisterSystem<ECS::TreeNormalsRenderSystem>(ECS::ISystem::DEFAULT_EXECUTION_ORDER - 1);
185 entityComponentSystem->RegisterSystem<ECS::GdrDeferredRenderSystem>(ECS::ISystem::DEFAULT_EXECUTION_ORDER);
186#elif OSK_NO_DEFAULT_RENDERER
187#else
188#error No hay un renderizador por defecto
189#endif
190
191 entityComponentSystem->RegisterSystem<ECS::PhysicsSystem>(SystemDependencies::Empty());
192
193 ECS::SystemDependencies collisionDependencies{};
194 collisionDependencies.executeAfterThese.insert(static_cast<std::string>(PhysicsSystem::GetSystemName()));
195 collisionDependencies.exclusiveExecution = true;
196 collisionDependencies.singleThreaded = true;
197 entityComponentSystem->RegisterSystem<ECS::CollisionSystem>(collisionDependencies);
198
199 ECS::SystemDependencies collisionResolverDependencies{};
200 collisionResolverDependencies.executeAfterThese.insert(static_cast<std::string>(CollisionSystem::GetSystemName()));
201 collisionResolverDependencies.exclusiveExecution = true;
202 collisionResolverDependencies.singleThreaded = true;
203 entityComponentSystem->RegisterSystem<ECS::PhysicsResolver>(collisionResolverDependencies);
204
205 ECS::SystemDependencies transformApplierDependencies{};
206 transformApplierDependencies.executeAfterThese.insert(static_cast<std::string>(PhysicsResolver::GetSystemName()));
207 transformApplierDependencies.singleThreaded = true;
208 entityComponentSystem->RegisterSystem<TransformApplierSystem>(transformApplierDependencies);
209
210 entityComponentSystem->RegisterSystem<ECS::SkyboxRenderSystem>(SystemDependencies::Empty());
211 entityComponentSystem->RegisterSystem<ECS::RenderSystem2D>(SystemDependencies::Empty());
