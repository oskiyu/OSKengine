96 OSK_ASSERT(targetCommandList, InvalidObjectStateException("No se ha establecido la lista de comandos. V√©ase: SpriteRenderer::SetCommandList()."));

99 // Optimizaci√≥n: comrpobamos si la imagen del sprite actual es la misma que la del sprite anterior.

170void SpriteRenderer::DrawString(const FontInstance& fontInstance, USize32 fontSize, std::string_view text, const ECS::Transform2D& transform, const Color& color) {
171 OSK_ASSERT(fontInstance.sprite.GetPointer() != nullptr, InvalidArgumentException("La fuente no tiene configurado su sprite."));
172
173 const Vector2f& position = transform.GetPosition();
174 float x = transform.GetPosition().x;
175 float y = transform.GetPosition().y;
176
177 for (UIndex32 i = 0; i < text.size(); i++) {
178 const FontCharacter& character = fontInstance.characters.at(text[i]);
179
180 if (text[i] == '\n') {
181 y += character.size.y + character.bearing.y;
182 x = position.x;
183
184 continue;
185 }
186
187 if (text[i] == '\t') {
188 x += (character.advance >> 6) * 4;// Font::SPACES_PER_TAB;
189
190 continue;
191 }
192
193 if (text[i] == ' ') {
194 x += (character.advance >> 6);
195
196 continue;
197 }
198
199 float sizeX = character.size.x;
200 float sizeY = character.size.y;
201
202 float posX = x + character.bearing.x;
203 float posY = y - (character.bearing.y);
204
205 Draw(
206 *fontInstance.sprite.GetPointer(),
207 TextureCoordinates2D::Pixels(character.texCoords.ToVector4f()),
208 Vector2f(posX, posY),
209 Vector2f(sizeX, sizeY) * transform.GetScale(),
210 transform.GetRotation(), color);
211
212 x += static_cast<float>(character.advance >> 6);
213 }
