64 OSK_ASSERT(targetCommandList, InvalidObjectStateException("No se ha establecido la lista de comandos. Véase: SpriteRenderer::SetCommandList()."));
65 OSK_ASSERT(isStarted, InvalidObjectStateException("No se ha llamado a SpriteRenderer::Begin()."));
66
67 // Optimización: comrpobamos si la imagen del sprite actual es la misma que la del sprite anterior.
68 // Si es la misma, no necesitamos volver a establecer el material slot.
69 // De lo contrario, debemos establecerlo.
70 if (sprite.GetView() != previousImage) {
71 targetCommandList->BindMaterialSlot(*sprite.GetTextureSlot());
72 previousImage = sprite.GetView();
73 }
74
75 PushConst2D pushConst{};
76
77 pushConst.color = sprite.color;
78 pushConst.matrix = currentCameraMatrix * transform.GetAsMatrix();
79 pushConst.texCoords = texCoords.type == TextureCoordsType::NORMALIZED
80 ? texCoords.texCoords
81 : texCoords.GetNormalized(sprite.GetView()->GetSize2D().ToVector2f());
82
83 targetCommandList->PushMaterialConstants("sprite", pushConst);
84
85 targetCommandList->DrawSingleInstance(6);

137void SpriteRenderer::DrawString(const FontInstance& fontInstance, USize32 fontSize, std::string_view text, const ECS::Transform2D& transform, const Color& color) {
138 OSK_ASSERT(fontInstance.sprite.GetPointer() != nullptr, InvalidArgumentException("La fuente no tiene configurado su sprite."));
139
140 const Vector2f& position = transform.GetPosition();
141 float x = transform.GetPosition().x;
142 float y = transform.GetPosition().y;
143
144 for (UIndex32 i = 0; i < text.size(); i++) {
145 const FontCharacter& character = fontInstance.characters.at(text[i]);
146
147 if (text[i] == '\n') {
148 y += character.size.y + character.bearing.y;
149 x = position.x;
150
151 continue;
152 }
153
154 if (text[i] == '\t') {
155 x += (character.advance >> 6) * 4;// Font::SPACES_PER_TAB;
156
157 continue;
158 }
159
160 if (text[i] == ' ') {
161 x += (character.advance >> 6);
162
163 continue;
164 }
165
166 float sizeX = character.size.x;
167 float sizeY = character.size.y;
168
169 float posX = x + character.bearing.x;
170 float posY = y - (character.bearing.y);
171
172 Draw(
173 *fontInstance.sprite.GetPointer(),
174 TextureCoordinates2D::Pixels(character.texCoords.ToVector4f()),
175 Vector2f(posX, posY),
176 Vector2f(sizeX, sizeY) * transform.GetScale(),
177 transform.GetRotation(), color);
178
179 x += static_cast<float>(character.advance >> 6);
180 }
181}
