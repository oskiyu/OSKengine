332void HybridRenderSystem::RenderGBuffer(GRAPHICS::ICommandList* commandList) {
333 // Sincronización con todos los targets de color sobre los que vamos a escribir.
334 for (const auto type : GBuffer::ColorTargetTypes)
335 commandList->SetGpuImageBarrier(
336 gBuffer.GetImage(type),
337 GpuImageLayout::COLOR_ATTACHMENT,
338 GpuBarrierInfo(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE));
339
340
341 gBuffer.BeginRenderpass(commandList, Color::Black * 0.0f);
342
343 SetupViewport(commandList);
344 commandList->BindMaterial(*gbufferMaterial);
345 commandList->BindMaterialSlot(*globalGbufferMaterialInstance->GetSlot("global"));
346
347 GBufferRenderLoop(commandList, ModelType::STATIC_MESH);
348 GBufferRenderLoop(commandList, ModelType::ANIMATED_MODEL);
349
350 commandList->EndGraphicsRenderpass();
351
352
353 // Copia al gbuffer historico
354 for (const auto type : GBuffer::ColorTargetTypes) {
355 GpuImage* previousImage = gBuffer.GetImage(type);
356 GpuImage* targetImage = historicalGBuffer.GetImage(type);
357
358 commandList->SetGpuImageBarrier(
359 previousImage,
360 GpuImageLayout::TRANSFER_SOURCE,
361 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
362 commandList->SetGpuImageBarrier(
363 targetImage,
364 GpuImageLayout::TRANSFER_DESTINATION,
365 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
366
367 const Vector2ui imgSize = targetImage->GetSize2D();
368 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(imgSize);
369 commandList->RawCopyImageToImage(*previousImage, targetImage, copyInfo);
370
371 commandList->SetGpuImageBarrier(
372 previousImage,
373 GpuImageLayout::GENERAL,
374 GpuBarrierInfo(GpuCommandStage::RAYTRACING_SHADER, GpuAccessStage::SHADER_READ));
375 commandList->SetGpuImageBarrier(

389void HybridRenderSystem::RenderShadows(GRAPHICS::ICommandList* cmdList) {
390 GpuImage* rayTracedImage = raytracedShadowsTarget.GetTargetImage();
391 GpuImage* reprojectedImage = reprojectedShadowsTarget.GetTargetImage();
392 GpuImage* denoisedImage = finalShadowsTarget.GetTargetImage();
393 GpuImage* previousDenoisedImage = finalShadowsTarget.GetTargetImage();
394 GpuImage* historicalImage = historicalShadowsTarget.GetTargetImage();
395
396 const Vector2ui resolution = finalShadowsTarget.GetSize();
397 const Vector2ui threadGroupSize = { 8u, 8u };
398 const Vector2ui dispatchRes = resolution / threadGroupSize + Vector2ui(1u, 1u);
399
400 // Ray traces
401
402 cmdList->SetGpuImageBarrier(rayTracedImage, GpuImageLayout::GENERAL,
403 GpuBarrierInfo(GpuCommandStage::RAYTRACING_SHADER, GpuAccessStage::SHADER_WRITE));
404
405 cmdList->BindMaterial(*shadowsMaterial);
406 cmdList->BindMaterialSlot(*shadowsMaterialInstance->GetSlot("rt"));
407 cmdList->BindMaterialSlot(*shadowsMaterialInstance->GetSlot("shadows"));
408
409 static int temporalIndex = 0;
410 temporalIndex++;
411 temporalIndex = temporalIndex % 64;
412
413 cmdList->PushMaterialConstants("frameInfo", temporalIndex);
414
415 cmdList->TraceRays(0, 0, 0, rayTracedImage->GetSize2D());
416
417 cmdList->SetGpuImageBarrier(rayTracedImage, GpuImageLayout::SAMPLED,
418 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
419
420
421 // Copia hacia la imagen histórica del frame anterior
422
423 cmdList->SetGpuImageBarrier(previousDenoisedImage, GpuImageLayout::TRANSFER_SOURCE,
424 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
425 cmdList->SetGpuImageBarrier(historicalImage, GpuImageLayout::TRANSFER_DESTINATION,
426 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
427
428 const Vector2ui imgSize = historicalImage->GetSize2D();
429 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(imgSize);
430 cmdList->RawCopyImageToImage(*previousDenoisedImage, historicalImage, copyInfo);
431
432 cmdList->SetGpuImageBarrier(previousDenoisedImage, GpuImageLayout::SAMPLED,
433 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
434 cmdList->SetGpuImageBarrier(historicalImage, GpuImageLayout::SAMPLED,
435 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
436
437
438 // Reproyección
439
440 cmdList->SetGpuImageBarrier(reprojectedImage, GpuImageLayout::GENERAL,
441 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
442
443 cmdList->BindMaterial(*shadowsReprojectionMaterial);
444 cmdList->BindMaterialSlot(*shadowsReprojectionMaterialInstance->GetSlot("shadows"));
445 cmdList->DispatchCompute({ dispatchRes.x, dispatchRes.y, 1 });
446
447 cmdList->SetGpuImageBarrier(reprojectedImage, GpuImageLayout::GENERAL,
448 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
449
450 // Denoising
451
452 cmdList->SetGpuImageBarrier(denoisedImage, GpuImageLayout::GENERAL,
453 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
454
455 cmdList->BindMaterial(*shadowsAtrousMaterial);
456 cmdList->BindMaterialSlot(*shadowsAtrousMaterialInstance->GetSlot("shadows"));
457 cmdList->DispatchCompute({ dispatchRes.x, dispatchRes.y, 1 });
458
459 cmdList->SetGpuImageBarrier(denoisedImage, GpuImageLayout::SAMPLED,
460 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
461}
