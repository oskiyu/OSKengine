278 // Sincronizaci√≥n con todos los targets de color sobre los que vamos a escribir.
279 for (const auto type : GBuffer::ColorTargetTypes)
280 commandList->SetGpuImageBarrier(
281 gBuffer.GetImage(type),
282 GpuImageLayout::COLOR_ATTACHMENT,
283 GpuBarrierInfo(GpuCommandStage::COLOR_ATTACHMENT_OUTPUT, GpuAccessStage::COLOR_ATTACHMENT_WRITE));
284
285
286 gBuffer.BeginRenderpass(commandList, Color::Black * 0.0f);

288 SetupViewport(commandList);
289 commandList->BindMaterial(*gbufferMaterial);
290 commandList->BindMaterialSlot(*globalGbufferMaterialInstance->GetSlot("global"));
291
292 for (auto& pass : m_shaderPasses.GetAllPasses()) {
293 pass->RenderLoop(
294 commandList,
295 m_shaderPasses.GetCompatibleObjects(pass->GetTypeName()),
296 &m_meshMapping,
297 0,
298 m_renderTarget.GetSize());
299 }
300
301 commandList->EndGraphicsRenderpass();
302
303
304 // Copia al gbuffer historico
305 for (const auto type : GBuffer::ColorTargetTypes) {
306 GpuImage* previousImage = gBuffer.GetImage(type);
307 GpuImage* targetImage = historicalGBuffer.GetImage(type);
308
309 commandList->SetGpuImageBarrier(
310 previousImage,
311 GpuImageLayout::TRANSFER_SOURCE,
312 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
313 commandList->SetGpuImageBarrier(
314 targetImage,
315 GpuImageLayout::TRANSFER_DESTINATION,
316 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
317
318 const Vector2ui imgSize = targetImage->GetSize2D();
319 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(imgSize);
320 commandList->RawCopyImageToImage(*previousImage, targetImage, copyInfo);
321
322 commandList->SetGpuImageBarrier(
323 previousImage,
324 GpuImageLayout::GENERAL,
325 GpuBarrierInfo(GpuCommandStage::RAYTRACING_SHADER, GpuAccessStage::SHADER_READ));
326 commandList->SetGpuImageBarrier(
327 targetImage,
328 GpuImageLayout::SAMPLED,
329 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
330 }

336 GpuBarrierInfo(GpuCommandStage::FRAGMENT_SHADER, GpuAccessStage::SHADER_READ), GpuBarrierInfo(GpuCommandStage::RAYTRACING_SHADER, GpuAccessStage::SHADER_READ),

340void HybridRenderSystem::RenderShadows(GRAPHICS::ICommandList* cmdList) {
341 GpuImage* rayTracedImage = raytracedShadowsTarget.GetTargetImage();
342 GpuImage* reprojectedImage = reprojectedShadowsTarget.GetTargetImage();
343 GpuImage* denoisedImage = finalShadowsTarget.GetTargetImage();
344 GpuImage* previousDenoisedImage = finalShadowsTarget.GetTargetImage();
345 GpuImage* historicalImage = historicalShadowsTarget.GetTargetImage();
346
347 const Vector2ui resolution = finalShadowsTarget.GetSize();
348 const Vector2ui threadGroupSize = { 8u, 8u };
349 const Vector2ui dispatchRes = resolution / threadGroupSize + Vector2ui(1u, 1u);
350
351 // Ray traces
352
353 cmdList->SetGpuImageBarrier(rayTracedImage, GpuImageLayout::GENERAL,
354 GpuBarrierInfo(GpuCommandStage::RAYTRACING_SHADER, GpuAccessStage::SHADER_WRITE));
355
356 cmdList->BindMaterial(*shadowsMaterial);
357 cmdList->BindMaterialSlot(*shadowsMaterialInstance->GetSlot("rt"));
358 cmdList->BindMaterialSlot(*shadowsMaterialInstance->GetSlot("shadows"));
359
360 static int temporalIndex = 0;
361 temporalIndex++;
362 temporalIndex = temporalIndex % 64;
363
364 cmdList->PushMaterialConstants("frameInfo", temporalIndex);
365
366 cmdList->TraceRays(0, 0, 0, rayTracedImage->GetSize2D());
367
368 cmdList->SetGpuImageBarrier(rayTracedImage, GpuImageLayout::SAMPLED,
369 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
370
371
372 // Copia hacia la imagen hist√≥rica del frame anterior
373
374 cmdList->SetGpuImageBarrier(previousDenoisedImage, GpuImageLayout::TRANSFER_SOURCE,
375 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_READ));
376 cmdList->SetGpuImageBarrier(historicalImage, GpuImageLayout::TRANSFER_DESTINATION,
377 GpuBarrierInfo(GpuCommandStage::TRANSFER, GpuAccessStage::TRANSFER_WRITE));
378
379 const Vector2ui imgSize = historicalImage->GetSize2D();
380 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(imgSize);
381 cmdList->RawCopyImageToImage(*previousDenoisedImage, historicalImage, copyInfo);
382
383 cmdList->SetGpuImageBarrier(previousDenoisedImage, GpuImageLayout::SAMPLED,
384 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
385 cmdList->SetGpuImageBarrier(historicalImage, GpuImageLayout::SAMPLED,
386 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
387
388
389 // Reproyecci√≥n
390
391 cmdList->SetGpuImageBarrier(reprojectedImage, GpuImageLayout::GENERAL,
392 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
393
394 cmdList->BindMaterial(*shadowsReprojectionMaterial);
395 cmdList->BindMaterialSlot(*shadowsReprojectionMaterialInstance->GetSlot("shadows"));
396 cmdList->DispatchCompute({ dispatchRes.x, dispatchRes.y, 1 });
397
398 cmdList->SetGpuImageBarrier(reprojectedImage, GpuImageLayout::GENERAL,
399 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
400
401 // Denoising
402
403 cmdList->SetGpuImageBarrier(denoisedImage, GpuImageLayout::GENERAL,
404 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_WRITE));
405
406 cmdList->BindMaterial(*shadowsAtrousMaterial);
407 cmdList->BindMaterialSlot(*shadowsAtrousMaterialInstance->GetSlot("shadows"));
408 cmdList->DispatchCompute({ dispatchRes.x, dispatchRes.y, 1 });
409
410 cmdList->SetGpuImageBarrier(denoisedImage, GpuImageLayout::SAMPLED,
411 GpuBarrierInfo(GpuCommandStage::COMPUTE_SHADER, GpuAccessStage::SHADER_READ));
412}

444void HybridRenderSystem::Render(GRAPHICS::ICommandList* commandList, std::span<const ECS::GameObjectIndex> objects) {
445 const UIndex32 resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
446
447 topLevelAccelerationStructures[resourceIndex]->Update(commandList);
448
449 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
450 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
451
452 cameraUbos[resourceIndex]->MapMemory();
453 cameraUbos[resourceIndex]->Write(camera.GetProjectionMatrix());
454 cameraUbos[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
455 cameraUbos[resourceIndex]->Write(cameraTransform.GetPosition());
456 cameraUbos[resourceIndex]->Unmap();
457
458 previousCameraUbos[resourceIndex]->MapMemory();
459 previousCameraUbos[resourceIndex]->Write(previousProjection);
460 previousCameraUbos[resourceIndex]->Write(previousView);
461 previousCameraUbos[resourceIndex]->Unmap();
462
463 previousProjection = camera.GetProjectionMatrix();
464 previousView = camera.GetViewMatrix(cameraTransform);
465
466 dirLightUbos[resourceIndex]->MapMemory();
467 dirLightUbos[resourceIndex]->Write(dirLight);
468 dirLightUbos[resourceIndex]->Unmap();
469
470 RenderGBuffer(commandList);
471 RenderShadows(commandList);
472 Resolve(commandList);
473
474 for (const GameObjectIndex obj : objects) {
475 previousModelMatrices[obj] = Engine::GetEcs()->GetComponent<Transform3D>(obj).GetAsMatrix();
476 }
477}
