238 auto it = bufferMemoryBlocks.Find({ size, usage, sharedType });
239
240 for (auto it : bufferMemoryBlocks) {
241 if (it.first.usage == usage && it.first.sharedType == sharedType) {
242 auto& list = it.second;
243
244 for (auto& i : list)
245 if (i->GetAvailableSpace() >= size)
246 return i.GetPointer();
247
248 auto i = GpuMemoryBlockVulkan::CreateNewBufferBlock(size, device, sharedType, usage);
