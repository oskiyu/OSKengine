65OwnedPtr<GpuImage> GpuMemoryAllocatorVulkan::CreateImage(const GpuImageCreateInfo& info) {
66 TSize numBytes = GetFormatNumberOfBytes(info.format);
67
68 switch (info.dimension) {
69 case OSK::GRAPHICS::GpuImageDimension::d1D: numBytes *= info.resolution.X; break;
70 case OSK::GRAPHICS::GpuImageDimension::d2D: numBytes *= info.resolution.X * info.resolution.Y; break;
71 case OSK::GRAPHICS::GpuImageDimension::d3D: numBytes *= info.resolution.X * info.resolution.Y * info.resolution.Z; break;
72 }
73
74 Vector3ui finalImageSize = info.resolution;
75
76 switch (info.dimension) {
77 case OSK::GRAPHICS::GpuImageDimension::d1D: finalImageSize = { info.resolution.X , 1, 1 }; break;
78 case OSK::GRAPHICS::GpuImageDimension::d2D: finalImageSize = { info.resolution.X , info.resolution.Y, 1 }; break;
79 }
80
81 if (finalImageSize.X == 0)
82 finalImageSize.X = 1;
83 if (finalImageSize.Y == 0)
84 finalImageSize.Y = 1;
85
86 VkSampler sampler = VK_NULL_HANDLE;
87
88 GpuImageVulkan* output = new GpuImageVulkan(info.resolution, info.dimension, info.usage, info.numLayers, info.format, info.msaaSamples, info.samplerDesc);
89
90 TSize numMipLevels = 0;
91 switch (info.samplerDesc.mipMapMode) {
92 case GpuImageMipmapMode::AUTO:
93 numMipLevels = output->GetMipLevels();
94 break;
95 case GpuImageMipmapMode::CUSTOM:
96 numMipLevels = info.samplerDesc.maxMipLevel;
97 break;
98 case GpuImageMipmapMode::NONE:
99 numMipLevels = 1;
100 break;
101 }
102
103 output->CreateVkImage();
104 auto block = GpuMemoryBlockVulkan::CreateNewImageBlock(output, device, info.memoryType, info.usage);
105 output->SetBlock(block.GetPointer());
106
107 imageMemoryBlocks.Insert(block.GetPointer());
108
109 // ------ IMAGE ---------- //
110 output->CreateVkSampler(info.samplerDesc);
111
112 return output;
113}
