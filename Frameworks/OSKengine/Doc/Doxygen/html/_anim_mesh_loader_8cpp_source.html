340void AnimMeshLoader::LoadAnimations() {
341 const TSize numAnimations = gltfModel.animations.size();
342
343 for (TSize animationId = 0; animationId < numAnimations; animationId++) {
344 tinygltf::Animation gltfAnimation = gltfModel.animations[animationId];
345 Animation animation{};
346
347 animation.name = gltfAnimation.name;
348 animation.samplers.Resize(gltfAnimation.samplers.size());
349 animation.channels.Resize(gltfAnimation.channels.size());
350
351 // Samplers
352 for (TSize samplerId = 0; samplerId < gltfAnimation.samplers.size(); samplerId++) {
353 tinygltf::AnimationSampler gltfSampler = gltfAnimation.samplers[samplerId];
354
355 AnimationSampler& sampler = animation.samplers[samplerId];
356 // TODO: sampler interpolation type
357
358 // Establece los valores de entrada del sampler, es decir,
359 // los timestamps en el que se produce una transforamci√≥n de alg√∫n tipo.
360 {
361 const tinygltf::Accessor& accessor = gltfModel.accessors[gltfSampler.input];
362 const tinygltf::BufferView& bufferView = gltfModel.bufferViews[accessor.bufferView];
363 const tinygltf::Buffer& buffer = gltfModel.buffers[bufferView.buffer];
364
365 const void* data = &buffer.data[accessor.byteOffset + bufferView.byteOffset];
366 const float* timestampsBuffer = static_cast<const float*>(data);
367
368 for (TSize i = 0; i < accessor.count; i++)
369 sampler.timestamps.Insert(timestampsBuffer[i]);
370
371 // Timestamps de inicio y fin de la animaci√≥n.
372 for (const TDeltaTime timestamp : sampler.timestamps) {
373 animation.startTime = glm::min(animation.startTime, timestamp);
374 animation.endTime = glm::max(animation.endTime, timestamp);
375 }
376 }
377
378 // Establece los valores de salida del sampler, es decir,
379 // la transformaci√≥n de cada timestamp.
380 {
381 const tinygltf::Accessor& accessor = gltfModel.accessors[gltfSampler.output];
382 const tinygltf::BufferView& bufferView = gltfModel.bufferViews[accessor.bufferView];
383 const tinygltf::Buffer& buffer = gltfModel.buffers[bufferView.buffer];
384 const void* data = &buffer.data[accessor.byteOffset + bufferView.byteOffset];
385
386 switch (accessor.type) {
387
388 case TINYGLTF_TYPE_VEC3: {
389 const glm::vec3* vecBuffer = static_cast<const glm::vec3*>(data);
390 for (TSize i = 0; i < accessor.count; i++)
391 sampler.outputs.Insert(glm::vec4(vecBuffer[i], 0.0));
392 }
393 break;
394
395 case TINYGLTF_TYPE_VEC4: {
396 const glm::vec4* vecBuffer = static_cast<const glm::vec4*>(data);
397 for (TSize i = 0; i < accessor.count; i++)
398 sampler.outputs.Insert(vecBuffer[i]);
399 }
400 break;
401
402 }
403 }
404 }
405
406 // Channels
407 for (TSize channelId = 0; channelId < gltfAnimation.channels.size(); channelId++) {
408 tinygltf::AnimationChannel gltfChannel = gltfAnimation.channels[channelId];
409
410 AnimationChannel& channel = animation.channels[channelId];
411 channel.samplerIndex = channelId;
412 channel.nodeId = gltfChannel.target_node;
413
414 if (gltfChannel.target_path == "translation") {
415 channel.type = AnimationChannel::ChannelType::TRANSLATION;
416 }
417 else if (gltfChannel.target_path == "rotation") {
418 channel.type = AnimationChannel::ChannelType::ROTATION;
419 }
420 else if (gltfChannel.target_path == "scale") {
421 channel.type = AnimationChannel::ChannelType::SCALE;
422 }
423 else {
424 OSK_CHECK(false, "El canal de animaci√≥n " + gltfChannel.target_path + " no est√° soportado.");
425 }
426 }
427
428 Engine::GetLogger()->InfoLog("Animaci√≥n cargada: " + animation.name);
429 tempAnimator._AddAnimation(animation);
