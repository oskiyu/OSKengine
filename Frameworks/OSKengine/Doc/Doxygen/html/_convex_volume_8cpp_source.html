18ConvexVolume ConvexVolume::CreateObb(const Vector3f& size, float bottomHeight) {

81void ConvexVolume::AddFace(const DynamicArray<Vector3f>& points) {
82 const TSize numOldVertices = vertices.GetSize();
83 const TSize numNewVertices = points.GetSize();
84
85 /*
86 // Primero, comprobamos si alguno de los v√©rtices
87 // ya existe en la lista de v√©rtices.
88 //
89 // Lista que mapea cada nuevo v√©rtice con el √≠ndice
90 // del v√©rtice antiguo equivalente.
91 // Es posible que un v√©rtice no tenga equivalente anterior.
92 DynamicArray<std::optional<TIndex>> existingIndices = DynamicArray<std::optional<TIndex>>::CreateResizedArray(numNewVertices);
93
94 for (TIndex newIndex = 0; newIndex < numNewVertices; newIndex++) {
95 const Vector3f newVertex = points[newIndex];
96
97 for (TIndex oldIndex = 0; oldIndex < numOldVertices; oldIndex++) {
98 const Vector3f oldVertex = vertices[oldIndex];
99
100 if (oldVertex.Equals(newVertex, 0.01f))
101 existingIndices[newIndex] = oldIndex;
102 }
103 }
104
105 // Lista con los nuevos v√©rtices (que no estaban presentes anteriormente).
106 DynamicArray<Vector3f> newVertices;
107 for (TIndex i = 0; i < numNewVertices; i++)
108 if (!existingIndices[i].has_value())
109 newVertices.Insert(points[i]);
110 */
111
112 FaceIndices indices = FaceIndices::CreateReservedArray(numNewVertices);
113 for (TIndex i = 0; i < numNewVertices; i++) {
114 indices.Insert(numOldVertices + i);
115 vertices.Insert(points[i]);
116 }
117
118 faces.Insert(indices);
119}

121void ConvexVolume::OptimizeAxes() {
122 if (faces.IsEmpty())
123 return;
124
125 DynamicArray<FaceIndices> optimizedFaces = {};
126 DynamicArray<TIndex> duplicatedIds = {};
127
128 for (TIndex i = 0; i < faces.GetSize() - 1; i++) {
129
130 // Por cada eje original, comprobamos si hay otros
131 // ejes equivalentes.
132
133 for (TIndex j = i + 1; j < faces.GetSize(); j++) {
134
135 const Axis axis1 = GetLocalSpaceAxis(i);
136 const Axis axis2 = GetLocalSpaceAxis(j);
137
138 // @todo epsilon?
139 const bool isSameAxis = axis1 == axis2 || axis1 == -axis2;
140
141 // Si el otro eje es igual, lo marcamos para no a√±adirlo
142 // a la lista optimizada.
143 if (isSameAxis)
144 duplicatedIds.Insert(j);
145 }
146 }
147
148 // A√±adimos las caras que no hayan sido marcadas.
149 for (TIndex i = 0; i < faces.GetSize(); i++)
150 if (!duplicatedIds.ContainsElement(i))
151 optimizedFaces.Insert(faces[i]);
152
153 faces = optimizedFaces;
154}

156ConvexVolume::Axis ConvexVolume::GetLocalSpaceAxis(TIndex faceId) const {
157 const FaceIndices& faceIndices = faces[faceId];
158
159 const Vector3f pointA = vertices.At(faceIndices.At(0));
160 const Vector3f pointB = vertices.At(faceIndices.At(1));
161 const Vector3f pointC = vertices.At(faceIndices.At(2));
162
163 const Vector3f vec1 = pointB - pointA;
164 const Vector3f vec2 = pointC - pointA;
165
166 return vec1.Cross(vec2).GetNormalized();

169ConvexVolume::Axis ConvexVolume::GetWorldSpaceAxis(TIndex faceId, const Transform3D& transform) const {
170 const FaceIndices& faceIndices = faces[faceId];
171
172 const Vector3f pointA = vertices[faceIndices[0]];
173 const Vector3f pointB = vertices[faceIndices[1]];
174 const Vector3f pointC = vertices[faceIndices[2]];
175
176 const Vector3f transformedA = transform.TransformPoint(pointA);
177 const Vector3f transformedB = transform.TransformPoint(pointB);
178 const Vector3f transformedC = transform.TransformPoint(pointC);
179
180 const Vector3f vec1 = transformedB - transformedA;
181 const Vector3f vec2 = transformedC - transformedB;
182
183 return vec1.Cross(vec2).GetNormalized();

186DetailedCollisionInfo ConvexVolume::GetCollisionInfo(const IBottomLevelCollider& other,
187 const Transform3D& thisTransform, const Transform3D& otherTransform) const {
188
189 const ConvexVolume& otherVolume = (const ConvexVolume&)other;
190
191 // N√∫mero de caras (y por lo tanto, de ejes) de ambos colliders.
192 const TSize thisFaceCount = this->faces.GetSize();
193 const TSize otherFaceCount = otherVolume.faces.GetSize();
194
195 // V√©rtices en world-space de cada collider.
196 const auto transformedVerticesA = this->GetWorldSpaceVertices(thisTransform);
197 const auto transformedVerticesB = otherVolume.GetWorldSpaceVertices(otherTransform);
198
199 // Vector m√≠nimo que se debe aplicar para separar las entidades,
200 // de tal manera que dejen de estar en colisi√≥n.
201
202 Vector3f currentAxis = 0.0f;
203 bool isAxisA = false;
204 bool isAxisB = false;
205 TIndex mtvFaceIndex = 0;
206
207 // Overlap del mtv.
208 float minimumOverlap = std::numeric_limits<float>::max();
209
210 // Comprobamos los ejes de este collider.
211 for (TIndex i = 0; i < thisFaceCount; i++) {
212 const Axis axis = this->GetWorldSpaceAxis(i, thisTransform);
213
214 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
215 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
216
217 if (!projA.Overlaps(projB))
218 return DetailedCollisionInfo::False();
219
220 const float overlap = projA.GetOverlap(projB);
221 if (glm::abs(overlap) < glm::abs(minimumOverlap)) {
222 minimumOverlap = overlap;
223 currentAxis = axis;
224
225 isAxisA = true;
226 isAxisB = false;
227
228 mtvFaceIndex = i;
229 }
230 }
231
232 // Comprobamos los ejes del otro collider.
233 for (TIndex i = 0; i < otherFaceCount; i++) {
234 const Axis axis = otherVolume.GetWorldSpaceAxis(i, otherTransform);
235
236 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
237 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
238
239 if (!projA.Overlaps(projB))
240 return DetailedCollisionInfo::False();
241
242 const float overlap = projA.GetOverlap(projB);
243 if (glm::abs(overlap) < glm::abs(minimumOverlap)) {
244 minimumOverlap = overlap;
245 currentAxis = axis;
246
247 isAxisA = false;
248 isAxisB = true;
249
250 mtvFaceIndex = i;
251 }
252 }
253
254 // A to B
255 const Vector3f mtv = currentAxis.GetNormalized() * minimumOverlap;
256
257
258 // Obtenemos los puntos de contacto.
259 // @see https://dyn4j.org/2011/11/contact-points-using-clipping/
260
261
262 // El collider primario es el collider a partir del cual hemos
263 // obtenido el MTV.
264 const ConvexVolume& primaryCollider = isAxisA
265 ? *this
266 : otherVolume;
267
268 // El collider secundario es el otro collider.
269 const ConvexVolume& secondaryCollider = isAxisA
270 ? otherVolume
271 : *this;
272
273 // Los v√©rtices, transfomrados a espacio del mundo,
274 // del collider de referencia.
275 const DynamicArray<Vector3f> primaryVertices = isAxisA
276 ? transformedVerticesA
277 : transformedVerticesB;
278
279 // Los v√©rtices, transfomrados a espacio del mundo,
280 // del collider incidente.
281 const DynamicArray<Vector3f> secondaryVertices = isAxisA
282 ? transformedVerticesB
283 : transformedVerticesA;
284
285
286 // MTV, expresado en la direcci√≥n Referencia -> Incidente.
287 const Vector3f directedMtv = isAxisA
288 ? mtv
289 : -mtv;
290
291 // MTV normalizado, expresado en la direcci√≥n Referencia -> Incidente.
292 const Vector3f normalizedDirectedMtv = directedMtv.GetNormalized();
293
294
295 // Transform del collider de referencia.
296 const Transform3D& primaryTransform = isAxisA
297 ? thisTransform
298 : otherTransform;
299
300 // Transform del collider incidente.
301 const Transform3D& secondaryTransform = isAxisA
302 ? otherTransform
303 : thisTransform;
304
305
306 // Debemos obtener el punto del collider de referencia que est√°
307 // m√°s hacia el collider incidente.
308 TIndex referenceFurthestPointIndex = 0;
309 float referenceFurthestPointMaxDot = std::numeric_limits<float>::lowest();
310 for (TIndex i = 0; i < primaryVertices.GetSize(); i++) {
311 const auto& vertex = primaryVertices[i];
312
313 const float dot = vertex.Dot(normalizedDirectedMtv);
314 if (dot > referenceFurthestPointMaxDot) {
315 referenceFurthestPointMaxDot = dot;
316 referenceFurthestPointIndex = i;
317 }
318 }
319
320 // Debemos obtener el punto del collider incidente que est√°
321 // m√°s hacia el collider de referencia.
322 TIndex secondaryFurthestPointIndex = 0;
323 float secondaryFurthestPointMaxDot = std::numeric_limits<float>::lowest();
324 for (TIndex i = 0; i < secondaryVertices.GetSize(); i++) {
325 const auto& vertex = secondaryVertices[i];
326
327 const float dot = vertex.Dot(-normalizedDirectedMtv);
328 if (dot > secondaryFurthestPointMaxDot) {
329 secondaryFurthestPointMaxDot = dot;
330 secondaryFurthestPointIndex = i;
331 }
332 }
333
334
335 // √çndices de las caras del collider de referencia que contienen
336 // el v√©rtice m√°s alejado.
337 const DynamicArray<TIndex> primaryFacesIndices = primaryCollider.GetFaceIndicesWithVertex(primaryCollider.vertices[referenceFurthestPointIndex]);
338
339 // √çndices de las caras del segundo collider que contienen
340 // el v√©rtice m√°s alejado.
341 const DynamicArray<TIndex> secondaryFacesIndices = secondaryCollider.GetFaceIndicesWithVertex(secondaryCollider.vertices[secondaryFurthestPointIndex]);
342
343
344 // Obtenemos la cara del collider incidente que est√° m√°s paralela a
345 // la cara del MTV.
346 TIndex mostParallelFaceIndex = 0;
347
348 {
349 float mostParallelDot = std::numeric_limits<float>::lowest();
350 for (const TIndex faceId : secondaryFacesIndices) {
351 const Vector3f faceNormal = secondaryCollider.GetWorldSpaceAxis(faceId, secondaryTransform);
352 const float dot = glm::abs(faceNormal.Dot(normalizedDirectedMtv));
353
354 if (dot > mostParallelDot) {
355 mostParallelFaceIndex = faceId;
356 mostParallelDot = dot;
357 }
358 }
359 }
360
361 // √çndice de la cara incidente.
362 const TIndex incidentFaceIndex = mostParallelFaceIndex;
363
364 // √çndice de la cara del collider de referencia a partir
365 // del que hemos sacado el MTV.
366 TIndex referenceFaceIndex = 0;
367
368 if (false or primaryFacesIndices.ContainsElement(mtvFaceIndex)) {
369 referenceFaceIndex = mtvFaceIndex;
370 }
371 else {
372 float mostParallelDot = std::numeric_limits<float>::lowest();
373 for (const TIndex faceId : primaryFacesIndices) {
374 const Vector3f faceNormal = primaryCollider.GetWorldSpaceAxis(faceId, primaryTransform);
375 const float dot = faceNormal.Dot(normalizedDirectedMtv);
376
377 if (dot > mostParallelDot) {
378 referenceFaceIndex = faceId;
379 mostParallelDot = dot;
380 }
381 }
382 }
383
384 // Puntos de contacto.
385 // Se inicializan a los puntos de la cara incidente, y despu√©s
386 // se van ajustando para representar el √°rea de colisi√≥n.
387 DynamicArray<Vector3f> contactPoints = DynamicArray<Vector3f>::CreateReservedArray(secondaryCollider.faces[incidentFaceIndex].GetSize());
388 for (const TIndex index : secondaryCollider.faces[incidentFaceIndex])
389 contactPoints.Insert(secondaryVertices[index]);
390
391
392 // --------------------------- //
393
394 // Cara de referencia.
395 const FaceIndices& referenceFace = primaryCollider.faces[referenceFaceIndex];
396
397 // N√∫mero de v√©rtices en la cara de referencia.
398 const TSize numReferenceVertices = referenceFace.GetSize();
399
400 // Obtenemos los pares de v√©rtices consecutivos de la cara de referencia.
401 for (TIndex v = 0; v < numReferenceVertices; v++) {
402 const TIndex firstReferenceVertexIndex = referenceFace[v];
403 const TIndex secondReferenceVertexIndex = referenceFace[(v + 1) % numReferenceVertices];
404
405 // Definimos un plano a partir de una l√≠nea (definida por dos v√©rtices consecutivos)
406 // en direcci√≥n al centro.
407 const Vector3f referenceVertex1 = primaryVertices[firstReferenceVertexIndex];
408 const Vector3f referenceVertex2 = primaryVertices[secondReferenceVertexIndex];
409
410 const Vector3f lineDirection = -(referenceVertex2 - referenceVertex1).GetNormalized();
411
412 // Direcci√≥n a la que deben est√°r los puntos.
413 const Vector3f direction = -normalizedDirectedMtv.Cross(lineDirection).GetNormalized();
414
415 for (TIndex j = 0; j < contactPoints.GetSize(); j++) {
416 Vector3f& point = contactPoints[j];
417
418 const Vector3f relativePosition = point - referenceVertex1;
419
420 const float dot = relativePosition.Dot(direction);
421
422 if (dot < 0.0f)
423 point += -direction * dot;
424 }
425 }
426
427
428 DynamicArray<Vector3f> finalContactPoints = DynamicArray<Vector3f>::CreateReservedArray(contactPoints.GetSize());
429 float currentFurthestDistance = -999.9f;
430 for (const Vector3f& point : contactPoints) {
431 // Mayor distanca = m√°s metido.
432 const float distance = point.Dot(-normalizedDirectedMtv);
433
434 if (distance > currentFurthestDistance) {
435 currentFurthestDistance = distance;
436
437 finalContactPoints.Empty();
438 finalContactPoints.Insert(point);
439 }
440 else if (glm::abs(distance - currentFurthestDistance) < 0.01f) {
441 finalContactPoints.Insert(point);
442 }
443 }
444
445 for (auto& point : finalContactPoints)
446 point += directedMtv * 0.5f;
447
448// #define PRIMARY_POINTS
449// #define SECONDARY_POINTS
450#ifdef PRIMARY_POINTS
451 finalContactPoints.Empty();
452 for (TIndex i = 0; i < referenceFace.GetSize(); i++)
453 finalContactPoints.Insert(primaryVertices[referenceFace[i]]);
454#elif defined(SECONDARY_POINTS)
455 finalContactPoints.Empty();
456 for (TIndex i = 0; i < secondaryCollider.faces[incidentFaceIndex].GetSize(); i++)
457 finalContactPoints.Insert(secondaryVertices[secondaryCollider.faces[incidentFaceIndex][i]]);
458#endif
459
460 return DetailedCollisionInfo::True(mtv, finalContactPoints);
461}
