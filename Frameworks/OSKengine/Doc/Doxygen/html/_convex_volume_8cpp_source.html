25void ConvexVolume::OptimizeAxes() {
26 if (faces.IsEmpty())
27 return;
28
29 DynamicArray<FaceIndices> optimizedFaces = {};
30 DynamicArray<TIndex> duplicatedIds = {};
31
32 for (TIndex i = 0; i < faces.GetSize() - 1; i++) {
33
34 // Por cada eje original, comprobamos si hay otros
35 // ejes equivalentes.
36
37 for (TIndex j = i + 1; j < faces.GetSize(); j++) {
38
39 const Axis axis1 = GetAxisFromFace(i);
40 const Axis axis2 = GetAxisFromFace(j);
41
42 const bool isSameAxis = axis1 == axis2 || axis1 == -axis2;
43
44 // Si el otro eje es igual, lo marcamos para no añadirlo
45 // a la lista optimizada.
46 if (isSameAxis)
47 duplicatedIds.Insert(j);
48 }
49 }
50
51 // Añadimos las caras que no hayan sido marcadas.
52 for (TIndex i = 0; i < faces.GetSize(); i++)
53 if (!duplicatedIds.ContainsElement(i))
54 optimizedFaces.Insert(faces[i]);
55
56 faces = optimizedFaces;
57}

59ConvexVolume::Axis ConvexVolume::GetAxisFromFace(TIndex faceId) const {
60 const FaceIndices& faceIndices = faces[faceId];
61 const Vector3f pointA = vertices.At(faceIndices.At(0));
62 const Vector3f pointB = vertices.At(faceIndices.At(1));
63 const Vector3f pointC = vertices.At(faceIndices.At(2));
64
65 const Vector3f vec1 = pointB - pointA;
66 const Vector3f vec2 = pointC - pointA;
67
68 return vec1.Cross(vec2).GetNormalized();

89bool ConvexVolume::IsColliding(const IBottomLevelCollider& other,
90 const Transform3D& thisTransform, const Transform3D& otherTransform) const {
91
92 const ConvexVolume& otherVolume = (const ConvexVolume&)other;
93
94 const TSize thisFaceCount = faces.GetSize();
95 const TSize otherFaceCount = otherVolume.faces.GetSize();
96
97 const auto transformedVerticesA = GetTransformedVertices(thisTransform);
98 const auto transformedVerticesB = otherVolume.GetTransformedVertices(otherTransform);
99
100 for (TIndex i = 0; i < thisFaceCount; i++) {
101 const Axis axis = GetAxisFromFace(i, thisTransform);
102
103 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
104 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
105
106 if (!projA.Overlaps(projB))
107 return false;
108 }
109
110 for (TIndex i = 0; i < otherFaceCount; i++) {
111 const Axis axis = GetAxisFromFace(i, thisTransform);
112
113 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
114 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
115
116 if (!projA.Overlaps(projB))
117 return false;
118 }
119
120 return true;
121}
