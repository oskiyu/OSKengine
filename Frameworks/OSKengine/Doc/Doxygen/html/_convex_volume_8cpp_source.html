25void ConvexVolume::OptimizeAxes() {
26 if (faces.IsEmpty())
27 return;
28
29 DynamicArray<FaceIndices> optimizedFaces = {};
30 DynamicArray<TIndex> duplicatedIds = {};
31
32 for (TIndex i = 0; i < faces.GetSize() - 1; i++) {
33
34 // Por cada eje original, comprobamos si hay otros
35 // ejes equivalentes.
36
37 for (TIndex j = i + 1; j < faces.GetSize(); j++) {
38
39 const Axis axis1 = GetLocalSpaceAxis(i);
40 const Axis axis2 = GetLocalSpaceAxis(j);
41
42 // @todo epsilon?
43 const bool isSameAxis = axis1 == axis2 || axis1 == -axis2;
44
45 // Si el otro eje es igual, lo marcamos para no añadirlo
46 // a la lista optimizada.
47 if (isSameAxis)
48 duplicatedIds.Insert(j);
49 }
50 }
51
52 // Añadimos las caras que no hayan sido marcadas.
53 for (TIndex i = 0; i < faces.GetSize(); i++)
54 if (!duplicatedIds.ContainsElement(i))
55 optimizedFaces.Insert(faces[i]);
56
57 faces = optimizedFaces;
58}

60ConvexVolume::Axis ConvexVolume::GetLocalSpaceAxis(TIndex faceId) const {
61 const FaceIndices& faceIndices = faces[faceId];
62
63 const Vector3f pointA = vertices.At(faceIndices.At(0));
64 const Vector3f pointB = vertices.At(faceIndices.At(1));
65 const Vector3f pointC = vertices.At(faceIndices.At(2));
66
67 const Vector3f vec1 = pointB - pointA;
68 const Vector3f vec2 = pointC - pointA;

90DetailedCollisionInfo ConvexVolume::GetCollisionInfo(const IBottomLevelCollider& other,
91 const Transform3D& thisTransform, const Transform3D& otherTransform) const {
92
93 const ConvexVolume& otherVolume = (const ConvexVolume&)other;
94
95 // Número de caras (y por lo tanto, de ejes) de ambos colliders.
96 const TSize thisFaceCount = this->faces.GetSize();
97 const TSize otherFaceCount = otherVolume.faces.GetSize();
98
99 // Vértices en world-space de cada collider.
100 const auto transformedVerticesA = this->GetWorldSpaceVertices(thisTransform);
101 const auto transformedVerticesB = otherVolume.GetWorldSpaceVertices(otherTransform);
102
103 // Vector mínimo que se debe aplicar para separar las entidades,
104 // de tal manera que dejen de estar en colisión.
105
106 Vector3f mtv = 0.0f;
107
108 // Overlap del mtv.
109 float minimumOverlap = std::numeric_limits<float>::max();
110
111 // Comprobamos los ejes de este collider.
112 for (TIndex i = 0; i < thisFaceCount; i++) {
113 const Axis axis = this->GetWorldSpaceAxis(i, thisTransform);
114
115 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
116 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
117
118 if (!projA.Overlaps(projB))
119 return DetailedCollisionInfo::False();
120
121 const float overlap = projA.GetOverlap(projB);
122 if (glm::abs(overlap) < glm::abs(minimumOverlap)) {
123 minimumOverlap = overlap;
124 mtv = axis;
125 }
126 }
127
128 // Comprobamos los ejes del otro collider.
129 for (TIndex i = 0; i < otherFaceCount; i++) {
130 const Axis axis = otherVolume.GetWorldSpaceAxis(i, otherTransform);
131
132 const FaceProjection projA = FaceProjection(transformedVerticesA, axis);
133 const FaceProjection projB = FaceProjection(transformedVerticesB, axis);
134
135 if (!projA.Overlaps(projB))
136 return DetailedCollisionInfo::False();
137
138 const float overlap = projA.GetOverlap(projB);
139 if (glm::abs(overlap) < glm::abs(minimumOverlap)) {
140 minimumOverlap = overlap;
141 mtv = axis;
142 }
143 }
144
145 return DetailedCollisionInfo::True(mtv.GetNormalized() * minimumOverlap);
146}
