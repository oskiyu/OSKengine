14D3D12_SHADER_VISIBILITY GetShaderStageDx12(ShaderStage stage) {
15 if (OSK::EFTraits::HasFlag(stage, ShaderStage::VERTEX)
16 && OSK::EFTraits::HasFlag(stage, ShaderStage::FRAGMENT))
17 return D3D12_SHADER_VISIBILITY_ALL;
18
19 switch (stage) {
20 case ShaderStage::VERTEX:
21 return D3D12_SHADER_VISIBILITY_VERTEX;
22 case ShaderStage::FRAGMENT:
23 return D3D12_SHADER_VISIBILITY_PIXEL;

25 return D3D12_SHADER_VISIBILITY_ALL;
26 case ShaderStage::TESSELATION_EVALUATION:
