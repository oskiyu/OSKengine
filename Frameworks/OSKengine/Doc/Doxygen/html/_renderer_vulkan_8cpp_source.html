173void RendererVulkan::CreateInstance(const std::string& appName, const Version& version) {
174 //Información de la app.
175 VkApplicationInfo appInfo{};
176 appInfo.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
177 appInfo.pApplicationName = appName.c_str();
178 appInfo.applicationVersion = VK_MAKE_VERSION((int)version.mayor, (int)version.menor, (int)version.parche);
179 appInfo.pEngineName = "OSKengine";
180 appInfo.engineVersion = VK_MAKE_VERSION(Engine::GetVersion().mayor, Engine::GetVersion().menor, Engine::GetVersion().parche);
181 appInfo.apiVersion = VK_API_VERSION_1_2;
182
183 //Create info.
184 VkInstanceCreateInfo createInfo{};
185 createInfo.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
186 createInfo.pApplicationInfo = &appInfo;
187
188 //Establecemos las extensiones.
189 //Extensiones de la ventana.
190 uint32_t glfwExtensionCount = 0;
191 const char** glfwExtensions;
192 glfwExtensions = glfwGetRequiredInstanceExtensions(&glfwExtensionCount);
193
194 //Extensiones totales.
195 auto extensions = DynamicArray<const char*>::CreateReservedArray(glfwExtensionCount);
196 for (size_t i = 0; i < glfwExtensionCount; i++)
197 extensions.Insert(glfwExtensions[i]);
198
199#ifdef OSK_DEBUG
200 extensions.Insert(VK_EXT_DEBUG_UTILS_EXTENSION_NAME);
201 //Capas de validación.
202
203 if (AreValidationLayersAvailable()) {
204 Engine::GetLogger()->DebugLog("Capas de validación activas.");
205
206 createInfo.enabledLayerCount = validationLayers.GetSize();
207 createInfo.ppEnabledLayerNames = validationLayers.GetData();
208 }
209 else {
210 Engine::GetLogger()->Log(IO::LogLevel::WARNING, "No se ha encontrado soporte para las capas de validación.");
211 }
212#endif
213
214 createInfo.enabledExtensionCount = extensions.GetSize();
215 createInfo.ppEnabledExtensionNames = extensions.GetData();
216 createInfo.pNext = nullptr;
217
218 Engine::GetLogger()->InfoLog("Extensiones del renderizador: ");
219 for (const auto& i : extensions)
220 Engine::GetLogger()->InfoLog(" " + std::string(i));
221
222 Engine::GetLogger()->InfoLog("Capas de validación del renderizador: ");
223 for (const auto& i : validationLayers)
224 Engine::GetLogger()->InfoLog(" " + std::string(i));
225
226 VkResult result = vkCreateInstance(&createInfo, nullptr, &instance);
227 OSK_ASSERT(result == VK_SUCCESS, "Crear instancia de Vulkan." + std::to_string(result));
228}

261 currentGpu = new GpuVulkan;
262
263 // --------------- Physical ----------------- //
264
265 //Obtiene el número de GPUs disponibles.
266 uint32_t count = 0;
267 vkEnumeratePhysicalDevices(instance, &count, nullptr);
268
269 OSK_ASSERT(count != 0, "No hay ninguna GPU compatible con Vulkan.");
270
271 //Obtiene las GPUs.
272 auto devices = DynamicArray<VkPhysicalDevice>::CreateResizedArray(count);
273 vkEnumeratePhysicalDevices(instance, &count, devices.GetData());
274
275 //Comprobar la compatibilidad de las GPUs.
276 //Obtener una GPU compatible.
277 DynamicArray<GpuVulkan::Info> gpus;
278 for (const auto& gpu : devices) {
279 auto info = GpuVulkan::Info::Get(gpu, surface);
280
281 gpus.Insert(info);
282 }
283
284 VkPhysicalDevice gpu = devices[0];
285 GpuVulkan::Info info = gpus[0];
286 vkGetPhysicalDeviceMemoryProperties(gpu, &info.memoryProperties);
287
288 OSK_ASSERT(gpu != VK_NULL_HANDLE, "No hay ninguna GPU compatible con Vulkan.");
289
290 Engine::GetLogger()->InfoLog("GPU elegida: " + std::string(info.properties.deviceName));
291
292 currentGpu->As<GpuVulkan>()->SetPhysicalDevice(gpu);
293 currentGpu->As<GpuVulkan>()->SetInfo(info);
294}

321bool RendererVulkan::AreValidationLayersAvailable() const {
322#ifdef OSK_DEBUG
323 // Obtenemos el número de capas.
324 uint32_t layerCount;
325 vkEnumerateInstanceLayerProperties(&layerCount, nullptr);
326
327 // Obtenemos las capas.
328 auto availableLayers = DynamicArray<VkLayerProperties>::CreateResizedArray(layerCount);
329
330 vkEnumerateInstanceLayerProperties(&layerCount, availableLayers.GetData());
331
332 //Capas de validación necesitadas.
333 for (auto layerName : validationLayers) {
334 bool layerFound = false;
335
336 for (const auto& layerProperties : availableLayers) {
337 if (strcmp(layerName, layerProperties.layerName) == 0) {
338 layerFound = true;
339
340 break;
341 }
342 }
343
344 if (layerFound)
345 return true;
346 }
347
348 return false;
349#else
350 return false;
351#endif
