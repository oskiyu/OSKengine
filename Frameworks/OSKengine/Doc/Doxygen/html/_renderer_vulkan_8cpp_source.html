68 // Message severity:
69 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT: "diagnostic" message.
70 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT: información.
71 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT: warning.
72 // VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT: error.
73
74 // Tipos de mensaje:
75 // VK_DEBUG_UTILS_MESSAGE_TYPE_GENERAL_BIT_EXT: algo ha ocurrido, no tiene que ver con la especificación o el rendimiento.
76 // VK_DEBUG_UTILS_MESSAGE_TYPE_VALIDATION_BIT_EXT: algo ha ocurrido, incumple la especificación.
77 // VK_DEBUG_UTILS_MESSAGE_TYPE_PERFORMANCE_BIT_EXT: algo ha ocurrido, uso no óptimo de vulkan.
78
79 if (ignoredValidationLayersMessages.ContainsElement(pCallbackData->messageIdNumber))
80 return 0;
81
82 IO::LogLevel level = IO::LogLevel::WARNING;
83
84 switch (messageType) {
85 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
86 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
87 return 0;
88
89 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
90 level = IO::LogLevel::WARNING;
91 break;
92
93 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
94 level = IO::LogLevel::L_ERROR;
95 break;
96
97 default:
98 level = IO::LogLevel::WARNING;

325void RendererVulkan::ChooseGpu() {
326 currentGpu = new GpuVulkan;
327
328 // --------------- Physical ----------------- //
329
330 //Obtiene el número de GPUs disponibles.
331 uint32_t count = 0;
332 vkEnumeratePhysicalDevices(instance, &count, nullptr);
333
334 OSK_ASSERT(count != 0, "No hay ninguna GPU compatible con Vulkan.");
335
336 //Obtiene las GPUs.
337 auto devices = DynamicArray<VkPhysicalDevice>::CreateResizedArray(count);
338 vkEnumeratePhysicalDevices(instance, &count, devices.GetData());
339
340 //Comprobar la compatibilidad de las GPUs.
341 //Obtener una GPU compatible.
342 DynamicArray<GpuVulkan::Info> gpus;
343 for (const auto& gpu : devices) {
344 auto info = GpuVulkan::Info::Get(gpu, surface);
345
346 if (info.isSuitable)
347 gpus.Insert(info);
348 }
349
350 OSK_ASSERT(!gpus.IsEmpty(), "No hay una GPU compatible.");
351
352 VkPhysicalDevice gpu = devices[0];
353 GpuVulkan::Info info = gpus[0];
354 vkGetPhysicalDeviceMemoryProperties(gpu, &info.memoryProperties);
355
356 OSK_ASSERT(gpu != VK_NULL_HANDLE, "No hay ninguna GPU compatible con Vulkan.");
357
358 Engine::GetLogger()->InfoLog("GPU elegida: " + std::string(info.properties.deviceName));
359
360 currentGpu->As<GpuVulkan>()->SetPhysicalDevice(gpu);
361 currentGpu->As<GpuVulkan>()->SetInfo(info);
362
363 currentGpu->As<GpuVulkan>()->CreateLogicalDevice(surface);
364}

438void RendererVulkan::CreateSyncDevice() {
439 const VkDevice logicalDevice = currentGpu->As<GpuVulkan>()->GetLogicalDevice();
440
441 imageAvailableSemaphores = DynamicArray<VkSemaphore>::CreateResizedArray(3, VK_NULL_HANDLE);
442 renderFinishedSemaphores = DynamicArray<VkSemaphore>::CreateResizedArray(3, VK_NULL_HANDLE);
443 preComputeFinishedSemaphores = DynamicArray<VkSemaphore>::CreateResizedArray(3, VK_NULL_HANDLE);
444 postComputeFinishedSemaphores = DynamicArray<VkSemaphore>::CreateResizedArray(3, VK_NULL_HANDLE);
445 frameBuildSemaphores = DynamicArray<VkSemaphore>::CreateResizedArray(3, VK_NULL_HANDLE);
446
447 fullyRenderedFences = DynamicArray<VkFence>::CreateResizedArray(3, VK_NULL_HANDLE);
448
449 VkSemaphoreCreateInfo semaphoreInfo{};
450 semaphoreInfo.sType = VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO;
451
452 VkFenceCreateInfo fenceInfo{};
453 fenceInfo.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
454 fenceInfo.flags = VK_FENCE_CREATE_SIGNALED_BIT;
455
456 for (TSize i = 0; i < 3; i++) {
457
458 VkResult result = vkCreateSemaphore(logicalDevice, &semaphoreInfo, nullptr, &imageAvailableSemaphores[i]);
459 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el semáforo.");
460
461 result = vkCreateSemaphore(logicalDevice, &semaphoreInfo, nullptr, &preComputeFinishedSemaphores[i]);
462 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el semáforo.");
463
464 result = vkCreateSemaphore(logicalDevice, &semaphoreInfo, nullptr, &renderFinishedSemaphores[i]);
465 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el semáforo.");
466
467 result = vkCreateSemaphore(logicalDevice, &semaphoreInfo, nullptr, &postComputeFinishedSemaphores[i]);
468 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el semáforo.");
469
470 result = vkCreateSemaphore(logicalDevice, &semaphoreInfo, nullptr, &frameBuildSemaphores[i]);
471 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el semáforo.");
472
473 // Fences
474 result = vkCreateFence(logicalDevice, &fenceInfo, nullptr, &fullyRenderedFences[i]);
475 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el fence.");
476
477 if (pvkSetDebugUtilsObjectNameEXT != nullptr) {

482 debugName.objectType = VK_OBJECT_TYPE_SEMAPHORE;

485 debugName.pObjectName = name.c_str();
486 debugName.objectHandle = (uint64_t)imageAvailableSemaphores[i];
487 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
488
489 name = "Render Finished Semaphore[" + std::to_string(i) + "]";
490 debugName.pObjectName = name.c_str();
491 debugName.objectHandle = (uint64_t)renderFinishedSemaphores[i];
492 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
493
494 name = "Pre-Compute Finished Semaphore[" + std::to_string(i) + "]";
495 debugName.pObjectName = name.c_str();
496 debugName.objectHandle = (uint64_t)preComputeFinishedSemaphores[i];
497 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
498
499 name = "Post-Compute Finished Semaphore[" + std::to_string(i) + "]";
500 debugName.pObjectName = name.c_str();
501 debugName.objectHandle = (uint64_t)postComputeFinishedSemaphores[i];
502 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
503
504 name = "Frame build Semaphore[" + std::to_string(i) + "]";
505 debugName.pObjectName = name.c_str();
506 debugName.objectHandle = (uint64_t)frameBuildSemaphores[i];
507 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
508
509
510 debugName.objectType = VK_OBJECT_TYPE_FENCE;
511
512 name = "Fully Rendered Fence[" + std::to_string(i) + "]";
513 debugName.pObjectName = name.c_str();
514 debugName.objectHandle = (uint64_t)fullyRenderedFences[i];
515 result = pvkSetDebugUtilsObjectNameEXT(logicalDevice, &debugName);
516 }
517
518 }

521 // pero las hemos creado signaled.
522 const VkFence firstFencesToReset[] = {
523 fullyRenderedFences[0]
524 };
525 vkResetFences(logicalDevice, _countof(firstFencesToReset), firstFencesToReset);
526}

534 AcquireNextFrame();

536 graphicsCommandList->Reset();
537 graphicsCommandList->Start();
538
539 if (!singleCommandQueue) {

542

545

554 if (!singleCommandQueue) {

556 postComputeCommandList->Close();
557 frameBuildCommandList->Close();
558 }
559

563 }

572
573 for (TSize i = 0; i < singleTimeCommandLists.GetSize(); i++)
574 singleTimeCommandLists.At(i)->DeleteAllStagingBuffers();
575 singleTimeCommandLists.Free();
576
577 //
578
579 graphicsCommandList->Reset();
580 graphicsCommandList->Start();
581
582 if (!singleCommandQueue) {
583 preComputeCommandList->Reset();
584 preComputeCommandList->Start();

586 postComputeCommandList->Reset();

594void RendererVulkan::SubmitPreComputeCommands() {
595 // Esperar a que la imagen esté disponible antes de renderizar.
596 const VkPipelineStageFlags waitStages[] = { VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT };
597
598 VkSubmitInfo submitInfo{};
599 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;

698void RendererVulkan::SubmitFrame() {
699 const VkSwapchainKHR swapChains[] = { swapchain->As<SwapchainVulkan>()->GetSwapchain() };
700 const VkSemaphore waitSemaphores[] = { frameBuildSemaphores[currentCommandBufferIndex] };
701
702 VkPresentInfoKHR presentInfo{};
703 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
704 presentInfo.waitSemaphoreCount = _countof(waitSemaphores);
705 presentInfo.pWaitSemaphores = waitSemaphores; // Debemos esperar a que la imagen actual termine de renderizarse.
706 presentInfo.swapchainCount = 1;
707 presentInfo.pSwapchains = swapChains;
708 presentInfo.pImageIndices = &currentFrameIndex; // Lo presentamos en esta imagen.
709 presentInfo.pResults = nullptr;
710
711 VkResult result = vkQueuePresentKHR(presentQueue->As<CommandQueueVulkan>()->GetQueue(), &presentInfo);
712 const bool resized = result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR;
713 if (resized) {
714 // Esperamos a que se terminen todas las listas de comandos para
715 // poder cambiar de tamaño los render targets.
716 vkDeviceWaitIdle(currentGpu->As<GpuVulkan>()->GetLogicalDevice());
718
719 swapchain->As<SwapchainVulkan>()->Resize();
720
721 if (window->GetWindowSize().X > 0 && window->GetWindowSize().Y > 0)
722 Engine::GetRenderer()->As<RendererVulkan>()->HandleResize();
723 }
724
725 currentCommandBufferIndex = (currentCommandBufferIndex + 1) % swapchain->GetImageCount();
726
727 // Si la siguiente imagen está siendo procesada, esperar a que termine.
728 // @todo vkGetSemaphoreState
729 vkWaitForFences(currentGpu->As<GpuVulkan>()->GetLogicalDevice(), 1, &fullyRenderedFences[currentCommandBufferIndex], VK_TRUE, UINT64_MAX);
730 vkResetFences(currentGpu->As<GpuVulkan>()->GetLogicalDevice(), 1, &fullyRenderedFences[currentCommandBufferIndex]);
731}
