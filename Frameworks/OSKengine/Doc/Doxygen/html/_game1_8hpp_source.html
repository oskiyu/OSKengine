179 void OnTick(TDeltaTime deltaTime) override {
180 //OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform3D>(ballObject).RotateLocalSpace(deltaTime, { 0 ,1 ,0 });

182 OSK::ECS::CameraComponent3D& camera = OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::CameraComponent3D>(cameraObject);
183 OSK::ECS::Transform3D& cameraTransform = OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform3D>(cameraObject);
184
185 float forwardMovement = 0.0f;
186 float rightMovement = 0.0f;
187 auto newKs = OSK::Engine::GetWindow()->GetKeyboardState();
188 auto oldKs = OSK::Engine::GetWindow()->GetPreviousKeyboardState();
189 if (newKs->IsKeyDown(OSK::IO::Key::W))
190 forwardMovement += 1.0f;
191 if (newKs->IsKeyDown(OSK::IO::Key::S))
192 forwardMovement -= 1.0f;
193 if (newKs->IsKeyDown(OSK::IO::Key::A))
194 rightMovement -= 1.0f;
195 if (newKs->IsKeyDown(OSK::IO::Key::D))
196 rightMovement += 1.0f;
197
198 if (newKs->IsKeyDown(OSK::IO::Key::F11) && oldKs->IsKeyUp(OSK::IO::Key::F11))
199 OSK::Engine::GetWindow()->ToggleFullScreen();
200
201 if (newKs->IsKeyDown(OSK::IO::Key::ESCAPE))
202 this->Exit();
203
204 camera.Rotate(
205 (OSK::Engine::GetWindow()->GetMouseState()->GetPosition().X - OSK::Engine::GetWindow()->GetPreviousMouseState()->GetPosition().X),
206 (OSK::Engine::GetWindow()->GetMouseState()->GetPosition().Y - OSK::Engine::GetWindow()->GetPreviousMouseState()->GetPosition().Y)
207 );
208
209 cameraTransform.AddPosition(cameraTransform.GetForwardVector().GetNormalized() * forwardMovement * deltaTime);
210 cameraTransform.AddPosition(cameraTransform.GetRightVector().GetNormalized() * rightMovement * deltaTime);
211 camera.UpdateTransform(&cameraTransform);
212
213 uniformBuffer->ResetCursor();
214 uniformBuffer->MapMemory();
215 uniformBuffer->Write(camera.GetProjectionMatrix(cameraTransform));
216 uniformBuffer->Write(camera.GetViewMatrix());
217 uniformBuffer->Write(cameraTransform.GetPosition());
218 uniformBuffer->Unmap();
219
220 OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::CameraComponent2D>(cameraObject2d).UpdateUniformBuffer(
221 OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform2D>(cameraObject2d)
222 );
