207 void OnTick(TDeltaTime deltaTime) override {
208 OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform3D>(ballObject).RotateLocalSpace(deltaTime, { 0 ,1 ,0 });
209
210 OSK::ECS::CameraComponent3D& camera = OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::CameraComponent3D>(cameraObject);
211 OSK::ECS::Transform3D& cameraTransform = OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform3D>(cameraObject);
212
213 float forwardMovement = 0.0f;
214 float rightMovement = 0.0f;
215 auto newKs = OSK::Engine::GetWindow()->GetKeyboardState();
216 auto oldKs = OSK::Engine::GetWindow()->GetPreviousKeyboardState();
217 if (newKs->IsKeyDown(OSK::IO::Key::W))
218 forwardMovement += 1.0f;
219 if (newKs->IsKeyDown(OSK::IO::Key::S))
220 forwardMovement -= 1.0f;
221 if (newKs->IsKeyDown(OSK::IO::Key::A))
222 rightMovement -= 1.0f;
223 if (newKs->IsKeyDown(OSK::IO::Key::D))
224 rightMovement += 1.0f;
225
226 // Car
227 auto& carTransform = OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform3D>(ballObject);
228 if (newKs->IsKeyDown(OSK::IO::Key::UP))
229 carTransform.AddPosition(carTransform.GetForwardVector() * deltaTime * 2);
230 if (newKs->IsKeyDown(OSK::IO::Key::DOWN))
231 carTransform.AddPosition(carTransform.GetForwardVector() * deltaTime * -2);
232 if (newKs->IsKeyDown(OSK::IO::Key::LEFT))
233 carTransform.RotateWorldSpace(deltaTime * 2, { 0, 1, 0 });
234 if (newKs->IsKeyDown(OSK::IO::Key::RIGHT))
235 carTransform.RotateWorldSpace(deltaTime * 2, { 0, -1, 0 });
236
237 if (newKs->IsKeyDown(OSK::IO::Key::F11) && oldKs->IsKeyUp(OSK::IO::Key::F11))
238 OSK::Engine::GetWindow()->ToggleFullScreen();
239
240 if (newKs->IsKeyDown(OSK::IO::Key::F1) && oldKs->IsKeyUp(OSK::IO::Key::F1)) {
241 if (terrainMaterial == terrainMaterialFill)
242 terrainMaterial = terrainMaterialLine;
243 else
244 terrainMaterial = terrainMaterialFill;
245 }
246
247 if (newKs->IsKeyDown(OSK::IO::Key::ESCAPE))
248 this->Exit();
249
250 camera.Rotate(
251 (OSK::Engine::GetWindow()->GetMouseState()->GetPosition().X - OSK::Engine::GetWindow()->GetPreviousMouseState()->GetPosition().X),
252 (OSK::Engine::GetWindow()->GetMouseState()->GetPosition().Y - OSK::Engine::GetWindow()->GetPreviousMouseState()->GetPosition().Y)
253 );
254
255 cameraTransform.AddPosition(cameraTransform.GetForwardVector().GetNormalized() * forwardMovement * deltaTime);
256 cameraTransform.AddPosition(cameraTransform.GetRightVector().GetNormalized() * rightMovement * deltaTime);
257 camera.UpdateTransform(&cameraTransform);
258
259 uniformBuffer->ResetCursor();
260 uniformBuffer->MapMemory();
261 uniformBuffer->Write(camera.GetProjectionMatrix(cameraTransform));
262 uniformBuffer->Write(camera.GetViewMatrix());
263 uniformBuffer->Write(cameraTransform.GetPosition());
264 uniformBuffer->Unmap();
265
266 OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::CameraComponent2D>(cameraObject2d).UpdateUniformBuffer(
267 OSK::Engine::GetEntityComponentSystem()->GetComponent<OSK::ECS::Transform2D>(cameraObject2d)
268 );
269
270 }
