26MaterialSlotVulkan::MaterialSlotVulkan(const std::string& name, const MaterialLayout* layout)
27 : IMaterialSlot(layout, name) {
28
29 TSize swapchainCount = Engine::GetRenderer()->GetSwapchainImagesCount();
30
31 descriptorSets.Resize(swapchainCount);
32 bindings.Resize(3);
33
34 descLayout = new DescriptorLayoutVulkan(&layout->GetSlot(name));
35 pool = new DescriptorPoolVulkan(descLayout.GetValue(), swapchainCount);
36
37 VkDescriptorSetAllocateInfo allocInfo{};
38 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
39 allocInfo.descriptorPool = pool->GetPool();
40 allocInfo.descriptorSetCount = static_cast<uint32_t>(swapchainCount);
41
42 VkDescriptorSetLayout layouts[] = { descLayout->GetLayout(), descLayout->GetLayout(), descLayout->GetLayout() };
43 allocInfo.pSetLayouts = layouts;
44
45 VkResult result = vkAllocateDescriptorSets(GetDevice(), &allocInfo, descriptorSets.GetData());
46 OSK_ASSERT(result == VK_SUCCESS, "Error al crear descriptor sets.");
47}
