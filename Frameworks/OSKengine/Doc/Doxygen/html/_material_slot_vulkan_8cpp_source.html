24MaterialSlotVulkan::MaterialSlotVulkan(const std::string& name, const MaterialLayout* layout)
25 : IMaterialSlot(layout, name) {
26
27 TSize swapchainCount = Engine::GetRenderer()->GetSwapchainImagesCount();
28
29 descriptorSets.Resize(swapchainCount);
30 bindings.Resize(3);
31
32 descLayout = new DescriptorLayoutVulkan(&layout->GetSlot(name));
33 pool = new DescriptorPoolVulkan(descLayout.GetValue(), swapchainCount);
34
35 VkDescriptorSetAllocateInfo allocInfo{};
36 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
37 allocInfo.descriptorPool = pool->GetPool();
38 allocInfo.descriptorSetCount = static_cast<uint32_t>(swapchainCount);
39
40 VkDescriptorSetLayout layouts[] = { descLayout->GetLayout(), descLayout->GetLayout(), descLayout->GetLayout() };
41 allocInfo.pSetLayouts = layouts;
42
43 VkResult result = vkAllocateDescriptorSets(GetDevice(), &allocInfo, descriptorSets.GetData());
44 OSK_ASSERT(result == VK_SUCCESS, "Error al crear descriptor sets.");
45}
