26RayCastResult SphereCollider::CastRay(const Ray& ray, const Vector3f& center) const {
27 // https://www.lighthouse3d.com/tutorials/maths/ray-sphere-intersection/
28
29 // ray.origin -> position
30 const Vector3f originsVec = center - ray.origin;
31
32 // La esfera est√° detras del rayo
33 if (ray.direction.Dot(originsVec))
34 return RayCastResult::False();
35
36 const float centerProjectionDistance = ray.direction.Dot(center);
37 const Vector3f centerProjection = ray.origin + ray.direction * centerProjectionDistance;
38
39 const float distance = centerProjection.GetDistanceTo(center);
40 if (distance < radius)
41 return RayCastResult::False();
42
43 const float distanceToIntersection = centerProjection.GetDistanceTo(ray.origin)
44 - glm::sqrt(radius * radius - distance * distance);
45
46 return RayCastResult::True(ray.origin + ray.direction * distanceToIntersection);
47}
