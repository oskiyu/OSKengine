42GpuMemoryBlockVulkan::GpuMemoryBlockVulkan(TSize reservedSize, IGpu* device, GpuSharedMemoryType type, GpuBufferUsage bufferUSage)
43 : IGpuMemoryBlock(reservedSize, device, type, GpuMemoryUsage::BUFFER) {
44
45 VkBufferCreateInfo bufferInfo{};
46 bufferInfo.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
47 bufferInfo.size = reservedSize;
48 bufferInfo.usage = GetGpuBufferUsageVulkan(bufferUSage);
49 bufferInfo.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
50
51 VkResult result = vkCreateBuffer(device->As<GpuVulkan>()->GetLogicalDevice(), &bufferInfo, nullptr, &buffer);
52 OSK_ASSERT(result == VK_SUCCESS, "Error al crear buffer en la GPU.");

54 VkMemoryRequirements memRequirements;

59 memoryAllocateFlagsInfo.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_FLAGS_INFO;
60 memoryAllocateFlagsInfo.flags = GetMemoryAllocateFlags(bufferUSage);
61
62 VkMemoryAllocateInfo allocInfo{};
63 allocInfo.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
64 allocInfo.allocationSize = memRequirements.size;
65 allocInfo.memoryTypeIndex = GetMemoryType(memRequirements.memoryTypeBits, device->As<GpuVulkan>(), type);
66 allocInfo.pNext = &memoryAllocateFlagsInfo;
67
68 result = vkAllocateMemory(device->As<GpuVulkan>()->GetLogicalDevice(), &allocInfo, nullptr, &memory);
69 OSK_ASSERT(result == VK_SUCCESS, "Error al asignar memoria en la GPU.");

103VkMemoryAllocateFlags GpuMemoryBlockVulkan::GetMemoryAllocateFlags(GpuBufferUsage usage) {
104 VkMemoryAllocateFlags output = 0;
105
106 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE))
107 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
108
109 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_ACCELERATION_STRUCTURE_BUILDING))
110 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
111
112 if (EFTraits::HasFlag(usage, GpuBufferUsage::RT_SHADER_BINDING_TABLE))
113 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
114
115 if (Engine::GetRenderer()->SupportsRaytracing() &&
116 (EFTraits::HasFlag(usage, GpuBufferUsage::VERTEX_BUFFER) || EFTraits::HasFlag(usage, GpuBufferUsage::INDEX_BUFFER)))
117 output |= VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT;
118
119 return output;
120}

122uint32_t GpuMemoryBlockVulkan::GetMemoryType(uint32_t memoryTypeFilter, GpuVulkan* device, GpuSharedMemoryType type) {
123 auto nativeSharedMode = GetGpuSharedMemoryTypeVulkan(type);
124
125 for (uint32_t i = 0; i < device->GetInfo().memoryProperties.memoryTypeCount; i++)
126 if ((memoryTypeFilter & (1 << i)) && ((device->GetInfo().memoryProperties.memoryTypes[i].propertyFlags & nativeSharedMode) == nativeSharedMode))
127 return i;
128
129 OSK_ASSERT(false, "No se encontró memoria en la GPU comptible");
130
131 return -1;
