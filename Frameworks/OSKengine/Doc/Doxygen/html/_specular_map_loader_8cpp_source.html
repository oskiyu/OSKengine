32void SpecularMapLoader::Load(const std::string& assetFilePath, IAsset** asset) {
33 // Asset file.
34 nlohmann::json assetInfo = nlohmann::json::parse(FileIO::ReadFromFile(assetFilePath));
35
36 OSK_ASSERT(assetInfo.contains("file_type"), "Archivo de textura incorrecto: no se encuentra 'file_type'.");
37 OSK_ASSERT(assetInfo.contains("spec_ver"), "Archivo de textura incorrecto: no se encuentra 'spec_ver'.");
38 OSK_ASSERT(assetInfo.contains("name"), "Archivo de textura incorrecto: no se encuentra 'name'.");
39 OSK_ASSERT(assetInfo.contains("asset_type"), "Archivo de textura incorrecto: no se encuentra 'asset_type'.");
40 OSK_ASSERT(assetInfo.contains("raw_asset_path"), "Archivo de textura incorrecto: no se encuentra 'raw_asset_path'.");
41
42 std::string texturePath = assetInfo["raw_asset_path"];
43
44 // Original texture loading
45 int width = 0;
46 int height = 0;
47 int numChannels = 0;
48
49 const float* pixels = stbi_loadf(texturePath.c_str(), &width, &height, &numChannels, 4);
50
51 const Vector2ui originalImageSize = {
52 (uint32_t)width,
53 (uint32_t)height
54 };
55
56
57 // Carga de la imagen HDR original
58 const GpuImageUsage originalImageUsage = GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED;
59 const GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(originalImageSize, Format::RGBA32_SFLOAT, originalImageUsage);
60 UniquePtr<GpuImage> originalImage = Engine::GetRenderer()->GetAllocator()->CreateImage(imageInfo).GetPointer();
61
62 UploadImage(originalImage.GetPointer(), pixels, { originalImageSize.X, originalImageSize.Y, 1 });
63
64 // Creación del cubemap.
65 // Cada mip level representa el reflejo para una rugosidad específica.
66 const TSize maxMipLevel = 5;
67
68 GpuImageSamplerDesc sampler{};
69 sampler.mipMapMode = GpuImageMipmapMode::CUSTOM;
70 sampler.minMipLevel = 0;
71 sampler.maxMipLevel = maxMipLevel;
72 sampler.filteringType = GpuImageFilteringType::LIENAR;
73
74 OwnedPtr<GpuImage> targetCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(maxResolution,
75 Format::RGBA32_SFLOAT, GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP,
76 GpuSharedMemoryType::GPU_ONLY, sampler);
77
78 // TODO: renderizar a cubemap
79
80 // Preprocesado
81 OwnedPtr<ICommandList> cmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
82
83 for (TIndex mipLevel = 0; mipLevel < maxMipLevel; mipLevel++) {
84 const float roughness = (float)mipLevel / (maxMipLevel - 1.0f);
85 DrawPreFilter(targetCubemap.GetPointer(), cmdList.GetPointer(), mipLevel, roughness);
86 }
87}

89void SpecularMapLoader::DrawPreFilter(GpuImage* cubemap, ICommandList* cmdList, TIndex mipLevel, float roughness) {
90 struct {
91 glm::mat4 cameraProjection = SpecularMapLoader::cameraProjection;
92 glm::mat4 cameraView = glm::mat4(1.0f);
93 } renderInfo;
94
95 for (TIndex faceId = 0; faceId < 6; faceId++) {
96 cmdList->BeginGraphicsRenderpass(&cubemapRenderTarget);
97
98 const float sizeRatio = glm::pow(0.5f, mipLevel);
99 const Viewport viewport{
100 .rectangle = {
101 0, 0,
102 (uint32_t)(cubemapRenderTarget.GetSize().X * sizeRatio),
103 (uint32_t)(cubemapRenderTarget.GetSize().Y * sizeRatio)
104 }
105 };
106
107 cmdList->SetViewport(viewport);
108 cmdList->BindMaterial(prefilterMaterial);
109 cmdList->BindMaterialSlot(prefilterMaterialInstance->GetSlot("global"));
110
111 renderInfo.cameraView = cameraViews[faceId];
112 cmdList->PushMaterialConstants("roughness", roughness);
113
114 cmdList->BindVertexBuffer(cubemapModel->GetVertexBuffer());
115 cmdList->BindIndexBuffer(cubemapModel->GetIndexBuffer());
116 cmdList->DrawSingleInstance(cubemapModel->GetIndexCount());
117
118 cmdList->EndGraphicsRenderpass();
119
120 const TIndex resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
121
122 cmdList->SetGpuImageBarrier(cubemapRenderTarget.GetMainColorImage(resourceIndex),
123 GpuImageLayout::TRANSFER_SOURCE,
124 GpuBarrierInfo(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE),
125 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ),
126 GpuImageBarrierInfo{ .baseLayer = 0, .numLayers = 1, .baseMipLevel = mipLevel, .numMipLevels = 1 });
127
128 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(viewport.rectangle.GetRectangleSize());
129 copyInfo.destinationArrayLevel = faceId;
130 copyInfo.destinationMipLevel = mipLevel;
131 cmdList->CopyImageToImage(cubemapRenderTarget.GetMainColorImage(resourceIndex), cubemap, copyInfo);
132 }
133}

135void SpecularMapLoader::UploadImage(GpuImage* img, const float* pixels, const Vector3ui& size) const {
136 OwnedPtr<ICommandList> uploadCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
137 uploadCmdList->Reset();
138 uploadCmdList->Start();
139
140 uploadCmdList->SetGpuImageBarrier(img, GpuImageLayout::UNDEFINED, GpuImageLayout::TRANSFER_DESTINATION,
141 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE),
142 GpuImageBarrierInfo{ .baseLayer = 0, .numLayers = ALL_IMAGE_LAYERS, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
143
144 Engine::GetRenderer()->UploadImageToGpu(img, (TByte*)pixels, size.X * size.Y * size.Z * GetFormatNumberOfBytes(Format::RGBA32_SFLOAT), uploadCmdList.GetPointer());
145
146 uploadCmdList->SetGpuImageBarrier(img, GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::SAMPLED,
147 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE), GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
148 GpuImageBarrierInfo{ .baseLayer = 0, .numLayers = ALL_IMAGE_LAYERS, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
149
150 uploadCmdList->Close();
151 Engine::GetRenderer()->SubmitSingleUseCommandList(uploadCmdList);
152}
