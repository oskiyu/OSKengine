58void SpecularMapLoader::Load(const std::string& assetFilePath, IAsset** asset) {
59 SpecularMap* output = (SpecularMap*)*asset;
60
61 // Asset file.
62 nlohmann::json assetInfo = nlohmann::json::parse(FileIO::ReadFromFile(assetFilePath));
63
64 OSK_ASSERT(assetInfo.contains("file_type"), "Archivo de textura incorrecto: no se encuentra 'file_type'.");
65 OSK_ASSERT(assetInfo.contains("spec_ver"), "Archivo de textura incorrecto: no se encuentra 'spec_ver'.");
66 OSK_ASSERT(assetInfo.contains("name"), "Archivo de textura incorrecto: no se encuentra 'name'.");
67 OSK_ASSERT(assetInfo.contains("asset_type"), "Archivo de textura incorrecto: no se encuentra 'asset_type'.");
68 OSK_ASSERT(assetInfo.contains("raw_asset_path"), "Archivo de textura incorrecto: no se encuentra 'raw_asset_path'.");
69
70 std::string texturePath = assetInfo["raw_asset_path"];
71 output->SetName(assetInfo["name"]);
72
73 // Original texture loading
74 int width = 0;
75 int height = 0;
76 int numChannels = 0;
77
78 const float* pixels = stbi_loadf(texturePath.c_str(), &width, &height, &numChannels, 4);
79
80 const Vector2ui originalImageSize = {
81 (uint32_t)width,
82 (uint32_t)height
83 };
84
85
86 // Carga de la imagen HDR original
87 const GpuImageUsage originalImageUsage = GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED;
88 const GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(originalImageSize, Format::RGBA32_SFLOAT, originalImageUsage);
89 UniquePtr<GpuImage> originalImage = Engine::GetRenderer()->GetAllocator()->CreateImage(imageInfo).GetPointer();
90 originalImage->SetDebugName("Original Specular 2D");
91
92 UploadImage(originalImage.GetPointer(), pixels, { originalImageSize.X, originalImageSize.Y, 1 });
93
94 // Creaci√≥n del cubemap.
95 // Cada mip level representa el reflejo para una rugosidad espec√≠fica.
96 const TSize maxMipLevel = 5;
97
98 GpuImageSamplerDesc origianlSampler{};
99 origianlSampler.mipMapMode = GpuImageMipmapMode::NONE;
100
101 GpuImageSamplerDesc prefilterSampler{};
102 prefilterSampler.mipMapMode = GpuImageMipmapMode::CUSTOM;
103 prefilterSampler.minMipLevel = 0;
104 prefilterSampler.maxMipLevel = maxMipLevel;
105 prefilterSampler.filteringType = GpuImageFilteringType::LIENAR;
106
107 UniquePtr<GpuImage> originalCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
108 maxResolution,
109 Format::RGBA32_SFLOAT,
110 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
111 GpuSharedMemoryType::GPU_ONLY,
112 origianlSampler).GetPointer();
113
114 OwnedPtr<GpuImage> targetCubemap = Engine::GetRenderer()->GetAllocator()->CreateCubemapImage(
115 maxResolution,
116 Format::RGBA32_SFLOAT,
117 GpuImageUsage::COLOR | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP | GpuImageUsage::TRANSFER_DESTINATION,
118 GpuSharedMemoryType::GPU_ONLY,
119 prefilterSampler);
120
121 originalCubemap->SetDebugName("Original Specular Cubemap");
122 targetCubemap->SetDebugName("Prefiltered Specular Cubemap");
123
124 generationMaterialInstance->GetSlot("global")->SetGpuImage("image", originalImage->GetView(GpuImageViewConfig::CreateSampled_SingleMipLevel(0)));
125 prefilterMaterialInstance->GetSlot("global")->SetGpuImage("image", originalCubemap->GetView(GpuImageViewConfig::CreateSampled_Cubemap()));
126
127 generationMaterialInstance->GetSlot("global")->FlushUpdate();
128 prefilterMaterialInstance->GetSlot("global")->FlushUpdate();
129
130 // Renderizar a cubemap
131 OwnedPtr<ICommandList> originalDrawCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
132 originalDrawCmdList->SetDebugName("Specular Cubemap Cmd List");
133
134 originalDrawCmdList->Reset();
135 originalDrawCmdList->Start();
136
137 DrawOriginal(originalCubemap.GetPointer(), originalDrawCmdList.GetPointer());
138
139 originalDrawCmdList->Close();
140
141 Engine::GetRenderer()->SubmitSingleUseCommandList(originalDrawCmdList);
142
143 // Preprocesado
144
145 OwnedPtr<ICommandList> prefilterCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
146 prefilterCmdList->SetDebugName("Prefilter and LUT generation");
147
148 prefilterCmdList->Reset();
149 prefilterCmdList->Start();
150
151 for (TIndex mipLevel = 0; mipLevel < maxMipLevel; mipLevel++) {
152 const float roughness = (float)mipLevel / (maxMipLevel - 1.0f);
153 DrawPreFilter(targetCubemap.GetPointer(), prefilterCmdList.GetPointer(), mipLevel, roughness);
154 }
155
156 // LUT
157 GenerateLut(prefilterCmdList.GetPointer());
158 GpuImageCreateInfo lutCreateInfo = GpuImageCreateInfo::CreateDefault2D(
159 lookUpTable.GetSize(),
160 lookUpTable.GetTargetImage(0)->GetFormat(),
161 GpuImageUsage::SAMPLED | GpuImageUsage::TRANSFER_DESTINATION);
162 OwnedPtr<GpuImage> lut = Engine::GetRenderer()->GetAllocator()->CreateImage(lutCreateInfo);
163 lut->SetDebugName("Specular Look-Up Table");
164
165 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::TRANSFER_DESTINATION,
166 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT),
167 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE));
168
169 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(lookUpTable.GetSize());
170 prefilterCmdList->CopyImageToImage(lookUpTable.GetTargetImage(Engine::GetRenderer()->GetCurrentResourceIndex()), lut.GetPointer(), copyInfo);
171
172 prefilterCmdList->SetGpuImageBarrier(lut.GetPointer(), GpuImageLayout::SAMPLED,
173 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE),
174 GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ));
175
176 prefilterCmdList->Close();
177
178 Engine::GetRenderer()->SubmitSingleUseCommandList(prefilterCmdList);
179
180 output->_SetCubemapImage(targetCubemap);
181 output->_SetLookUpTable(lut);
182}

184void SpecularMapLoader::DrawOriginal(GRAPHICS::GpuImage* cubemap, GRAPHICS::ICommandList* cmdList) {
185 struct {
186 glm::mat4 cameraProjection = SpecularMapLoader::cameraProjection;
187 glm::mat4 cameraView = glm::mat4(1.0f);
188 } renderInfo;
189
190 cmdList->BindMaterial(generationMaterial);
191 cmdList->BindMaterialSlot(generationMaterialInstance->GetSlot("global"));
192
193 for (TIndex faceId = 0; faceId < 6; faceId++) {
194 const TIndex resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
195
196 cmdList->BeginGraphicsRenderpass(&cubemapRenderTarget);
197
198 const Viewport viewport{
199 .rectangle = {
200 0, 0,
201 (uint32_t)(cubemapRenderTarget.GetSize().X),
202 (uint32_t)(cubemapRenderTarget.GetSize().Y)
203 }
204 };
205
206 cmdList->SetViewport(viewport);
207 cmdList->SetScissor(viewport.rectangle);
208
209 renderInfo.cameraView = cameraViews[faceId];
210 cmdList->PushMaterialConstants("info", renderInfo);
211
212 cmdList->BindVertexBuffer(cubemapModel->GetVertexBuffer());
213 cmdList->BindIndexBuffer(cubemapModel->GetIndexBuffer());
214 cmdList->DrawSingleInstance(cubemapModel->GetIndexCount());
215
216 cmdList->EndGraphicsRenderpass();
217
218 cmdList->SetGpuImageBarrier(cubemapRenderTarget.GetMainColorImage(resourceIndex),
219 GpuImageLayout::TRANSFER_SOURCE,
220 GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
221 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ),
222 GpuImageBarrierInfo{ .baseLayer = 0, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
223
224 cmdList->SetGpuImageBarrier(cubemap,
225 GpuImageLayout::TRANSFER_DESTINATION,
226 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT),
227 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE),
228 GpuImageBarrierInfo{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
229
230 CopyImageInfo copyInfo = CopyImageInfo::CreateDefault2D(viewport.rectangle.GetRectangleSize());
231 copyInfo.destinationArrayLevel = faceId;
232 cmdList->CopyImageToImage(cubemapRenderTarget.GetMainColorImage(resourceIndex), cubemap, copyInfo);
233
234 cmdList->SetGpuImageBarrier(cubemap,
235 GpuImageLayout::SAMPLED,
236 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE),
237 GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
238 GpuImageBarrierInfo{ .baseLayer = faceId, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
239 }

303void SpecularMapLoader::GenerateLut(ICommandList* cmdList) {
304 const TIndex resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
305 GpuImage* img = lookUpTable.GetTargetImage(resourceIndex);
306
307 cmdList->SetGpuImageBarrier(img, GpuImageLayout::GENERAL,
308 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT),
309 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_WRITE));
310
311 cmdList->BindMaterial(lutGenerationMaterial);
312 cmdList->BindMaterialSlot(lutGenerationMaterialInstance->GetSlot("global"));
313
314 const Vector2ui imageResolution = lookUpTable.GetSize();
315 const Vector2ui threadBlockSize = { 8u, 8u };
316 const Vector2ui dispatchResolution = imageResolution / threadBlockSize;
317
318 cmdList->DispatchCompute({ dispatchResolution.X, dispatchResolution.Y, 1 });
319
320 cmdList->SetGpuImageBarrier(img, GpuImageLayout::TRANSFER_SOURCE,
321 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_WRITE),
322 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ));
323}
