20VkFormat GetSupportedFormat(const DynamicArray<VkFormat>& candidates, VkImageTiling tiling, VkFormatFeatureFlags features) {
21 for (VkFormat format : candidates) {
22 VkFormatProperties properties;
23 vkGetPhysicalDeviceFormatProperties(
24 OSK::Engine::GetRenderer()->As<RendererVulkan>()->GetGpu()->As<GpuVulkan>()->GetPhysicalDevice(),
25 format, &properties);
26
27 if (tiling == VK_IMAGE_TILING_LINEAR && (properties.linearTilingFeatures & features) == features)
28 return format;
29
30 if (tiling == VK_IMAGE_TILING_OPTIMAL && (properties.optimalTilingFeatures & features) == features)
31 return format;
32 }
33
34 return candidates.At(0);
35}
