73void ColliderRenderSystem::Render(GRAPHICS::ICommandList* commandList) {
74 if (cameraObject == EMPTY_GAME_OBJECT)
75 return;
76
77 const TSize resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
78
79 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
80 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
81
82 cameraUbos[resourceIndex]->MapMemory();
83 cameraUbos[resourceIndex]->Write(camera.GetProjectionMatrix());
84 cameraUbos[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
85 cameraUbos[resourceIndex]->Write(cameraTransform.GetPosition());
86 cameraUbos[resourceIndex]->Unmap();
87
88 // Informaci칩n por cada entidad.
89 struct RenderInfo {
90 glm::mat4 modelMatrix;
91 Color color;
92 } renderInfo {};
93
94 commandList->SetGpuImageBarrier(renderTarget.GetMainColorImage(resourceIndex),
95 GpuImageLayout::SAMPLED,
96 GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
97 GpuBarrierInfo(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE));
98
99 commandList->BeginGraphicsRenderpass(&renderTarget, Color::BLACK() * 0.0f);
100 SetupViewport(commandList);
101
102 commandList->StartDebugSection("Collision Render", Color::RED());
103
104 // Obtenemos todas las entidades que est치n colisionando,
105 // para poder cambiarles de color.
106 const auto& eventQueue = Engine::GetEcs()->GetEventQueue<CollisionEvent>();
107 std::set<GameObjectIndex> collisionObjects;
108 for (const CollisionEvent& ev : eventQueue) {
109 collisionObjects.insert(ev.firstEntity);
110 collisionObjects.insert(ev.secondEntity);
111 }
112
113 for (GameObjectIndex gameObject : GetObjects()) {
114 commandList->BindMaterial(material);
115 commandList->BindMaterialSlot(materialInstance->GetSlot("global"));
116
117 const Transform3D& originalTransform = Engine::GetEcs()->GetComponent<Transform3D>(gameObject);
118 Transform3D topLevelTransform = originalTransform;
119 topLevelTransform.SetRotation({});
120
121 const Collider& collider = Engine::GetEcs()->GetComponent<Collider>(gameObject);
122 const ITopLevelCollider* topLevelCollider = collider.GetTopLevelCollider();
123
124 if (collisionObjects.contains(gameObject))
125 renderInfo.color = Color::RED() * 0.75f;
126 else
127 renderInfo.color = Color::YELLOW() * 0.75f;
128
129 if (auto* box = dynamic_cast<const AxisAlignedBoundingBox*>(topLevelCollider)) {
130 topLevelTransform.SetScale(box->GetSize());
131 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
132 commandList->PushMaterialConstants("pushConstants", renderInfo);
133
134 commandList->BindVertexBuffer(cubeModel->GetVertexBuffer());
135 commandList->BindIndexBuffer(cubeModel->GetIndexBuffer());
136 commandList->DrawSingleInstance(cubeModel->GetIndexCount());
137 } else
138 if (auto* sphere = dynamic_cast<const SphereCollider*>(topLevelCollider)) {
139 topLevelTransform.SetScale(sphere->GetRadius());
140 renderInfo.modelMatrix = topLevelTransform.GetAsMatrix();
141 commandList->PushMaterialConstants("pushConstants", renderInfo);
142
143 commandList->BindVertexBuffer(sphereModel->GetVertexBuffer());
144 commandList->BindIndexBuffer(sphereModel->GetIndexBuffer());
145 commandList->DrawSingleInstance(sphereModel->GetIndexCount());
146 }
147
148 if (bottomLevelVertexBuffers.ContainsKey(gameObject)) {
149 const auto& vertexBuffers = bottomLevelVertexBuffers.Get(gameObject);
150 const auto& indexBuffers = bottomLevelIndexBuffers.Get(gameObject);
151
152 if (collisionObjects.contains(gameObject))
153 renderInfo.color = Color::RED();
154 else
155 renderInfo.color = Color::YELLOW();
156
157 renderInfo.modelMatrix = originalTransform.GetAsMatrix();
158 commandList->PushMaterialConstants("pushConstants", renderInfo);
159
160 commandList->BindMaterial(lowLevelMaterial);
161 for (TIndex i = 0; i < vertexBuffers.GetSize(); i++) {
162 commandList->BindVertexBuffer(vertexBuffers[i].GetPointer());
163 commandList->BindIndexBuffer(indexBuffers[i].GetPointer());
164 commandList->DrawSingleInstance(indexBuffers[i]->GetNumIndices());
165 }
166 }
167 }
168
169 commandList->EndDebugSection();
170
171 commandList->EndGraphicsRenderpass();
172}

185void ColliderRenderSystem::SetupBottomLevelModel(GameObjectIndex obj) {
186 const Collider& collider = Engine::GetEcs()->GetComponent<Collider>(obj);
187
188 if (bottomLevelVertexBuffers.ContainsKey(obj)) {
189 bottomLevelVertexBuffers.Get(obj).Empty();
190 bottomLevelIndexBuffers.Get(obj).Empty();
191 }
192
193 // Listas con los vertex e index buffers de la entidad.
194 // Cada ConvexVolume tendr치 uno.
195 DynamicArray<OwnedPtr<IGpuVertexBuffer>> vertexBuffers;
196 DynamicArray<OwnedPtr<IGpuIndexBuffer>> indexBuffers;
197
198
199 for (TIndex c = 0; c < collider.GetBottomLevelCollidersCount(); c++) {
200 const auto& blc = *collider.GetBottomLevelCollider(c)->As<ConvexVolume>();
201
202 // "Convertimos" los v칠rtices del collider
203 // en v칠rtices de renderizado.
204 DynamicArray<Vertex3D> vertices;
205 for (const auto& cVertex : blc.GetLocalSpaceVertices()){
206 Vertex3D vertex{};
207
208 vertex.position = cVertex;
209 vertex.normal = 0.0f;
210 vertex.color = Color::WHITE();
211 vertex.texCoords = 0.0f;
212
213 vertices.Insert(vertex);
214 }
215
216 // 칈ndices de v칠rtices.
217 DynamicArray<TIndexSize> indices;
218 for (const auto& face : blc.GetFaceIndices()) {
219 for (const auto index : face)
220 indices.Insert(index);
221
222 // Re-a침adimos el 칰ltimo v칠rtice de la geometr칤a,
223 // para "cerrar" el pol칤gono, creando una
224 // l칤nea entre el primer y el 칰ltimo v칠rtice.
225 indices.Insert(face[0]);
226 }
227
228 // Creamos los buffers.
229 vertexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateVertexBuffer(vertices, Vertex3D::GetVertexInfo()).GetPointer());
230 indexBuffers.Insert(Engine::GetRenderer()->GetAllocator()->CreateIndexBuffer(indices).GetPointer());
231 }
232
233 // Si no exist칤an las listas, las introducimos primero.
234 if (!bottomLevelVertexBuffers.ContainsKey(obj)) {
235 bottomLevelVertexBuffers.Insert(obj, {});
236 bottomLevelIndexBuffers.Insert(obj, {});
237 }
238
239 for (const auto& v : vertexBuffers)
240 bottomLevelVertexBuffers.Get(obj).Insert(v.GetPointer());
241
242 for (const auto& i : indexBuffers)
243 bottomLevelIndexBuffers.Get(obj).Insert(i.GetPointer());
244}
