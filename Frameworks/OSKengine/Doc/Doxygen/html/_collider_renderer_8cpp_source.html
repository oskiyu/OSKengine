24ColliderRenderSystem::ColliderRenderSystem() {
25 Signature signature{};
26 signature.SetTrue(Engine::GetEcs()->GetComponentType<Transform3D>());
27 signature.SetTrue(Engine::GetEcs()->GetComponentType<Collider>());
28 _SetSignature(signature);
29
30 // Material load
31 material = Engine::GetRenderer()->GetMaterialSystem()
32 ->LoadMaterial("Resources/Materials/Collision/collision_material.json");
33 materialInstance = material->CreateInstance().GetPointer();
34
35 // Asset load
36 cubeModel = Engine::GetAssetManager()->Load<Model3D>("", "ColliderRenderSystem");
37 sphereModel = Engine::GetAssetManager()->Load<Model3D>("", "ColliderRenderSystem");
38
39 // Material setup
40 const IGpuUniformBuffer* _cameraUbos[NUM_RESOURCES_IN_FLIGHT]{};
41 for (TIndex i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
42 cameraUbos[i] = Engine::GetRenderer()->GetAllocator()
43 ->CreateUniformBuffer(sizeof(glm::mat4) * 2 + sizeof(glm::vec4))
44 .GetPointer();
45 _cameraUbos[i] = cameraUbos[i].GetPointer();
46 }
47
48 materialInstance->GetSlot("global")->SetUniformBuffers("Camera", _cameraUbos);
49 materialInstance->GetSlot("global")->FlushUpdate();
50}

62void ColliderRenderSystem::Render(GRAPHICS::ICommandList* commandList) {
63 const TSize resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
64
65 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
66 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
67
68 cameraUbos[resourceIndex]->MapMemory();
69 cameraUbos[resourceIndex]->Write(camera.GetProjectionMatrix());
70 cameraUbos[resourceIndex]->Write(camera.GetViewMatrix(cameraTransform));
71 cameraUbos[resourceIndex]->Write(cameraTransform.GetPosition());
72 cameraUbos[resourceIndex]->Unmap();
73
74 // Render
75 struct RenderInfo {
76 glm::mat4 modelMatrix;
77 Color color;
78 } renderInfo;
79
80 commandList->SetGpuImageBarrier(renderTarget.GetMainColorImage(resourceIndex),
81 GpuImageLayout::SAMPLED,
82 GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
83 GpuBarrierInfo(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE));
84
85 commandList->BeginGraphicsRenderpass(&renderTarget, Color::BLACK() * 0.0f);
86 SetupViewport(commandList);
87
88 commandList->BindMaterial(material);
89 commandList->BindMaterialSlot(materialInstance->GetSlot("global"));
90
91 for (GameObjectIndex gameObject : GetObjects()) {
92 Transform3D transform = Engine::GetEcs()->GetComponent<Transform3D>(gameObject);
93 // transform.SetRotation(Quaternion::CreateFromEulerAngles(0.0f));
94
95 const Collider& collider = Engine::GetEcs()->GetComponent<Collider>(gameObject);
96 const ITopLevelCollider* topLevelCollider = collider.GetTopLevelCollider();
97
98 renderInfo.color = Color::RED();
99
100 commandList->PushMaterialConstants("Model", renderInfo);
101
102 if (auto* box = reinterpret_cast<const AxisAlignedBoundingBox*>(topLevelCollider)) {
103 transform.SetScale(box->GetSize());
104 renderInfo.modelMatrix = transform.GetAsMatrix();
105
106 commandList->BindVertexBuffer(cubeModel->GetVertexBuffer());
107 commandList->BindIndexBuffer(cubeModel->GetIndexBuffer());
108 commandList->DrawSingleInstance(cubeModel->GetIndexCount());
109 } else
110 if (auto* sphere = reinterpret_cast<const SphereCollider*>(topLevelCollider)) {
111 transform.SetScale(sphere->GetRadius());
112 renderInfo.modelMatrix = transform.GetAsMatrix();
113
114 commandList->BindVertexBuffer(sphereModel->GetVertexBuffer());
115 commandList->BindIndexBuffer(sphereModel->GetIndexBuffer());
116 commandList->DrawSingleInstance(sphereModel->GetIndexCount());
117 }
118
119 for (TIndex i = 0; i < collider.GetBottomLevelCollidersCount(); i++) {
120 // TODO
121
122 // const IBottomLevelCollider* bottomLevelCollider = collider.GetBottomLevelCollider(i);
123 // const ConvexVolume*;
124 }
125 }
126
127 commandList->EndGraphicsRenderpass();
128}
