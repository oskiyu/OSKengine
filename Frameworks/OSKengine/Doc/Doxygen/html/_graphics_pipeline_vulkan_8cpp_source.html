33void GraphicsPipelineVulkan::Create(const MaterialLayout* materialLayout, IGpu* device, const PipelineCreateInfo& info, const VertexInfo& vertexInfo) {
34 gpu = device;
35
36 layout = new PipelineLayoutVulkan(materialLayout);
37
38 LoadVertexShader(info.vertexPath);
39 LoadFragmentShader(info.fragmentPath);
40
41 if (info.tesselationControlPath != "") {
42 LoadTesselationControlShader(info.tesselationControlPath);
43 LoadTesselationEvaluationShader(info.tesselationEvaluationPath);
44 }
45
46 // Información básica del input del pipeline.
47 VkPipelineInputAssemblyStateCreateInfo inputAssemblyInfo{};
48 inputAssemblyInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_INPUT_ASSEMBLY_STATE_CREATE_INFO;
49 if (info.tesselationControlPath == "")
50 inputAssemblyInfo.topology = VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST;
51 else
52 inputAssemblyInfo.topology = VK_PRIMITIVE_TOPOLOGY_PATCH_LIST;
53 inputAssemblyInfo.primitiveRestartEnable = VK_FALSE;
54
55 // Vertex
56 VkVertexInputBindingDescription vertexBindingDesc = GetBindingDescription(vertexInfo);
57 DynamicArray<VkVertexInputAttributeDescription> vertexAttribDescription = GetAttributeDescription(vertexInfo);
58
59 VkPipelineVertexInputStateCreateInfo vertexInputInfo{};
60 vertexInputInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
61 vertexInputInfo.vertexBindingDescriptionCount = 1;
62 vertexInputInfo.pVertexBindingDescriptions = &vertexBindingDesc;
63 vertexInputInfo.vertexAttributeDescriptionCount = static_cast<uint32_t>(vertexAttribDescription.GetSize());
64 vertexInputInfo.pVertexAttributeDescriptions = vertexAttribDescription.GetData();
65
66 // Color blending.
67 const VkPipelineColorBlendAttachmentState colorBlendAttachment = GetColorBlendInfo(info);
68
69 DynamicArray<VkPipelineColorBlendAttachmentState> colorBlends;
70 for (TSize i = 0; i < info.formats.GetSize(); i++)
71 colorBlends.Insert(colorBlendAttachment);
72
73 VkPipelineColorBlendStateCreateInfo colorBlendCreateInfo{};
74 colorBlendCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_COLOR_BLEND_STATE_CREATE_INFO;
75 colorBlendCreateInfo.logicOpEnable = VK_FALSE;
76 colorBlendCreateInfo.logicOp = VK_LOGIC_OP_COPY;
77 colorBlendCreateInfo.attachmentCount = colorBlends.GetSize();
78 colorBlendCreateInfo.pAttachments = colorBlends.GetData();
79 colorBlendCreateInfo.blendConstants[0] = 0.0f;
80 colorBlendCreateInfo.blendConstants[1] = 0.0f;
81 colorBlendCreateInfo.blendConstants[2] = 0.0f;
82 colorBlendCreateInfo.blendConstants[3] = 0.0f;
83
84 // Rasterizer
85 VkPipelineRasterizationStateCreateInfo rasterizerInfo = GetResterizerInfo(info);
86
87 // Depth-stencil.
88 VkPipelineDepthStencilStateCreateInfo depthInfo = GetDepthInfo(info);
89
90 VkPipelineMultisampleStateCreateInfo msaaCreateInfo = GetMsaaInfo(info, *gpu->As<GpuVulkan>());
91
92 VkPipelineTessellationStateCreateInfo tesselationInfo = GetTesselationInfo(info);
93
94 // Estructuras dinámicas
95 const VkDynamicState states[] = {
96 VK_DYNAMIC_STATE_VIEWPORT,
97 VK_DYNAMIC_STATE_SCISSOR
98 };
99
100 VkPipelineDynamicStateCreateInfo dynamicCreateInfo{};
101 dynamicCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_DYNAMIC_STATE_CREATE_INFO;
102 dynamicCreateInfo.pDynamicStates = states;
103 dynamicCreateInfo.dynamicStateCount = _countof(states);
104 dynamicCreateInfo.flags = 0;
105
106 // Viewport (será dinámico)
107 VkViewport viewport{};
108 VkRect2D scissor{};
109
110 VkPipelineViewportStateCreateInfo viewportInfo{};
111 viewportInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
112 viewportInfo.viewportCount = 1;
113 viewportInfo.pViewports = &viewport;
114 viewportInfo.scissorCount = 1;
115 viewportInfo.pScissors = &scissor;
116
117 // Renderpass
118 DynamicArray<VkFormat> colorFormats;
119 for (const auto format : info.formats)
120 colorFormats.Insert(GetFormatVulkan(format));
121
122 VkPipelineRenderingCreateInfoKHR renderingCreateInfo{};
123 renderingCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_RENDERING_CREATE_INFO_KHR;
124 renderingCreateInfo.colorAttachmentCount = colorFormats.GetSize();
125 renderingCreateInfo.pColorAttachmentFormats = colorFormats.GetData();
126 renderingCreateInfo.depthAttachmentFormat = GetFormatVulkan(Format::D32S8_SFLOAT_SUINT);
127 renderingCreateInfo.stencilAttachmentFormat = GetFormatVulkan(Format::D32S8_SFLOAT_SUINT);
128
129 // Pipeline
130 VkGraphicsPipelineCreateInfo pipelineCreateInfo{};
131 pipelineCreateInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
132 pipelineCreateInfo.stageCount = shaderStagesInfo.GetSize();
133 pipelineCreateInfo.pStages = shaderStagesInfo.GetData();
134 pipelineCreateInfo.pVertexInputState = &vertexInputInfo;
135 pipelineCreateInfo.pInputAssemblyState = &inputAssemblyInfo;
136 pipelineCreateInfo.pViewportState = &viewportInfo;
137 pipelineCreateInfo.pRasterizationState = &rasterizerInfo;
138 pipelineCreateInfo.pMultisampleState = &msaaCreateInfo;
139 pipelineCreateInfo.pDepthStencilState = &depthInfo;
140 pipelineCreateInfo.pColorBlendState = &colorBlendCreateInfo;
141 pipelineCreateInfo.pDynamicState = &dynamicCreateInfo;
142 if (info.tesselationControlPath == "")
143 pipelineCreateInfo.pTessellationState = nullptr;
144 else
145 pipelineCreateInfo.pTessellationState = &tesselationInfo;
146 pipelineCreateInfo.layout = layout->As<PipelineLayoutVulkan>()->GetLayout();
147 pipelineCreateInfo.renderPass = VK_NULL_HANDLE;
148 pipelineCreateInfo.subpass = 0;
149 pipelineCreateInfo.basePipelineHandle = nullptr;
150 pipelineCreateInfo.basePipelineIndex = -1;
151 pipelineCreateInfo.pNext = &renderingCreateInfo;
152
153 VkResult result = vkCreateGraphicsPipelines(gpu->As<GpuVulkan>()->GetLogicalDevice(),
154 nullptr, 1, &pipelineCreateInfo, nullptr, &pipeline);
155 OSK_ASSERT(result == VK_SUCCESS, "Error al crear el pipeline.");
156}

158void GraphicsPipelineVulkan::LoadVertexShader(const std::string& path) {
159 VkShaderModule shaderModule = VK_NULL_HANDLE;
160
161 // Lee el código SPIR-V.
162 const DynamicArray<char> vertexCode = IO::FileIO::ReadBinaryFromFile(path);
163
164 VkShaderModuleCreateInfo createInfo{};
165 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
166 createInfo.codeSize = vertexCode.GetSize();
167 createInfo.pCode = reinterpret_cast<const uint32_t*>(vertexCode.GetData());
168
169 VkResult result = vkCreateShaderModule(gpu->As<GpuVulkan>()->GetLogicalDevice(),
170 &createInfo, nullptr, &shaderModule);
171 OSK_ASSERT(result == VK_SUCCESS, "No se ha podido cargar el vertex shader.");
172
173 // Insertar el stage en el array, para poder insertarlo al crear el pipeline.
174 VkPipelineShaderStageCreateInfo vertexShaderStageInfo{};
175 vertexShaderStageInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
176 vertexShaderStageInfo.stage = VK_SHADER_STAGE_VERTEX_BIT;
177 vertexShaderStageInfo.module = shaderModule;
178 vertexShaderStageInfo.pName = "main";
179
180 shaderStagesInfo.Insert(vertexShaderStageInfo);
181 shaderModulesToDelete.Insert(shaderModule);
182}
