52void IRenderer::UploadLayeredImageToGpu(GpuImage* destination, const TByte* data, TSize numBytes, TSize numLayers, ICommandList* cmdList) {
53 TSize gpuNumBytes = destination->GetPhysicalNumberOfBytes() * numLayers;
54
55 const TByte* uploadableData = data;
56
57 auto stagingBuffer = GetAllocator()->CreateStagingBuffer(gpuNumBytes);
58 stagingBuffer->MapMemory();
59 stagingBuffer->Write(uploadableData, gpuNumBytes);
60 stagingBuffer->Unmap();
61
62 TSize offsetPerIteration = gpuNumBytes / numLayers;
63 for (TSize i = 0; i < numLayers; i++)
64 cmdList->CopyBufferToImage(stagingBuffer.GetPointer(), destination, i, offsetPerIteration * i);
65
66 cmdList->RegisterStagingBuffer(stagingBuffer);
67}
