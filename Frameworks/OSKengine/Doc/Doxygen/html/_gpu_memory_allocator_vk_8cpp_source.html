70 TSize numBytes = GetFormatNumberOfBytes(info.format);
71
72 switch (info.dimension) {

74 case OSK::GRAPHICS::GpuImageDimension::d2D: numBytes *= info.resolution.X * info.resolution.Y; break;

82 case OSK::GRAPHICS::GpuImageDimension::d2D: finalImageSize = { info.resolution.X , info.resolution.Y, 1 }; break;
83 }
84
85 if (finalImageSize.X == 0)
86 finalImageSize.X = 1;
87 if (finalImageSize.Y == 0)
88 finalImageSize.Y = 1;
89
90 VkSampler sampler = VK_NULL_HANDLE;
91
92 GpuImageVk* output = new GpuImageVk(info.resolution, info.dimension, info.usage, info.numLayers, info.format, info.msaaSamples, info.samplerDesc);
93
94 TSize numMipLevels = 0;
95 switch (info.samplerDesc.mipMapMode) {
96 case GpuImageMipmapMode::AUTO:
97 numMipLevels = output->GetMipLevels();
98 break;
99 case GpuImageMipmapMode::CUSTOM:
100 numMipLevels = info.samplerDesc.maxMipLevel;
101 break;
102 case GpuImageMipmapMode::NONE:
103 numMipLevels = 1;
104 break;
