62void CommandListVulkan::TransitionImageLayout(GpuImage* image, GpuImageLayout previous, GpuImageLayout next, TSize baseLayer, TSize numLayers, TSize baseMipLevel, TSize numMipLevels) {
63 VkImageMemoryBarrier barrier{};
64 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
65 barrier.oldLayout = GetGpuImageLayoutVulkan(previous);
66 barrier.newLayout = GetGpuImageLayoutVulkan(next);
67 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
68 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
69 barrier.image = image->As<GpuImageVulkan>()->GetImage();
70 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
71 barrier.subresourceRange.baseMipLevel = baseMipLevel;
72 barrier.subresourceRange.levelCount = numMipLevels == 0 ? VK_REMAINING_MIP_LEVELS : numMipLevels;
73 barrier.subresourceRange.baseArrayLayer = baseLayer;
74 barrier.subresourceRange.layerCount = numLayers;
75 barrier.srcAccessMask = 0;
76 barrier.dstAccessMask = 0;
77
78 VkPipelineStageFlags sourceStage{};
79 VkPipelineStageFlags destinationStage{};
80
81 switch (GetGpuImageLayoutVulkan(previous)) {
82 case VK_IMAGE_LAYOUT_UNDEFINED:
83 barrier.srcAccessMask = 0;
84 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
85 break;
86
87 case VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL:
88 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
89 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
90 break;
91
92 case VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL:
93 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
94 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
95 break;
96
97 case VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL:
98 barrier.srcAccessMask = VK_ACCESS_SHADER_READ_BIT;
99 sourceStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
100 break;
101
102 case VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL:
103 barrier.srcAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
104 sourceStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
105 break;
106
107 case VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL:/*/*/
108 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT | VK_IMAGE_ASPECT_STENCIL_BIT;
109 barrier.srcAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
110 sourceStage = VK_PIPELINE_STAGE_LATE_FRAGMENT_TESTS_BIT;
111 break;
112
113 case VK_IMAGE_LAYOUT_GENERAL:
114 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
115 barrier.srcAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
116 sourceStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
117
118 default:
119 OSK_ASSERT(false, "Layout no soportado.");
120 break;
121 }
122
123 switch (GetGpuImageLayoutVulkan(next)) {
124 case VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL:
125 barrier.dstAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
126 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
127 break;
128
129 case VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL:
130 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
131 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
132 break;
133
134 case VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL:
135 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
136 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
137 break;
138
139 case VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL:
140 barrier.dstAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
141 destinationStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
142 break;
143
144 case VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL:
145 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT | VK_IMAGE_ASPECT_STENCIL_BIT;
146 barrier.dstAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
147 destinationStage = VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT;
148 break;
149
150 case VK_IMAGE_LAYOUT_PRESENT_SRC_KHR:
151 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
152 destinationStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
153 break;
154
155 case VK_IMAGE_LAYOUT_GENERAL:
156 destinationStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
157 break;
158
159 default:
160 OSK_ASSERT(false, "Layout no soportado.");
161 break;
162 }
163
164 vkCmdPipelineBarrier(commandBuffers[GetCommandListIndex()], sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
165
166 image->SetLayout(next);
167}

169void CommandListVulkan::CopyBufferToImage(const GpuDataBuffer* source, GpuImage* dest, TSize layer, TSize offset) {
170 VkBufferImageCopy region{};
171
172 region.bufferOffset = source->GetMemorySubblock()->GetOffsetFromBlock() + offset;
173 region.bufferRowLength = 0;
174 region.bufferImageHeight = 0;
175
176 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
177 region.imageSubresource.mipLevel = 0;
178 region.imageSubresource.baseArrayLayer = layer;
179 region.imageSubresource.layerCount = 1;
180
181 region.imageOffset = { 0, 0, 0 };
182 region.imageExtent = {
183 dest->GetSize().X,
184 dest->GetSize().Y,
185 1
186 };
187
188 vkCmdCopyBufferToImage(commandBuffers[GetCommandListIndex()],
189 source->GetMemoryBlock()->As<GpuMemoryBlockVulkan>()->GetVulkanBuffer(),
190 dest->As<GpuImageVulkan>()->GetImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
191
192 // Mip levels
193 Vector2i mipSize = { static_cast<int>(dest->GetSize().X), static_cast<int>(dest->GetSize().Y) };
194 // El nivel 0 ya está lleno.
195 for (TSize mipLevel = 1; mipLevel < dest->GetMipLevels(); mipLevel++) {
196 TransitionImageLayout(dest, GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::TRANSFER_SOURCE, layer, 1, mipLevel - 1, 1);
197
198 VkImageBlit blit{};
199
200 blit.srcOffsets[0] = { 0, 0, 0 };
201 blit.srcOffsets[1] = { mipSize.X, mipSize.Y, 1 };
202
203 // Origen: mip level anterior.
204 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
205 blit.srcSubresource.mipLevel = mipLevel - 1;
206 blit.srcSubresource.baseArrayLayer = layer;
207 blit.srcSubresource.layerCount = 1;
208
209 blit.dstOffsets[0] = { 0, 0, 0 };
210 blit.dstOffsets[1] = { mipSize.X > 1 ? mipSize.X / 2 : 1, mipSize.Y > 1 ? mipSize.Y / 2 : 1, 1 };
211
212 // Destino.
213 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
214 blit.dstSubresource.mipLevel = mipLevel;
215 blit.dstSubresource.baseArrayLayer = layer;
216 blit.dstSubresource.layerCount = 1;
217
218 vkCmdBlitImage(commandBuffers[GetCommandListIndex()],
219 dest->As<GpuImageVulkan>()->GetImage(), VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
220 dest->As<GpuImageVulkan>()->GetImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
221 1, &blit, VK_FILTER_LINEAR);
222
223 if (mipSize.X > 1)
224 mipSize.X /= 2;
225 if (mipSize.Y > 1)
226 mipSize.Y /= 2;
227 }
228 TransitionImageLayout(dest, GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::TRANSFER_SOURCE, layer, 1, dest->GetMipLevels() - 1, 1);
229
230 TransitionImageLayout(dest, GpuImageLayout::TRANSFER_SOURCE, GpuImageLayout::TRANSFER_DESTINATION, layer, 1, 0, 0);
231}
