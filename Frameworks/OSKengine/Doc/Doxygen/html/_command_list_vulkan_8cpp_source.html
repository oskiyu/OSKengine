66void CommandListVulkan::SetGpuImageBarrier(GpuImage* image, GpuImageLayout previousLayout, GpuImageLayout nextLayout, GpuBarrierInfo previous, GpuBarrierInfo next, const GpuImageBarrierInfo& prevImageInfo) {
67 VkImageMemoryBarrier barrier{};
68 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
69 barrier.oldLayout = GetGpuImageLayoutVulkan(previousLayout);
70 barrier.newLayout = GetGpuImageLayoutVulkan(nextLayout);
71 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
72 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
73 barrier.image = image->As<GpuImageVulkan>()->GetVkImage();
74 barrier.subresourceRange.aspectMask = 0;
75 barrier.subresourceRange.baseMipLevel = prevImageInfo.baseMipLevel;
76 barrier.subresourceRange.levelCount = prevImageInfo.numMipLevels;
77 barrier.subresourceRange.baseArrayLayer = prevImageInfo.baseLayer;
78 barrier.subresourceRange.layerCount = prevImageInfo.numLayers;
79
80 if (EFTraits::HasFlag(prevImageInfo.channel, SampledChannel::COLOR))
81 barrier.subresourceRange.aspectMask = barrier.subresourceRange.aspectMask | VK_IMAGE_ASPECT_COLOR_BIT;
82 if (EFTraits::HasFlag(prevImageInfo.channel, SampledChannel::DEPTH))
83 barrier.subresourceRange.aspectMask = barrier.subresourceRange.aspectMask | VK_IMAGE_ASPECT_DEPTH_BIT;
84 if (EFTraits::HasFlag(prevImageInfo.channel, SampledChannel::STENCIL))
85 barrier.subresourceRange.aspectMask = barrier.subresourceRange.aspectMask | VK_IMAGE_ASPECT_STENCIL_BIT;
86
87 barrier.srcAccessMask = GetPipelineAccess(previous.accessStage);
88 barrier.dstAccessMask = GetPipelineAccess(next.accessStage);
89
90 const VkPipelineStageFlags sourceStage = GetPipelineStage(previous.stage);
91 const VkPipelineStageFlags destinationStage = GetPipelineStage(next.stage);
92
93 vkCmdPipelineBarrier(commandBuffers[GetCommandListIndex()], sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
94}

96void CommandListVulkan::CopyBufferToImage(const GpuDataBuffer* source, GpuImage* dest, TSize layer, TSize offset) {
97 VkBufferImageCopy region{};
98
99 region.bufferOffset = source->GetMemorySubblock()->GetOffsetFromBlock() + offset;
100 region.bufferRowLength = 0;
101 region.bufferImageHeight = 0;
102
103 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
104 region.imageSubresource.mipLevel = 0;
105 region.imageSubresource.baseArrayLayer = layer;
106 region.imageSubresource.layerCount = 1;
107
108 region.imageOffset = { 0, 0, 0 };
109 region.imageExtent = {
110 dest->GetSize().X,
111 dest->GetSize().Y,
112 1
113 };
114
115 // Layout[0] = TRANSFER_DEST
116 vkCmdCopyBufferToImage(commandBuffers[GetCommandListIndex()],
117 source->GetMemoryBlock()->As<GpuMemoryBlockVulkan>()->GetVulkanBuffer(),
118 dest->As<GpuImageVulkan>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
119
120 // Mip levels
121 Vector2i mipSize = { static_cast<int>(dest->GetSize().X), static_cast<int>(dest->GetSize().Y) };
122 // El nivel 0 ya está lleno.
123 for (TSize mipLevel = 1; mipLevel < dest->GetMipLevels(); mipLevel++) {
124
125 // Mip anterior: layout DESTINATION -> SOURCE
126 SetGpuImageBarrier(dest, GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::TRANSFER_SOURCE,
127 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ),
128 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = mipLevel - 1, .numMipLevels = 1 });
129
130 VkImageBlit blit{};
131
132 blit.srcOffsets[0] = { 0, 0, 0 };
133 blit.srcOffsets[1] = { mipSize.X, mipSize.Y, 1 };
134
135 // Origen: mip level anterior.
136 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
137 blit.srcSubresource.mipLevel = mipLevel - 1;
138 blit.srcSubresource.baseArrayLayer = layer;
139 blit.srcSubresource.layerCount = 1;
140
141 blit.dstOffsets[0] = { 0, 0, 0 };
142 blit.dstOffsets[1] = { mipSize.X > 1 ? mipSize.X / 2 : 1, mipSize.Y > 1 ? mipSize.Y / 2 : 1, 1 };
143
144 // Destino.
145 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
146 blit.dstSubresource.mipLevel = mipLevel;
147 blit.dstSubresource.baseArrayLayer = layer;
148 blit.dstSubresource.layerCount = 1;
149
150 // Mip actual: layout DESTINATION
151 vkCmdBlitImage(commandBuffers[GetCommandListIndex()],
152 dest->As<GpuImageVulkan>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
153 dest->As<GpuImageVulkan>()->GetVkImage(), VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
154 1, &blit, VK_FILTER_LINEAR);
155
156 if (mipSize.X > 1)
157 mipSize.X /= 2;
158 if (mipSize.Y > 1)
159 mipSize.Y /= 2;
160 }
161
162 // Establecer layout del último nivel = TRANSFER_SOURCE
163 SetGpuImageBarrier(dest, GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::TRANSFER_SOURCE,
164 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ),
165 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = dest->GetMipLevels() - 1, .numMipLevels = 1 });
166
167 // Establecer todo TRANSFER_SOURCE -> TRANSFER_DESTINATION
168 SetGpuImageBarrier(dest, GpuImageLayout::TRANSFER_SOURCE, GpuImageLayout::TRANSFER_DESTINATION,
169 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_READ), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE),
170 { .baseLayer = layer, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS });
171}

214void CommandListVulkan::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color) {
215 const Vector2ui targetSize = { colorImages[0].targetImage->GetSize().X, colorImages[0].targetImage->GetSize().Y };
216
217 for (const auto& img : colorImages) {
218 SetGpuImageBarrier(img.targetImage, GpuImageLayout::UNDEFINED, GpuImageLayout::COLOR_ATTACHMENT,
219 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT), GpuBarrierInfo(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE),
220 { .baseLayer = img.arrayLevel, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS, });
221 }
222
223 DynamicArray<VkRenderingAttachmentInfo> colorAttachments = DynamicArray<VkRenderingAttachmentInfo>::CreateResizedArray(colorImages.GetSize());
224 for (TSize i = 0; i < colorImages.GetSize(); i++) {
225 colorAttachments[i] = {};
226 colorAttachments[i].sType = VK_STRUCTURE_TYPE_RENDERING_ATTACHMENT_INFO_KHR;
227 colorAttachments[i].loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
228 colorAttachments[i].storeOp = VK_ATTACHMENT_STORE_OP_STORE;
229 colorAttachments[i].imageLayout = VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL;
230 colorAttachments[i].imageView = colorImages[i].targetImage->GetView(SampledChannel::COLOR, SampledArrayType::SINGLE_LAYER, colorImages[i].arrayLevel, 1, ViewUsage::COLOR_TARGET)->As<GpuImageViewVulkan>()->GetVkView();
231 colorAttachments[i].resolveMode = VK_RESOLVE_MODE_NONE;
232 colorAttachments[i].resolveImageLayout = VK_IMAGE_LAYOUT_UNDEFINED;
233 colorAttachments[i].resolveImageView = VK_NULL_HANDLE;
234 colorAttachments[i].clearValue.color = { color.Red, color.Green, color.Blue, color.Alpha };
235 }
236
237 OSK_ASSERT(colorAttachments.GetSize() == colorImages.GetSize(), "Error al iniciar el renderpass.");
238
239 VkRenderingAttachmentInfo depthAttachment{};
240 depthAttachment.sType = VK_STRUCTURE_TYPE_RENDERING_ATTACHMENT_INFO_KHR;
241 depthAttachment.loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
242 depthAttachment.storeOp = VK_ATTACHMENT_STORE_OP_STORE;
243 depthAttachment.imageLayout = VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
244 depthAttachment.imageView = depthImage.targetImage->GetView(SampledChannel::DEPTH | SampledChannel::STENCIL, SampledArrayType::SINGLE_LAYER, depthImage.arrayLevel, 1, ViewUsage::DEPTH_STENCIL_TARGET)
245 ->As<GpuImageViewVulkan>()->GetVkView();
246 depthAttachment.resolveMode = VK_RESOLVE_MODE_NONE;
247 depthAttachment.resolveImageLayout = VK_IMAGE_LAYOUT_UNDEFINED;
248 depthAttachment.resolveImageView = VK_NULL_HANDLE;
249 depthAttachment.clearValue.depthStencil = { 1.0f, 0 };
250
251 VkRenderingInfo renderpassInfo{};
252 renderpassInfo.sType = VK_STRUCTURE_TYPE_RENDERING_INFO;
253 renderpassInfo.renderArea = { 0, 0, targetSize.X, targetSize.Y };
254 renderpassInfo.layerCount = 1;
255 renderpassInfo.colorAttachmentCount = colorAttachments.GetSize();
256 renderpassInfo.pColorAttachments = colorAttachments.GetData();
257 renderpassInfo.pDepthAttachment = &depthAttachment;
258 renderpassInfo.pStencilAttachment = &depthAttachment;
259
260 RendererVulkan::pvkCmdBeginRendering(commandBuffers.At(GetCommandListIndex()), &renderpassInfo);
261
262 currentColorImages = colorImages;
263 currentDepthImage = depthImage;
264}

266void CommandListVulkan::EndGraphicsRenderpass() {
267 RendererVulkan::pvkCmdEndRendering(commandBuffers[GetCommandListIndex()]);
268
269 const bool isFinal = currentRenderpassType == RenderpassType::FINAL;
270
271 const GpuImageLayout finalLayout = isFinal
272 ? GpuImageLayout::PRESENT
273 : GpuImageLayout::SAMPLED;
274
275 for (const auto& img : currentColorImages) {
276 SetGpuImageBarrier(img.targetImage, GpuImageLayout::COLOR_ATTACHMENT, finalLayout,
277 GpuBarrierInfo(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE), GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ),
278 { .baseLayer = img.arrayLevel, .numLayers = 1, .baseMipLevel = 0, .numMipLevels = ALL_MIP_LEVELS, });
279 }
280
281 if (currentRenderpassType != RenderpassType::INTERMEDIATE)
282 currentRenderpassType = RenderpassType::INTERMEDIATE;
283}

285void CommandListVulkan::BindMaterial(Material* material) {
286 currentMaterial = material;
287
288 switch (material->GetMaterialType()) {
289 case MaterialType::GRAPHICS: {
290 DynamicArray<Format> colorFormats{};
291 for (const auto& colorImg : currentColorImages)
292 colorFormats.Insert(colorImg.targetImage->GetFormat());
293
294 currentPipeline.graphics = material->GetGraphicsPipeline(colorFormats);
295
296 vkCmdBindPipeline(commandBuffers[GetCommandListIndex()], VK_PIPELINE_BIND_POINT_GRAPHICS, currentPipeline.graphics->As<GraphicsPipelineVulkan>()->GetPipeline());
297 }
298 break;
299
300 case MaterialType::RAYTRACING:
301 currentPipeline.raytracing = material->GetRaytracingPipeline();
302 vkCmdBindPipeline(commandBuffers[GetCommandListIndex()], VK_PIPELINE_BIND_POINT_RAY_TRACING_KHR, currentPipeline.raytracing->As<RaytracingPipelineVulkan>()->GetPipeline());
303
304 break;
305
306 case MaterialType::COMPUTE:
307 currentPipeline.compute = material->GetComputePipeline();
308 vkCmdBindPipeline(commandBuffers[GetCommandListIndex()], VK_PIPELINE_BIND_POINT_COMPUTE, currentPipeline.compute->As<ComputePipelineVulkan>()->GetPipeline());
309
310 break;
311 }
312}

329void CommandListVulkan::BindMaterialSlot(const IMaterialSlot* slot) {
330 VkDescriptorSet sets[] = { slot->As<MaterialSlotVulkan>()->GetDescriptorSet(Engine::GetRenderer()->GetCurrentResourceIndex()) };
331
332 VkPipelineBindPoint bindPoint{};
333 switch (currentMaterial->GetMaterialType()) {
334 case MaterialType::COMPUTE: bindPoint = VK_PIPELINE_BIND_POINT_COMPUTE; break;
335 case MaterialType::GRAPHICS: bindPoint = VK_PIPELINE_BIND_POINT_GRAPHICS; break;
336 case MaterialType::RAYTRACING: bindPoint = VK_PIPELINE_BIND_POINT_RAY_TRACING_KHR; break;
337 }
338
339 VkPipelineLayout layout = VK_NULL_HANDLE;
340 switch (currentMaterial->GetMaterialType()) {
341 case MaterialType::COMPUTE: layout = currentPipeline.compute->GetLayout()->As<PipelineLayoutVulkan>()->GetLayout(); break;
342 case MaterialType::GRAPHICS: layout = currentPipeline.graphics->GetLayout()->As<PipelineLayoutVulkan>()->GetLayout(); break;
343 case MaterialType::RAYTRACING: layout = currentPipeline.raytracing->GetLayout()->As<PipelineLayoutVulkan>()->GetLayout(); break;
344 }
345
346 const TIndex glslSetIndex = currentMaterial->GetLayout()->GetSlot(slot->GetName()).glslSetIndex;
347 vkCmdBindDescriptorSets(commandBuffers[GetCommandListIndex()], bindPoint, layout, glslSetIndex, 1, sets, 0, nullptr);
348}

470VkPipelineStageFlagBits CommandListVulkan::GetPipelineStage(GpuBarrierStage stage) const {
471 switch (stage) {
472 case OSK::GRAPHICS::GpuBarrierStage::VERTEX_SHADER:
473 return VK_PIPELINE_STAGE_VERTEX_SHADER_BIT;
474 case OSK::GRAPHICS::GpuBarrierStage::FRAGMENT_SHADER:
