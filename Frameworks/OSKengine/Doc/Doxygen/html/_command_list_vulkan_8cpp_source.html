60void CommandListVulkan::TransitionImageLayout(GpuImage* image, GpuImageLayout previous, GpuImageLayout next, TSize baseLayer, TSize numLayers) {
61 VkImageMemoryBarrier barrier{};
62 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
63 barrier.oldLayout = GetGpuImageLayoutVulkan(previous);
64 barrier.newLayout = GetGpuImageLayoutVulkan(next);
65 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
66 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
67 barrier.image = image->As<GpuImageVulkan>()->GetImage();
68 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
69 barrier.subresourceRange.baseMipLevel = 0;
70 barrier.subresourceRange.levelCount = VK_REMAINING_MIP_LEVELS;
71 barrier.subresourceRange.baseArrayLayer = baseLayer;
72 barrier.subresourceRange.layerCount = numLayers;
73 barrier.srcAccessMask = 0;
74 barrier.dstAccessMask = 0;
75
76 VkPipelineStageFlags sourceStage;
77 VkPipelineStageFlags destinationStage;
78
79 switch (GetGpuImageLayoutVulkan(previous)) {
80 case VK_IMAGE_LAYOUT_UNDEFINED:
81 barrier.srcAccessMask = 0;
82 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
83 break;
84
85 case VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL:
86 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
87 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
88 break;
89
90 case VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL:
91 barrier.srcAccessMask = VK_ACCESS_SHADER_READ_BIT;
92 sourceStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
93 break;
94
95 case VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL:
96 barrier.srcAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
97 sourceStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
98 break;
99
100 case VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL:/*/*/
101 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
102 barrier.srcAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
103 sourceStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
104 break;
105
106 }
107
108 switch (GetGpuImageLayoutVulkan(next)) {
109 case VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL:
110 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
111 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
112 break;
113
114 case VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL:
115 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
116 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
117 break;
118
119 case VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL:
120 barrier.dstAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
121 destinationStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
122 break;
123
124 case VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL:
125 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
126 barrier.dstAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
127 destinationStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
128 break;
129
130 case VK_IMAGE_LAYOUT_PRESENT_SRC_KHR:
131 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
132 destinationStage = VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT;
133 break;
134 }
135
136 vkCmdPipelineBarrier(commandBuffers[Engine::GetRenderer()->GetCurrentCommandListIndex()], sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
137
138 image->SetLayout(next);
139}
