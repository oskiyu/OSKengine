103void BloomPass::DownscaleBloom(GRAPHICS::ICommandList* computeCmdList, Vector2f* res) {
104 computeCmdList->BindMaterial(downscaleMaterial);
105
106 for (TSize i = 0; i < numPasses; i++) {
107 TSize inputIndex = i - 1;
108 TSize outputIndex = i;
109
110 if (i == 0) {
111 inputIndex = numPasses;
112 outputIndex = 0;
113 }
114
115 ExecuteSinglePass(computeCmdList, *res, *res / 2.0f, inputIndex, outputIndex);
116
117 *res /= 2.0f;
118 }

121void BloomPass::UpscaleBloom(GRAPHICS::ICommandList* computeCmdList, Vector2f* res) {
122 computeCmdList->BindMaterial(upscaleMaterial);
123 for (TSize i = numPasses - 1; i > 0; i--) {
124 const TSize inputIndex = i;
125 const TSize outputIndex = inputIndex - 1;
126
127 ExecuteSinglePass(computeCmdList, *res, *res * 2.0f, inputIndex, outputIndex);
128
129 *res *= 2.0f;
130 }
131
132 computeCmdList->BindMaterial(finalMaterial);
133 ExecuteSinglePass(computeCmdList, *res, *res * 2.0f, 0, 3);
