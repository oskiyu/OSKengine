23void BloomPass::Create(const Vector2ui& size) {
24 // Carga de materiales.
25 downscaleMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/PbrMaterials/Bloom/downscale.json");
26 upscaleMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/PbrMaterials/Bloom/upscale.json");
27 resolveMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/PbrMaterials/Bloom/final.json");
28
29 // Final render target
30 GpuImageSamplerDesc renderTargetSampler{};
31 renderTargetSampler.addressMode = GpuImageAddressMode::EDGE; // Para evitar que los bordes tengan leaks de bloom.
32 renderTargetSampler.filteringType = GpuImageFilteringType::LIENAR; // Para los tap que agrupan varios píxeles.
33 renderTargetSampler.mipMapMode = GpuImageMipmapMode::CUSTOM; // Cada nivel de bloom se ejecuta en un mipmap.
34 renderTargetSampler.minMipLevel = 0;
35 renderTargetSampler.maxMipLevel = maxNumPasses;
36
37 RenderTargetAttachmentInfo resolveInfo{};
38 resolveInfo.format = Format::RGBA16_SFLOAT;
39 resolveInfo.sampler = renderTargetSampler;
40 resolveInfo.name = "Bloom Output";
41 resolveInfo.usage =
42 GpuImageUsage::TRANSFER_DESTINATION |
43 GpuImageUsage::COMPUTE |
44 GpuImageUsage::SAMPLED;
45
46
47 // Inicialización de instancias del material.
48 resolveRenderTarget.Create(size, resolveInfo);
49 resolveInstance = resolveMaterial->CreateInstance().GetPointer();
50
51 for (TIndex pass = 0; pass < maxNumPasses; pass++) {
52 downscalingMaterialInstance[pass] = downscaleMaterial->CreateInstance().GetPointer();
53 upscalingMaterialInstance[pass] = upscaleMaterial->CreateInstance().GetPointer();
54 }

61void BloomPass::SetupMaterialInstances() {
62 const IGpuImageView* sourceImages[NUM_RESOURCES_IN_FLIGHT]{};
63 const IGpuImageView* targetImages[NUM_RESOURCES_IN_FLIGHT]{};
64
65 GpuImageViewConfig sourceViewConfig = GpuImageViewConfig::CreateSampled_MipLevelRanged(0, 0);
66 GpuImageViewConfig destViewConfig = GpuImageViewConfig::CreateStorage_Default();
67
68 const TIndex lowestLevel = GetNumPasses();
69
70 for (TIndex pass = 0; pass < lowestLevel - 1; pass++) {
71
72 sourceViewConfig.baseMipLevel = pass;
73 sourceViewConfig.topMipLevel = pass;
74
75 destViewConfig.baseMipLevel = pass + 1;
76 destViewConfig.topMipLevel = pass + 1;
77
78 for (TIndex i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
79 sourceImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(sourceViewConfig);
80 targetImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(destViewConfig);
81 }
82
83 downscalingMaterialInstance[pass]->GetSlot("texture")->SetGpuImages("inputImg", sourceImages);
84 downscalingMaterialInstance[pass]->GetSlot("texture")->SetStorageImages("outputImg", targetImages);
85 downscalingMaterialInstance[pass]->GetSlot("texture")->FlushUpdate();
86 }
87
88 for (TIndex pass = lowestLevel - 2; pass > 0; pass--) {
89
90 sourceViewConfig.baseMipLevel = pass + 1;
91 sourceViewConfig.topMipLevel = pass + 1;
92
93 destViewConfig.baseMipLevel = pass;
94 destViewConfig.topMipLevel = pass;
95
96 for (TIndex i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
97 targetImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(destViewConfig);
98 sourceImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(sourceViewConfig);
99 }
100
101 upscalingMaterialInstance[pass]->GetSlot("texture")->SetGpuImages("inputImg", sourceImages);
102 upscalingMaterialInstance[pass]->GetSlot("texture")->SetStorageImages("outputImg", targetImages);
103 upscalingMaterialInstance[pass]->GetSlot("texture")->FlushUpdate();
104 }
105
106 // Resolve
107 sourceViewConfig.baseMipLevel = 1;
108 sourceViewConfig.topMipLevel = 1;
109
110 destViewConfig.baseMipLevel = 0;
111 destViewConfig.topMipLevel = 0;
112
113 for (TSize i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++)
114 sourceImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(sourceViewConfig);
115
116 const IGpuImageView* sceneImages[NUM_RESOURCES_IN_FLIGHT]{};
117 for (TSize i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++)
118 sceneImages[i] = inputImages[i]->GetView(destViewConfig);
119
120 for (TSize i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++)
121 targetImages[i] = resolveRenderTarget.GetTargetImage(i)->GetView(destViewConfig);
122
123 resolveInstance->GetSlot("texture")->SetGpuImages("bloomImg", sourceImages);
124 resolveInstance->GetSlot("texture")->SetGpuImages("sceneImg", sceneImages);
125 resolveInstance->GetSlot("texture")->SetStorageImages("outputImg", targetImages);
126 resolveInstance->GetSlot("texture")->FlushUpdate();
127}

129void BloomPass::ExecuteSinglePass(ICommandList* computeCmdList, TIndex sourceMipLevel, TIndex destMipLevel) {
130 const TIndex resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
131
132 Vector2ui sourceResolution = resolveRenderTarget.GetTargetImage(resourceIndex)->GetMipLevelSize2D(sourceMipLevel);
133 Vector2ui destResolution = resolveRenderTarget.GetTargetImage(resourceIndex)->GetMipLevelSize2D(destMipLevel);
134
135 const Vector2ui dispatchRes = {
136 glm::max(sourceResolution.X, destResolution.X) / static_cast<TSize>(BLOCK_SIZE),
137 glm::max(sourceResolution.Y, destResolution.Y) / static_cast<TSize>(BLOCK_SIZE)
138 };
139
140 computeCmdList->SetGpuImageBarrier(
141 resolveRenderTarget.GetTargetImage(resourceIndex),
142 GpuImageLayout::SAMPLED,
143 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_WRITE),
144 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_READ),
145 { .baseLayer = 0, .numLayers = 1, .baseMipLevel = sourceMipLevel, .numMipLevels = 1 });
146
147 computeCmdList->SetGpuImageBarrier(
148 resolveRenderTarget.GetTargetImage(resourceIndex),
149 GpuImageLayout::GENERAL,
150 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_READ),
151 GpuBarrierInfo(GpuBarrierStage::COMPUTE_SHADER, GpuBarrierAccessStage::SHADER_WRITE),
152 { .baseLayer = 0, .numLayers = 1, .baseMipLevel = destMipLevel, .numMipLevels = 1 });
153
154 computeCmdList->DispatchCompute({ dispatchRes.X, dispatchRes.Y, 1 });
155}

157void BloomPass::DownscaleBloom(GRAPHICS::ICommandList* computeCmdList) {
158 computeCmdList->BindMaterial(downscaleMaterial);
159 const TIndex resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
160
161 for (TIndex i = 0; i < GetNumPasses() - 1; i++) {
162 computeCmdList->BindMaterialSlot(downscalingMaterialInstance[i]->GetSlot("texture"));
163
164 if (i == 0)
165 computeCmdList->PushMaterialConstants("info", 1);
166 else
167 computeCmdList->PushMaterialConstants("info", 0);
168
169 const TIndex sourceMipLevel = i;
170 const TIndex destinationMipLevel = sourceMipLevel + 1;
171
172 ExecuteSinglePass(computeCmdList, sourceMipLevel, destinationMipLevel);
173 }
