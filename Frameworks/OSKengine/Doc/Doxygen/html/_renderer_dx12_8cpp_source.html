141const TByte* RendererDx12::FormatImageDataForGpu(const GpuImage* image, const TByte* data, TSize numLayers) {
142 TByte* output = new TByte[image->GetPhysicalNumberOfBytes() * numLayers];
143
144 const TSize numBytesPerPixel = GetFormatNumberOfBytes(image->GetFormat());
145 const TSize numBytesPerLayer = image->GetPhysicalNumberOfBytes();
146
147 for (TSize i = 0; i < numLayers + 0; i++) {
148 const TSize layerOffset = numBytesPerLayer * i;
149
150 for (TSize y = 0; y < image->GetPhysicalSize().Y; y++)
151 memcpy(&output[layerOffset + y * image->GetPhysicalSize().X * numBytesPerPixel], &data[layerOffset + y * image->GetSize().X * numBytesPerPixel], image->GetSize().X * numBytesPerPixel);
152 }
153
154 return output;
155}

178void RendererDx12::ChooseGpu() {
179 currentGpu = new GpuDx12;
180 ComPtr<IDXGIAdapter1> adapter;
181
182 bool found = false;
183 // Iteramos sobre todas las gpu disponibles.
184 for (uint32_t i = 0; !found && factory->EnumAdapters1(i, &adapter) != DXGI_ERROR_NOT_FOUND; ++i) {
185 DXGI_ADAPTER_DESC1 description;
186 adapter->GetDesc1(&description);
187
188 // que este adaptador no sea emulado por software
189 if (description.Flags & DXGI_ADAPTER_FLAG_SOFTWARE)
190 continue;
191
192 // decidir con bAdapterFound si es este adaptador el que queramos
193 auto isValid = D3D12CreateDevice(adapter.Get(), D3D_FEATURE_LEVEL_12_1, _uuidof(ID3D12Device), nullptr);
194 if (SUCCEEDED(isValid)) {
195 Engine::GetLogger()->InfoLog("GPU elegida: " + std::string(CW2A(description.Description)));
196
197 found = true;
198 }
199 }
200
201 OSK_ASSERT(found, "No se ha encontrado ninguna GPU compatible con DirectX 12");
202
203 ComPtr<ID3D12Device5> device;
204 D3D12CreateDevice(adapter.Get(), D3D_FEATURE_LEVEL_12_1, IID_PPV_ARGS(&device));
205
206 currentGpu->As<GpuDx12>()->SetAdapter(adapter);
207 currentGpu->As<GpuDx12>()->SetDevice(device);
208
209 if (useDebugConsole) {
210 ID3D12InfoQueue* infoQueue1 = nullptr;
211 HRESULT result = device->QueryInterface(IID_ID3D12InfoQueue, (LPVOID*)&infoQueue1);
212
213 if (SUCCEEDED(result)) {
214 result = device->QueryInterface(IID_ID3D12InfoQueue1, (LPVOID*)&debugMessageQueue);
215 debugMessageQueue->RegisterMessageCallback(RendererDx12::DebugCallback, D3D12_MESSAGE_CALLBACK_FLAG_NONE, nullptr, NULL);
216 }
217 else {
218 Engine::GetLogger()->InfoLog("NO se ha podido establecer el callback de mensajes de log.");
219 }
220 }
221}

223void RendererDx12::CreateCommandQueues() {
224 graphicsQueue = new CommandQueueDx12;
225
226 ComPtr<ID3D12CommandQueue> commandQ;
227
228 D3D12_COMMAND_QUEUE_DESC createInfo{};
229
230 // Hay distintos tipos de colas como:
231 // D3D12_COMMAND_LIST_TYPE_COMPUTE
232 // para temas relacionados con compute shader
233 // pero nosotros queremos usar la cola gr√°fica:
234 createInfo.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
235
236 // para temas de usar m√∫ltiples nodos GPU.
237 // para uso de una sola GPU se pone a 0
238 createInfo.NodeMask = 0;
239
240 // La prioridad de la cola, normalmente NORMAL
241 createInfo.Priority = D3D12_COMMAND_QUEUE_PRIORITY_NORMAL;
242
243 // Flags adicionales. En estos momentos solo hay dos:
244 // D3D12_COMMAND_LIST_TYPE_BUNDLE
245 // D3D12_COMMAND_QUEUE_FLAG_DISABLE_GPU_TIMEOUT
246 createInfo.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
247
248 // Una vez rellenada la estructura, crear el CommandQueue
249 currentGpu->As<GpuDx12>()->GetDevice()->CreateCommandQueue(&createInfo, IID_PPV_ARGS(&commandQ));
250
251 graphicsQueue->As<CommandQueueDx12>()->SetCommandQueue(commandQ);
252 graphicsCommandPool = currentGpu->CreateGraphicsCommandPool().GetPointer();
253
254 graphicsCommandList = graphicsCommandPool->CreateCommandList(currentGpu.GetValue()).GetPointer();
255 graphicsCommandList->As<CommandListDx12>()->SetCommandPool(*graphicsCommandPool->As<CommandPoolDx12>());
256 Engine::GetLogger()->InfoLog("Creada la lista de comandos.");
