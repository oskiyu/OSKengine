28void ShadowMap::Create(const Vector2ui& imageSize) {

31
32 IGpuMemoryAllocator* memAllocator = Engine::GetRenderer()->GetAllocator();
33
34 for (TSize i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
35 GpuImageCreateInfo imageInfo = GpuImageCreateInfo::CreateDefault2D(imageSize, Format::RGBA8_UNORM, GpuImageUsage::COLOR | GpuImageUsage::SAMPLED);
36 imageInfo.numLayers = 4;

38
39 unusedColorArrayAttachment[i] = memAllocator->CreateImage(imageInfo).GetPointer();
40 unusedColorArrayAttachment[i]->SetDebugName("Shadow Map [" + std::to_string(i) + "] Unused Attachment");
41
42 imageInfo.format = Format::D32S8_SFLOAT_SUINT;
43 imageInfo.usage = GpuImageUsage::DEPTH_STENCIL | GpuImageUsage::SAMPLED | GpuImageUsage::SAMPLED_ARRAY;
44
45 depthArrayAttachment[i] = memAllocator->CreateImage(imageInfo).GetPointer();
46 depthArrayAttachment[i]->SetDebugName("Shadow Map [" + std::to_string(i) + "] Depth");
47 }
48
49 shadowsGenMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/material_shadows.json");
50 shadowsGenAnimMaterial = Engine::GetRenderer()->GetMaterialSystem()->LoadMaterial("Resources/material_shadows_anim.json");
51 shadowsGenMaterialInstance = shadowsGenMaterial->CreateInstance().GetPointer();
52
53 const IGpuUniformBuffer* lightUbos[3]{};
54 for (TSize i = 0; i < NUM_RESOURCES_IN_FLIGHT; i++) {
55 lightUniformBuffer[i] = Engine::GetRenderer()->GetAllocator()->CreateUniformBuffer(sizeof(glm::mat4) * 4 + sizeof(Vector3f)).GetPointer();

58
59 shadowsGenMaterialInstance->GetSlot("global")->SetUniformBuffers("dirLight", lightUbos);

61}

72void ShadowMap::UpdateLightMatrixBuffer() {
73 const TSize resourceIndex = Engine::GetRenderer()->GetCurrentResourceIndex();
74
75 ShadowsBufferContent bufferContent{};
76
77 const CameraComponent3D& camera = Engine::GetEcs()->GetComponent<CameraComponent3D>(cameraObject);
78 const Transform3D& cameraTransform = Engine::GetEcs()->GetComponent<Transform3D>(cameraObject);
79
80 float cascadeSplits[numMaps]{ 1, 5, 15, 50 };
81 for (TSize i = 0; i < numMaps; i++)
82 cascadeSplits[i] -= camera.GetNearPlane();
83
84 bufferContent.cascadeSplits = Vector4f(cascadeSplits[0], cascadeSplits[1], cascadeSplits[2], cascadeSplits[3]);
85
86 float lastClip = camera.GetNearPlane();
87
88// #define OSK_SHADOWS_NVIDIA
89#ifndef OSK_SHADOWS_NVIDIA
90#define OSK_SHADOWS_CUSTOM
91#endif
92
93 // NVIDIA:
94#ifdef OSK_SHADOWS_NVIDIA
95 for (TSize i = 0; i < numMaps; i++) {
96
97 // Esquinas del frustum en world-space.
98 DynamicArray<Vector3f> worldSpaceFrustumCorners = GetFrustumCorners(cameraProjection * cameraView);
99
100 for (TSize corner = 0; corner < worldSpaceFrustumCorners.GetSize() / 2; corner++) {
101 const Vector3f distanceToCorner = worldSpaceFrustumCorners[corner + 4] - worldSpaceFrustumCorners[corner];
102
103 worldSpaceFrustumCorners[corner + 4] = worldSpaceFrustumCorners[corner] + (distanceToCorner * cascadeSplits[i]);
104 worldSpaceFrustumCorners[corner] += (distanceToCorner * lastClip);
105 }
106
107 Vector3f frustumCenter = 0.0f;
108 for (const auto& corner : worldSpaceFrustumCorners)
109 frustumCenter += corner;
110 frustumCenter /= static_cast<float>(worldSpaceFrustumCorners.GetSize());
111
112 float radius = 0.0f;
113 for (const auto& corner : worldSpaceFrustumCorners)
114 radius = glm::max(radius, frustumCenter.GetDistanceTo(corner));
115 radius = glm::ceil(radius * 16.0f) / 16.0f;
116
117 const Vector3f maxExtent = radius;
118 const Vector3f minExtent = -radius;
119
120 const glm::mat4 lightView = glm::lookAt((frustumCenter - lightDirection.GetNormalized() * radius * 0.5f).ToGLM(), frustumCenter.ToGLM(), glm::vec3(0.0f, 1.0f, 0.0f));
121 const glm::mat4 lightProjection = glm::ortho(minExtent.X, maxExtent.X, minExtent.Y, maxExtent.Y, minExtent.Z, maxExtent.Z);
122
123 bufferContent.matrices[i] = lightProjection * lightView;
124 lastClip = cascadeSplits[i];
125
126 cascadeSplits[i] = ((nearClip + lastClip * clipRange) * -1.0f);
127 }
128#endif
129#ifdef OSK_SHADOWS_CUSTOM
130
131 // CUSTOM:
132 for (TSize i = 0; i < numMaps; i++) {
133
134 // Creamos un frustum con NEAR = CascadeSplits[i - 1] y FAR = CascadeSplits[i],
135 // de tal manera que quede dentro toda la zona renderizada que está dentro del split.
136 CameraComponent3D clipCamera = camera;
137 clipCamera.SetPlanes(lastClip, cascadeSplits[i]);
138 // Obtiene las esquinas en el espacio del mundo.
139 const DynamicArray<Vector3f> worldSpaceFrustumCorners =
140 GetFrustumCorners(clipCamera.GetProjectionMatrix() * clipCamera.GetViewMatrix(cameraTransform));
141
142 // Calculamos el centro del frustum.
143 Vector3f frustumCenter = 0.0f;
144 for (const auto& corner : worldSpaceFrustumCorners)
145 frustumCenter += corner;
146 frustumCenter /= static_cast<float>(worldSpaceFrustumCorners.GetSize());
147
148 // Movemos para que el centro este en Y = 0.
149 for (auto& corner : worldSpaceFrustumCorners)
150 corner += lightDirection * frustumCenter.Y;
151 frustumCenter += lightDirection * frustumCenter.Y;
152
153 // Calculamos el radio, que será el radio del frustum que renderizará el mapa de sombras.
154 float radius = 0.0f;
155 for (const auto& corner : worldSpaceFrustumCorners)
156 radius = glm::max(radius, frustumCenter.GetDistanceTo(corner));
157 radius = glm::ceil(radius * 16.0f) / 16.0f;
158
159 // Obtenemos alturas min y max, para poder establecer correctamente los
160 // planos near y far del frustum final.
161 const float yPerUnit = 1.0f / glm::radians(90.0f - lightDirection.GetAngle({ 0.0f, -1.0f, 0.0f }));
162
163 const float minCornerHeight = -yPerUnit * radius;
164 const float maxCornerHeight = yPerUnit * radius;
165
166 const float shadowNearClip = (minCornerHeight) / yPerUnit;
167 const float shadowFarClip = maxCornerHeight / yPerUnit;
168
169 // Extents del frustum
170 const Vector2f maxExtent = radius;
171 const Vector2f minExtent = -maxExtent;
172
173 // 'Cámara' virtual para renderizar el mapa de sombras.
174 const glm::mat4 lightProjection = glm::ortho(minExtent.X, maxExtent.X, maxExtent.Y, minExtent.Y, nearPlane * (i + 1), farPlane * (i + 1));
175 const glm::mat4 lightView = glm::lookAt((frustumCenter - lightDirection).ToGLM(), frustumCenter.ToGLM(), glm::vec3(0.0f, 1.0f, 0.0f));
176
177 bufferContent.matrices[i] = lightProjection * lightView;
178
179 lastClip = cascadeSplits[i];
180 }
181#endif
182
183 lightUniformBuffer[resourceIndex]->ResetCursor();
184 lightUniformBuffer[resourceIndex]->MapMemory();
185 lightUniformBuffer[resourceIndex]->Write(bufferContent.matrices[0]);
186 lightUniformBuffer[resourceIndex]->Write(bufferContent.matrices[1]);
187 lightUniformBuffer[resourceIndex]->Write(bufferContent.matrices[2]);
188 lightUniformBuffer[resourceIndex]->Write(bufferContent.matrices[3]);
189 lightUniformBuffer[resourceIndex]->Write(bufferContent.cascadeSplits);
190 lightUniformBuffer[resourceIndex]->Unmap();
191}

236DynamicArray<Vector3f> ShadowMap::GetFrustumCorners(const glm::mat4& cameraMatrix) {
237 const glm::mat4 inverseMatrix = glm::inverse(cameraMatrix);
238
239 // Normalized Device Coordinates
240 DynamicArray<Vector3f> corners = {
241 Vector3f(-1.0f, 1.0f, -1.0f),
242 Vector3f( 1.0f, 1.0f, -1.0f),
243 Vector3f(-1.0f, 1.0f, 1.0f),
244 Vector3f( 1.0f, 1.0f, 1.0f),
245 Vector3f( 1.0f, -1.0f, -1.0f),
246 Vector3f(-1.0f, -1.0f, -1.0f),
247 Vector3f( 1.0f, -1.0f, 1.0f),
248 Vector3f(-1.0f, -1.0f, 1.0f),
249 };
250
251 for (auto& corner : corners) {
252 const glm::vec4 inverseResult = inverseMatrix * glm::vec4(corner.ToGLM(), 1.0f);
253 corner = glm::vec3(inverseResult / inverseResult.w);
254 }
255
256 return corners;
257}
