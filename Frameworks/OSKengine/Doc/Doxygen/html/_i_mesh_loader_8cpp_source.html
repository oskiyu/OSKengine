44 for (TSize i = 0; i < meshes.GetSize(); i++) {

67glm::mat4 IMeshLoader::GetNodeMatrix(const tinygltf::Node& node) {
68 // Obtenemos el transform del nodo.
69 // Puede estar definido de varias maneras:
70
71 glm::mat4 nodeMatrix = glm::mat4(1.0f);
72
73 // Definido por una posición.
74 if (node.translation.size() == 3) {
75 nodeMatrix = glm::translate(nodeMatrix, glm::vec3(node.translation[0], node.translation[1], node.translation[2]));
76 }
77 // Definido por una rotación.
78 if (node.rotation.size() == 4) {
79 glm::quat quaternion = glm::make_quat(node.rotation.data());
80 nodeMatrix *= glm::mat4(quaternion);
81 }
82 // Definido por una escala.
83 if (node.scale.size() == 3) {
84 nodeMatrix = glm::scale(nodeMatrix, glm::vec3(glm::make_vec3(node.scale.data())));
85 }
86 // Definido por una matriz.
87 if (node.matrix.size() == 16) {
88 nodeMatrix = glm::make_mat4x4(node.matrix.data());
89 }
90
91 return nodeMatrix;
92}

94DynamicArray<OwnedPtr<GpuImage>> IMeshLoader::LoadImages() {
95 DynamicArray<OwnedPtr<GpuImage>> output;
96
97 for (TSize i = 0; i < gltfModel.images.size(); i++) {
98 const tinygltf::Image& originalImage = gltfModel.images[i];
99
100 auto image = Engine::GetRenderer()->GetAllocator()->CreateImage(GpuImageCreateInfo::CreateDefault2D(
101 { (unsigned int)originalImage.width, (unsigned int)originalImage.height },
102 Format::RGBA8_UNORM,
103 GpuImageUsage::TRANSFER_SOURCE | GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED)
104 );
105 image->SetDebugName("Mesh texture " + std::to_string(i));
106
107 OwnedPtr<ICommandList> uploadCmdList = Engine::GetRenderer()->CreateSingleUseCommandList();
108 uploadCmdList->Reset();
109 uploadCmdList->Start();
110
111 const TSize numBytes = originalImage.width * originalImage.height * 4;
112 if (originalImage.component == 3) {
113 TByte* data = (TByte*)malloc(numBytes);
114 memset(data, 255, numBytes);
115 OSK_ASSERT(data != NULL, "No se pudo reservar memoria para las imágenes del modelo 3d.");
116
117 TSize rgbPos = 0;
118 for (TSize i = 0; i < numBytes; i += 4) {
119 memcpy(data + i, originalImage.image.data() + rgbPos, 3);
120 rgbPos += 3;
121 }
122
123 uploadCmdList->SetGpuImageBarrier(image.GetPointer(), GpuImageLayout::UNDEFINED, GpuImageLayout::TRANSFER_DESTINATION,
124 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE));
125 Engine::GetRenderer()->UploadImageToGpu(image.GetPointer(), originalImage.image.data(), numBytes, uploadCmdList.GetPointer());
126 uploadCmdList->SetGpuImageBarrier(image.GetPointer(), GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::SAMPLED,
127 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE), GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ));
128
129 free(data);
130 }
131 else if (originalImage.component == 4) {
132 uploadCmdList->SetGpuImageBarrier(image.GetPointer(), GpuImageLayout::UNDEFINED, GpuImageLayout::TRANSFER_DESTINATION,
133 GpuBarrierInfo(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT), GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE));
134 Engine::GetRenderer()->UploadImageToGpu(image.GetPointer(), originalImage.image.data(), numBytes, uploadCmdList.GetPointer());
135 uploadCmdList->SetGpuImageBarrier(image.GetPointer(), GpuImageLayout::TRANSFER_DESTINATION, GpuImageLayout::SAMPLED,
136 GpuBarrierInfo(GpuBarrierStage::TRANSFER, GpuBarrierAccessStage::TRANSFER_WRITE), GpuBarrierInfo(GpuBarrierStage::FRAGMENT_SHADER, GpuBarrierAccessStage::SHADER_READ));
137 }
138 else {
139 OSK_ASSERT(false, "Formato de imagen no soportado en gltf2.0.");
140 }
141
142 uploadCmdList->Close();
143 Engine::GetRenderer()->SubmitSingleUseCommandList(uploadCmdList.GetPointer());
144
145 output.Insert(image);
146 }
147
148 return output;
149}

151DynamicArray<GltfMaterialInfo> IMeshLoader::LoadMaterials() {
152 auto output = DynamicArray<GltfMaterialInfo>::CreateResizedArray((TSize)gltfModel.materials.size());
153
154 for (TSize i = 0; i < gltfModel.materials.size(); i++) {
155 GltfMaterialInfo info{};
156 info.baseColor = {
157 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[0]),
158 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[1]),
159 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[2]),
160 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[3])
161 };
162
163 info.metallicFactor = static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.metallicFactor);
164 info.roughnessFactor = static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.roughnessFactor);
165
166 if (info.roughnessFactor >= 1.0f)
167 info.roughnessFactor = 1.0f;
168
169 if (gltfModel.materials[i].pbrMetallicRoughness.baseColorTexture.index != -1) {
170 info.hasBaseTexture = true;
171 info.baseTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.baseColorTexture.index].source;
172 }
173
174 if (gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index != -1) {
175 info.hasMetallicTexture = true;
176 info.metallicTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
177
178 Engine::GetLogger()->InfoLog("El material tiene textura metálica.");
179 }
180
181 if (gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index != -1) {
182 info.hasRoughnessTexture = true;
183 info.metallicTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
184
185 Engine::GetLogger()->InfoLog("El material tiene textura de rugosidad.");
186 }
187
188 output[i] = info;
189 }
190
191 return output;
192}

222DynamicArray<Vector3f> IMeshLoader::GetVertexPositions(const tinygltf::Primitive& primitive, const glm::mat4& nodeMatrix) const {
223 // Comprobamos que tiene almacenado info de posición.
224 OSK_ASSERT(HasPositions(primitive), "No se encontraron posiciones de vértices.");
225
226 // Para poder acceder a la información en forma de buffer.
227 const tinygltf::Accessor& accessor = gltfModel.accessors[primitive.attributes.find("POSITION")->second];
228 const tinygltf::BufferView& view = gltfModel.bufferViews[accessor.bufferView];
229
230 // Leemos el buffer.
231 const float* positionsBuffer = reinterpret_cast<const float*>(&(gltfModel.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
232 const TSize numVertices = static_cast<TSize>(accessor.count);
233
234
235 // Transformamos a vértices
236 DynamicArray<Vector3f> output = DynamicArray<Vector3f>::CreateResizedArray(numVertices);
237
238 for (TIndex v = 0; v < numVertices; v++) {
239 const glm::vec4 vertexPosition = glm::vec4(
240 positionsBuffer[v * 3 + 0],
241 positionsBuffer[v * 3 + 1],
242 positionsBuffer[v * 3 + 2],
243 globalScale
244 );
245
246 output[v] = glm::vec3(nodeMatrix * vertexPosition);
247 }
248
249 return output;
250}

301DynamicArray<Color> IMeshLoader::GetVertexColors(const tinygltf::Primitive& primitive) const {
302 if (!HasColors(primitive))
303 return DynamicArray<Color>{};
304
305 // Para poder acceder a la información en forma de buffer.
306 const tinygltf::Accessor& accessor = gltfModel.accessors[primitive.attributes.find("COLOR_0")->second];
307 const tinygltf::BufferView& view = gltfModel.bufferViews[accessor.bufferView];
308
309 // Leemos el buffer.
310 const void* colorBuffer = &(gltfModel.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]);
311 const TSize numVertices = static_cast<TSize>(accessor.count);
312
313 const auto colorVectorType = accessor.type;
314 const auto colorDataType = accessor.componentType;
315
316 const int stride = colorVectorType;
317
318 DynamicArray<Color> output = DynamicArray<Color>::CreateResizedArray(numVertices);
319
320 for (TIndex v = 0; v < numVertices; v++) {
321
322 switch (colorDataType) {
323
324 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
325 constexpr float denom = std::numeric_limits<uint8_t>::max();
326 output[v] = {
327 static_cast<float>(static_cast<const uint8_t*>(colorBuffer)[v * stride + 0] / denom),
328 static_cast<float>(static_cast<const uint8_t*>(colorBuffer)[v * stride + 1] / denom),
329 static_cast<float>(static_cast<const uint8_t*>(colorBuffer)[v * stride + 2] / denom),
330 1.0f
331 };
332
333 break;
334 }
335
336 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
337 constexpr float denom = std::numeric_limits<uint16_t>::max();
338 output[v] = {
339 static_cast<float>(static_cast<const uint16_t*>(colorBuffer)[v * stride + 0] / denom),
340 static_cast<float>(static_cast<const uint16_t*>(colorBuffer)[v * stride + 1] / denom),
341 static_cast<float>(static_cast<const uint16_t*>(colorBuffer)[v * stride + 2] / denom),
342 1.0f
343 };
344
345 break;
346 }
347
348 case TINYGLTF_PARAMETER_TYPE_FLOAT: {
349 output[v] = Color{
350 static_cast<float>(static_cast<const float*>(colorBuffer)[v * stride + 0]),
351 static_cast<float>(static_cast<const float*>(colorBuffer)[v * stride + 1]),
352 static_cast<float>(static_cast<const float*>(colorBuffer)[v * stride + 2]),
353 1.0f
354 };
355
356 break;
357 }
358 }
359 }

362DynamicArray<TIndexSize> IMeshLoader::GetIndices(const tinygltf::Primitive& primitive, TIndexSize startOffset) const {
363 // Índices
364 const tinygltf::Accessor& indicesAccesor = gltfModel.accessors[primitive.indices];
365 const tinygltf::BufferView& indicesView = gltfModel.bufferViews[indicesAccesor.bufferView];
366 const tinygltf::Buffer& indicesBuffer = gltfModel.buffers[indicesView.buffer];
367
368 const TSize numIndices = indicesAccesor.count;
369
370 DynamicArray<TIndexSize> output = DynamicArray<TIndexSize>::CreateResizedArray(numIndices);
371
372 // Los índices tambien se guardan en buffers.
373 // Necesitamos saber su tipo para procesar el buffer.
374 // El índice es el ID del vértice dentro de la primitiva.
375 // Para obtener el ID real, debemos tener en cuenta todos los vértices anteriormente procesados.
376 switch (indicesAccesor.componentType) {
377 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_INT: {
378 const uint32_t* indBuffer = (const uint32_t*)(&indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset]);
379 for (TSize index = 0; index < indicesAccesor.count; index++)
380 output.Insert((TIndexSize)indBuffer[index] + startOffset);
381
382 break;
383 }
384
385 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
386 const uint16_t* indBuffer = (const uint16_t*)(&indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset]);
387 for (size_t index = 0; index < indicesAccesor.count; index++)
388 output.Insert((TIndexSize)indBuffer[index] + startOffset);
389
390 break;
391 }
392
393 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
394 const uint8_t* indBuffer = (const uint8_t*)(&indicesBuffer.data[indicesAccesor.byteOffset + indicesView.byteOffset]);
395 for (size_t index = 0; index < indicesAccesor.count; index++)
396 output.Insert((TIndexSize)indBuffer[index] + startOffset);
397
398 break;
399 }
400
401 }
402
403 return output;
404}
