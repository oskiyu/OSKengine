67glm::mat4 IMeshLoader::GetNodeMatrix(const tinygltf::Node& node) {
68 // Obtenemos el transform del nodo.
69 // Puede estar definido de varias maneras:
70
71 glm::mat4 nodeMatrix = glm::mat4(1.0f);
72
73 // Definido por una posición.
74 if (node.translation.size() == 3) {
75 nodeMatrix = glm::translate(nodeMatrix, glm::vec3(node.translation[0], node.translation[1], node.translation[2]));
76 }
77 // Definido por una rotación.
78 if (node.rotation.size() == 4) {
79 glm::quat quaternion = glm::make_quat(node.rotation.data());
80 nodeMatrix *= glm::mat4(quaternion);
81 }
82 // Definido por una escala.
83 if (node.scale.size() == 3) {
84 nodeMatrix = glm::scale(nodeMatrix, glm::vec3(glm::make_vec3(node.scale.data())));
85 }
86 // Definido por una matriz.
87 if (node.matrix.size() == 16) {
88 nodeMatrix = glm::make_mat4x4(node.matrix.data());
89 }
90
91 return nodeMatrix;
92}

151DynamicArray<GltfMaterialInfo> IMeshLoader::LoadMaterials() {
152 auto output = DynamicArray<GltfMaterialInfo>::CreateResizedArray((TSize)gltfModel.materials.size());
153
154 for (TSize i = 0; i < gltfModel.materials.size(); i++) {
155 GltfMaterialInfo info{};
156 info.baseColor = {
157 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[0]),
158 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[1]),
159 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[2]),
160 static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.baseColorFactor[3])
161 };
162
163 info.metallicFactor = static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.metallicFactor);
164 info.roughnessFactor = static_cast<float>(gltfModel.materials[i].pbrMetallicRoughness.roughnessFactor);
165
166 if (info.roughnessFactor == 1.0f)
167 info.roughnessFactor = 0.7f;
168
169 if (gltfModel.materials[i].pbrMetallicRoughness.baseColorTexture.index != -1) {
170 info.hasBaseTexture = true;
171 info.baseTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.baseColorTexture.index].source;
172 }
173
174 if (gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index != -1) {
175 info.hasMetallicTexture = true;
176 info.metallicTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
177
178 Engine::GetLogger()->InfoLog("El material tiene textura metálica.");
179 }
180
181 if (gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index != -1) {
182 info.hasRoughnessTexture = true;
183 info.metallicTextureIndex = gltfModel.textures[gltfModel.materials[i].pbrMetallicRoughness.metallicRoughnessTexture.index].source;
184
185 Engine::GetLogger()->InfoLog("El material tiene textura de rugosidad.");
186 }
187
188 output[i] = info;
189 }
190
191 return output;
