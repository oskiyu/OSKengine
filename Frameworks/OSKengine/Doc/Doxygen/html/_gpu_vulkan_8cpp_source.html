193QueueFamilyIndices GpuVulkan::GetQueueFamilyIndices(VkSurfaceKHR surface) const {
194 QueueFamilyIndices indices{};
195
196 // Número de colas.
197 uint32_t queueFamilyCount = 0;
198 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, nullptr);
199
200 // Obtener las colas.
201 auto queueFamilies = DynamicArray<VkQueueFamilyProperties>::CreateResizedArray(queueFamilyCount);
202
203 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, queueFamilies.GetData());
204
205 // Comprobar el soporte de distintos tipos de cola.
206 int i = 0;
207 for (const auto& q : queueFamilies) {
208 if (q.queueFlags & VK_QUEUE_GRAPHICS_BIT)
209 indices.graphicsFamily = i;
210
211 // Soporte para presentación.
212 VkBool32 presentSupport = false;
213 vkGetPhysicalDeviceSurfaceSupportKHR(physicalDevice, i, surface, &presentSupport);
214
215 if (presentSupport)
216 indices.presentFamily = i;
217
218 if (q.queueFlags & VK_QUEUE_COMPUTE_BIT)
219 indices.computeFamily = i;
220
221 if (indices.IsComplete())
222 break;
223
224 i++;
225 }
226
227 return indices;
228}
