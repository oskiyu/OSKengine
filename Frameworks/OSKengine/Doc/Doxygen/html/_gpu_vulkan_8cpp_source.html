160QueueFamilyIndices GpuVulkan::GetQueueFamilyIndices(VkSurfaceKHR surface) const {
161 QueueFamilyIndices indices{};
162
163 // N√∫mero de colas.
164 uint32_t queueFamilyCount = 0;
165 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, nullptr);
166
167 // Obtener las colas.
168 auto queueFamilies = DynamicArray<VkQueueFamilyProperties>::CreateResizedArray(queueFamilyCount);
169
170 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, queueFamilies.GetData());
171
172 // Comprobar el soporte de distintos tipos de cola.
173 int i = 0;
174 for (const auto& q : queueFamilies) {
175 if (q.queueFlags & VK_QUEUE_GRAPHICS_BIT)
176 indices.graphicsFamily = i;
177
178 // Soporte para presentaci√≥n.
179 VkBool32 presentSupport = false;
180 vkGetPhysicalDeviceSurfaceSupportKHR(physicalDevice, i, surface, &presentSupport);
181
182 if (presentSupport)
183 indices.presentFamily = i;
184
185 if (indices.IsComplete())
186 break;
187
188 i++;
189 }
190
191 return indices;
192}
