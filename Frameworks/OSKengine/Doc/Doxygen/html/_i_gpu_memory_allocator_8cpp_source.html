40OwnedPtr<IGpuVertexBuffer> IGpuMemoryAllocator::CreateVertexBuffer(const void* data, TSize vertexSize, TSize numVertices, const VertexInfo& vertexInfo, GpuBufferUsage usage) {
41 const TSize bufferSize = numVertices * vertexSize;
42 IGpuMemoryBlock* block = GetNextBufferMemoryBlock(
43 bufferSize,
44 usage | GpuBufferUsage::VERTEX_BUFFER | GpuBufferUsage::TRANSFER_DESTINATION,
45 GpuSharedMemoryType::GPU_ONLY);
46
47 // Buffer temporal sobre el que escribimos los vértices.
48 GpuDataBuffer* stagingBuffer = CreateStagingBuffer(bufferSize).GetPointer();
49 stagingBuffer->MapMemory();
50 stagingBuffer->Write(data, bufferSize);
51 stagingBuffer->Unmap();
52
53 OwnedPtr<IGpuVertexBuffer> output = _CreateVertexBuffer(
54 block->GetNextMemorySubblock(bufferSize, 0),
55 bufferSize,
56 0,
57 numVertices,
58 vertexInfo);
59
60
61 // Debemos copiar el buffer temporal al buffer final.
62
63 OwnedPtr<ICommandList> singleTimeCommandList = Engine::GetRenderer()->CreateSingleUseCommandList();
64 singleTimeCommandList->Reset();
65 singleTimeCommandList->Start();
66
67 singleTimeCommandList->CopyBufferToBuffer(stagingBuffer, output.GetPointer(), bufferSize, 0, 0);
68
69 singleTimeCommandList->Close();
70
71 singleTimeCommandList->RegisterStagingBuffer(stagingBuffer);
72
73 Engine::GetRenderer()->SubmitSingleUseCommandList(singleTimeCommandList);
74
75 return output;
76}

107void IGpuMemoryAllocator::RemoveMemoryBlock(IGpuMemoryBlock* iblock) {
108 std::optional<TIndex> index;
109 for (TIndex i = 0; i < imageMemoryBlocks.GetSize(); i++) {
110 if (imageMemoryBlocks[i].GetPointer() == iblock) {
111 index = i;
112 break;
113 }
114 }
115
116 if (index.has_value()) {
117 imageMemoryBlocks.RemoveAt(index.value());
118 return;
119 }
120
121 for (auto& list : bufferMemoryBlocks) {
122 for (TIndex i = 0; i < list.GetSize(); i++) {
123 if (list[i].GetPointer() == iblock) {
124 index = i;
125 break;
126 }
127 }
128
129 if (index.has_value()) {
130 imageMemoryBlocks.RemoveAt(index.value());
131 return;
132 }
133 }
134}

136void IGpuMemoryAllocator::FreeStagingMemory() {
137 for (TIndex i = 0; i < bufferMemoryBlocksInfo.GetSize(); i++) {
138 const auto& info = bufferMemoryBlocksInfo[i];
139
140 if (info.sharedType == GpuSharedMemoryType::GPU_AND_CPU
141 && EFTraits::HasFlag(info.usage, GpuBufferUsage::UPLOAD_ONLY)) {
142
143 DynamicArray<TIndex> freeableIndices{};
144
145 for (TIndex b = 0; b < bufferMemoryBlocks[i].GetSize(); b++) {
146 const auto block = bufferMemoryBlocks[i][b].GetPointer();
147
148 if (block->IsUnused())
149 freeableIndices.Insert(b);
150 }
151
152 if (freeableIndices.GetSize() > 0)
153 for (TIndex b = freeableIndices.GetSize() - 1; b > 0; b--)
154 bufferMemoryBlocks[i].RemoveAt(b);
155 }
156 }
157}

227IGpuMemoryBlock* IGpuMemoryAllocator::GetNextBufferMemoryBlock(TSize size, GpuBufferUsage usage, GpuSharedMemoryType sharedType) {
228 // Miramos todas las listas de bloques para comprobar si hay alguna que
229 // tiene bloques con las características necesarias.
230 for (TIndex i = 0; i < bufferMemoryBlocksInfo.GetSize(); i++) {
231 const auto& blockInfo = bufferMemoryBlocksInfo[i];
232
233 // TODO: permitir reutilización.
234 if (blockInfo.usage == usage && blockInfo.sharedType == sharedType) {
235 for (const auto& block : bufferMemoryBlocks[i]) {
236 if (block->GetAvailableSpace() >= size)
237 return block.GetPointer();
238 }
239 }
240 }
241
242 // Creamos uno nuevo
243 if (size < IGpuMemoryAllocator::SizeOfMemoryBlockInMb * 1024 * 1024)
244 size = IGpuMemoryAllocator::SizeOfMemoryBlockInMb * 1024 * 1024;
245
246 // Creamos un nuevo bloque.
247 OwnedPtr<IGpuMemoryBlock> newBlock = CreateNewBufferBlock(size, usage, sharedType);
248
249 // Engine::GetLogger()->InfoLog("Creado bloque de memoria GPU.");
250 // Engine::GetLogger()->InfoLog(" Tipo: " + ToString(sharedType)); @bug ???
251 // Engine::GetLogger()->InfoLog(" Uso: " + ToString(usage)); @bug ???
252 // Engine::GetLogger()->InfoLog(" Tamaño: " + std::to_string(size));
253
254 bufferMemoryBlocksInfo.Insert(GpuBufferMemoryBlockInfo{ size, usage, sharedType });
255 bufferMemoryBlocks.Insert({ });
256 bufferMemoryBlocks.Peek().Insert(newBlock.GetPointer());
257
258 return newBlock.GetPointer();
259}

274 output = glm::max(output, minIndexBufferAlignment);
