45void Engine::Create(GRAPHICS::RenderApiType type) {
46 logger = new IO::Logger;
47 window = new IO::Window;
48 entityComponentSystem = new ECS::EntityComponentSystem;
49 inputManager = new IO::InputManager;
50
51 logger->Start("LOG_LAST.txt");
52 window->SetRenderApiType(type);
53
54 logger->InfoLog("Iniciando OSKengine.");
55 logger->InfoLog(" Version: "
56 + std::to_string(GetVersion().mayor) + "."
57 + std::to_string(GetVersion().menor) + "."
58 + std::to_string(GetVersion().parche) + ".");
59 logger->InfoLog(" Build " + GetBuild() + ".");
60
61 switch (type) {
62 case GRAPHICS::RenderApiType::OPENGL:
63 //renderer = new GRAPHICS::RendererOgl;
64 OSK_ASSERT(false, "OpenGL no soportado");
65
66 break;
67
68 case OSK::GRAPHICS::RenderApiType::VULKAN:
69 renderer = new GRAPHICS::RendererVulkan;
70
71 break;
72
73 case OSK::GRAPHICS::RenderApiType::DX12:
74 renderer = new GRAPHICS::RendererDx12;
75
76 break;
77
78 case OSK::GRAPHICS::RenderApiType::OPENGL_ES:
79 OSK_ASSERT(false, "El renderizador de OPENGL ES no está implementado.");

81 break;
82 }

84 assetManager = new ASSETS::AssetManager();
