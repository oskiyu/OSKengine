57void Engine::Create(GRAPHICS::RenderApiType type) {
58 logger = new IO::Logger;
59 display = new IO::Window;
60 entityComponentSystem = new ECS::EntityComponentSystem;
61 input = new IO::PcUserInput;
62 inputManager = new IO::InputManager;
63
64 logger->Start("LOG_LAST.txt");
65
66 logger->InfoLog("Iniciando OSKengine.");
67 logger->InfoLog(" Version: "
68 + std::to_string(GetVersion().mayor) + "."
69 + std::to_string(GetVersion().menor) + "."
70 + std::to_string(GetVersion().parche) + ".");
71 logger->InfoLog(" Build " + GetBuild() + ".");
72
73 const nlohmann::json engineConfig = nlohmann::json::parse(OSK::IO::FileIO::ReadFromFile("engine_config.json"));
74 const bool requestRayTracing = engineConfig.contains("use_rt") && engineConfig["use_rt"] == 1;
75
76 switch (type) {
77
78 case OSK::GRAPHICS::RenderApiType::VULKAN:
79 renderer = new GRAPHICS::RendererVulkan(requestRayTracing);
80
81 break;
82
83 case OSK::GRAPHICS::RenderApiType::DX12:
84 renderer = new GRAPHICS::RendererDx12(requestRayTracing);
85
86 break;
87 }
88
89 assetManager = new ASSETS::AssetManager();
90}
