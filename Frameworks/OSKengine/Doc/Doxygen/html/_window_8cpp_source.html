44void Window::Create(int32_t sizeX, int32_t sizeY, const std::string& title) {
45 // Creación de la ventana.
46 window = glfwCreateWindow(sizeX, sizeY, title.c_str(), NULL, NULL);
47 OSK_ASSERT(window.HasValue(), "No se ha podido iniciar la ventana.");
48 glfwMakeContextCurrent(window.GetPointer());
49
50 this->sizeX = sizeX;
51 this->sizeY = sizeY;
52
53 UpdateScreenRatio();
54
55 // Obtención de variables de glfw.
56 glfwSetWindowUserPointer(window.GetPointer(), this);
57 monitor = glfwGetPrimaryMonitor();
58 monitorInfo = glfwGetVideoMode(monitor.GetPointer());
59
60 // Establece los callbacks.
61 glfwSetFramebufferSizeCallback(window.GetPointer(), Window::GlfwResizeCallback);
62 glfwSetCursorPosCallback(window.GetPointer(), Window::GlfwMouseInputCallback);
63 glfwSetScrollCallback(window.GetPointer(), Window::GlfwMouseScrollCallback);
64
65 // Info-log.
66 Engine::GetLogger()->InfoLog("Ventana iniciada correctamente.");
67 Engine::GetLogger()->InfoLog(" Monitor sizeX: " + std::to_string(monitorInfo->width));
68 Engine::GetLogger()->InfoLog(" Monitor sizeY: " + std::to_string(monitorInfo->height));
69 Engine::GetLogger()->InfoLog(" Monitor refresh rate: " + std::to_string(monitorInfo->refreshRate));
70
71 int x = 0;
72 int y = 0;
73 int nmChannels = 0;
74 stbi_uc* icon = stbi_load("Resources/Icons/engineIcon.png", &x, &y, &nmChannels, 4);
75
76 GLFWimage glfwIcon[1]{};
77 glfwIcon[0].width = x;
78 glfwIcon[0].height = y;
79 glfwIcon[0].pixels = icon;
80
81 glfwSetWindowIcon(window.GetPointer(), _countof(glfwIcon), glfwIcon);
82
83 stbi_image_free(icon);
84
85 isOpen = true;
86}

299void Window::UpdateGamepadState(GamepadState* gamepad) {
300 const auto id = GLFW_JOYSTICK_1 + gamepad->GetIdentifier();
301
302 if (glfwJoystickPresent(id)) {
303 int count;
304 const float* axes = glfwGetJoystickAxes(id, &count);
305
306 gamepad->_SetConnectionState(true);
307
308 for (int axis = 0; axis < 6; axis++)
309 gamepad->_SetAxisState((GamepadAxis)axis, axes[axis]);
310
311 const unsigned char* buttons = glfwGetJoystickButtons(id, &count);
312 for (int button = 0; button < 4; button++)
313 gamepad->_SetButtonState((GamepadButton)button, (bool)buttons[button] ? GamepadButtonState::PRESSED : GamepadButtonState::RELEASED);
314 }
315 else {
316 gamepad->_SetConnectionState(false);
317 }
318}
