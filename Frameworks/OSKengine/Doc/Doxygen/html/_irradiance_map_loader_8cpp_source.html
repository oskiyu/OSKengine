95 OwnedPtr<GpuImage> finalImage = Engine::GetRenderer()->GetMemoryAllocator()->CreateCubemapImage(irradianceLayerSize, Format::RGBA32_SFLOAT, GpuImageUsage::TRANSFER_DESTINATION | GpuImageUsage::SAMPLED | GpuImageUsage::CUBEMAP, GpuSharedMemoryType::GPU_ONLY);
96
97 // Material

99 cubemapGenMaterialInstance->GetSlot("global")->FlushUpdate();
100
101 cubemapConvolutionMaterialInstance->GetSlot("global")->SetGpuImage("image", intermediateCubemap.GetPointer());
102 cubemapConvolutionMaterialInstance->GetSlot("global")->FlushUpdate();
103
104 // Renderizado
105 OwnedPtr<ICommandList> cmdList = Engine::GetRenderer()->CreateSingleUseCommandList().GetPointer();
106 cmdList->Start();
107 GenCubemap(intermediateCubemap.GetPointer(), cmdList.GetPointer());
108 ConvoluteCubemap(finalImage.GetPointer(), cmdList.GetPointer());
109 cmdList->Close();
110 Engine::GetRenderer()->SubmitSingleUseCommandList(cmdList.GetPointer());
111
112 output->_SetGpuImage(finalImage);
113 output->_SetOriginalCubemap(intermediateCubemap);

115 stbi_image_free(pixels);
