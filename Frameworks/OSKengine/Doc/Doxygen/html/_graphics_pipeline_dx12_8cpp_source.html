33DXGI_FORMAT GetVertexAttribFormatDx12(const VertexInfo::Entry& entry) {
34 switch (entry.type) {
35
36 case VertexInfo::Entry::Type::INT:
37 if (entry.size == 2 * sizeof(int)) return DXGI_FORMAT_R32G32_SINT;
38 if (entry.size == 3 * sizeof(int)) return DXGI_FORMAT_R32G32B32_SINT;
39 if (entry.size == 4 * sizeof(int)) return DXGI_FORMAT_R32G32B32A32_SINT;
40
41 case VertexInfo::Entry::Type::FLOAT:
42 if (entry.size == 2 * sizeof(float)) return DXGI_FORMAT_R32G32_FLOAT;
43 if (entry.size == 3 * sizeof(float)) return DXGI_FORMAT_R32G32B32_FLOAT;
44 if (entry.size == 4 * sizeof(float)) return DXGI_FORMAT_R32G32B32A32_FLOAT;
45 }
46
47 OSK_ASSERT(false, "Formato de vértice incorecto.");
48 return DXGI_FORMAT_UNKNOWN;
49}

51InputLayoutDescDx12 GetInputLayoutDescDx12(const VertexInfo& info) {
52 InputLayoutDescDx12 output{};
53
54 TSize offset = 0;
55 for (TSize i = 0; i < info.entries.GetSize(); i++) {
56 const VertexInfo::Entry& entry = info.entries.At(i);
57
58 output.stringNames.Insert(entry.name);
59
60 D3D12_INPUT_ELEMENT_DESC desc = {
61 NULL,
62 0,
63 GetVertexAttribFormatDx12(entry),
64 0,
65 offset,
66 D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA,
67 0
68 };
69
70 output.inputElements.Insert(desc);
71
72 offset += info.entries.At(i).size;
73 }
74
75 return output;
76}
