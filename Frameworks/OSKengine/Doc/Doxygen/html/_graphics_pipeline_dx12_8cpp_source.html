32DXGI_FORMAT GetVertexAttribFormatDx12(const VertexInfo::Entry& entry) {
33 switch (entry.type) {
34
35 case VertexInfo::Entry::Type::INT:
36 if (entry.size == 2 * sizeof(int)) return DXGI_FORMAT_R32G32_SINT;
37 if (entry.size == 3 * sizeof(int)) return DXGI_FORMAT_R32G32B32_SINT;
38 if (entry.size == 4 * sizeof(int)) return DXGI_FORMAT_R32G32B32A32_SINT;
39
40 case VertexInfo::Entry::Type::FLOAT:
41 if (entry.size == 2 * sizeof(float)) return DXGI_FORMAT_R32G32_FLOAT;
42 if (entry.size == 3 * sizeof(float)) return DXGI_FORMAT_R32G32B32_FLOAT;
43 if (entry.size == 4 * sizeof(float)) return DXGI_FORMAT_R32G32B32A32_FLOAT;
44 }
45
46 OSK_ASSERT(false, "Formato de vértice incorecto.");
47 return DXGI_FORMAT_UNKNOWN;
48}

50DynamicArray<D3D12_INPUT_ELEMENT_DESC> GetInputLayoutDescDx12(const VertexInfo& info) {
51 DynamicArray<D3D12_INPUT_ELEMENT_DESC> output{};
52
53 TSize offset = 0;
54 for (TSize i = 0; i < info.entries.GetSize(); i++) {
55 const VertexInfo::Entry& entry = info.entries.At(i);
56
57 D3D12_INPUT_ELEMENT_DESC desc = {
58 NULL,
59 0,
60 GetVertexAttribFormatDx12(entry),
61 0,
62 offset,
63 D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA,
64 0
65 };
66
67 output.Insert(desc);
68
69 offset += entry.size;
70 }
71
72 return output;
73}

113ComPtr<IDxcBlob> CompileShaderSpv(const std::string& spirvPath, const VertexInfo& vertexInfo, const std::string& hlslProfile, const MaterialLayout& layout) {
114 const auto bytecode = FileIO::ReadBinaryFromFile(spirvPath);
115
116 // Generación del código HLSL.
117 SpirvToHlsl compilerToHlsl = SpirvToHlsl(bytecode);
118 compilerToHlsl.SetHlslTargetProfile(6, 1);
119 compilerToHlsl.SetVertexAttributesMapping(vertexInfo);
120 compilerToHlsl.SetLayoutMapping(layout);
121
122 const std::string hlslSourceCode = compilerToHlsl.CreateHlsl();
123
124 //Engine::GetLogger()->DebugLog(hlslSourceCode);
125 FileIO::WriteFile("./temp_shader", hlslSourceCode);
126
127 HlslRuntimeCompiler compiler;
128 return compiler.CompileFromFile("./temp_shader", hlslProfile);
129}

131void GraphicsPipelineDx12::Create(const MaterialLayout* materialLayout, IGpu* device, const PipelineCreateInfo& info, const VertexInfo& vertexInfo) {
132 layout = new PipelineLayoutDx12(materialLayout);
133
134 D3D12_GRAPHICS_PIPELINE_STATE_DESC createInfo{};
135
136 auto layoutDesc = GetInputLayoutDescDx12(vertexInfo);
137
138 for (TSize i = 0; i < layoutDesc.GetSize(); i++)
139 layoutDesc.At(i).SemanticName = vertexInfo.entries[i].GetName().c_str();
140
141 D3D12_INPUT_LAYOUT_DESC inputLayout{};
142 inputLayout.NumElements = layoutDesc.GetSize();
143 inputLayout.pInputElementDescs = layoutDesc.GetData();
144
145 createInfo.InputLayout = inputLayout;
146
147 LoadVertexShader(info, vertexInfo, *materialLayout);
148 LoadFragmentShader(info, vertexInfo, *materialLayout);
149
150 if (info.tesselationControlPath != "") {
151 LoadTeselationControlShader(info, vertexInfo, *materialLayout);
152 createInfo.HS = tesselationControlShaderBytecode;
153
154 topologyType = D3D_PRIMITIVE_TOPOLOGY_3_CONTROL_POINT_PATCHLIST;
155 }
156 if (info.tesselationEvaluationPath != "") {
157 LoadTesselationEvaluationShader(info, vertexInfo, *materialLayout);
158 createInfo.DS = tesselationEvaluationShaderBytecode;
159 }
160
161 createInfo.VS = vertexShaderBytecode;
162 createInfo.PS = fragmentShaderBytecode;
163
164 createInfo.RasterizerState = GetRasterizerDesc(info);
165 createInfo.BlendState = GetBlendDesc(info);
166 createInfo.DepthStencilState = GetDepthStencilDesc(info);
167 createInfo.SampleMask = UINT_MAX;
168 if (info.tesselationControlPath != "")
169 createInfo.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_PATCH;
170 else
171 createInfo.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
172 createInfo.NumRenderTargets = 1;
173 createInfo.RTVFormats[0] = DXGI_FORMAT_R8G8B8A8_UNORM;
174 createInfo.DSVFormat = GetFormatDx12(Format::D32S8_SFLOAT_SUINT);
175 createInfo.SampleDesc.Count = 1;
176
177 createInfo.pRootSignature = layout->As<PipelineLayoutDx12>()->GetSignature();
178
179 HRESULT result = device->As<GpuDx12>()->GetDevice()->CreateGraphicsPipelineState(&createInfo, IID_PPV_ARGS(&pipeline));
180 OSK_ASSERT(SUCCEEDED(result), "Error al crear el pipeline. Code: " + std::to_string(result));
181}
