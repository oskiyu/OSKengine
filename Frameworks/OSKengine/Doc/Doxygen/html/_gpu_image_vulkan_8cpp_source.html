36void GpuImageVulkan::CreateVkImage() {
37 bool usesMultipleQueues = EFTraits::HasFlag(GetUsage(), GpuImageUsage::COMPUTE) && GetUsage() != GpuImageUsage::COMPUTE;
38
39 VkImageCreateInfo imageInfo{};
40
41 imageInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
42 imageInfo.imageType = GetVkImageType();
43
44 imageInfo.extent.width = GetSize().X;
45 imageInfo.extent.height = GetSize().Y;
46 imageInfo.extent.depth = GetSize().Z;
47
48 imageInfo.mipLevels = GetMipLevels();
49 imageInfo.arrayLayers = GetNumLayers();
50
51 imageInfo.format = GetFormatVulkan(GetFormat());
52 imageInfo.tiling = VK_IMAGE_TILING_OPTIMAL;
53 imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
54 imageInfo.usage = GetGpuImageUsageVulkan(GetUsage());
55 imageInfo.samples = (VkSampleCountFlagBits)GetNumSamples();
56
57 DynamicArray<uint32_t> queueIndices;
58 if (usesMultipleQueues) {
59 queueIndices.Insert(Engine::GetRenderer()->GetGraphicsCommandQueue()->As<CommandQueueVulkan>()->GetQueueIndex());
60 queueIndices.Insert(Engine::GetRenderer()->GetComputeCommandQueue()->As<CommandQueueVulkan>()->GetQueueIndex());
61
62 if (queueIndices[0] == queueIndices[1])
63 usesMultipleQueues = false;
64 }
65 else if (GetUsage() == GpuImageUsage::COMPUTE) {
66 queueIndices.Insert(Engine::GetRenderer()->GetComputeCommandQueue()->As<CommandQueueVulkan>()->GetQueueIndex());
67 }
68 else {
69 queueIndices.Insert(Engine::GetRenderer()->GetGraphicsCommandQueue()->As<CommandQueueVulkan>()->GetQueueIndex());
70 }
71
72 imageInfo.queueFamilyIndexCount = usesMultipleQueues ? queueIndices.GetSize() : 1;
73 imageInfo.pQueueFamilyIndices = queueIndices.GetData();
74 imageInfo.sharingMode = usesMultipleQueues
75 ? VK_SHARING_MODE_CONCURRENT
76 : VK_SHARING_MODE_EXCLUSIVE;
77
78 // Si la imagen se usar√° como cubemap, debemos especificarlo.
79 imageInfo.flags = EFTraits::HasFlag(GetUsage(), GpuImageUsage::CUBEMAP) ? VK_IMAGE_CREATE_CUBE_COMPATIBLE_BIT : 0;
80
81 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVulkan>()->GetLogicalDevice();
82 VkResult result = vkCreateImage(device, &imageInfo, nullptr, &image);
83 OSK_ASSERT(result == VK_SUCCESS, "No se pudo crear la imagen en la GPU.");
84}

90void GpuImageVulkan::CreateVkSampler(const GpuImageSamplerDesc& samplerDesc) {
91 //Info del sampler.
92 VkSamplerCreateInfo samplerInfo{};
93 samplerInfo.sType = VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO;
94 //Filtro:
95 // VK_FILTER_LINEAR: suavizado.
96 // VK_FILTER_NEAREST: pixelado.
97 samplerInfo.minFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
98 samplerInfo.magFilter = GetFilterTypeVulkan(samplerDesc.filteringType);
99 //AddressMode: como se accede a la imagen con TexCoords fuera de los l√≠mites.
100 samplerInfo.addressModeU = GetAddressModeVulkan(samplerDesc.addressMode);
101 samplerInfo.addressModeV = GetAddressModeVulkan(samplerDesc.addressMode);
102 samplerInfo.addressModeW = GetAddressModeVulkan(samplerDesc.addressMode);
103
104 samplerInfo.anisotropyEnable = VK_TRUE;
105 samplerInfo.maxAnisotropy = 16.0f;
106
107 samplerInfo.borderColor = VK_BORDER_COLOR_INT_OPAQUE_BLACK;
108 if (samplerDesc.addressMode == GpuImageAddressMode::BACKGROUND_WHITE)
109 samplerInfo.borderColor = VK_BORDER_COLOR_INT_OPAQUE_WHITE;
110
111 samplerInfo.unnormalizedCoordinates = VK_FALSE;
112
113 samplerInfo.compareEnable = VK_FALSE;
114 samplerInfo.compareOp = VK_COMPARE_OP_ALWAYS;
115
116 switch (samplerDesc.mipMapMode) {
117 case GpuImageMipmapMode::AUTO:
118 samplerInfo.minLod = 0.0f;
119 samplerInfo.maxLod = static_cast<float>(GetMipLevels());
120 break;
121
122 case GpuImageMipmapMode::CUSTOM:
123 samplerInfo.minLod = static_cast<float>(samplerDesc.minMipLevel);
124 samplerInfo.maxLod = static_cast<float>(samplerDesc.maxMipLevel);
125 break;
126
127 case GpuImageMipmapMode::NONE:
128 samplerInfo.minLod = 0.0f;
129 samplerInfo.maxLod = 0.0f;
130 break;
131 }
132
133 samplerInfo.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
134 samplerInfo.mipLodBias = 0.0f;
135
136 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVulkan>()->GetLogicalDevice();
137 vkCreateSampler(device, &samplerInfo, nullptr, &sampler);
138}

156OwnedPtr<IGpuImageView> GpuImageVulkan::CreateView(SampledChannel channel, SampledArrayType arrayType, TSize baseArrayLevel, TSize layerCount, ViewUsage usage) const {
157 VkImageView view = VK_NULL_HANDLE;
158
159 VkImageViewCreateInfo viewInfo{};
160
161 viewInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
162 viewInfo.image = image;
163 viewInfo.viewType = arrayType == SampledArrayType::SINGLE_LAYER ? GetVkImageViewType() : GetVkImageArrayViewType();
164 viewInfo.format = GetFormatVulkan(GetFormat());
165
166 VkImageAspectFlags aspectMask = 0;
167 if (EFTraits::HasFlag(channel, SampledChannel::COLOR))
168 aspectMask |= VK_IMAGE_ASPECT_COLOR_BIT;
169 if (EFTraits::HasFlag(channel, SampledChannel::DEPTH))
170 aspectMask |= VK_IMAGE_ASPECT_DEPTH_BIT;
171 if (EFTraits::HasFlag(channel, SampledChannel::STENCIL))
172 aspectMask |= VK_IMAGE_ASPECT_STENCIL_BIT;
173
174 viewInfo.subresourceRange.aspectMask = aspectMask;
175
176 viewInfo.subresourceRange.baseMipLevel = 0;
177 viewInfo.subresourceRange.levelCount = usage == ViewUsage::COLOR_TARGET ? 1 : GetMipLevels();
178
179 viewInfo.subresourceRange.baseArrayLayer = baseArrayLevel;
180 viewInfo.subresourceRange.layerCount = layerCount;
181
182 const VkDevice device = Engine::GetRenderer()->GetGpu()->As<GpuVulkan>()->GetLogicalDevice();
183 VkResult result = vkCreateImageView(device, &viewInfo, nullptr, &view);
184 OSK_ASSERT(result == VK_SUCCESS, "No se pudo crear el view de stencil.");
185
186 return new GpuImageViewVulkan(view, channel, arrayType, baseArrayLevel, layerCount, usage);
187}
