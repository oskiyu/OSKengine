125void CommandListDx12::BeginGraphicsRenderpass(DynamicArray<RenderPassImageInfo> colorImages, RenderPassImageInfo depthImage, const Color& color) {
126 DynamicArray<D3D12_CPU_DESCRIPTOR_HANDLE> colorAttachments{};
127 D3D12_CPU_DESCRIPTOR_HANDLE depthStencilDesc{};
128
129 if (currentRenderpassType== RenderpassType::FINAL) {
130 const SwapchainDx12* swapchain = Engine::GetRenderer()->_GetSwapchain()->As<SwapchainDx12>();
131
132 colorAttachments.Insert(swapchain->GetRenderTargetMemory()->GetCPUDescriptorHandleForHeapStart());
133 colorAttachments[0].ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
134 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
135
136 depthStencilDesc = swapchain->GetDepthStencilMemory()->GetCPUDescriptorHandleForHeapStart();
137 depthStencilDesc.ptr += ((SIZE_T)swapchain->GetCurrentFrameIndex()) *
138 Engine::GetRenderer()->As<RendererDx12>()->GetGpu()->As<GpuDx12>()->GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
139 }
140 else {
141 for (const auto& [image, layer] : colorImages)
142 colorAttachments.Insert(image->GetView(SampledChannel::COLOR, SampledArrayType::SINGLE_LAYER, layer, 1, ViewUsage::COLOR_TARGET)
143 ->As<GpuImageViewDx12>()->GetCpuAddress());
144
145 depthStencilDesc = depthImage.targetImage->GetView(SampledChannel::DEPTH | SampledChannel::STENCIL, SampledArrayType::SINGLE_LAYER, depthImage.arrayLevel, 1, ViewUsage::DEPTH_STENCIL_TARGET)
146 ->As<GpuImageViewDx12>()->GetCpuAddress();
147 }
148
149 for (const auto& img : colorImages) {
150 const GpuBarrierInfo prev(GpuBarrierStage::DEFAULT, GpuBarrierAccessStage::DEFAULT);
151 const GpuBarrierInfo next(GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT, GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE);
152 const GpuImageBarrierInfo imgInfo = {
153 .baseLayer = img.arrayLevel,
154 .numLayers = 1,
155 .baseMipLevel = 0,
156 .numMipLevels = 1,
157 };
158
159 SetGpuImageBarrier(img.targetImage, GpuImageLayout::UNDEFINED, GpuImageLayout::COLOR_ATTACHMENT, prev, next, imgInfo);
160 }
161
162 const FLOAT clearValue[] = { color.Red, color.Green, color.Blue, color.Alpha };
163
164 for (auto colorAttachment : colorAttachments)
165 commandList->ClearRenderTargetView(colorAttachment, clearValue, 0, nullptr);
166 commandList->ClearDepthStencilView(depthStencilDesc, D3D12_CLEAR_FLAG_DEPTH, 1.0f, 0, 0, nullptr);
167
168 commandList->OMSetRenderTargets(colorAttachments.GetSize(), colorAttachments.GetData(), FALSE, &depthStencilDesc);
169
170 currentColorImages = std::move(colorImages);
171 currentDepthImage = depthImage;
172}

174void CommandListDx12::EndGraphicsRenderpass() {
175 GpuImageLayout finalLayout = GpuImageLayout::SAMPLED;
176 if (currentRenderpassType == RenderpassType::FINAL)
177 finalLayout = GpuImageLayout::PRESENT;
178
179 for (const auto& img : currentColorImages) {
180 GpuBarrierInfo prevBarrier{};
181 prevBarrier.stage = GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT;
182 prevBarrier.accessStage = GpuBarrierAccessStage::COLOR_ATTACHMENT_WRITE;
183
184 GpuBarrierInfo nextBarrier{};
185 nextBarrier.stage = GpuBarrierStage::COLOR_ATTACHMENT_OUTPUT;
186 nextBarrier.accessStage = GpuBarrierAccessStage::SHADER_READ;
187
188 GpuImageBarrierInfo imgBarrier{};
189 imgBarrier.baseLayer = img.arrayLevel;
190 imgBarrier.numLayers = 1;
191 imgBarrier.baseMipLevel = 0;
192 imgBarrier.numMipLevels = ALL_MIP_LEVELS;
193
194 SetGpuImageBarrier(img.targetImage, GpuImageLayout::COLOR_ATTACHMENT, finalLayout, prevBarrier, nextBarrier, imgBarrier);
195 }
196
197 if (currentRenderpassType != RenderpassType::INTERMEDIATE)
198 currentRenderpassType = RenderpassType::INTERMEDIATE;
199}
