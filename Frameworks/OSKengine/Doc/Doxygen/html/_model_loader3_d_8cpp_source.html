74glm::mat4 GetNodeMatrix(const tinygltf::Node& node) {
75 // Obtenemos el transform del nodo.
76 // Puede estar definido de varias maneras:
77
78 glm::mat4 nodeMatrix = glm::mat4(1.0f);
79
80 // Definido por una posici√≥n.
81 if (node.translation.size() == 3) {
82 nodeMatrix = glm::translate(nodeMatrix, glm::vec3(node.translation[0], node.translation[1], node.translation[2]));
83 }
84 // Definido por una rotaci√≥n.
85 if (node.rotation.size() == 4) {
86 glm::quat quaternion = glm::make_quat(node.rotation.data());
87 nodeMatrix *= glm::mat4(quaternion);
88 }
89 // Definido por una escala.
90 if (node.scale.size() == 3) {
91 nodeMatrix = glm::scale(nodeMatrix, glm::vec3(glm::make_vec3(node.scale.data())));
92 }
93 // Definido por una matriz.
94 if (node.matrix.size() == 16) {
95 nodeMatrix = glm::make_mat4x4(node.matrix.data());
96 }
97
98 return nodeMatrix;
99}

273 auto image = Engine::GetRenderer()->GetMemoryAllocator()->CreateImage({ (unsigned int)originalImage.width, (unsigned int)originalImage.height, 1 },

291 }
