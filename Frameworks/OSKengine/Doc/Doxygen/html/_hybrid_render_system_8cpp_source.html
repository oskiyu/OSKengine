258void HybridRenderSystem::GBufferRenderLoop(GRAPHICS::ICommandList* commandList, ASSETS::ModelType modelType) {
259 IGpuVertexBuffer* previousVertexBuffer = nullptr;
260 IGpuIndexBuffer* previousIndexBuffer = nullptr;
261
262 for (GameObjectIndex obj : GetObjects()) {
263 const ModelComponent3D& model = Engine::GetEcs()->GetComponent<ModelComponent3D>(obj);
264 const Transform3D& transform = Engine::GetEcs()->GetComponent<Transform3D>(obj);
265
266 if (modelType != model.GetModel()->GetType())
267 continue;
268
269 if (modelType == ModelType::STATIC_MESH)
270 commandList->BindMaterial(gbufferMaterial);
271 else
272 commandList->BindMaterial(animatedGbufferMaterial);
273
274 // Actualizamos el modelo 3D, si es necesario.
275 if (previousVertexBuffer != model.GetModel()->GetVertexBuffer()) {
276 commandList->BindVertexBuffer(model.GetModel()->GetVertexBuffer());
277 previousVertexBuffer = model.GetModel()->GetVertexBuffer();
278 }
279 if (previousIndexBuffer != model.GetModel()->GetIndexBuffer()) {
280 commandList->BindIndexBuffer(model.GetModel()->GetIndexBuffer());
281 previousIndexBuffer = model.GetModel()->GetIndexBuffer();
282 }
283
284 if (modelType == ModelType::ANIMATED_MODEL)
285 commandList->BindMaterialSlot(model.GetModel()->GetAnimator()->GetMaterialInstance()->GetSlot("animation"));
286
287 for (TSize i = 0; i < model.GetModel()->GetMeshes().GetSize(); i++) {
288 commandList->BindMaterialSlot(model.GetMeshMaterialInstance(i)->GetSlot("texture"));
289
290 const glm::mat4 previousModelMatrix = previousModelMatrices.ContainsKey(obj)
291 ? previousModelMatrices.Get(obj) : glm::mat4(1.0f);
292
293 const Vector4f materialInfo{
294 model.GetModel()->GetMetadata().meshesMetadata[i].metallicFactor,
295 model.GetModel()->GetMetadata().meshesMetadata[i].roughnessFactor,
296 0.0f,
297 0.0f
298 };
299 struct {
300 glm::mat4 model;
301 glm::mat4 previousModel;
302 Vector4f materialInfo;
303 } modelConsts{
304 .model = transform.GetAsMatrix(),
305 .previousModel = previousModelMatrix,
306 .materialInfo = materialInfo
307 };
308 commandList->PushMaterialConstants("model", modelConsts);
309
310 commandList->DrawSingleMesh(model.GetModel()->GetMeshes()[i].GetFirstIndexId(), model.GetModel()->GetMeshes()[i].GetNumberOfIndices());
311 }
312 }
