22void GpuImageDx12::FillResourceDesc() {
23 resourceDesc.Width = GetSize2D().x;
24 resourceDesc.Height = GetSize2D().y;
25 resourceDesc.DepthOrArraySize = GetNumLayers() == 1 ? GetPhysicalSize().z : GetNumLayers();
26 resourceDesc.Dimension = (D3D12_RESOURCE_DIMENSION)((USize64)GetDimension() + 1);
27 resourceDesc.Alignment = D3D12_DEFAULT_RESOURCE_PLACEMENT_ALIGNMENT;
28 resourceDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
29 resourceDesc.MipLevels = GetMipLevels();
30 resourceDesc.SampleDesc.Count = 1;
31 resourceDesc.SampleDesc.Quality = 0;
32 resourceDesc.Format = GetFormatDx12(GetFormat());
33
34 D3D12_RESOURCE_FLAGS flags = D3D12_RESOURCE_FLAG_NONE;
35 if (EFTraits::HasFlag(GetUsage(), GpuImageUsage::COLOR))
36 flags |= D3D12_RESOURCE_FLAG_ALLOW_RENDER_TARGET;
37 if (EFTraits::HasFlag(GetUsage(), GpuImageUsage::STENCIL))
38 flags |= D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
39 if (EFTraits::HasFlag(GetUsage(), GpuImageUsage::DEPTH))
40 flags |= D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
41 if (EFTraits::HasFlag(GetUsage(), GpuImageUsage::COMPUTE))
42 flags |= D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS;
43
44 resourceDesc.Flags = flags;
