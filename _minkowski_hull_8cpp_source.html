23void MinkowskiHull::BuildFullHull(const ConvexVolume& first, const ConvexVolume& second) {
24 const auto& firstVertices = first.GetVertices();
25 const auto& secondVertices = second.GetVertices();
26
27 const auto& firstAxes = first.GetAxes();
28 const auto& secondAxes = second.GetAxes();
29
30 const auto& firstFaces = first.GetFaceIndices();
31 const auto& secondFaces = second.GetFaceIndices();
32
33 float minDistance = std::numeric_limits<float>::max();
34
35 m_faces.Insert({ 0, 1, 2 });
36 m_vertices.InsertAll({ {},{},{} });
37
38 for (UIndex64 i = 0; i < firstAxes.GetSize(); i++) {
39 const auto& axis = firstAxes[i];
40 const auto& refVertex = firstVertices[firstFaces[i][0]];
41
42 const auto secondSupport = second.GetSupport(-axis);
43
44 const float positiveProjection = refVertex.Dot(axis);
45 const float negativeProjection = secondSupport.point.Dot(axis);
46
47 const float minkowskiProjection = positiveProjection - negativeProjection;
48
49 if (minkowskiProjection < minDistance) {
50 minDistance = minkowskiProjection;
51
52 m_mtv = axis * minkowskiProjection;
53
54 m_vertices[0].originalVertexIdA = firstFaces[i][0];
55 m_vertices[0].originalVertexIdB = secondSupport.originalVertexId;
56
57 m_vertices[1].originalVertexIdA = firstFaces[i][1];
58 m_vertices[1].originalVertexIdB = secondSupport.originalVertexId;
59
60 m_vertices[2].originalVertexIdA = firstFaces[i][2];
61 m_vertices[2].originalVertexIdB = secondSupport.originalVertexId;
62
63 m_referenceVolume = Volume::A;
64 }
65 }
66
67 for (UIndex64 i = 0; i < secondAxes.GetSize(); i++) {
68 const auto& axis = secondAxes[i];
69 const auto& refVertex = secondVertices[secondFaces[i][0]];
70
71 const auto firstSupport = first.GetSupport(-axis);
72
73 const float positiveProjection = refVertex.Dot(axis);
74 const float negativeProjection = firstSupport.point.Dot(axis);
75
76 const float minkowskiProjection = positiveProjection - negativeProjection;
77
78 if (minkowskiProjection < minDistance) {
79 minDistance = minkowskiProjection;
80
81 m_mtv = -axis * minkowskiProjection;
82
83 m_vertices[0].originalVertexIdA = firstSupport.originalVertexId;
84 m_vertices[0].originalVertexIdB = secondFaces[i][0];
85
86 m_vertices[1].originalVertexIdA = firstSupport.originalVertexId;
87 m_vertices[1].originalVertexIdB = secondFaces[i][1];
88
89 m_vertices[2].originalVertexIdA = firstSupport.originalVertexId;
90 m_vertices[2].originalVertexIdB = secondFaces[i][2];
91
92 m_referenceVolume = Volume::B;
93 }
94 }

125void Polytope::Expand(const ConvexVolume& first, const ConvexVolume& second) {
126#define OSK_EPA_DEBUG
127 constexpr static USize64 numIterations = 24;
128
129 UIndex64 iteration = 0;
130 for (; iteration < numIterations; iteration++) {
131 Update();
132
133 if (!m_surpassesMinDepth) {
134 break;
135 }
136
137 if (m_earlyExit) {
138#ifdef OSK_EPA_DEBUG
139 Engine::GetLogger()->InfoLog("\tSALIDA: m_earlyExit (NaN)");
140#endif
141
142 break;
143 }
144
145
146 const double currentDistance = m_distances[m_minDistanceIndex];
147 const Vector3d currentNormal = m_normals[m_minDistanceIndex];
148
149
150 // Si la distancia está ya por debajo de un número bajo,
151 // lo consideramos válido.
152 if (currentDistance < MinDistanceThreshold) {
153#ifdef OSK_EPA_DEBUG
154 Engine::GetLogger()->InfoLog("\tSALIDA: currentDistance < MinDistanceThreshold");
155#endif
156 m_surpassesMinDepth = false;
157
158 break;
159 }
160
161#ifdef OSK_EPA_DEBUG
162 Engine::GetLogger()->DebugLog(std::format("Iteración {}", iteration));
163 Engine::GetLogger()->DebugLog(std::format("\tNormal mínimo: {:.3f} {:.3f} {:.3f}", currentNormal.x, currentNormal.y, currentNormal.z));
164 Engine::GetLogger()->DebugLog(std::format("\tDistancia mínima: {:.3f}", m_distances[m_minDistanceIndex]));
165#endif
166
167 // Punto del hull más hacia afuera en la dirección
168 // actual (normal de la cara con distancia mínima).
169 MinkowskiSupport support = GetMinkowskiSupport(first, second, currentNormal);
170
171 // Punto actual más hacia afuera en la dirección actual.
172 const Vector3d furthestPoint = this->GetSupport(currentNormal);
173
174#ifdef OSK_EPA_DEBUG
175 Engine::GetLogger()->DebugLog(std::format("\tSoporte teórico: {:.3f} {:.3f} {:.3f}", support.point.x, support.point.y, support.point.z));
176 Engine::GetLogger()->DebugLog(std::format("\tSoporte actual: {:.3f} {:.3f} {:.3f}", furthestPoint.x, furthestPoint.y, furthestPoint.z));
177#endif
178
179
180 // Distancia del soporte respecto al origen en la dirección actual.
181 const float supportProjection = support.point.Dot(currentNormal);
182
183#ifdef OSK_EPA_DEBUG
184 Engine::GetLogger()->DebugLog(std::format("\tDistancia teórica (proj): {:.3f}", supportProjection));
185#endif
186
187 // Diferencia entre las distancias nueva y actual.
188 const float newDistanceDiff = glm::abs(currentDistance - supportProjection);
189 bool alreadyContainsPoint = false;
190 for (const auto& vertex : m_vertices) {
191 if (vertex.point.Equals(support.point, MinDistanceThreshold)) {
192 alreadyContainsPoint = true;
193 break;
194 }
195 }
196
197
198 if (newDistanceDiff < MinDistanceThreshold) {
199#ifdef OSK_EPA_DEBUG
200 Engine::GetLogger()->InfoLog("\tSALIDA: newDistanceDiff < MinDistanceThreshold");
201#endif
202
203 break;
204 }
205
206 if (alreadyContainsPoint) {
207#ifdef OSK_EPA_DEBUG
208 Engine::GetLogger()->InfoLog("\tSALIDA: alreadyContainsPoint");
209#endif
210
211 break;
212 }
213
214
215 // Aristas que forma el borde de las caras que están
216 // mirando hacia el nuevo punto.
217 DynamicArray<Edge> uniqueEdges;
218
219 // ELiminamos las caras que estén mirando hacia el nuevo punto.
220 for (int64_t faceId = 0; faceId < m_faces.GetSize(); faceId++) {
221 const auto& face = m_faces[faceId];
222 const auto vertexIndex = face[0];
223
224 const Vector3d faceVertex = m_vertices[vertexIndex].point;
225 const Vector3d faceNormal = m_normals[faceId];
226
227 const double faceVertexDot = faceVertex.Dot(faceNormal);
228 const double supportDot = support.point.Dot(faceNormal);
229
230 const bool isSameDirection = supportDot > faceVertexDot;
231
232 // Si la cara no apunta hacia el nuevo punto,
233 // no hace falta hacer nada.
234 if (!isSameDirection) {
235 continue;
236 }
237
238 AddUniqueEdge(&uniqueEdges, face[0], face[1]);
239 AddUniqueEdge(&uniqueEdges, face[1], face[2]);
240 AddUniqueEdge(&uniqueEdges, face[2], face[0]);
241
242 // Eliminamos la cara.
243 m_faces[faceId] = m_faces.Pop();
244 m_normals[faceId] = m_normals.Pop();
245 m_distances[faceId] = m_distances.Pop();
246
247 if (faceId == m_minDistanceIndex) {
248 m_minDistanceIndex = std::numeric_limits<UIndex64>::max();
249 }
250
251 faceId--;
252 }
253
254 // Nuevas caras construidas a partir de las aristas
255 // y el nuevo vértice.
256 m_faces.InsertAll(BuildNewFaces(uniqueEdges, &m_vertices, support));
257 }
258
259#ifdef OSK_EPA_DEBUG
260 Engine::GetLogger()->InfoLog(std::format("\tNum. final de iteraciones: {}", iteration));
261 if (GetMtv().GetLenght() > 0.08f) {
262 Engine::GetLogger()->InfoLog("\n\n\n\n\n\n!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!");
263 }
264#endif // OSK_EPA_DEBUG
265}

267void Polytope::Update() {
268 // Construir nuevos normales:
269 for (UIndex64 i = m_normals.GetSize(); i < m_faces.GetSize(); i++) {
270 const auto& face = m_faces[i];
271
272 const Vector3d& a = m_vertices[face[0]].point;
273 const Vector3d& b = m_vertices[face[1]].point;
274 const Vector3d& c = m_vertices[face[2]].point;
275
276 const Vector3d ab = b - a;
277 const Vector3d ac = c - a;
278
279 m_normals.Insert(ab.Cross(ac).GetNormalized());
280
281 if (m_normals[i].IsNaN()) {
282 m_earlyExit = true;
283 return;
284 }
285
286 m_distances.Insert(a.Dot(m_normals[i]));
287
288 OSK_ASSERT(!m_normals[i].IsNaN(), InvalidObjectStateException("Vector NAN"));
289
290 if (m_distances[i] < 0.0f) {
291 m_normals[i] *= -1.0f;
292 m_distances[i] *= -1.0f;
293 }
294
295 }
296
297 double minDistance = std::numeric_limits<double>::max();
298 for (UIndex64 i = 0; i < m_distances.GetSize(); i++) {
299 if (m_distances[i] < minDistance) {
300 minDistance = m_distances[i];
301 m_minDistanceIndex = i;
302 }
303 }
304}

306void Polytope::AddUniqueEdge(DynamicArray<Edge>* edges, UIndex64 vertexA, UIndex64 vertexB) {
307 const Edge normalEdge = { .a = vertexA, .b = vertexB };
308 const Edge inversedEdge = { .a = vertexB, .b = vertexA };
309
310 const UIndex64 normalEdgeIndex = edges->GetIndexOf(normalEdge);
311 const UIndex64 reveresedEdgeIndex = edges->GetIndexOf(inversedEdge);
312
313 if (normalEdgeIndex != DynamicArray<Polytope::Edge>::NotFoundIndex) {
314 edges->RemoveAt(normalEdgeIndex);
315 }
316 if (reveresedEdgeIndex != DynamicArray<Polytope::Edge>::NotFoundIndex) {
317 edges->RemoveAt(reveresedEdgeIndex);
318 }
319
320 if (normalEdgeIndex == DynamicArray<Polytope::Edge>::NotFoundIndex && reveresedEdgeIndex == DynamicArray<Polytope::Edge>::NotFoundIndex) {
321 edges->Insert({ vertexA, vertexB });
322 }
323}

339Vector3f Polytope::GetSupport(const Vector3f& normalizedDirection) const {
340 Vector3f output = Vector3f::Zero;
341 double maxProjection = std::numeric_limits<float>::lowest();
342
343 for (const MinkowskiSupport& vertex : m_vertices) {
344 const float iProjection = vertex.point.Dot(normalizedDirection);
345
346 if (iProjection > maxProjection) {
347 maxProjection = iProjection;
348 output = vertex.point;
